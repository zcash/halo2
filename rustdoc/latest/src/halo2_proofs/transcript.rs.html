13/// Prefix to a prover's message soliciting a challenge
14const BLAKE2B_PREFIX_CHALLENGE: u8 = 0;
15
16/// Prefix to a prover's message containing a curve point
17const BLAKE2B_PREFIX_POINT: u8 = 1;
18
19/// Prefix to a prover's message containing a scalar
20const BLAKE2B_PREFIX_SCALAR: u8 = 2;
21
22/// Generic transcript view (from either the prover or verifier's perspective)
23pub trait Transcript<C: CurveAffine, E: EncodedChallenge<C>> {
24    /// Squeeze an encoded verifier challenge from the transcript.
25    fn squeeze_challenge(&mut self) -> E;
26
27    /// Squeeze a typed challenge (in the scalar field) from the transcript.
28    fn squeeze_challenge_scalar<T>(&mut self) -> ChallengeScalar<C, T> {
29        ChallengeScalar {
30            inner: self.squeeze_challenge().get_scalar(),
31            _marker: PhantomData,
32        }
33    }
34
35    /// Writing the point to the transcript without writing it to the proof,
36    /// treating it as a common input.
37    fn common_point(&mut self, point: C) -> io::Result<()>;
38
39    /// Writing the scalar to the transcript without writing it to the proof,
40    /// treating it as a common input.
41    fn common_scalar(&mut self, scalar: C::Scalar) -> io::Result<()>;
42}
43
44/// Transcript view from the perspective of a verifier that has access to an
45/// input stream of data from the prover to the verifier.
46pub trait TranscriptRead<C: CurveAffine, E: EncodedChallenge<C>>: Transcript<C, E> {
47    /// Read a curve point from the prover.
48    fn read_point(&mut self) -> io::Result<C>;
49
50    /// Read a curve scalar from the prover.
51    fn read_scalar(&mut self) -> io::Result<C::Scalar>;
52}
53
54/// Transcript view from the perspective of a prover that has access to an
55/// output stream of messages from the prover to the verifier.
56pub trait TranscriptWrite<C: CurveAffine, E: EncodedChallenge<C>>: Transcript<C, E> {
57    /// Write a curve point to the proof and the transcript.
58    fn write_point(&mut self, point: C) -> io::Result<()>;
59
60    /// Write a scalar to the proof and the transcript.
61    fn write_scalar(&mut self, scalar: C::Scalar) -> io::Result<()>;
62}

64/// We will replace BLAKE2b with an algebraic hash function in a later version.
65#[derive(Debug, Clone)]
66pub struct Blake2bRead<R: Read, C: CurveAffine, E: EncodedChallenge<C>> {
67    state: Blake2bState,
68    reader: R,
69    _marker: PhantomData<(C, E)>,
70}
71
72impl<R: Read, C: CurveAffine, E: EncodedChallenge<C>> Blake2bRead<R, C, E> {
73    /// Initialize a transcript given an input buffer.
74    pub fn init(reader: R) -> Self {
75        Blake2bRead {
76            state: Blake2bParams::new()
77                .hash_length(64)
78                .personal(b"Halo2-Transcript")
79                .to_state(),
80            reader,
81            _marker: PhantomData,
82        }
83    }
84}

86impl<R: Read, C: CurveAffine> TranscriptRead<C, Challenge255<C>>
87    for Blake2bRead<R, C, Challenge255<C>>
88where
89    C::Scalar: FromUniformBytes<64>,
90{
91    fn read_point(&mut self) -> io::Result<C> {
92        let mut compressed = C::Repr::default();
93        self.reader.read_exact(compressed.as_mut())?;
94        let point: C = Option::from(C::from_bytes(&compressed)).ok_or_else(|| {
95            io::Error::new(io::ErrorKind::Other, "invalid point encoding in proof")
96        })?;
97        self.common_point(point)?;
98
99        Ok(point)
100    }
101
102    fn read_scalar(&mut self) -> io::Result<C::Scalar> {
103        let mut data = <C::Scalar as PrimeField>::Repr::default();
104        self.reader.read_exact(data.as_mut())?;
105        let scalar: C::Scalar = Option::from(C::Scalar::from_repr(data)).ok_or_else(|| {
106            io::Error::new(
107                io::ErrorKind::Other,
108                "invalid field element encoding in proof",
109            )
110        })?;
111        self.common_scalar(scalar)?;
112
113        Ok(scalar)
114    }
115}

117impl<R: Read, C: CurveAffine> Transcript<C, Challenge255<C>> for Blake2bRead<R, C, Challenge255<C>>
118where
119    C::Scalar: FromUniformBytes<64>,
120{
121    fn squeeze_challenge(&mut self) -> Challenge255<C> {
122        self.state.update(&[BLAKE2B_PREFIX_CHALLENGE]);
123        let hasher = self.state.clone();
124        let result: [u8; 64] = hasher.finalize().as_bytes().try_into().unwrap();
125        Challenge255::<C>::new(&result)
126    }
127
128    fn common_point(&mut self, point: C) -> io::Result<()> {
129        self.state.update(&[BLAKE2B_PREFIX_POINT]);
130        let coords: Coordinates<C> = Option::from(point.coordinates()).ok_or_else(|| {
131            io::Error::new(
132                io::ErrorKind::Other,
133                "cannot write points at infinity to the transcript",
134            )
135        })?;
136        self.state.update(coords.x().to_repr().as_ref());
137        self.state.update(coords.y().to_repr().as_ref());
138
139        Ok(())
140    }
141
142    fn common_scalar(&mut self, scalar: C::Scalar) -> io::Result<()> {
143        self.state.update(&[BLAKE2B_PREFIX_SCALAR]);
144        self.state.update(scalar.to_repr().as_ref());
145
146        Ok(())
147    }
148}

150/// We will replace BLAKE2b with an algebraic hash function in a later version.
151#[derive(Debug, Clone)]
152pub struct Blake2bWrite<W: Write, C: CurveAffine, E: EncodedChallenge<C>> {
153    state: Blake2bState,
154    writer: W,
155    _marker: PhantomData<(C, E)>,
156}
157
158impl<W: Write, C: CurveAffine, E: EncodedChallenge<C>> Blake2bWrite<W, C, E> {
159    /// Initialize a transcript given an output buffer.
160    pub fn init(writer: W) -> Self {
161        Blake2bWrite {
162            state: Blake2bParams::new()
163                .hash_length(64)
164                .personal(b"Halo2-Transcript")
165                .to_state(),
166            writer,
167            _marker: PhantomData,
168        }
169    }
170
171    /// Conclude the interaction and return the output buffer (writer).
172    pub fn finalize(self) -> W {
173        // TODO: handle outstanding scalars? see issue #138
174        self.writer
175    }
176}

178impl<W: Write, C: CurveAffine> TranscriptWrite<C, Challenge255<C>>
179    for Blake2bWrite<W, C, Challenge255<C>>
180where
181    C::Scalar: FromUniformBytes<64>,
182{
183    fn write_point(&mut self, point: C) -> io::Result<()> {
184        self.common_point(point)?;
185        let compressed = point.to_bytes();
186        self.writer.write_all(compressed.as_ref())
187    }
188    fn write_scalar(&mut self, scalar: C::Scalar) -> io::Result<()> {
189        self.common_scalar(scalar)?;
190        let data = scalar.to_repr();
191        self.writer.write_all(data.as_ref())
192    }
193}

229/// The scalar representation of a verifier challenge.
230///
231/// The `Type` type can be used to scope the challenge to a specific context, or
232/// set to `()` if no context is required.
233#[derive(Copy, Clone, Debug)]
234pub struct ChallengeScalar<C: CurveAffine, T> {
235    inner: C::Scalar,
236    _marker: PhantomData<T>,
237}
238
239impl<C: CurveAffine, T> std::ops::Deref for ChallengeScalar<C, T> {
240    type Target = C::Scalar;
241
242    fn deref(&self) -> &Self::Target {
243        &self.inner
244    }
245}

247/// `EncodedChallenge<C>` defines a challenge encoding with a [`Self::Input`]
248/// that is used to derive the challenge encoding and `get_challenge` obtains
249/// the _real_ `C::Scalar` that the challenge encoding represents.
250pub trait EncodedChallenge<C: CurveAffine> {
251    /// The Input type used to derive the challenge encoding. For example,
252    /// an input from the Poseidon hash would be a base field element;
253    /// an input from the Blake2b hash would be a [u8; 64].
254    type Input;
255
256    /// Get an encoded challenge from a given input challenge.
257    fn new(challenge_input: &Self::Input) -> Self;
258
259    /// Get a scalar field element from an encoded challenge.
260    fn get_scalar(&self) -> C::Scalar;
261
262    /// Cast an encoded challenge as a typed `ChallengeScalar`.
263    fn as_challenge_scalar<T>(&self) -> ChallengeScalar<C, T> {
264        ChallengeScalar {
265            inner: self.get_scalar(),
266            _marker: PhantomData,
267        }
268    }
269}

271/// A 255-bit challenge.
272#[derive(Copy, Clone, Debug)]
273pub struct Challenge255<C: CurveAffine>([u8; 32], PhantomData<C>);
274
275impl<C: CurveAffine> std::ops::Deref for Challenge255<C> {
276    type Target = [u8; 32];
277
278    fn deref(&self) -> &Self::Target {
279        &self.0
280    }
281}
282
283impl<C: CurveAffine> EncodedChallenge<C> for Challenge255<C>
284where
285    C::Scalar: FromUniformBytes<64>,
286{
287    type Input = [u8; 64];
288
289    fn new(challenge_input: &[u8; 64]) -> Self {
290        Challenge255(
291            C::Scalar::from_uniform_bytes(challenge_input)
292                .to_repr()
293                .as_ref()
294                .try_into()
295                .expect("Scalar fits into 256 bits"),
296            PhantomData,
297        )
298    }
299    fn get_scalar(&self) -> C::Scalar {
300        let mut repr = <C::Scalar as PrimeField>::Repr::default();
301        repr.as_mut().copy_from_slice(&self.0);
302        C::Scalar::from_repr(repr).unwrap()
303    }
304}
