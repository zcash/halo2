23/// A column with an index and type
24#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
25pub struct Column<C: ColumnType> {
26    index: usize,
27    column_type: C,
28}
29
30impl<C: ColumnType> Column<C> {
31    #[cfg(test)]
32    pub(crate) fn new(index: usize, column_type: C) -> Self {
33        Column { index, column_type }
34    }
35
36    pub(crate) fn index(&self) -> usize {
37        self.index
38    }
39
40    /// Type of this column.
41    pub fn column_type(&self) -> &C {
42        &self.column_type
43    }
44}
45
46impl<C: ColumnType> Ord for Column<C> {
47    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
48        // This ordering is consensus-critical! The layouters rely on deterministic column
49        // orderings.
50        match self.column_type.into().cmp(&other.column_type.into()) {
51            // Indices are assigned within column types.
52            std::cmp::Ordering::Equal => self.index.cmp(&other.index),
53            order => order,
54        }
55    }
56}
57
58impl<C: ColumnType> PartialOrd for Column<C> {
59    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
60        Some(self.cmp(other))
61    }
62}

76/// An enum over the Advice, Fixed, Instance structs
77#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
78pub enum Any {
79    /// An Advice variant
80    Advice,
81    /// A Fixed variant
82    Fixed,
83    /// An Instance variant
84    Instance,
85}
86
87impl Ord for Any {
88    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
89        // This ordering is consensus-critical! The layouters rely on deterministic column
90        // orderings.
91        match (self, other) {
92            (Any::Instance, Any::Instance)
93            | (Any::Advice, Any::Advice)
94            | (Any::Fixed, Any::Fixed) => std::cmp::Ordering::Equal,
95            // Across column types, sort Instance < Advice < Fixed.
96            (Any::Instance, Any::Advice)
97            | (Any::Advice, Any::Fixed)
98            | (Any::Instance, Any::Fixed) => std::cmp::Ordering::Less,
99            (Any::Fixed, Any::Instance)
100            | (Any::Fixed, Any::Advice)
101            | (Any::Advice, Any::Instance) => std::cmp::Ordering::Greater,
102        }
103    }
104}
105
106impl PartialOrd for Any {
107    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
108        Some(self.cmp(other))
109    }
110}

204/// A selector, representing a fixed boolean value per row of the circuit.
205///
206/// Selectors can be used to conditionally enable (portions of) gates:
207/// ```
208/// use halo2_proofs::poly::Rotation;
209/// # use halo2_proofs::pasta::Fp;
210/// # use halo2_proofs::plonk::ConstraintSystem;
211///
212/// # let mut meta = ConstraintSystem::<Fp>::default();
213/// let a = meta.advice_column();
214/// let b = meta.advice_column();
215/// let s = meta.selector();
216///
217/// meta.create_gate("foo", |meta| {
218///     let a = meta.query_advice(a, Rotation::prev());
219///     let b = meta.query_advice(b, Rotation::cur());
220///     let s = meta.query_selector(s);
221///
222///     // On rows where the selector is enabled, a is constrained to equal b.
223///     // On rows where the selector is disabled, a and b can take any value.
224///     vec![s * (a - b)]
225/// });
226/// ```
227///
228/// Selectors are disabled on all rows by default, and must be explicitly enabled on each
229/// row when required:
230/// ```
231/// use group::ff::Field;
232/// use halo2_proofs::{
233///     circuit::{Chip, Layouter, Value},
234///     plonk::{Advice, Column, Error, Selector},
235/// };
236/// # use halo2_proofs::plonk::Fixed;
237///
238/// struct Config {
239///     a: Column<Advice>,
240///     b: Column<Advice>,
241///     s: Selector,
242/// }
243///
244/// fn circuit_logic<F: Field, C: Chip<F>>(chip: C, mut layouter: impl Layouter<F>) -> Result<(), Error> {
245///     let config = chip.config();
246///     # let config: Config = todo!();
247///     layouter.assign_region(|| "bar", |mut region| {
248///         region.assign_advice(|| "a", config.a, 0, || Value::known(F::ONE))?;
249///         region.assign_advice(|| "b", config.b, 1, || Value::known(F::ONE))?;
250///         config.s.enable(&mut region, 1)
251///     })?;
252///     Ok(())
253/// }
254/// ```
255#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
256pub struct Selector(pub(crate) usize, bool);
257
258impl Selector {
259    /// Enable this selector at the given offset within the given region.
260    pub fn enable<F: Field>(&self, region: &mut Region<F>, offset: usize) -> Result<(), Error> {
261        region.enable_selector(|| "", self, offset)
262    }
263
264    /// Is this selector "simple"? Simple selectors can only be multiplied
265    /// by expressions that contain no other simple selectors.
266    pub fn is_simple(&self) -> bool {
267        self.1
268    }
269}

340    /// Creates a new region and enters into it.

342    /// Panics if we are currently in a region (if `exit_region` was not called).

440/// A floor planning strategy for a circuit.
441///
442/// The floor planner is chip-agnostic and applies its strategy to the circuit it is used
443/// within.
444pub trait FloorPlanner {
445    /// Given the provided `cs`, synthesize the given circuit.
446    ///
447    /// `constants` is the list of fixed columns that the layouter may use to assign
448    /// global constant values. These columns will all have been equality-enabled.
449    ///
450    /// Internally, a floor planner will perform the following operations:
451    /// - Instantiate a [`Layouter`] for this floor planner.
452    /// - Perform any necessary setup or measurement tasks, which may involve one or more
453    ///   calls to `Circuit::default().synthesize(config, &mut layouter)`.
454    /// - Call `circuit.synthesize(config, &mut layouter)` exactly once.
455    fn synthesize<F: Field, CS: Assignment<F>, C: Circuit<F>>(
456        cs: &mut CS,
457        circuit: &C,
458        config: C::Config,
459        constants: Vec<Column<Fixed>>,
460    ) -> Result<(), Error>;
461}

463/// This is a trait that circuits provide implementations for so that the
464/// backend prover can ask the circuit to synthesize using some given
465/// [`ConstraintSystem`] implementation.
466pub trait Circuit<F: Field> {
467    /// This is a configuration object that stores things like columns.
468    type Config: Clone;
469    /// The floor planner used for this circuit. This is an associated type of the
470    /// `Circuit` trait because its behaviour is circuit-critical.
471    type FloorPlanner: FloorPlanner;
472
473    /// Returns a copy of this circuit with no witness values (i.e. all witnesses set to
474    /// `None`). For most circuits, this will be equal to `Self::default()`.
475    fn without_witnesses(&self) -> Self;
476
477    /// The circuit is given an opportunity to describe the exact gate
478    /// arrangement, column arrangement, etc.
479    fn configure(meta: &mut ConstraintSystem<F>) -> Self::Config;
480
481    /// Given the provided `cs`, synthesize the circuit. The concrete type of
482    /// the caller will be different depending on the context, and they may or
483    /// may not expect to have a witness present.
484    fn synthesize(&self, config: Self::Config, layouter: impl Layouter<F>) -> Result<(), Error>;
485}

487/// Low-degree expression representing an identity that must hold over the committed columns.
488#[derive(Clone)]
489pub enum Expression<F> {
490    /// This is a constant polynomial
491    Constant(F),
492    /// This is a virtual selector
493    Selector(Selector),
494    /// This is a fixed column queried at a certain relative location
495    Fixed(FixedQuery),
496    /// This is an advice (witness) column queried at a certain relative location
497    Advice(AdviceQuery),
498    /// This is an instance (external) column queried at a certain relative location
499    Instance(InstanceQuery),
500    /// This is a negated polynomial
501    Negated(Box<Expression<F>>),
502    /// This is the sum of two polynomials
503    Sum(Box<Expression<F>>, Box<Expression<F>>),
504    /// This is the product of two polynomials
505    Product(Box<Expression<F>>, Box<Expression<F>>),
506    /// This is a scaled polynomial
507    Scaled(Box<Expression<F>>, F),
508}
509
510impl<F: Field> Expression<F> {
511    /// Evaluate the polynomial using the provided closures to perform the
512    /// operations.
513    #[allow(clippy::too_many_arguments)]
514    pub fn evaluate<T>(
515        &self,
516        constant: &impl Fn(F) -> T,
517        selector_column: &impl Fn(Selector) -> T,
518        fixed_column: &impl Fn(FixedQuery) -> T,
519        advice_column: &impl Fn(AdviceQuery) -> T,
520        instance_column: &impl Fn(InstanceQuery) -> T,
521        negated: &impl Fn(T) -> T,
522        sum: &impl Fn(T, T) -> T,
523        product: &impl Fn(T, T) -> T,
524        scaled: &impl Fn(T, F) -> T,
525    ) -> T {
526        match self {
527            Expression::Constant(scalar) => constant(*scalar),
528            Expression::Selector(selector) => selector_column(*selector),
529            Expression::Fixed(query) => fixed_column(*query),
530            Expression::Advice(query) => advice_column(*query),
531            Expression::Instance(query) => instance_column(*query),
532            Expression::Negated(a) => {
533                let a = a.evaluate(
534                    constant,
535                    selector_column,
536                    fixed_column,
537                    advice_column,
538                    instance_column,
539                    negated,
540                    sum,
541                    product,
542                    scaled,
543                );
544                negated(a)
545            }
546            Expression::Sum(a, b) => {
547                let a = a.evaluate(
548                    constant,
549                    selector_column,
550                    fixed_column,
551                    advice_column,
552                    instance_column,
553                    negated,
554                    sum,
555                    product,
556                    scaled,
557                );
558                let b = b.evaluate(
559                    constant,
560                    selector_column,
561                    fixed_column,
562                    advice_column,
563                    instance_column,
564                    negated,
565                    sum,
566                    product,
567                    scaled,
568                );
569                sum(a, b)
570            }
571            Expression::Product(a, b) => {
572                let a = a.evaluate(
573                    constant,
574                    selector_column,
575                    fixed_column,
576                    advice_column,
577                    instance_column,
578                    negated,
579                    sum,
580                    product,
581                    scaled,
582                );
583                let b = b.evaluate(
584                    constant,
585                    selector_column,
586                    fixed_column,
587                    advice_column,
588                    instance_column,
589                    negated,
590                    sum,
591                    product,
592                    scaled,
593                );
594                product(a, b)
595            }
596            Expression::Scaled(a, f) => {
597                let a = a.evaluate(
598                    constant,
599                    selector_column,
600                    fixed_column,
601                    advice_column,
602                    instance_column,
603                    negated,
604                    sum,
605                    product,
606                    scaled,
607                );
608                scaled(a, *f)
609            }
610        }
611    }
612
613    /// Compute the degree of this polynomial
614    pub fn degree(&self) -> usize {
615        match self {
616            Expression::Constant(_) => 0,
617            Expression::Selector(_) => 1,
618            Expression::Fixed { .. } => 1,
619            Expression::Advice { .. } => 1,
620            Expression::Instance { .. } => 1,
621            Expression::Negated(poly) => poly.degree(),
622            Expression::Sum(a, b) => max(a.degree(), b.degree()),
623            Expression::Product(a, b) => a.degree() + b.degree(),
624            Expression::Scaled(poly, _) => poly.degree(),
625        }
626    }
627
628    /// Square this expression.
629    pub fn square(self) -> Self {
630        self.clone() * self
631    }
632
633    /// Returns whether or not this expression contains a simple `Selector`.
634    fn contains_simple_selector(&self) -> bool {
635        self.evaluate(
636            &|_| false,
637            &|selector| selector.is_simple(),
638            &|_| false,
639            &|_| false,
640            &|_| false,
641            &|a| a,
642            &|a, b| a || b,
643            &|a, b| a || b,
644            &|a, _| a,
645        )
646    }
647
648    /// Extracts a simple selector from this gate, if present
649    fn extract_simple_selector(&self) -> Option<Selector> {
650        let op = |a, b| match (a, b) {
651            (Some(a), None) | (None, Some(a)) => Some(a),
652            (Some(_), Some(_)) => panic!("two simple selectors cannot be in the same expression"),
653            _ => None,
654        };
655
656        self.evaluate(
657            &|_| None,
658            &|selector| {
659                if selector.is_simple() {
660                    Some(selector)
661                } else {
662                    None
663                }
664            },
665            &|_| None,
666            &|_| None,
667            &|_| None,
668            &|a| a,
669            &op,
670            &op,
671            &|a, _| a,
672        )
673    }
674}

676impl<F: std::fmt::Debug> std::fmt::Debug for Expression<F> {
677    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
678        match self {
679            Expression::Constant(scalar) => f.debug_tuple("Constant").field(scalar).finish(),
680            Expression::Selector(selector) => f.debug_tuple("Selector").field(selector).finish(),
681            // Skip enum variant and print query struct directly to maintain backwards compatibility.
682            Expression::Fixed(FixedQuery {
683                index,
684                column_index,
685                rotation,
686            }) => f
687                .debug_struct("Fixed")
688                .field("query_index", index)
689                .field("column_index", column_index)
690                .field("rotation", rotation)
691                .finish(),
692            Expression::Advice(AdviceQuery {
693                index,
694                column_index,
695                rotation,
696            }) => f
697                .debug_struct("Advice")
698                .field("query_index", index)
699                .field("column_index", column_index)
700                .field("rotation", rotation)
701                .finish(),
702            Expression::Instance(InstanceQuery {
703                index,
704                column_index,
705                rotation,
706            }) => f
707                .debug_struct("Instance")
708                .field("query_index", index)
709                .field("column_index", column_index)
710                .field("rotation", rotation)
711                .finish(),
712            Expression::Negated(poly) => f.debug_tuple("Negated").field(poly).finish(),
713            Expression::Sum(a, b) => f.debug_tuple("Sum").field(a).field(b).finish(),
714            Expression::Product(a, b) => f.debug_tuple("Product").field(a).field(b).finish(),
715            Expression::Scaled(poly, scalar) => {
716                f.debug_tuple("Scaled").field(poly).field(scalar).finish()
717            }
718        }
719    }
720}

729impl<F: Field> Add for Expression<F> {
730    type Output = Expression<F>;
731    fn add(self, rhs: Expression<F>) -> Expression<F> {
732        if self.contains_simple_selector() || rhs.contains_simple_selector() {
733            panic!("attempted to use a simple selector in an addition");
734        }
735        Expression::Sum(Box::new(self), Box::new(rhs))
736    }

741    fn sub(self, rhs: Expression<F>) -> Expression<F> {
742        if self.contains_simple_selector() || rhs.contains_simple_selector() {
743            panic!("attempted to use a simple selector in a subtraction");
744        }
745        Expression::Sum(Box::new(self), Box::new(-rhs))
746    }
747}
748
749impl<F: Field> Mul for Expression<F> {
750    type Output = Expression<F>;
751    fn mul(self, rhs: Expression<F>) -> Expression<F> {
752        if self.contains_simple_selector() && rhs.contains_simple_selector() {
753            panic!("attempted to multiply two expressions containing simple selectors");
754        }
755        Expression::Product(Box::new(self), Box::new(rhs))
756    }
757}
758
759impl<F: Field> Mul<F> for Expression<F> {
760    type Output = Expression<F>;
761    fn mul(self, rhs: F) -> Expression<F> {
762        Expression::Scaled(Box::new(self), rhs)
763    }
764}

815/// A set of polynomial constraints with a common selector.
816///
817/// ```
818/// use group::ff::Field;
819/// use halo2_proofs::{pasta::Fp, plonk::{Constraints, Expression}, poly::Rotation};
820/// # use halo2_proofs::plonk::ConstraintSystem;
821///
822/// # let mut meta = ConstraintSystem::<Fp>::default();
823/// let a = meta.advice_column();
824/// let b = meta.advice_column();
825/// let c = meta.advice_column();
826/// let s = meta.selector();
827///
828/// meta.create_gate("foo", |meta| {
829///     let next = meta.query_advice(a, Rotation::next());
830///     let a = meta.query_advice(a, Rotation::cur());
831///     let b = meta.query_advice(b, Rotation::cur());
832///     let c = meta.query_advice(c, Rotation::cur());
833///     let s_ternary = meta.query_selector(s);
834///
835///     let one_minus_a = Expression::Constant(Fp::ONE) - a.clone();
836///
837///     Constraints::with_selector(
838///         s_ternary,
839///         std::array::IntoIter::new([
840///             ("a is boolean", a.clone() * one_minus_a.clone()),
841///             ("next == a ? b : c", next - (a * b + one_minus_a * c)),
842///         ]),
843///     )
844/// });
845/// ```
846///
847/// Note that the use of `std::array::IntoIter::new` is only necessary if you need to
848/// support Rust 1.51 or 1.52. If your minimum supported Rust version is 1.53 or greater,
849/// you can pass an array directly.
850#[derive(Debug)]
851pub struct Constraints<F: Field, C: Into<Constraint<F>>, Iter: IntoIterator<Item = C>> {
852    selector: Expression<F>,
853    constraints: Iter,
854}
855
856impl<F: Field, C: Into<Constraint<F>>, Iter: IntoIterator<Item = C>> Constraints<F, C, Iter> {
857    /// Constructs a set of constraints that are controlled by the given selector.
858    ///
859    /// Each constraint `c` in `iterator` will be converted into the constraint
860    /// `selector * c`.
861    pub fn with_selector(selector: Expression<F>, constraints: Iter) -> Self {
862        Constraints {
863            selector,
864            constraints,
865        }
866    }
867}
868
869fn apply_selector_to_constraint<F: Field, C: Into<Constraint<F>>>(
870    (selector, c): (Expression<F>, C),
871) -> Constraint<F> {
872    let constraint: Constraint<F> = c.into();
873    Constraint {
874        name: constraint.name,
875        poly: selector * constraint.poly,
876    }
877}
878
879type ApplySelectorToConstraint<F, C> = fn((Expression<F>, C)) -> Constraint<F>;
880type ConstraintsIterator<F, C, I> = std::iter::Map<
881    std::iter::Zip<std::iter::Repeat<Expression<F>>, I>,
882    ApplySelectorToConstraint<F, C>,
883>;
884
885impl<F: Field, C: Into<Constraint<F>>, Iter: IntoIterator<Item = C>> IntoIterator
886    for Constraints<F, C, Iter>
887{
888    type Item = Constraint<F>;
889    type IntoIter = ConstraintsIterator<F, C, Iter::IntoIter>;
890
891    fn into_iter(self) -> Self::IntoIter {
892        std::iter::repeat(self.selector)
893            .zip(self.constraints.into_iter())
894            .map(apply_selector_to_constraint)
895    }
896}

1200    pub fn create_gate<C: Into<Constraint<F>>, Iter: IntoIterator<Item = C>>(
1201        &mut self,
1202        name: &'static str,
1203        constraints: impl FnOnce(&mut VirtualCells<'_, F>) -> Iter,
1204    ) {
1205        let mut cells = VirtualCells::new(self);
1206        let constraints = constraints(&mut cells);
1207        let queried_selectors = cells.queried_selectors;
1208        let queried_cells = cells.queried_cells;
1209

1211            .into_iter()
1212            .map(|c| c.into())

1230    /// This will compress selectors together depending on their provided
1231    /// assignments. This `ConstraintSystem` will then be modified to add new
1232    /// fixed columns (representing the actual selectors) and will return the
1233    /// polynomials for those columns. Finally, an internal map is updated to
1234    /// find which fixed column corresponds with a given `Selector`.
1235    ///
1236    /// Do not call this twice. Yes, this should be a builder pattern instead.
1237    pub(crate) fn compress_selectors(mut self, selectors: Vec<Vec<bool>>) -> (Self, Vec<Vec<F>>) {
1238        // The number of provided selector assignments must be the number we
1239        // counted for this constraint system.
1240        assert_eq!(selectors.len(), self.num_selectors);
1241
1242        // Compute the maximal degree of every selector. We only consider the
1243        // expressions in gates, as lookup arguments cannot support simple
1244        // selectors. Selectors that are complex or do not appear in any gates
1245        // will have degree zero.
1246        let mut degrees = vec![0; selectors.len()];
1247        for expr in self.gates.iter().flat_map(|gate| gate.polys.iter()) {
1248            if let Some(selector) = expr.extract_simple_selector() {
1249                degrees[selector.0] = max(degrees[selector.0], expr.degree());
1250            }
1251        }
1252
1253        // We will not increase the degree of the constraint system, so we limit
1254        // ourselves to the largest existing degree constraint.
1255        let max_degree = self.degree();
1256
1257        let mut new_columns = vec![];
1258        let (polys, selector_assignment) = compress_selectors::process(
1259            selectors
1260                .into_iter()
1261                .zip(degrees.into_iter())
1262                .enumerate()
1263                .map(
1264                    |(i, (activations, max_degree))| compress_selectors::SelectorDescription {
1265                        selector: i,
1266                        activations,
1267                        max_degree,
1268                    },
1269                )
1270                .collect(),
1271            max_degree,
1272            || {
1273                let column = self.fixed_column();
1274                new_columns.push(column);
1275                Expression::Fixed(FixedQuery {
1276                    index: self.query_fixed_index(column),
1277                    column_index: column.index,
1278                    rotation: Rotation::cur(),
1279                })
1280            },
1281        );
1282
1283        let mut selector_map = vec![None; selector_assignment.len()];
1284        let mut selector_replacements = vec![None; selector_assignment.len()];
1285        for assignment in selector_assignment {
1286            selector_replacements[assignment.selector] = Some(assignment.expression);
1287            selector_map[assignment.selector] = Some(new_columns[assignment.combination_index]);
1288        }
1289
1290        self.selector_map = selector_map
1291            .into_iter()
1292            .map(|a| a.unwrap())
1293            .collect::<Vec<_>>();
1294        let selector_replacements = selector_replacements
1295            .into_iter()
1296            .map(|a| a.unwrap())
1297            .collect::<Vec<_>>();
1298
1299        fn replace_selectors<F: Field>(
1300            expr: &mut Expression<F>,
1301            selector_replacements: &[Expression<F>],
1302            must_be_nonsimple: bool,
1303        ) {
1304            *expr = expr.evaluate(
1305                &|constant| Expression::Constant(constant),
1306                &|selector| {
1307                    if must_be_nonsimple {
1308                        // Simple selectors are prohibited from appearing in
1309                        // expressions in the lookup argument by
1310                        // `ConstraintSystem`.
1311                        assert!(!selector.is_simple());
1312                    }
1313
1314                    selector_replacements[selector.0].clone()
1315                },
1316                &|query| Expression::Fixed(query),
1317                &|query| Expression::Advice(query),
1318                &|query| Expression::Instance(query),
1319                &|a| -a,
1320                &|a, b| a + b,
1321                &|a, b| a * b,
1322                &|a, f| a * f,
1323            );
1324        }
1325
1326        // Substitute selectors for the real fixed columns in all gates
1327        for expr in self.gates.iter_mut().flat_map(|gate| gate.polys.iter_mut()) {
1328            replace_selectors(expr, &selector_replacements, false);
1329        }
1330
1331        // Substitute non-simple selectors for the real fixed columns in all
1332        // lookup expressions
1333        for expr in self.lookups.iter_mut().flat_map(|lookup| {
1334            lookup
1335                .input_expressions
1336                .iter_mut()
1337                .chain(lookup.table_expressions.iter_mut())
1338        }) {
1339            replace_selectors(expr, &selector_replacements, true);
1340        }
1341
1342        (self, polys)
1343    }

1401    /// Compute the degree of the constraint system (the maximum degree of all
1402    /// constraints).
1403    pub fn degree(&self) -> usize {
1404        // The permutation argument will serve alongside the gates, so must be
1405        // accounted for.
1406        let mut degree = self.permutation.required_degree();
1407
1408        // The lookup argument also serves alongside the gates and must be accounted
1409        // for.
1410        degree = std::cmp::max(
1411            degree,
1412            self.lookups
1413                .iter()
1414                .map(|l| l.required_degree())
1415                .max()
1416                .unwrap_or(1),
1417        );
1418
1419        // Account for each gate to ensure our quotient polynomial is the
1420        // correct degree and that our extended domain is the right size.
1421        degree = std::cmp::max(
1422            degree,
1423            self.gates
1424                .iter()
1425                .flat_map(|gate| gate.polynomials().iter().map(|poly| poly.degree()))
1426                .max()
1427                .unwrap_or(0),
1428        );
1429
1430        std::cmp::max(degree, self.minimum_degree.unwrap_or(1))
1431    }

1433    /// Compute the number of blinding factors necessary to perfectly blind
1434    /// each of the prover's witness polynomials.
1435    pub fn blinding_factors(&self) -> usize {
1436        // All of the prover's advice columns are evaluated at no more than
1437        let factors = *self.num_advice_queries.iter().max().unwrap_or(&1);
1438        // distinct points during gate checks.
1439
1440        // - The permutation argument witness polynomials are evaluated at most 3 times.
1441        // - Each lookup argument has independent witness polynomials, and they are
1442        //   evaluated at most 2 times.
1443        let factors = std::cmp::max(3, factors);
1444
1445        // Each polynomial is evaluated at most an additional time during
1446        // multiopen (at x_3 to produce q_evals):
1447        let factors = factors + 1;
1448
1449        // h(x) is derived by the other evaluations so it does not reveal
1450        // anything; in fact it does not even appear in the proof.
1451
1452        // h(x_3) is also not revealed; the verifier only learns a single
1453        // evaluation of a polynomial in x_1 which has h(x_3) and another random
1454        // polynomial evaluated at x_3 as coefficients -- this random polynomial
1455        // is "random_poly" in the vanishing argument.
1456
1457        // Add an additional blinding factor as a slight defense against
1458        // off-by-one errors.
1459        factors + 1
1460    }
1461
1462    /// Returns the minimum necessary rows that need to exist in order to
1463    /// account for e.g. blinding factors.
1464    pub fn minimum_rows(&self) -> usize {
1465        self.blinding_factors() // m blinding factors
1466            + 1 // for l_{-(m + 1)} (l_last)
1467            + 1 // for l_0 (just for extra breathing room for the permutation
1468                // argument, to essentially force a separation in the
1469                // permutation polynomial between the roles of l_last, l_0
1470                // and the interstitial values.)
1471            + 1 // for at least one row
1472    }
1473}

1475/// Exposes the "virtual cells" that can be queried while creating a custom gate or lookup
1476/// table.
1477#[derive(Debug)]
1478pub struct VirtualCells<'a, F: Field> {
1479    meta: &'a mut ConstraintSystem<F>,
1480    queried_selectors: Vec<Selector>,
1481    queried_cells: Vec<VirtualCell>,
1482}
1483
1484impl<'a, F: Field> VirtualCells<'a, F> {
1485    fn new(meta: &'a mut ConstraintSystem<F>) -> Self {
1486        VirtualCells {
1487            meta,
1488            queried_selectors: vec![],
1489            queried_cells: vec![],
1490        }
1491    }
1492
1493    /// Query a selector at the current position.
1494    pub fn query_selector(&mut self, selector: Selector) -> Expression<F> {
1495        self.queried_selectors.push(selector);
1496        Expression::Selector(selector)
1497    }
1498
1499    /// Query a fixed column at a relative position
1500    pub fn query_fixed(&mut self, column: Column<Fixed>) -> Expression<F> {
1501        self.queried_cells.push((column, Rotation::cur()).into());
1502        Expression::Fixed(FixedQuery {
1503            index: self.meta.query_fixed_index(column),
1504            column_index: column.index,
1505            rotation: Rotation::cur(),
1506        })
1507    }
1508
1509    /// Query an advice column at a relative position
1510    pub fn query_advice(&mut self, column: Column<Advice>, at: Rotation) -> Expression<F> {
1511        self.queried_cells.push((column, at).into());
1512        Expression::Advice(AdviceQuery {
1513            index: self.meta.query_advice_index(column, at),
1514            column_index: column.index,
1515            rotation: at,
1516        })
1517    }
1518
1519    /// Query an instance column at a relative position
1520    pub fn query_instance(&mut self, column: Column<Instance>, at: Rotation) -> Expression<F> {
1521        self.queried_cells.push((column, at).into());
1522        Expression::Instance(InstanceQuery {
1523            index: self.meta.query_instance_index(column, at),
1524            column_index: column.index,
1525            rotation: at,
1526        })
1527    }
1528
1529    /// Query an Any column at a relative position
1530    ///
1531    /// # Panics
1532    ///
1533    /// Panics if query_fixed is called with a non-cur Rotation.
1534    pub fn query_any<C: Into<Column<Any>>>(&mut self, column: C, at: Rotation) -> Expression<F> {
1535        let column = column.into();
1536        match column.column_type() {
1537            Any::Advice => self.query_advice(Column::<Advice>::try_from(column).unwrap(), at),
1538            Any::Fixed => {
1539                if at != Rotation::cur() {
1540                    panic!("Fixed columns can only be queried at the current rotation");
1541                }
1542                self.query_fixed(Column::<Fixed>::try_from(column).unwrap())
1543            }
1544            Any::Instance => self.query_instance(Column::<Instance>::try_from(column).unwrap(), at),
1545        }
1546    }
