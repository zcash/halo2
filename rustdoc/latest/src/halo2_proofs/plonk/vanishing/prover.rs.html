37impl<C: CurveAffine> Argument<C> {
38    pub(in crate::plonk) fn commit<E: EncodedChallenge<C>, R: RngCore, T: TranscriptWrite<C, E>>(
39        params: &Params<C>,
40        domain: &EvaluationDomain<C::Scalar>,
41        mut rng: R,
42        transcript: &mut T,
43    ) -> Result<Committed<C>, Error> {
44        // Sample a random polynomial of degree n - 1
45        let mut random_poly = domain.empty_coeff();
46        for coeff in random_poly.iter_mut() {
47            *coeff = C::Scalar::random(&mut rng);
48        }
49        // Sample a random blinding factor
50        let random_blind = Blind(C::Scalar::random(rng));
51
52        // Commit
53        let c = params.commit(&random_poly, random_blind).to_affine();
54        transcript.write_point(c)?;
55
56        Ok(Committed {
57            random_poly,
58            random_blind,
59        })
60    }
61}

63impl<C: CurveAffine> Committed<C> {
64    #[allow(clippy::too_many_arguments)]
65    pub(in crate::plonk) fn construct<
66        E: EncodedChallenge<C>,
67        Ev: Copy + Send + Sync,
68        R: RngCore,
69        T: TranscriptWrite<C, E>,
70    >(
71        self,
72        params: &Params<C>,
73        domain: &EvaluationDomain<C::Scalar>,
74        evaluator: poly::Evaluator<Ev, C::Scalar, ExtendedLagrangeCoeff>,
75        expressions: impl Iterator<Item = poly::Ast<Ev, C::Scalar, ExtendedLagrangeCoeff>>,
76        y: ChallengeY<C>,
77        mut rng: R,
78        transcript: &mut T,
79    ) -> Result<Constructed<C>, Error> {
80        // Evaluate the h(X) polynomial's constraint system expressions for the constraints provided
81        let h_poly = poly::Ast::distribute_powers(expressions, *y); // Fold the gates together with the y challenge
82        let h_poly = evaluator.evaluate(&h_poly, domain); // Evaluate the h(X) polynomial
83
84        // Divide by t(X) = X^{params.n} - 1.
85        let h_poly = domain.divide_by_vanishing_poly(h_poly);
86
87        // Obtain final h(X) polynomial
88        let h_poly = domain.extended_to_coeff(h_poly);
89
90        // Split h(X) up into pieces
91        let h_pieces = h_poly
92            .chunks_exact(params.n as usize)
93            .map(|v| domain.coeff_from_vec(v.to_vec()))
94            .collect::<Vec<_>>();
95        drop(h_poly);
96        let h_blinds: Vec<_> = h_pieces
97            .iter()
98            .map(|_| Blind(C::Scalar::random(&mut rng)))
99            .collect();
100
101        // Compute commitments to each h(X) piece
102        let h_commitments_projective: Vec<_> = h_pieces
103            .iter()
104            .zip(h_blinds.iter())
105            .map(|(h_piece, blind)| params.commit(h_piece, *blind))
106            .collect();
107        let mut h_commitments = vec![C::identity(); h_commitments_projective.len()];
108        C::Curve::batch_normalize(&h_commitments_projective, &mut h_commitments);
109        let h_commitments = h_commitments;
110
111        // Hash each h(X) piece
112        for c in h_commitments.iter() {
113            transcript.write_point(*c)?;
114        }
115
116        Ok(Constructed {
117            h_pieces,
118            h_blinds,
119            committed: self,
120        })
121    }
122}

124impl<C: CurveAffine> Constructed<C> {
125    pub(in crate::plonk) fn evaluate<E: EncodedChallenge<C>, T: TranscriptWrite<C, E>>(
126        self,
127        x: ChallengeX<C>,
128        xn: C::Scalar,
129        domain: &EvaluationDomain<C::Scalar>,
130        transcript: &mut T,
131    ) -> Result<Evaluated<C>, Error> {
132        let h_poly = self
133            .h_pieces
134            .iter()
135            .rev()
136            .fold(domain.empty_coeff(), |acc, eval| acc * xn + eval);
137
138        let h_blind = self
139            .h_blinds
140            .iter()
141            .rev()
142            .fold(Blind(C::Scalar::ZERO), |acc, eval| acc * Blind(xn) + *eval);
143
144        let random_eval = eval_polynomial(&self.committed.random_poly, *x);
145        transcript.write_scalar(random_eval)?;
146
147        Ok(Evaluated {
148            h_poly,
149            h_blind,
150            committed: self.committed,
151        })
152    }
153}
