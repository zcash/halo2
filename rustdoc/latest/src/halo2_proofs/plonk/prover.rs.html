31/// This creates a proof for the provided `circuit` when given the public
32/// parameters `params` and the proving key [`ProvingKey`] that was
33/// generated previously for the same circuit. The provided `instances`
34/// are zero-padded internally.
35pub fn create_proof<
36    C: CurveAffine,
37    E: EncodedChallenge<C>,
38    R: RngCore,
39    T: TranscriptWrite<C, E>,
40    ConcreteCircuit: Circuit<C::Scalar>,
41>(
42    params: &Params<C>,
43    pk: &ProvingKey<C>,
44    circuits: &[ConcreteCircuit],
45    instances: &[&[&[C::Scalar]]],
46    mut rng: R,
47    transcript: &mut T,
48) -> Result<(), Error> {
49    if circuits.len() != instances.len() {
50        return Err(Error::InvalidInstances);
51    }
52
53    for instance in instances.iter() {
54        if instance.len() != pk.vk.cs.num_instance_columns {
55            return Err(Error::InvalidInstances);
56        }
57    }
58
59    // Hash verification key into transcript
60    pk.vk.hash_into(transcript)?;
61
62    let domain = &pk.vk.domain;
63    let mut meta = ConstraintSystem::default();
64    let config = ConcreteCircuit::configure(&mut meta);
65
66    // Selector optimizations cannot be applied here; use the ConstraintSystem
67    // from the verification key.
68    let meta = &pk.vk.cs;
69
70    struct InstanceSingle<C: CurveAffine> {
71        pub instance_values: Vec<Polynomial<C::Scalar, LagrangeCoeff>>,
72        pub instance_polys: Vec<Polynomial<C::Scalar, Coeff>>,
73        pub instance_cosets: Vec<Polynomial<C::Scalar, ExtendedLagrangeCoeff>>,
74    }
75
76    let instance: Vec<InstanceSingle<C>> = instances
77        .iter()
78        .map(|instance| -> Result<InstanceSingle<C>, Error> {
79            let instance_values = instance
80                .iter()
81                .map(|values| {
82                    let mut poly = domain.empty_lagrange();
83                    assert_eq!(poly.len(), params.n as usize);
84                    if values.len() > (poly.len() - (meta.blinding_factors() + 1)) {
85                        return Err(Error::InstanceTooLarge);
86                    }
87                    for (poly, value) in poly.iter_mut().zip(values.iter()) {
88                        *poly = *value;
89                    }
90                    Ok(poly)
91                })
92                .collect::<Result<Vec<_>, _>>()?;
93            let instance_commitments_projective: Vec<_> = instance_values
94                .iter()
95                .map(|poly| params.commit_lagrange(poly, Blind::default()))
96                .collect();
97            let mut instance_commitments =
98                vec![C::identity(); instance_commitments_projective.len()];
99            C::Curve::batch_normalize(&instance_commitments_projective, &mut instance_commitments);
100            let instance_commitments = instance_commitments;
101            drop(instance_commitments_projective);
102
103            for commitment in &instance_commitments {
104                transcript.common_point(*commitment)?;
105            }
106
107            let instance_polys: Vec<_> = instance_values
108                .iter()
109                .map(|poly| {
110                    let lagrange_vec = domain.lagrange_from_vec(poly.to_vec());
111                    domain.lagrange_to_coeff(lagrange_vec)
112                })
113                .collect();
114
115            let instance_cosets: Vec<_> = instance_polys
116                .iter()
117                .map(|poly| domain.coeff_to_extended(poly.clone()))
118                .collect();
119
120            Ok(InstanceSingle {
121                instance_values,
122                instance_polys,
123                instance_cosets,
124            })
125        })
126        .collect::<Result<Vec<_>, _>>()?;
127
128    struct AdviceSingle<C: CurveAffine> {
129        pub advice_values: Vec<Polynomial<C::Scalar, LagrangeCoeff>>,
130        pub advice_polys: Vec<Polynomial<C::Scalar, Coeff>>,
131        pub advice_cosets: Vec<Polynomial<C::Scalar, ExtendedLagrangeCoeff>>,
132        pub advice_blinds: Vec<Blind<C::Scalar>>,
133    }
134
135    let advice: Vec<AdviceSingle<C>> = circuits
136        .iter()
137        .zip(instances.iter())
138        .map(|(circuit, instances)| -> Result<AdviceSingle<C>, Error> {
139            struct WitnessCollection<'a, F: Field> {
140                k: u32,
141                pub advice: Vec<Polynomial<Assigned<F>, LagrangeCoeff>>,
142                instances: &'a [&'a [F]],
143                usable_rows: RangeTo<usize>,
144                _marker: std::marker::PhantomData<F>,
145            }
146
147            impl<'a, F: Field> Assignment<F> for WitnessCollection<'a, F> {
148                fn enter_region<NR, N>(&mut self, _: N)
149                where
150                    NR: Into<String>,
151                    N: FnOnce() -> NR,
152                {
153                    // Do nothing; we don't care about regions in this context.
154                }
155
156                fn exit_region(&mut self) {
157                    // Do nothing; we don't care about regions in this context.
158                }
159
160                fn enable_selector<A, AR>(
161                    &mut self,
162                    _: A,
163                    _: &Selector,
164                    _: usize,
165                ) -> Result<(), Error>
166                where
167                    A: FnOnce() -> AR,
168                    AR: Into<String>,
169                {
170                    // We only care about advice columns here
171
172                    Ok(())
173                }
174
175                fn query_instance(
176                    &self,
177                    column: Column<Instance>,
178                    row: usize,
179                ) -> Result<Value<F>, Error> {
180                    if !self.usable_rows.contains(&row) {
181                        return Err(Error::not_enough_rows_available(self.k));
182                    }
183
184                    self.instances
185                        .get(column.index())
186                        .and_then(|column| column.get(row))
187                        .map(|v| Value::known(*v))
188                        .ok_or(Error::BoundsFailure)
189                }
190
191                fn assign_advice<V, VR, A, AR>(
192                    &mut self,
193                    _: A,
194                    column: Column<Advice>,
195                    row: usize,
196                    to: V,
197                ) -> Result<(), Error>
198                where
199                    V: FnOnce() -> Value<VR>,
200                    VR: Into<Assigned<F>>,
201                    A: FnOnce() -> AR,
202                    AR: Into<String>,
203                {
204                    if !self.usable_rows.contains(&row) {
205                        return Err(Error::not_enough_rows_available(self.k));
206                    }
207
208                    *self
209                        .advice
210                        .get_mut(column.index())
211                        .and_then(|v| v.get_mut(row))
212                        .ok_or(Error::BoundsFailure)? = to().into_field().assign()?;
213
214                    Ok(())
215                }
216
217                fn assign_fixed<V, VR, A, AR>(
218                    &mut self,
219                    _: A,
220                    _: Column<Fixed>,
221                    _: usize,
222                    _: V,
223                ) -> Result<(), Error>
224                where
225                    V: FnOnce() -> Value<VR>,
226                    VR: Into<Assigned<F>>,
227                    A: FnOnce() -> AR,
228                    AR: Into<String>,
229                {
230                    // We only care about advice columns here
231
232                    Ok(())
233                }
234
235                fn copy(
236                    &mut self,
237                    _: Column<Any>,
238                    _: usize,
239                    _: Column<Any>,
240                    _: usize,
241                ) -> Result<(), Error> {
242                    // We only care about advice columns here
243
244                    Ok(())
245                }
246
247                fn fill_from_row(
248                    &mut self,
249                    _: Column<Fixed>,
250                    _: usize,
251                    _: Value<Assigned<F>>,
252                ) -> Result<(), Error> {
253                    Ok(())
254                }
255
256                fn push_namespace<NR, N>(&mut self, _: N)
257                where
258                    NR: Into<String>,
259                    N: FnOnce() -> NR,
260                {
261                    // Do nothing; we don't care about namespaces in this context.
262                }
263
264                fn pop_namespace(&mut self, _: Option<String>) {
265                    // Do nothing; we don't care about namespaces in this context.
266                }
267            }
268
269            let unusable_rows_start = params.n as usize - (meta.blinding_factors() + 1);
270
271            let mut witness = WitnessCollection {
272                k: params.k,
273                advice: vec![domain.empty_lagrange_assigned(); meta.num_advice_columns],
274                instances,
275                // The prover will not be allowed to assign values to advice
276                // cells that exist within inactive rows, which include some
277                // number of blinding factors and an extra row for use in the
278                // permutation argument.
279                usable_rows: ..unusable_rows_start,
280                _marker: std::marker::PhantomData,
281            };
282
283            // Synthesize the circuit to obtain the witness and other information.
284            ConcreteCircuit::FloorPlanner::synthesize(
285                &mut witness,
286                circuit,
287                config.clone(),
288                meta.constants.clone(),
289            )?;
290
291            let mut advice = batch_invert_assigned(witness.advice);
292
293            // Add blinding factors to advice columns
294            for advice in &mut advice {
295                for cell in &mut advice[unusable_rows_start..] {
296                    *cell = C::Scalar::random(&mut rng);
297                }
298            }
299
300            // Compute commitments to advice column polynomials
301            let advice_blinds: Vec<_> = advice
302                .iter()
303                .map(|_| Blind(C::Scalar::random(&mut rng)))
304                .collect();
305            let advice_commitments_projective: Vec<_> = advice
306                .iter()
307                .zip(advice_blinds.iter())
308                .map(|(poly, blind)| params.commit_lagrange(poly, *blind))
309                .collect();
310            let mut advice_commitments = vec![C::identity(); advice_commitments_projective.len()];
311            C::Curve::batch_normalize(&advice_commitments_projective, &mut advice_commitments);
312            let advice_commitments = advice_commitments;
313            drop(advice_commitments_projective);
314
315            for commitment in &advice_commitments {
316                transcript.write_point(*commitment)?;
317            }
318
319            let advice_polys: Vec<_> = advice
320                .clone()
321                .into_iter()
322                .map(|poly| domain.lagrange_to_coeff(poly))
323                .collect();
324
325            let advice_cosets: Vec<_> = advice_polys
326                .iter()
327                .map(|poly| domain.coeff_to_extended(poly.clone()))
328                .collect();
329
330            Ok(AdviceSingle {
331                advice_values: advice,
332                advice_polys,
333                advice_cosets,
334                advice_blinds,
335            })
336        })
337        .collect::<Result<Vec<_>, _>>()?;
338
339    // Create polynomial evaluator context for values.
340    let mut value_evaluator = poly::new_evaluator(|| {});
341
342    // Register fixed values with the polynomial evaluator.
343    let fixed_values: Vec<_> = pk
344        .fixed_values
345        .iter()
346        .map(|poly| value_evaluator.register_poly(poly.clone()))
347        .collect();
348
349    // Register advice values with the polynomial evaluator.
350    let advice_values: Vec<_> = advice
351        .iter()
352        .map(|advice| {
353            advice
354                .advice_values
355                .iter()
356                .map(|poly| value_evaluator.register_poly(poly.clone()))
357                .collect::<Vec<_>>()
358        })
359        .collect();
360
361    // Register instance values with the polynomial evaluator.
362    let instance_values: Vec<_> = instance
363        .iter()
364        .map(|instance| {
365            instance
366                .instance_values
367                .iter()
368                .map(|poly| value_evaluator.register_poly(poly.clone()))
369                .collect::<Vec<_>>()
370        })
371        .collect();
372
373    // Create polynomial evaluator context for cosets.
374    let mut coset_evaluator = poly::new_evaluator(|| {});
375
376    // Register fixed cosets with the polynomial evaluator.
377    let fixed_cosets: Vec<_> = pk
378        .fixed_cosets
379        .iter()
380        .map(|poly| coset_evaluator.register_poly(poly.clone()))
381        .collect();
382
383    // Register advice cosets with the polynomial evaluator.
384    let advice_cosets: Vec<_> = advice
385        .iter()
386        .map(|advice| {
387            advice
388                .advice_cosets
389                .iter()
390                .map(|poly| coset_evaluator.register_poly(poly.clone()))
391                .collect::<Vec<_>>()
392        })
393        .collect();
394
395    // Register instance cosets with the polynomial evaluator.
396    let instance_cosets: Vec<_> = instance
397        .iter()
398        .map(|instance| {
399            instance
400                .instance_cosets
401                .iter()
402                .map(|poly| coset_evaluator.register_poly(poly.clone()))
403                .collect::<Vec<_>>()
404        })
405        .collect();
406
407    // Register permutation cosets with the polynomial evaluator.
408    let permutation_cosets: Vec<_> = pk
409        .permutation
410        .cosets
411        .iter()
412        .map(|poly| coset_evaluator.register_poly(poly.clone()))
413        .collect();
414
415    // Register boundary polynomials used in the lookup and permutation arguments.
416    let l0 = coset_evaluator.register_poly(pk.l0.clone());
417    let l_blind = coset_evaluator.register_poly(pk.l_blind.clone());
418    let l_last = coset_evaluator.register_poly(pk.l_last.clone());
419
420    // Sample theta challenge for keeping lookup columns linearly independent
421    let theta: ChallengeTheta<_> = transcript.squeeze_challenge_scalar();
422
423    let lookups: Vec<Vec<lookup::prover::Permuted<C, _>>> = instance_values
424        .iter()
425        .zip(instance_cosets.iter())
426        .zip(advice_values.iter())
427        .zip(advice_cosets.iter())
428        .map(|(((instance_values, instance_cosets), advice_values), advice_cosets)| -> Result<Vec<_>, Error> {
429            // Construct and commit to permuted values for each lookup
430            pk.vk
431                .cs
432                .lookups
433                .iter()
434                .map(|lookup| {
435                    lookup.commit_permuted(
436                        pk,
437                        params,
438                        domain,
439                        &value_evaluator,
440                        &mut coset_evaluator,
441                        theta,
442                        advice_values,
443                        &fixed_values,
444                        instance_values,
445                        advice_cosets,
446                        &fixed_cosets,
447                        instance_cosets,
448                        &mut rng,
449                        transcript,
450                    )
451                })
452                .collect()
453        })
454        .collect::<Result<Vec<_>, _>>()?;
455
456    // Sample beta challenge
457    let beta: ChallengeBeta<_> = transcript.squeeze_challenge_scalar();
458
459    // Sample gamma challenge
460    let gamma: ChallengeGamma<_> = transcript.squeeze_challenge_scalar();
461
462    // Commit to permutations.
463    let permutations: Vec<permutation::prover::Committed<C, _>> = instance
464        .iter()
465        .zip(advice.iter())
466        .map(|(instance, advice)| {
467            pk.vk.cs.permutation.commit(
468                params,
469                pk,
470                &pk.permutation,
471                &advice.advice_values,
472                &pk.fixed_values,
473                &instance.instance_values,
474                beta,
475                gamma,
476                &mut coset_evaluator,
477                &mut rng,
478                transcript,
479            )
480        })
481        .collect::<Result<Vec<_>, _>>()?;
482
483    let lookups: Vec<Vec<lookup::prover::Committed<C, _>>> = lookups
484        .into_iter()
485        .map(|lookups| -> Result<Vec<_>, _> {
486            // Construct and commit to products for each lookup
487            lookups
488                .into_iter()
489                .map(|lookup| {
490                    lookup.commit_product(
491                        pk,
492                        params,
493                        beta,
494                        gamma,
495                        &mut coset_evaluator,
496                        &mut rng,
497                        transcript,
498                    )
499                })
500                .collect::<Result<Vec<_>, _>>()
501        })
502        .collect::<Result<Vec<_>, _>>()?;
503
504    // Commit to the vanishing argument's random polynomial for blinding h(x_3)
505    let vanishing = vanishing::Argument::commit(params, domain, &mut rng, transcript)?;
506
507    // Obtain challenge for keeping all separate gates linearly independent
508    let y: ChallengeY<_> = transcript.squeeze_challenge_scalar();
509
510    // Evaluate the h(X) polynomial's constraint system expressions for the permutation constraints.
511    let (permutations, permutation_expressions): (Vec<_>, Vec<_>) = permutations
512        .into_iter()
513        .zip(advice_cosets.iter())
514        .zip(instance_cosets.iter())
515        .map(|((permutation, advice), instance)| {
516            permutation.construct(
517                pk,
518                &pk.vk.cs.permutation,
519                advice,
520                &fixed_cosets,
521                instance,
522                &permutation_cosets,
523                l0,
524                l_blind,
525                l_last,
526                beta,
527                gamma,
528            )
529        })
530        .unzip();
531
532    let (lookups, lookup_expressions): (Vec<Vec<_>>, Vec<Vec<_>>) = lookups
533        .into_iter()
534        .map(|lookups| {
535            // Evaluate the h(X) polynomial's constraint system expressions for the lookup constraints, if any.
536            lookups
537                .into_iter()
538                .map(|p| p.construct(beta, gamma, l0, l_blind, l_last))
539                .unzip()
540        })
541        .unzip();
542
543    let expressions = advice_cosets
544        .iter()
545        .zip(instance_cosets.iter())
546        .zip(permutation_expressions.into_iter())
547        .zip(lookup_expressions.into_iter())
548        .flat_map(
549            |(((advice_cosets, instance_cosets), permutation_expressions), lookup_expressions)| {
550                let fixed_cosets = &fixed_cosets;
551                iter::empty()
552                    // Custom constraints
553                    .chain(meta.gates.iter().flat_map(move |gate| {
554                        gate.polynomials().iter().map(move |expr| {
555                            expr.evaluate(
556                                &poly::Ast::ConstantTerm,
557                                &|_| panic!("virtual selectors are removed during optimization"),
558                                &|query| {
559                                    fixed_cosets[query.column_index]
560                                        .with_rotation(query.rotation)
561                                        .into()
562                                },
563                                &|query| {
564                                    advice_cosets[query.column_index]
565                                        .with_rotation(query.rotation)
566                                        .into()
567                                },
568                                &|query| {
569                                    instance_cosets[query.column_index]
570                                        .with_rotation(query.rotation)
571                                        .into()
572                                },
573                                &|a| -a,
574                                &|a, b| a + b,
575                                &|a, b| a * b,
576                                &|a, scalar| a * scalar,
577                            )
578                        })
579                    }))
580                    // Permutation constraints, if any.
581                    .chain(permutation_expressions.into_iter())
582                    // Lookup constraints, if any.
583                    .chain(lookup_expressions.into_iter().flatten())
584            },
585        );
586
587    // Construct the vanishing argument's h(X) commitments
588    let vanishing = vanishing.construct(
589        params,
590        domain,
591        coset_evaluator,
592        expressions,
593        y,
594        &mut rng,
595        transcript,
596    )?;
597
598    let x: ChallengeX<_> = transcript.squeeze_challenge_scalar();
599    let xn = x.pow(&[params.n, 0, 0, 0]);
600
601    // Compute and hash instance evals for each circuit instance
602    for instance in instance.iter() {
603        // Evaluate polynomials at omega^i x
604        let instance_evals: Vec<_> = meta
605            .instance_queries
606            .iter()
607            .map(|&(column, at)| {
608                eval_polynomial(
609                    &instance.instance_polys[column.index()],
610                    domain.rotate_omega(*x, at),
611                )
612            })
613            .collect();
614
615        // Hash each instance column evaluation
616        for eval in instance_evals.iter() {
617            transcript.write_scalar(*eval)?;
618        }
619    }
620
621    // Compute and hash advice evals for each circuit instance
622    for advice in advice.iter() {
623        // Evaluate polynomials at omega^i x
624        let advice_evals: Vec<_> = meta
625            .advice_queries
626            .iter()
627            .map(|&(column, at)| {
628                eval_polynomial(
629                    &advice.advice_polys[column.index()],
630                    domain.rotate_omega(*x, at),
631                )
632            })
633            .collect();
634
635        // Hash each advice column evaluation
636        for eval in advice_evals.iter() {
637            transcript.write_scalar(*eval)?;
638        }
639    }
640
641    // Compute and hash fixed evals (shared across all circuit instances)
642    let fixed_evals: Vec<_> = meta
643        .fixed_queries
644        .iter()
645        .map(|&(column, at)| {
646            eval_polynomial(&pk.fixed_polys[column.index()], domain.rotate_omega(*x, at))
647        })
648        .collect();
649
650    // Hash each fixed column evaluation
651    for eval in fixed_evals.iter() {
652        transcript.write_scalar(*eval)?;
653    }
654
655    let vanishing = vanishing.evaluate(x, xn, domain, transcript)?;
656
657    // Evaluate common permutation data
658    pk.permutation.evaluate(x, transcript)?;
659
660    // Evaluate the permutations, if any, at omega^i x.
661    let permutations: Vec<permutation::prover::Evaluated<C>> = permutations
662        .into_iter()
663        .map(|permutation| -> Result<_, _> { permutation.evaluate(pk, x, transcript) })
664        .collect::<Result<Vec<_>, _>>()?;
665
666    // Evaluate the lookups, if any, at omega^i x.
667    let lookups: Vec<Vec<lookup::prover::Evaluated<C>>> = lookups
668        .into_iter()
669        .map(|lookups| -> Result<Vec<_>, _> {
670            lookups
671                .into_iter()
672                .map(|p| p.evaluate(pk, x, transcript))
673                .collect::<Result<Vec<_>, _>>()
674        })
675        .collect::<Result<Vec<_>, _>>()?;
676
677    let instances = instance
678        .iter()
679        .zip(advice.iter())
680        .zip(permutations.iter())
681        .zip(lookups.iter())
682        .flat_map(|(((instance, advice), permutation), lookups)| {
683            iter::empty()
684                .chain(
685                    pk.vk
686                        .cs
687                        .instance_queries
688                        .iter()
689                        .map(move |&(column, at)| ProverQuery {
690                            point: domain.rotate_omega(*x, at),
691                            poly: &instance.instance_polys[column.index()],
692                            blind: Blind::default(),
693                        }),
694                )
695                .chain(
696                    pk.vk
697                        .cs
698                        .advice_queries
699                        .iter()
700                        .map(move |&(column, at)| ProverQuery {
701                            point: domain.rotate_omega(*x, at),
702                            poly: &advice.advice_polys[column.index()],
703                            blind: advice.advice_blinds[column.index()],
704                        }),
705                )
706                .chain(permutation.open(pk, x))
707                .chain(lookups.iter().flat_map(move |p| p.open(pk, x)).into_iter())
708        })
709        .chain(
710            pk.vk
711                .cs
712                .fixed_queries
713                .iter()
714                .map(|&(column, at)| ProverQuery {
715                    point: domain.rotate_omega(*x, at),
716                    poly: &pk.fixed_polys[column.index()],
717                    blind: Blind::default(),
718                }),
719        )
720        .chain(pk.permutation.open(x))
721        // We query the h(X) polynomial at x
722        .chain(vanishing.open(x));
723
724    multiopen::create_proof(params, rng, transcript, instances).map_err(|_| Error::Opening)
725}

728fn test_create_proof() {
729    use crate::{
730        circuit::SimpleFloorPlanner,
731        plonk::{keygen_pk, keygen_vk},
