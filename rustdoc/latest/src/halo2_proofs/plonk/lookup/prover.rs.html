65impl<F: WithSmallOrderMulGroup<3>> Argument<F> {
66    /// Given a Lookup with input expressions [A_0, A_1, ..., A_{m-1}] and table expressions
67    /// [S_0, S_1, ..., S_{m-1}], this method
68    /// - constructs A_compressed = \theta^{m-1} A_0 + theta^{m-2} A_1 + ... + \theta A_{m-2} + A_{m-1}
69    ///   and S_compressed = \theta^{m-1} S_0 + theta^{m-2} S_1 + ... + \theta S_{m-2} + S_{m-1},
70    /// - permutes A_compressed and S_compressed using permute_expression_pair() helper,
71    ///   obtaining A' and S', and
72    /// - constructs Permuted<C> struct using permuted_input_value = A', and
73    ///   permuted_table_expression = S'.
74    /// The Permuted<C> struct is used to update the Lookup, and is then returned.
75    #[allow(clippy::too_many_arguments)]
76    pub(in crate::plonk) fn commit_permuted<
77        'a,
78        C,
79        E: EncodedChallenge<C>,
80        Ev: Copy + Send + Sync,
81        Ec: Copy + Send + Sync,
82        R: RngCore,
83        T: TranscriptWrite<C, E>,
84    >(
85        &self,
86        pk: &ProvingKey<C>,
87        params: &Params<C>,
88        domain: &EvaluationDomain<C::Scalar>,
89        value_evaluator: &poly::Evaluator<Ev, C::Scalar, LagrangeCoeff>,
90        coset_evaluator: &mut poly::Evaluator<Ec, C::Scalar, ExtendedLagrangeCoeff>,
91        theta: ChallengeTheta<C>,
92        advice_values: &'a [poly::AstLeaf<Ev, LagrangeCoeff>],
93        fixed_values: &'a [poly::AstLeaf<Ev, LagrangeCoeff>],
94        instance_values: &'a [poly::AstLeaf<Ev, LagrangeCoeff>],
95        advice_cosets: &'a [poly::AstLeaf<Ec, ExtendedLagrangeCoeff>],
96        fixed_cosets: &'a [poly::AstLeaf<Ec, ExtendedLagrangeCoeff>],
97        instance_cosets: &'a [poly::AstLeaf<Ec, ExtendedLagrangeCoeff>],
98        mut rng: R,
99        transcript: &mut T,
100    ) -> Result<Permuted<C, Ec>, Error>
101    where
102        C: CurveAffine<ScalarExt = F>,
103        C::Curve: Mul<F, Output = C::Curve> + MulAssign<F>,
104    {
105        // Closure to get values of expressions and compress them
106        let compress_expressions = |expressions: &[Expression<C::Scalar>]| {
107            // Values of input expressions involved in the lookup
108            let unpermuted_expressions: Vec<_> = expressions
109                .iter()
110                .map(|expression| {
111                    expression.evaluate(
112                        &|scalar| poly::Ast::ConstantTerm(scalar),
113                        &|_| panic!("virtual selectors are removed during optimization"),
114                        &|query| {
115                            fixed_values[query.column_index]
116                                .with_rotation(query.rotation)
117                                .into()
118                        },
119                        &|query| {
120                            advice_values[query.column_index]
121                                .with_rotation(query.rotation)
122                                .into()
123                        },
124                        &|query| {
125                            instance_values[query.column_index]
126                                .with_rotation(query.rotation)
127                                .into()
128                        },
129                        &|a| -a,
130                        &|a, b| a + b,
131                        &|a, b| a * b,
132                        &|a, scalar| a * scalar,
133                    )
134                })
135                .collect();
136
137            let unpermuted_cosets: Vec<_> = expressions
138                .iter()
139                .map(|expression| {
140                    expression.evaluate(
141                        &|scalar| poly::Ast::ConstantTerm(scalar),
142                        &|_| panic!("virtual selectors are removed during optimization"),
143                        &|query| {
144                            fixed_cosets[query.column_index]
145                                .with_rotation(query.rotation)
146                                .into()
147                        },
148                        &|query| {
149                            advice_cosets[query.column_index]
150                                .with_rotation(query.rotation)
151                                .into()
152                        },
153                        &|query| {
154                            instance_cosets[query.column_index]
155                                .with_rotation(query.rotation)
156                                .into()
157                        },
158                        &|a| -a,
159                        &|a, b| a + b,
160                        &|a, b| a * b,
161                        &|a, scalar| a * scalar,
162                    )
163                })
164                .collect();
165
166            // Compressed version of expressions
167            let compressed_expression = unpermuted_expressions.iter().fold(
168                poly::Ast::ConstantTerm(C::Scalar::ZERO),
169                |acc, expression| &(acc * *theta) + expression,
170            );
171
172            // Compressed version of cosets
173            let compressed_coset = unpermuted_cosets.iter().fold(
174                poly::Ast::<_, _, ExtendedLagrangeCoeff>::ConstantTerm(C::Scalar::ZERO),
175                |acc, eval| acc * poly::Ast::ConstantTerm(*theta) + eval.clone(),
176            );
177
178            (
179                compressed_coset,
180                value_evaluator.evaluate(&compressed_expression, domain),
181            )
182        };
183
184        // Get values of input expressions involved in the lookup and compress them
185        let (compressed_input_coset, compressed_input_expression) =
186            compress_expressions(&self.input_expressions);
187
188        // Get values of table expressions involved in the lookup and compress them
189        let (compressed_table_coset, compressed_table_expression) =
190            compress_expressions(&self.table_expressions);
191
192        // Permute compressed (InputExpression, TableExpression) pair
193        let (permuted_input_expression, permuted_table_expression) = permute_expression_pair::<C, _>(
194            pk,
195            params,
196            domain,
197            &mut rng,
198            &compressed_input_expression,
199            &compressed_table_expression,
200        )?;
201
202        // Closure to construct commitment to vector of values
203        let mut commit_values = |values: &Polynomial<C::Scalar, LagrangeCoeff>| {
204            let poly = pk.vk.domain.lagrange_to_coeff(values.clone());
205            let blind = Blind(C::Scalar::random(&mut rng));
206            let commitment = params.commit_lagrange(values, blind).to_affine();
207            (poly, blind, commitment)
208        };
209
210        // Commit to permuted input expression
211        let (permuted_input_poly, permuted_input_blind, permuted_input_commitment) =
212            commit_values(&permuted_input_expression);
213
214        // Commit to permuted table expression
215        let (permuted_table_poly, permuted_table_blind, permuted_table_commitment) =
216            commit_values(&permuted_table_expression);
217
218        // Hash permuted input commitment
219        transcript.write_point(permuted_input_commitment)?;
220
221        // Hash permuted table commitment
222        transcript.write_point(permuted_table_commitment)?;
223
224        let permuted_input_coset = coset_evaluator
225            .register_poly(pk.vk.domain.coeff_to_extended(permuted_input_poly.clone()));
226        let permuted_table_coset = coset_evaluator
227            .register_poly(pk.vk.domain.coeff_to_extended(permuted_table_poly.clone()));
228
229        Ok(Permuted {
230            compressed_input_expression,
231            compressed_input_coset,
232            permuted_input_expression,
233            permuted_input_poly,
234            permuted_input_coset,
235            permuted_input_blind,
236            compressed_table_expression,
237            compressed_table_coset,
238            permuted_table_expression,
239            permuted_table_poly,
240            permuted_table_coset,
241            permuted_table_blind,
242        })
243    }
244}

246impl<C: CurveAffine, Ev: Copy + Send + Sync> Permuted<C, Ev> {
247    /// Given a Lookup with input expressions, table expressions, and the permuted
248    /// input expression and permuted table expression, this method constructs the
249    /// grand product polynomial over the lookup. The grand product polynomial
250    /// is used to populate the Product<C> struct. The Product<C> struct is
251    /// added to the Lookup and finally returned by the method.
252    #[allow(clippy::too_many_arguments)]
253    pub(in crate::plonk) fn commit_product<
254        E: EncodedChallenge<C>,
255        R: RngCore,
256        T: TranscriptWrite<C, E>,
257    >(
258        self,
259        pk: &ProvingKey<C>,
260        params: &Params<C>,
261        beta: ChallengeBeta<C>,
262        gamma: ChallengeGamma<C>,
263        evaluator: &mut poly::Evaluator<Ev, C::Scalar, ExtendedLagrangeCoeff>,
264        mut rng: R,
265        transcript: &mut T,
266    ) -> Result<Committed<C, Ev>, Error> {
267        let blinding_factors = pk.vk.cs.blinding_factors();
268        // Goal is to compute the products of fractions
269        //
270        // Numerator: (\theta^{m-1} a_0(\omega^i) + \theta^{m-2} a_1(\omega^i) + ... + \theta a_{m-2}(\omega^i) + a_{m-1}(\omega^i) + \beta)
271        //            * (\theta^{m-1} s_0(\omega^i) + \theta^{m-2} s_1(\omega^i) + ... + \theta s_{m-2}(\omega^i) + s_{m-1}(\omega^i) + \gamma)
272        // Denominator: (a'(\omega^i) + \beta) (s'(\omega^i) + \gamma)
273        //
274        // where a_j(X) is the jth input expression in this lookup,
275        // where a'(X) is the compression of the permuted input expressions,
276        // s_j(X) is the jth table expression in this lookup,
277        // s'(X) is the compression of the permuted table expressions,
278        // and i is the ith row of the expression.
279        let mut lookup_product = vec![C::Scalar::ZERO; params.n as usize];
280        // Denominator uses the permuted input expression and permuted table expression
281        parallelize(&mut lookup_product, |lookup_product, start| {
282            for ((lookup_product, permuted_input_value), permuted_table_value) in lookup_product
283                .iter_mut()
284                .zip(self.permuted_input_expression[start..].iter())
285                .zip(self.permuted_table_expression[start..].iter())
286            {
287                *lookup_product = (*beta + permuted_input_value) * &(*gamma + permuted_table_value);
288            }
289        });
290
291        // Batch invert to obtain the denominators for the lookup product
292        // polynomials
293        lookup_product.iter_mut().batch_invert();
294
295        // Finish the computation of the entire fraction by computing the numerators
296        // (\theta^{m-1} a_0(\omega^i) + \theta^{m-2} a_1(\omega^i) + ... + \theta a_{m-2}(\omega^i) + a_{m-1}(\omega^i) + \beta)
297        // * (\theta^{m-1} s_0(\omega^i) + \theta^{m-2} s_1(\omega^i) + ... + \theta s_{m-2}(\omega^i) + s_{m-1}(\omega^i) + \gamma)
298        parallelize(&mut lookup_product, |product, start| {
299            for ((product, &input_term), &table_term) in product
300                .iter_mut()
301                .zip(self.compressed_input_expression[start..].iter())
302                .zip(self.compressed_table_expression[start..].iter())
303            {
304                *product *= &(input_term + &*beta);
305                *product *= &(table_term + &*gamma);
306            }
307        });
308
309        // The product vector is a vector of products of fractions of the form
310        //
311        // Numerator: (\theta^{m-1} a_0(\omega^i) + \theta^{m-2} a_1(\omega^i) + ... + \theta a_{m-2}(\omega^i) + a_{m-1}(\omega^i) + \beta)
312        //            * (\theta^{m-1} s_0(\omega^i) + \theta^{m-2} s_1(\omega^i) + ... + \theta s_{m-2}(\omega^i) + s_{m-1}(\omega^i) + \gamma)
313        // Denominator: (a'(\omega^i) + \beta) (s'(\omega^i) + \gamma)
314        //
315        // where there are m input expressions and m table expressions,
316        // a_j(\omega^i) is the jth input expression in this lookup,
317        // a'j(\omega^i) is the permuted input expression,
318        // s_j(\omega^i) is the jth table expression in this lookup,
319        // s'(\omega^i) is the permuted table expression,
320        // and i is the ith row of the expression.
321
322        // Compute the evaluations of the lookup product polynomial
323        // over our domain, starting with z[0] = 1
324        let z = iter::once(C::Scalar::ONE)
325            .chain(lookup_product)
326            .scan(C::Scalar::ONE, |state, cur| {
327                *state *= &cur;
328                Some(*state)
329            })
330            // Take all rows including the "last" row which should
331            // be a boolean (and ideally 1, else soundness is broken)
332            .take(params.n as usize - blinding_factors)
333            // Chain random blinding factors.
334            .chain((0..blinding_factors).map(|_| C::Scalar::random(&mut rng)))
335            .collect::<Vec<_>>();
336        assert_eq!(z.len(), params.n as usize);
337        let z = pk.vk.domain.lagrange_from_vec(z);
338
339        #[cfg(feature = "sanity-checks")]
340        // This test works only with intermediate representations in this method.
341        // It can be used for debugging purposes.
342        {
343            // While in Lagrange basis, check that product is correctly constructed
344            let u = (params.n as usize) - (blinding_factors + 1);
345
346            // l_0(X) * (1 - z(X)) = 0
347            assert_eq!(z[0], C::Scalar::ONE);
348
349            // z(\omega X) (a'(X) + \beta) (s'(X) + \gamma)
350            // - z(X) (\theta^{m-1} a_0(X) + ... + a_{m-1}(X) + \beta) (\theta^{m-1} s_0(X) + ... + s_{m-1}(X) + \gamma)
351            for i in 0..u {
352                let mut left = z[i + 1];
353                let permuted_input_value = &self.permuted_input_expression[i];
354
355                let permuted_table_value = &self.permuted_table_expression[i];
356
357                left *= &(*beta + permuted_input_value);
358                left *= &(*gamma + permuted_table_value);
359
360                let mut right = z[i];
361                let mut input_term = self.compressed_input_expression[i];
362
363                let mut table_term = self.compressed_table_expression[i];
364
365                input_term += &(*beta);
366                table_term += &(*gamma);
367                right *= &(input_term * &table_term);
368
369                assert_eq!(left, right);
370            }
371
372            // l_last(X) * (z(X)^2 - z(X)) = 0
373            // Assertion will fail only when soundness is broken, in which
374            // case this z[u] value will be zero. (bad!)
375            assert_eq!(z[u], C::Scalar::ONE);
376        }
377
378        let product_blind = Blind(C::Scalar::random(rng));
379        let product_commitment = params.commit_lagrange(&z, product_blind).to_affine();
380        let z = pk.vk.domain.lagrange_to_coeff(z);
381        let product_coset = evaluator.register_poly(pk.vk.domain.coeff_to_extended(z.clone()));
382
383        // Hash product commitment
384        transcript.write_point(product_commitment)?;
385
386        Ok(Committed::<C, _> {
387            permuted: self,
388            product_poly: z,
389            product_coset,
390            product_blind,
391        })
392    }
393}

395impl<'a, C: CurveAffine, Ev: Copy + Send + Sync + 'a> Committed<C, Ev> {
396    /// Given a Lookup with input expressions, table expressions, permuted input
397    /// expression, permuted table expression, and grand product polynomial, this
398    /// method constructs constraints that must hold between these values.
399    /// This method returns the constraints as a vector of ASTs for polynomials in
400    /// the extended evaluation domain.
401    pub(in crate::plonk) fn construct(
402        self,
403        beta: ChallengeBeta<C>,
404        gamma: ChallengeGamma<C>,
405        l0: poly::AstLeaf<Ev, ExtendedLagrangeCoeff>,
406        l_blind: poly::AstLeaf<Ev, ExtendedLagrangeCoeff>,
407        l_last: poly::AstLeaf<Ev, ExtendedLagrangeCoeff>,
408    ) -> (
409        Constructed<C>,
410        impl Iterator<Item = poly::Ast<Ev, C::Scalar, ExtendedLagrangeCoeff>> + 'a,
411    ) {
412        let permuted = self.permuted;
413
414        let active_rows = poly::Ast::one() - (poly::Ast::from(l_last) + l_blind);
415        let beta = poly::Ast::ConstantTerm(*beta);
416        let gamma = poly::Ast::ConstantTerm(*gamma);
417
418        let expressions = iter::empty()
419            // l_0(X) * (1 - z(X)) = 0
420            .chain(Some((poly::Ast::one() - self.product_coset) * l0))
421            // l_last(X) * (z(X)^2 - z(X)) = 0
422            .chain(Some(
423                (poly::Ast::from(self.product_coset) * self.product_coset - self.product_coset)
424                    * l_last,
425            ))
426            // (1 - (l_last(X) + l_blind(X))) * (
427            //   z(\omega X) (a'(X) + \beta) (s'(X) + \gamma)
428            //   - z(X) (\theta^{m-1} a_0(X) + ... + a_{m-1}(X) + \beta) (\theta^{m-1} s_0(X) + ... + s_{m-1}(X) + \gamma)
429            // ) = 0
430            .chain({
431                // z(\omega X) (a'(X) + \beta) (s'(X) + \gamma)
432                let left: poly::Ast<_, _, _> = poly::Ast::<_, C::Scalar, _>::from(
433                    self.product_coset.with_rotation(Rotation::next()),
434                ) * (poly::Ast::from(permuted.permuted_input_coset)
435                    + beta.clone())
436                    * (poly::Ast::from(permuted.permuted_table_coset) + gamma.clone());
437
438                //  z(X) (\theta^{m-1} a_0(X) + ... + a_{m-1}(X) + \beta) (\theta^{m-1} s_0(X) + ... + s_{m-1}(X) + \gamma)
439                let right: poly::Ast<_, _, _> = poly::Ast::from(self.product_coset)
440                    * (permuted.compressed_input_coset + beta)
441                    * (permuted.compressed_table_coset + gamma);
442
443                Some((left - right) * active_rows.clone())
444            })
445            // Check that the first values in the permuted input expression and permuted
446            // fixed expression are the same.
447            // l_0(X) * (a'(X) - s'(X)) = 0
448            .chain(Some(
449                (poly::Ast::from(permuted.permuted_input_coset) - permuted.permuted_table_coset)
450                    * l0,
451            ))
452            // Check that each value in the permuted lookup input expression is either
453            // equal to the value above it, or the value at the same index in the
454            // permuted table expression.
455            // (1 - (l_last + l_blind)) * (a(X)  s(X))(a(X)  a(\omega^{-1} X)) = 0
456            .chain(Some(
457                (poly::Ast::<_, C::Scalar, _>::from(permuted.permuted_input_coset)
458                    - permuted.permuted_table_coset)
459                    * (poly::Ast::from(permuted.permuted_input_coset)
460                        - permuted
461                            .permuted_input_coset
462                            .with_rotation(Rotation::prev()))
463                    * active_rows,
464            ));
465
466        (
467            Constructed {
468                permuted_input_poly: permuted.permuted_input_poly,
469                permuted_input_blind: permuted.permuted_input_blind,
470                permuted_table_poly: permuted.permuted_table_poly,
471                permuted_table_blind: permuted.permuted_table_blind,
472                product_poly: self.product_poly,
473                product_blind: self.product_blind,
474            },
475            expressions,
476        )
477    }
478}

480impl<C: CurveAffine> Constructed<C> {
481    pub(in crate::plonk) fn evaluate<E: EncodedChallenge<C>, T: TranscriptWrite<C, E>>(
482        self,
483        pk: &ProvingKey<C>,
484        x: ChallengeX<C>,
485        transcript: &mut T,
486    ) -> Result<Evaluated<C>, Error> {
487        let domain = &pk.vk.domain;
488        let x_inv = domain.rotate_omega(*x, Rotation::prev());
489        let x_next = domain.rotate_omega(*x, Rotation::next());
490
491        let product_eval = eval_polynomial(&self.product_poly, *x);
492        let product_next_eval = eval_polynomial(&self.product_poly, x_next);
493        let permuted_input_eval = eval_polynomial(&self.permuted_input_poly, *x);
494        let permuted_input_inv_eval = eval_polynomial(&self.permuted_input_poly, x_inv);
495        let permuted_table_eval = eval_polynomial(&self.permuted_table_poly, *x);
496
497        // Hash each advice evaluation
498        for eval in iter::empty()
499            .chain(Some(product_eval))
500            .chain(Some(product_next_eval))
501            .chain(Some(permuted_input_eval))
502            .chain(Some(permuted_input_inv_eval))
503            .chain(Some(permuted_table_eval))
504        {
505            transcript.write_scalar(eval)?;
506        }
507
508        Ok(Evaluated { constructed: self })
509    }
510}

512impl<C: CurveAffine> Evaluated<C> {
513    pub(in crate::plonk) fn open<'a>(
514        &'a self,
515        pk: &'a ProvingKey<C>,
516        x: ChallengeX<C>,
517    ) -> impl Iterator<Item = ProverQuery<'a, C>> + Clone {
518        let x_inv = pk.vk.domain.rotate_omega(*x, Rotation::prev());
519        let x_next = pk.vk.domain.rotate_omega(*x, Rotation::next());
520
521        iter::empty()
522            // Open lookup product commitments at x
523            .chain(Some(ProverQuery {
524                point: *x,
525                poly: &self.constructed.product_poly,
526                blind: self.constructed.product_blind,
527            }))
528            // Open lookup input commitments at x
529            .chain(Some(ProverQuery {
530                point: *x,
531                poly: &self.constructed.permuted_input_poly,
532                blind: self.constructed.permuted_input_blind,
533            }))
534            // Open lookup table commitments at x
535            .chain(Some(ProverQuery {
536                point: *x,
537                poly: &self.constructed.permuted_table_poly,
538                blind: self.constructed.permuted_table_blind,
539            }))
540            // Open lookup input commitments at x_inv
541            .chain(Some(ProverQuery {
542                point: x_inv,
543                poly: &self.constructed.permuted_input_poly,
544                blind: self.constructed.permuted_input_blind,
545            }))
546            // Open lookup product commitments at x_next
547            .chain(Some(ProverQuery {
548                point: x_next,
549                poly: &self.constructed.product_poly,
550                blind: self.constructed.product_blind,
551            }))
552    }
553}
554
555type ExpressionPair<F> = (Polynomial<F, LagrangeCoeff>, Polynomial<F, LagrangeCoeff>);
556
557/// Given a vector of input values A and a vector of table values S,
558/// this method permutes A and S to produce A' and S', such that:
559/// - like values in A' are vertically adjacent to each other; and
560/// - the first row in a sequence of like values in A' is the row
561///   that has the corresponding value in S'.
562/// This method returns (A', S') if no errors are encountered.
563fn permute_expression_pair<C: CurveAffine, R: RngCore>(
564    pk: &ProvingKey<C>,
565    params: &Params<C>,
566    domain: &EvaluationDomain<C::Scalar>,
567    mut rng: R,
568    input_expression: &Polynomial<C::Scalar, LagrangeCoeff>,
569    table_expression: &Polynomial<C::Scalar, LagrangeCoeff>,
570) -> Result<ExpressionPair<C::Scalar>, Error> {
571    let blinding_factors = pk.vk.cs.blinding_factors();
572    let usable_rows = params.n as usize - (blinding_factors + 1);
573
574    let mut permuted_input_expression: Vec<C::Scalar> = input_expression.to_vec();
575    permuted_input_expression.truncate(usable_rows);
576
577    // Sort input lookup expression values
578    permuted_input_expression.sort();
579
580    // A BTreeMap of each unique element in the table expression and its count
581    let mut leftover_table_map: BTreeMap<C::Scalar, u32> = table_expression
582        .iter()
583        .take(usable_rows)
584        .fold(BTreeMap::new(), |mut acc, coeff| {
585            *acc.entry(*coeff).or_insert(0) += 1;
586            acc
587        });
588    let mut permuted_table_coeffs = vec![C::Scalar::ZERO; usable_rows];
589
590    let mut repeated_input_rows = permuted_input_expression
591        .iter()
592        .zip(permuted_table_coeffs.iter_mut())
593        .enumerate()
594        .filter_map(|(row, (input_value, table_value))| {
595            // If this is the first occurrence of `input_value` in the input expression
596            if row == 0 || *input_value != permuted_input_expression[row - 1] {
597                *table_value = *input_value;
598                // Remove one instance of input_value from leftover_table_map
599                if let Some(count) = leftover_table_map.get_mut(input_value) {
600                    assert!(*count > 0);
601                    *count -= 1;
602                    None
603                } else {
604                    // Return error if input_value not found
605                    Some(Err(Error::ConstraintSystemFailure))
606                }
607            // If input value is repeated
608            } else {
609                Some(Ok(row))
610            }
611        })
612        .collect::<Result<Vec<_>, _>>()?;
613
614    // Populate permuted table at unfilled rows with leftover table elements
615    for (coeff, count) in leftover_table_map.iter() {
616        for _ in 0..*count {
617            permuted_table_coeffs[repeated_input_rows.pop().unwrap()] = *coeff;
618        }
619    }
620    assert!(repeated_input_rows.is_empty());
621
622    permuted_input_expression
623        .extend((0..(blinding_factors + 1)).map(|_| C::Scalar::random(&mut rng)));
624    permuted_table_coeffs.extend((0..(blinding_factors + 1)).map(|_| C::Scalar::random(&mut rng)));
625    assert_eq!(permuted_input_expression.len(), params.n as usize);
626    assert_eq!(permuted_table_coeffs.len(), params.n as usize);
627
628    #[cfg(feature = "sanity-checks")]
629    {
630        let mut last = None;
631        for (a, b) in permuted_input_expression
632            .iter()
633            .zip(permuted_table_coeffs.iter())
634            .take(usable_rows)
635        {
636            if *a != *b {
637                assert_eq!(*a, last.unwrap());
638            }
639            last = Some(*a);
640        }
641    }
642
643    Ok((
644        domain.lagrange_from_vec(permuted_input_expression),
645        domain.lagrange_from_vec(permuted_table_coeffs),
646    ))
647}
