25pub(crate) fn create_domain<C, ConcreteCircuit>(
26    params: &Params<C>,
27) -> (
28    EvaluationDomain<C::Scalar>,
29    ConstraintSystem<C::Scalar>,
30    ConcreteCircuit::Config,
31)
32where
33    C: CurveAffine,
34    ConcreteCircuit: Circuit<C::Scalar>,
35{
36    let mut cs = ConstraintSystem::default();
37    let config = ConcreteCircuit::configure(&mut cs);
38
39    let degree = cs.degree();
40
41    let domain = EvaluationDomain::new(degree as u32, params.k);
42
43    (domain, cs, config)
44}

46/// Assembly to be used in circuit synthesis.
47#[derive(Debug)]
48struct Assembly<F: Field> {
49    k: u32,
50    fixed: Vec<Polynomial<Assigned<F>, LagrangeCoeff>>,
51    permutation: permutation::keygen::Assembly,
52    selectors: Vec<Vec<bool>>,
53    // A range of available rows for assignment and copies.
54    usable_rows: Range<usize>,
55    _marker: std::marker::PhantomData<F>,
56}
57
58impl<F: Field> Assignment<F> for Assembly<F> {
59    fn enter_region<NR, N>(&mut self, _: N)
60    where
61        NR: Into<String>,
62        N: FnOnce() -> NR,
63    {
64        // Do nothing; we don't care about regions in this context.
65    }
66
67    fn exit_region(&mut self) {
68        // Do nothing; we don't care about regions in this context.
69    }
70
71    fn enable_selector<A, AR>(&mut self, _: A, selector: &Selector, row: usize) -> Result<(), Error>
72    where
73        A: FnOnce() -> AR,
74        AR: Into<String>,
75    {
76        if !self.usable_rows.contains(&row) {
77            return Err(Error::not_enough_rows_available(self.k));
78        }
79
80        self.selectors[selector.0][row] = true;
81
82        Ok(())
83    }
84
85    fn query_instance(&self, _: Column<Instance>, row: usize) -> Result<Value<F>, Error> {
86        if !self.usable_rows.contains(&row) {
87            return Err(Error::not_enough_rows_available(self.k));
88        }
89
90        // There is no instance in this context.
91        Ok(Value::unknown())
92    }
93
94    fn assign_advice<V, VR, A, AR>(
95        &mut self,
96        _: A,
97        _: Column<Advice>,
98        _: usize,
99        _: V,
100    ) -> Result<(), Error>
101    where
102        V: FnOnce() -> Value<VR>,
103        VR: Into<Assigned<F>>,
104        A: FnOnce() -> AR,
105        AR: Into<String>,
106    {
107        // We only care about fixed columns here
108        Ok(())
109    }
110
111    fn assign_fixed<V, VR, A, AR>(
112        &mut self,
113        _: A,
114        column: Column<Fixed>,
115        row: usize,
116        to: V,
117    ) -> Result<(), Error>
118    where
119        V: FnOnce() -> Value<VR>,
120        VR: Into<Assigned<F>>,
121        A: FnOnce() -> AR,
122        AR: Into<String>,
123    {
124        if !self.usable_rows.contains(&row) {
125            return Err(Error::not_enough_rows_available(self.k));
126        }
127
128        *self
129            .fixed
130            .get_mut(column.index())
131            .and_then(|v| v.get_mut(row))
132            .ok_or(Error::BoundsFailure)? = to().into_field().assign()?;
133
134        Ok(())
135    }
136
137    fn copy(
138        &mut self,
139        left_column: Column<Any>,
140        left_row: usize,
141        right_column: Column<Any>,
142        right_row: usize,
143    ) -> Result<(), Error> {
144        if !self.usable_rows.contains(&left_row) || !self.usable_rows.contains(&right_row) {
145            return Err(Error::not_enough_rows_available(self.k));
146        }
147
148        self.permutation
149            .copy(left_column, left_row, right_column, right_row)
150    }
151
152    fn fill_from_row(
153        &mut self,
154        column: Column<Fixed>,
155        from_row: usize,
156        to: Value<Assigned<F>>,
157    ) -> Result<(), Error> {
158        if !self.usable_rows.contains(&from_row) {
159            return Err(Error::not_enough_rows_available(self.k));
160        }
161
162        let col = self
163            .fixed
164            .get_mut(column.index())
165            .ok_or(Error::BoundsFailure)?;
166
167        let filler = to.assign()?;
168        for row in self.usable_rows.clone().skip(from_row) {
169            col[row] = filler;
170        }
171
172        Ok(())
173    }
174
175    fn push_namespace<NR, N>(&mut self, _: N)
176    where
177        NR: Into<String>,
178        N: FnOnce() -> NR,
179    {
180        // Do nothing; we don't care about namespaces in this context.
181    }
182
183    fn pop_namespace(&mut self, _: Option<String>) {
184        // Do nothing; we don't care about namespaces in this context.
185    }
186}

188/// Generate a `VerifyingKey` from an instance of `Circuit`.
189pub fn keygen_vk<C, ConcreteCircuit>(
190    params: &Params<C>,
191    circuit: &ConcreteCircuit,
192) -> Result<VerifyingKey<C>, Error>
193where
194    C: CurveAffine,
195    C::Scalar: FromUniformBytes<64>,
196    ConcreteCircuit: Circuit<C::Scalar>,
197{
198    let (domain, cs, config) = create_domain::<C, ConcreteCircuit>(params);
199
200    if (params.n as usize) < cs.minimum_rows() {
201        return Err(Error::not_enough_rows_available(params.k));
202    }
203
204    let mut assembly: Assembly<C::Scalar> = Assembly {
205        k: params.k,
206        fixed: vec![domain.empty_lagrange_assigned(); cs.num_fixed_columns],
207        permutation: permutation::keygen::Assembly::new(params.n as usize, &cs.permutation),
208        selectors: vec![vec![false; params.n as usize]; cs.num_selectors],
209        usable_rows: 0..params.n as usize - (cs.blinding_factors() + 1),
210        _marker: std::marker::PhantomData,
211    };
212
213    // Synthesize the circuit to obtain URS
214    ConcreteCircuit::FloorPlanner::synthesize(
215        &mut assembly,
216        circuit,
217        config,
218        cs.constants.clone(),
219    )?;
220
221    let mut fixed = batch_invert_assigned(assembly.fixed);
222    let (cs, selector_polys) = cs.compress_selectors(assembly.selectors);
223    fixed.extend(
224        selector_polys
225            .into_iter()
226            .map(|poly| domain.lagrange_from_vec(poly)),
227    );
228
229    let permutation_vk = assembly
230        .permutation
231        .build_vk(params, &domain, &cs.permutation);
232
233    let fixed_commitments = fixed
234        .iter()
235        .map(|poly| params.commit_lagrange(poly, Blind::default()).to_affine())
236        .collect();
237
238    Ok(VerifyingKey::from_parts(
239        domain,
240        fixed_commitments,
241        permutation_vk,
242        cs,
243    ))
244}

246/// Generate a `ProvingKey` from a `VerifyingKey` and an instance of `Circuit`.
247pub fn keygen_pk<C, ConcreteCircuit>(
248    params: &Params<C>,
249    vk: VerifyingKey<C>,
250    circuit: &ConcreteCircuit,
251) -> Result<ProvingKey<C>, Error>
252where
253    C: CurveAffine,
254    ConcreteCircuit: Circuit<C::Scalar>,
255{
256    let mut cs = ConstraintSystem::default();
257    let config = ConcreteCircuit::configure(&mut cs);
258
259    let cs = cs;
260
261    if (params.n as usize) < cs.minimum_rows() {
262        return Err(Error::not_enough_rows_available(params.k));
263    }
264
265    let mut assembly: Assembly<C::Scalar> = Assembly {
266        k: params.k,
267        fixed: vec![vk.domain.empty_lagrange_assigned(); cs.num_fixed_columns],
268        permutation: permutation::keygen::Assembly::new(params.n as usize, &cs.permutation),
269        selectors: vec![vec![false; params.n as usize]; cs.num_selectors],
270        usable_rows: 0..params.n as usize - (cs.blinding_factors() + 1),
271        _marker: std::marker::PhantomData,
272    };
273
274    // Synthesize the circuit to obtain URS
275    ConcreteCircuit::FloorPlanner::synthesize(
276        &mut assembly,
277        circuit,
278        config,
279        cs.constants.clone(),
280    )?;
281
282    let mut fixed = batch_invert_assigned(assembly.fixed);
283    let (cs, selector_polys) = cs.compress_selectors(assembly.selectors);
284    fixed.extend(
285        selector_polys
286            .into_iter()
287            .map(|poly| vk.domain.lagrange_from_vec(poly)),
288    );
289
290    let fixed_polys: Vec<_> = fixed
291        .iter()
292        .map(|poly| vk.domain.lagrange_to_coeff(poly.clone()))
293        .collect();
294
295    let fixed_cosets = fixed_polys
296        .iter()
297        .map(|poly| vk.domain.coeff_to_extended(poly.clone()))
298        .collect();
299
300    let permutation_pk = assembly
301        .permutation
302        .build_pk(params, &vk.domain, &cs.permutation);
303
304    // Compute l_0(X)
305    // TODO: this can be done more efficiently
306    let mut l0 = vk.domain.empty_lagrange();
307    l0[0] = C::Scalar::ONE;
308    let l0 = vk.domain.lagrange_to_coeff(l0);
309    let l0 = vk.domain.coeff_to_extended(l0);
310
311    // Compute l_blind(X) which evaluates to 1 for each blinding factor row
312    // and 0 otherwise over the domain.
313    let mut l_blind = vk.domain.empty_lagrange();
314    for evaluation in l_blind[..].iter_mut().rev().take(cs.blinding_factors()) {
315        *evaluation = C::Scalar::ONE;
316    }
317    let l_blind = vk.domain.lagrange_to_coeff(l_blind);
318    let l_blind = vk.domain.coeff_to_extended(l_blind);
319
320    // Compute l_last(X) which evaluates to 1 on the first inactive row (just
321    // before the blinding factors) and 0 otherwise over the domain
322    let mut l_last = vk.domain.empty_lagrange();
323    l_last[params.n as usize - cs.blinding_factors() - 1] = C::Scalar::ONE;
324    let l_last = vk.domain.lagrange_to_coeff(l_last);
325    let l_last = vk.domain.coeff_to_extended(l_last);
326
327    Ok(ProvingKey {
328        vk,
329        l0,
330        l_blind,
331        l_last,
332        fixed_values: fixed,
333        fixed_polys,
334        fixed_cosets,
335        permutation: permutation_pk,
336    })
