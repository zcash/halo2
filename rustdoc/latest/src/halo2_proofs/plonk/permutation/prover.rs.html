45impl Argument {
46    #[allow(clippy::too_many_arguments)]
47    pub(in crate::plonk) fn commit<
48        C: CurveAffine,
49        E: EncodedChallenge<C>,
50        Ev: Copy + Send + Sync,
51        R: RngCore,
52        T: TranscriptWrite<C, E>,
53    >(
54        &self,
55        params: &Params<C>,
56        pk: &plonk::ProvingKey<C>,
57        pkey: &ProvingKey<C>,
58        advice: &[Polynomial<C::Scalar, LagrangeCoeff>],
59        fixed: &[Polynomial<C::Scalar, LagrangeCoeff>],
60        instance: &[Polynomial<C::Scalar, LagrangeCoeff>],
61        beta: ChallengeBeta<C>,
62        gamma: ChallengeGamma<C>,
63        evaluator: &mut poly::Evaluator<Ev, C::Scalar, ExtendedLagrangeCoeff>,
64        mut rng: R,
65        transcript: &mut T,
66    ) -> Result<Committed<C, Ev>, Error> {
67        let domain = &pk.vk.domain;
68
69        // How many columns can be included in a single permutation polynomial?
70        // We need to multiply by z(X) and (1 - (l_last(X) + l_blind(X))). This
71        // will never underflow because of the requirement of at least a degree
72        // 3 circuit for the permutation argument.
73        assert!(pk.vk.cs_degree >= 3);
74        let chunk_len = pk.vk.cs_degree - 2;
75        let blinding_factors = pk.vk.cs.blinding_factors();
76
77        // Each column gets its own delta power.
78        let mut deltaomega = C::Scalar::ONE;
79
80        // Track the "last" value from the previous column set
81        let mut last_z = C::Scalar::ONE;
82
83        let mut sets = vec![];
84
85        for (columns, permutations) in self
86            .columns
87            .chunks(chunk_len)
88            .zip(pkey.permutations.chunks(chunk_len))
89        {
90            // Goal is to compute the products of fractions
91            //
92            // (p_j(\omega^i) + \delta^j \omega^i \beta + \gamma) /
93            // (p_j(\omega^i) + \beta s_j(\omega^i) + \gamma)
94            //
95            // where p_j(X) is the jth column in this permutation,
96            // and i is the ith row of the column.
97
98            let mut modified_values = vec![C::Scalar::ONE; params.n as usize];
99
100            // Iterate over each column of the permutation
101            for (&column, permuted_column_values) in columns.iter().zip(permutations.iter()) {
102                let values = match column.column_type() {
103                    Any::Advice => advice,
104                    Any::Fixed => fixed,
105                    Any::Instance => instance,
106                };
107                parallelize(&mut modified_values, |modified_values, start| {
108                    for ((modified_values, value), permuted_value) in modified_values
109                        .iter_mut()
110                        .zip(values[column.index()][start..].iter())
111                        .zip(permuted_column_values[start..].iter())
112                    {
113                        *modified_values *= &(*beta * permuted_value + &*gamma + value);
114                    }
115                });
116            }
117
118            // Invert to obtain the denominator for the permutation product polynomial
119            modified_values.batch_invert();
120
121            // Iterate over each column again, this time finishing the computation
122            // of the entire fraction by computing the numerators
123            for &column in columns.iter() {
124                let omega = domain.get_omega();
125                let values = match column.column_type() {
126                    Any::Advice => advice,
127                    Any::Fixed => fixed,
128                    Any::Instance => instance,
129                };
130                parallelize(&mut modified_values, |modified_values, start| {
131                    let mut deltaomega = deltaomega * &omega.pow_vartime([start as u64]);
132                    for (modified_values, value) in modified_values
133                        .iter_mut()
134                        .zip(values[column.index()][start..].iter())
135                    {
136                        // Multiply by p_j(\omega^i) + \delta^j \omega^i \beta
137                        *modified_values *= &(deltaomega * &*beta + &*gamma + value);
138                        deltaomega *= &omega;
139                    }
140                });
141                deltaomega *= &C::Scalar::DELTA;
142            }
143
144            // The modified_values vector is a vector of products of fractions
145            // of the form
146            //
147            // (p_j(\omega^i) + \delta^j \omega^i \beta + \gamma) /
148            // (p_j(\omega^i) + \beta s_j(\omega^i) + \gamma)
149            //
150            // where i is the index into modified_values, for the jth column in
151            // the permutation
152
153            // Compute the evaluations of the permutation product polynomial
154            // over our domain, starting with z[0] = 1
155            let mut z = vec![last_z];
156            for row in 1..(params.n as usize) {
157                let mut tmp = z[row - 1];
158
159                tmp *= &modified_values[row - 1];
160                z.push(tmp);
161            }
162            let mut z = domain.lagrange_from_vec(z);
163            // Set blinding factors
164            for z in &mut z[params.n as usize - blinding_factors..] {
165                *z = C::Scalar::random(&mut rng);
166            }
167            // Set new last_z
168            last_z = z[params.n as usize - (blinding_factors + 1)];
169
170            let blind = Blind(C::Scalar::random(&mut rng));
171
172            let permutation_product_commitment_projective = params.commit_lagrange(&z, blind);
173            let permutation_product_blind = blind;
174            let z = domain.lagrange_to_coeff(z);
175            let permutation_product_poly = z.clone();
176
177            let permutation_product_coset =
178                evaluator.register_poly(domain.coeff_to_extended(z.clone()));
179
180            let permutation_product_commitment =
181                permutation_product_commitment_projective.to_affine();
182
183            // Hash the permutation product commitment
184            transcript.write_point(permutation_product_commitment)?;
185
186            sets.push(CommittedSet {
187                permutation_product_poly,
188                permutation_product_coset,
189                permutation_product_blind,
190            });
191        }
192
193        Ok(Committed { sets })
194    }
195}

197impl<C: CurveAffine, Ev: Copy + Send + Sync> Committed<C, Ev> {
198    #[allow(clippy::too_many_arguments)]
199    pub(in crate::plonk) fn construct<'a>(
200        self,
201        pk: &'a plonk::ProvingKey<C>,
202        p: &'a Argument,
203        advice_cosets: &'a [poly::AstLeaf<Ev, ExtendedLagrangeCoeff>],
204        fixed_cosets: &'a [poly::AstLeaf<Ev, ExtendedLagrangeCoeff>],
205        instance_cosets: &'a [poly::AstLeaf<Ev, ExtendedLagrangeCoeff>],
206        permutation_cosets: &'a [poly::AstLeaf<Ev, ExtendedLagrangeCoeff>],
207        l0: poly::AstLeaf<Ev, ExtendedLagrangeCoeff>,
208        l_blind: poly::AstLeaf<Ev, ExtendedLagrangeCoeff>,
209        l_last: poly::AstLeaf<Ev, ExtendedLagrangeCoeff>,
210        beta: ChallengeBeta<C>,
211        gamma: ChallengeGamma<C>,
212    ) -> (
213        Constructed<C>,
214        impl Iterator<Item = poly::Ast<Ev, C::Scalar, ExtendedLagrangeCoeff>> + 'a,
215    ) {
216        let chunk_len = pk.vk.cs_degree - 2;
217        let blinding_factors = pk.vk.cs.blinding_factors();
218        let last_rotation = Rotation(-((blinding_factors + 1) as i32));
219
220        let constructed = Constructed {
221            sets: self
222                .sets
223                .iter()
224                .map(|set| ConstructedSet {
225                    permutation_product_poly: set.permutation_product_poly.clone(),
226                    permutation_product_blind: set.permutation_product_blind,
227                })
228                .collect(),
229        };
230
231        let expressions = iter::empty()
232            // Enforce only for the first set.
233            // l_0(X) * (1 - z_0(X)) = 0
234            .chain(
235                self.sets
236                    .first()
237                    .map(|first_set| (poly::Ast::one() - first_set.permutation_product_coset) * l0),
238            )
239            // Enforce only for the last set.
240            // l_last(X) * (z_l(X)^2 - z_l(X)) = 0
241            .chain(self.sets.last().map(|last_set| {
242                ((poly::Ast::from(last_set.permutation_product_coset)
243                    * last_set.permutation_product_coset)
244                    - last_set.permutation_product_coset)
245                    * l_last
246            }))
247            // Except for the first set, enforce.
248            // l_0(X) * (z_i(X) - z_{i-1}(\omega^(last) X)) = 0
249            .chain(
250                self.sets
251                    .iter()
252                    .skip(1)
253                    .zip(self.sets.iter())
254                    .map(|(set, last_set)| {
255                        (poly::Ast::from(set.permutation_product_coset)
256                            - last_set
257                                .permutation_product_coset
258                                .with_rotation(last_rotation))
259                            * l0
260                    })
261                    .collect::<Vec<_>>(),
262            )
263            // And for all the sets we enforce:
264            // (1 - (l_last(X) + l_blind(X))) * (
265            //   z_i(\omega X) \prod_j (p(X) + \beta s_j(X) + \gamma)
266            // - z_i(X) \prod_j (p(X) + \delta^j \beta X + \gamma)
267            // )
268            .chain(
269                self.sets
270                    .into_iter()
271                    .zip(p.columns.chunks(chunk_len))
272                    .zip(permutation_cosets.chunks(chunk_len))
273                    .enumerate()
274                    .map(move |(chunk_index, ((set, columns), cosets))| {
275                        let mut left = poly::Ast::<_, C::Scalar, _>::from(
276                            set.permutation_product_coset
277                                .with_rotation(Rotation::next()),
278                        );
279                        for (values, permutation) in columns
280                            .iter()
281                            .map(|&column| match column.column_type() {
282                                Any::Advice => &advice_cosets[column.index()],
283                                Any::Fixed => &fixed_cosets[column.index()],
284                                Any::Instance => &instance_cosets[column.index()],
285                            })
286                            .zip(cosets.iter())
287                        {
288                            left *= poly::Ast::<_, C::Scalar, _>::from(*values)
289                                + (poly::Ast::ConstantTerm(*beta) * poly::Ast::from(*permutation))
290                                + poly::Ast::ConstantTerm(*gamma);
291                        }
292
293                        let mut right = poly::Ast::from(set.permutation_product_coset);
294                        let mut current_delta = *beta
295                            * &(C::Scalar::DELTA.pow_vartime([(chunk_index * chunk_len) as u64]));
296                        for values in columns.iter().map(|&column| match column.column_type() {
297                            Any::Advice => &advice_cosets[column.index()],
298                            Any::Fixed => &fixed_cosets[column.index()],
299                            Any::Instance => &instance_cosets[column.index()],
300                        }) {
301                            right *= poly::Ast::from(*values)
302                                + poly::Ast::LinearTerm(current_delta)
303                                + poly::Ast::ConstantTerm(*gamma);
304                            current_delta *= &C::Scalar::DELTA;
305                        }
306
307                        (left - right) * (poly::Ast::one() - (poly::Ast::from(l_last) + l_blind))
308                    }),
309            );
310
311        (constructed, expressions)
312    }
313}

315impl<C: CurveAffine> super::ProvingKey<C> {
316    pub(in crate::plonk) fn open(
317        &self,
318        x: ChallengeX<C>,
319    ) -> impl Iterator<Item = ProverQuery<'_, C>> + Clone {
320        self.polys.iter().map(move |poly| ProverQuery {
321            point: *x,
322            poly,
323            blind: Blind::default(),
324        })
325    }
326
327    pub(in crate::plonk) fn evaluate<E: EncodedChallenge<C>, T: TranscriptWrite<C, E>>(
328        &self,
329        x: ChallengeX<C>,
330        transcript: &mut T,
331    ) -> Result<(), Error> {
332        // Hash permutation evals
333        for eval in self.polys.iter().map(|poly| eval_polynomial(poly, *x)) {
334            transcript.write_scalar(eval)?;
335        }
336
337        Ok(())
338    }
339}

341impl<C: CurveAffine> Constructed<C> {
342    pub(in crate::plonk) fn evaluate<E: EncodedChallenge<C>, T: TranscriptWrite<C, E>>(
343        self,
344        pk: &plonk::ProvingKey<C>,
345        x: ChallengeX<C>,
346        transcript: &mut T,
347    ) -> Result<Evaluated<C>, Error> {
348        let domain = &pk.vk.domain;
349        let blinding_factors = pk.vk.cs.blinding_factors();
350
351        {
352            let mut sets = self.sets.iter();
353
354            while let Some(set) = sets.next() {
355                let permutation_product_eval = eval_polynomial(&set.permutation_product_poly, *x);
356
357                let permutation_product_next_eval = eval_polynomial(
358                    &set.permutation_product_poly,
359                    domain.rotate_omega(*x, Rotation::next()),
360                );
361
362                // Hash permutation product evals
363                for eval in iter::empty()
364                    .chain(Some(&permutation_product_eval))
365                    .chain(Some(&permutation_product_next_eval))
366                {
367                    transcript.write_scalar(*eval)?;
368                }
369
370                // If we have any remaining sets to process, evaluate this set at omega^u
371                // so we can constrain the last value of its running product to equal the
372                // first value of the next set's running product, chaining them together.
373                if sets.len() > 0 {
374                    let permutation_product_last_eval = eval_polynomial(
375                        &set.permutation_product_poly,
376                        domain.rotate_omega(*x, Rotation(-((blinding_factors + 1) as i32))),
377                    );
378
379                    transcript.write_scalar(permutation_product_last_eval)?;
380                }
381            }
382        }
383
384        Ok(Evaluated { constructed: self })
385    }
386}

388impl<C: CurveAffine> Evaluated<C> {
389    pub(in crate::plonk) fn open<'a>(
390        &'a self,
391        pk: &'a plonk::ProvingKey<C>,
392        x: ChallengeX<C>,
393    ) -> impl Iterator<Item = ProverQuery<'a, C>> + Clone {
394        let blinding_factors = pk.vk.cs.blinding_factors();
395        let x_next = pk.vk.domain.rotate_omega(*x, Rotation::next());
396        let x_last = pk
397            .vk
398            .domain
399            .rotate_omega(*x, Rotation(-((blinding_factors + 1) as i32)));
400
401        iter::empty()
402            .chain(self.constructed.sets.iter().flat_map(move |set| {
403                iter::empty()
404                    // Open permutation product commitments at x and \omega x
405                    .chain(Some(ProverQuery {
406                        point: *x,
407                        poly: &set.permutation_product_poly,
408                        blind: set.permutation_product_blind,
409                    }))
410                    .chain(Some(ProverQuery {
411                        point: x_next,
412                        poly: &set.permutation_product_poly,
413                        blind: set.permutation_product_blind,
414                    }))
415            }))
416            // Open it at \omega^{last} x for all but the last set. This rotation is only
417            // sensical for the first row, but we only use this rotation in a constraint
418            // that is gated on l_0.
419            .chain(
420                self.constructed
421                    .sets
422                    .iter()
423                    .rev()
424                    .skip(1)
425                    .flat_map(move |set| {
426                        Some(ProverQuery {
427                            point: x_last,
428                            poly: &set.permutation_product_poly,
429                            blind: set.permutation_product_blind,
430                        })
431                    }),
432            )
433    }
