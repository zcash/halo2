33impl Argument {
34    pub(crate) fn read_product_commitments<
35        C: CurveAffine,
36        E: EncodedChallenge<C>,
37        T: TranscriptRead<C, E>,
38    >(
39        &self,
40        vk: &plonk::VerifyingKey<C>,
41        transcript: &mut T,
42    ) -> Result<Committed<C>, Error> {
43        let chunk_len = vk.cs_degree - 2;
44
45        let permutation_product_commitments = self
46            .columns
47            .chunks(chunk_len)
48            .map(|_| transcript.read_point())
49            .collect::<Result<Vec<_>, _>>()?;
50
51        Ok(Committed {
52            permutation_product_commitments,
53        })
54    }
55}

59        &self,
60        transcript: &mut T,
61    ) -> Result<CommonEvaluated<C>, Error> {
62        let permutation_evals = self
63            .commitments
64            .iter()

66            .collect::<Result<Vec<_>, _>>()?;

72impl<C: CurveAffine> Committed<C> {
73    pub(crate) fn evaluate<E: EncodedChallenge<C>, T: TranscriptRead<C, E>>(
74        self,
75        transcript: &mut T,
76    ) -> Result<Evaluated<C>, Error> {
77        let mut sets = vec![];
78
79        let mut iter = self.permutation_product_commitments.into_iter();
80
81        while let Some(permutation_product_commitment) = iter.next() {
82            let permutation_product_eval = transcript.read_scalar()?;
83            let permutation_product_next_eval = transcript.read_scalar()?;
84            let permutation_product_last_eval = if iter.len() > 0 {
85                Some(transcript.read_scalar()?)
86            } else {
87                None
88            };
89
90            sets.push(EvaluatedSet {
91                permutation_product_commitment,
92                permutation_product_eval,
93                permutation_product_next_eval,
94                permutation_product_last_eval,
95            });
96        }
97
98        Ok(Evaluated { sets })
99    }
100}

102impl<C: CurveAffine> Evaluated<C> {
103    #[allow(clippy::too_many_arguments)]
104    pub(in crate::plonk) fn expressions<'a>(
105        &'a self,
106        vk: &'a plonk::VerifyingKey<C>,
107        p: &'a Argument,
108        common: &'a CommonEvaluated<C>,
109        advice_evals: &'a [C::Scalar],
110        fixed_evals: &'a [C::Scalar],
111        instance_evals: &'a [C::Scalar],
112        l_0: C::Scalar,
113        l_last: C::Scalar,
114        l_blind: C::Scalar,
115        beta: ChallengeBeta<C>,
116        gamma: ChallengeGamma<C>,
117        x: ChallengeX<C>,
118    ) -> impl Iterator<Item = C::Scalar> + 'a {
119        let chunk_len = vk.cs_degree - 2;
120        iter::empty()
121            // Enforce only for the first set.
122            // l_0(X) * (1 - z_0(X)) = 0
123            .chain(
124                self.sets
125                    .first()
126                    .map(|first_set| l_0 * &(C::Scalar::ONE - &first_set.permutation_product_eval)),
127            )
128            // Enforce only for the last set.
129            // l_last(X) * (z_l(X)^2 - z_l(X)) = 0
130            .chain(self.sets.last().map(|last_set| {
131                (last_set.permutation_product_eval.square() - &last_set.permutation_product_eval)
132                    * &l_last
133            }))
134            // Except for the first set, enforce.
135            // l_0(X) * (z_i(X) - z_{i-1}(\omega^(last) X)) = 0
136            .chain(
137                self.sets
138                    .iter()
139                    .skip(1)
140                    .zip(self.sets.iter())
141                    .map(|(set, last_set)| {
142                        (
143                            set.permutation_product_eval,
144                            last_set.permutation_product_last_eval.unwrap(),
145                        )
146                    })
147                    .map(move |(set, prev_last)| (set - &prev_last) * &l_0),
148            )
149            // And for all the sets we enforce:
150            // (1 - (l_last(X) + l_blind(X))) * (
151            //   z_i(\omega X) \prod (p(X) + \beta s_i(X) + \gamma)
152            // - z_i(X) \prod (p(X) + \delta^i \beta X + \gamma)
153            // )
154            .chain(
155                self.sets
156                    .iter()
157                    .zip(p.columns.chunks(chunk_len))
158                    .zip(common.permutation_evals.chunks(chunk_len))
159                    .enumerate()
160                    .map(move |(chunk_index, ((set, columns), permutation_evals))| {
161                        let mut left = set.permutation_product_next_eval;
162                        for (eval, permutation_eval) in columns
163                            .iter()
164                            .map(|&column| match column.column_type() {
165                                Any::Advice => advice_evals[vk.cs.get_any_query_index(column)],
166                                Any::Fixed => fixed_evals[vk.cs.get_any_query_index(column)],
167                                Any::Instance => instance_evals[vk.cs.get_any_query_index(column)],
168                            })
169                            .zip(permutation_evals.iter())
170                        {
171                            left *= &(eval + &(*beta * permutation_eval) + &*gamma);
172                        }
173
174                        let mut right = set.permutation_product_eval;
175                        let mut current_delta = (*beta * &*x)
176                            * &(C::Scalar::DELTA.pow_vartime([(chunk_index * chunk_len) as u64]));
177                        for eval in columns.iter().map(|&column| match column.column_type() {
178                            Any::Advice => advice_evals[vk.cs.get_any_query_index(column)],
179                            Any::Fixed => fixed_evals[vk.cs.get_any_query_index(column)],
180                            Any::Instance => instance_evals[vk.cs.get_any_query_index(column)],
181                        }) {
182                            right *= &(eval + &current_delta + &*gamma);
183                            current_delta *= &C::Scalar::DELTA;
184                        }
185
186                        (left - &right) * (C::Scalar::ONE - &(l_last + &l_blind))
187                    }),
188            )
189    }

191    pub(in crate::plonk) fn queries<'r, 'params: 'r>(
192        &'r self,
193        vk: &'r plonk::VerifyingKey<C>,
194        x: ChallengeX<C>,
195    ) -> impl Iterator<Item = VerifierQuery<'r, 'params, C>> + Clone {
196        let blinding_factors = vk.cs.blinding_factors();
197        let x_next = vk.domain.rotate_omega(*x, Rotation::next());
198        let x_last = vk
199            .domain
200            .rotate_omega(*x, Rotation(-((blinding_factors + 1) as i32)));
201
202        iter::empty()
203            .chain(self.sets.iter().flat_map(move |set| {
204                iter::empty()
205                    // Open permutation product commitments at x and \omega^{-1} x
206                    // Open permutation product commitments at x and \omega x
207                    .chain(Some(VerifierQuery::new_commitment(
208                        &set.permutation_product_commitment,
209                        *x,
210                        set.permutation_product_eval,
211                    )))
212                    .chain(Some(VerifierQuery::new_commitment(
213                        &set.permutation_product_commitment,
214                        x_next,
215                        set.permutation_product_next_eval,
216                    )))
217            }))
218            // Open it at \omega^{last} x for all but the last set
219            .chain(self.sets.iter().rev().skip(1).flat_map(move |set| {
220                Some(VerifierQuery::new_commitment(
221                    &set.permutation_product_commitment,
222                    x_last,
223                    set.permutation_product_last_eval.unwrap(),
224                ))
225            }))
226    }

229impl<C: CurveAffine> CommonEvaluated<C> {
230    pub(in crate::plonk) fn queries<'r, 'params: 'r>(
231        &'r self,
232        vkey: &'r VerifyingKey<C>,
233        x: ChallengeX<C>,
234    ) -> impl Iterator<Item = VerifierQuery<'r, 'params, C>> + Clone {
235        // Open permutation commitments for each permutation argument at x
236        vkey.commitments
237            .iter()
238            .zip(self.permutation_evals.iter())
239            .map(move |(commitment, &eval)| VerifierQuery::new_commitment(commitment, *x, eval))
