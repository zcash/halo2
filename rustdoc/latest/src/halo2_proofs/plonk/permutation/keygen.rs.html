17pub(crate) struct Assembly {
18    columns: Vec<Column<Any>>,
19    pub(crate) mapping: Vec<Vec<(usize, usize)>>,
20    aux: Vec<Vec<(usize, usize)>>,
21    sizes: Vec<Vec<usize>>,
22}

24impl Assembly {
25    pub(crate) fn new(n: usize, p: &Argument) -> Self {
26        // Initialize the copy vector to keep track of copy constraints in all
27        // the permutation arguments.
28        let mut columns = vec![];
29        for i in 0..p.columns.len() {
30            // Computes [(i, 0), (i, 1), ..., (i, n - 1)]
31            columns.push((0..n).map(|j| (i, j)).collect());
32        }
33
34        // Before any equality constraints are applied, every cell in the permutation is
35        // in a 1-cycle; therefore mapping and aux are identical, because every cell is
36        // its own distinguished element.
37        Assembly {
38            columns: p.columns.clone(),
39            mapping: columns.clone(),
40            aux: columns,
41            sizes: vec![vec![1usize; n]; p.columns.len()],
42        }
43    }
44
45    pub(crate) fn copy(
46        &mut self,
47        left_column: Column<Any>,
48        left_row: usize,
49        right_column: Column<Any>,
50        right_row: usize,
51    ) -> Result<(), Error> {
52        let left_column = self
53            .columns
54            .iter()
55            .position(|c| c == &left_column)
56            .ok_or(Error::ColumnNotInPermutation(left_column))?;
57        let right_column = self
58            .columns
59            .iter()
60            .position(|c| c == &right_column)
61            .ok_or(Error::ColumnNotInPermutation(right_column))?;
62
63        // Check bounds
64        if left_row >= self.mapping[left_column].len()
65            || right_row >= self.mapping[right_column].len()
66        {
67            return Err(Error::BoundsFailure);
68        }
69
70        // See book/src/design/permutation.md for a description of this algorithm.
71
72        let mut left_cycle = self.aux[left_column][left_row];
73        let mut right_cycle = self.aux[right_column][right_row];
74
75        // If left and right are in the same cycle, do nothing.
76        if left_cycle == right_cycle {
77            return Ok(());
78        }
79
80        if self.sizes[left_cycle.0][left_cycle.1] < self.sizes[right_cycle.0][right_cycle.1] {
81            std::mem::swap(&mut left_cycle, &mut right_cycle);
82        }
83
84        // Merge the right cycle into the left one.
85        self.sizes[left_cycle.0][left_cycle.1] += self.sizes[right_cycle.0][right_cycle.1];
86        let mut i = right_cycle;
87        loop {
88            self.aux[i.0][i.1] = left_cycle;
89            i = self.mapping[i.0][i.1];
90            if i == right_cycle {
91                break;
92            }
93        }
94
95        let tmp = self.mapping[left_column][left_row];
96        self.mapping[left_column][left_row] = self.mapping[right_column][right_row];
97        self.mapping[right_column][right_row] = tmp;
98
99        Ok(())
100    }

102    pub(crate) fn build_vk<C: CurveAffine>(
103        self,
104        params: &Params<C>,
105        domain: &EvaluationDomain<C::Scalar>,
106        p: &Argument,
107    ) -> VerifyingKey<C> {
108        // Compute [omega^0, omega^1, ..., omega^{params.n - 1}]
109        let mut omega_powers = Vec::with_capacity(params.n as usize);
110        {
111            let mut cur = C::Scalar::ONE;
112            for _ in 0..params.n {
113                omega_powers.push(cur);
114                cur *= &domain.get_omega();
115            }
116        }
117
118        // Compute [omega_powers * \delta^0, omega_powers * \delta^1, ..., omega_powers * \delta^m]
119        let mut deltaomega = Vec::with_capacity(p.columns.len());
120        {
121            let mut cur = C::Scalar::ONE;
122            for _ in 0..p.columns.len() {
123                let mut omega_powers = omega_powers.clone();
124                for o in &mut omega_powers {
125                    *o *= &cur;
126                }
127
128                deltaomega.push(omega_powers);
129
130                cur *= &C::Scalar::DELTA;
131            }
132        }
133
134        // Pre-compute commitments for the URS.
135        let mut commitments = Vec::with_capacity(p.columns.len());
136        for i in 0..p.columns.len() {
137            // Computes the permutation polynomial based on the permutation
138            // description in the assembly.
139            let mut permutation_poly = domain.empty_lagrange();
140            for (j, p) in permutation_poly.iter_mut().enumerate() {
141                let (permuted_i, permuted_j) = self.mapping[i][j];
142                *p = deltaomega[permuted_i][permuted_j];
143            }
144
145            // Compute commitment to permutation polynomial
146            commitments.push(
147                params
148                    .commit_lagrange(&permutation_poly, Blind::default())
149                    .to_affine(),
150            );
151        }
152        VerifyingKey { commitments }
153    }

155    pub(crate) fn build_pk<C: CurveAffine>(
156        self,
157        params: &Params<C>,
158        domain: &EvaluationDomain<C::Scalar>,
159        p: &Argument,
160    ) -> ProvingKey<C> {
161        // Compute [omega^0, omega^1, ..., omega^{params.n - 1}]
162        let mut omega_powers = Vec::with_capacity(params.n as usize);
163        {
164            let mut cur = C::Scalar::ONE;
165            for _ in 0..params.n {
166                omega_powers.push(cur);
167                cur *= &domain.get_omega();
168            }
169        }
170
171        // Compute [omega_powers * \delta^0, omega_powers * \delta^1, ..., omega_powers * \delta^m]
172        let mut deltaomega = Vec::with_capacity(p.columns.len());
173        {
174            let mut cur = C::Scalar::ONE;
175            for _ in 0..p.columns.len() {
176                let mut omega_powers = omega_powers.clone();
177                for o in &mut omega_powers {
178                    *o *= &cur;
179                }
180
181                deltaomega.push(omega_powers);
182
183                cur *= &C::Scalar::DELTA;
184            }
185        }
186
187        // Compute permutation polynomials, convert to coset form.
188        let mut permutations = vec![];
189        let mut polys = vec![];
190        let mut cosets = vec![];
191        for i in 0..p.columns.len() {
192            // Computes the permutation polynomial based on the permutation
193            // description in the assembly.
194            let mut permutation_poly = domain.empty_lagrange();
195            for (j, p) in permutation_poly.iter_mut().enumerate() {
196                let (permuted_i, permuted_j) = self.mapping[i][j];
197                *p = deltaomega[permuted_i][permuted_j];
198            }
199
200            // Store permutation polynomial and precompute its coset evaluation
201            permutations.push(permutation_poly.clone());
202            let poly = domain.lagrange_to_coeff(permutation_poly);
203            polys.push(poly.clone());
204            cosets.push(domain.coeff_to_extended(poly));
205        }
206        ProvingKey {
207            permutations,
208            polys,
209            cosets,
210        }
211    }
