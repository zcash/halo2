21/// Trait representing a strategy for verifying Halo 2 proofs.
22pub trait VerificationStrategy<'params, C: CurveAffine> {
23    /// The output type of this verification strategy after processing a proof.
24    type Output;
25
26    /// Obtains an MSM from the verifier strategy and yields back the strategy's
27    /// output.
28    fn process<E: EncodedChallenge<C>>(
29        self,
30        f: impl FnOnce(MSM<'params, C>) -> Result<Guard<'params, C, E>, Error>,
31    ) -> Result<Self::Output, Error>;
32}
33
34/// A verifier that checks a single proof at a time.
35#[derive(Debug)]
36pub struct SingleVerifier<'params, C: CurveAffine> {
37    msm: MSM<'params, C>,
38}
39
40impl<'params, C: CurveAffine> SingleVerifier<'params, C> {
41    /// Constructs a new single proof verifier.
42    pub fn new(params: &'params Params<C>) -> Self {
43        SingleVerifier {
44            msm: MSM::new(params),
45        }
46    }
47}
48
49impl<'params, C: CurveAffine> VerificationStrategy<'params, C> for SingleVerifier<'params, C> {
50    type Output = ();
51
52    fn process<E: EncodedChallenge<C>>(
53        self,
54        f: impl FnOnce(MSM<'params, C>) -> Result<Guard<'params, C, E>, Error>,
55    ) -> Result<Self::Output, Error> {
56        let guard = f(self.msm)?;
57        let msm = guard.use_challenges();
58        if msm.eval() {
59            Ok(())
60        } else {
61            Err(Error::ConstraintSystemFailure)
62        }
63    }
64}

66/// Returns a boolean indicating whether or not the proof is valid
67pub fn verify_proof<
68    'params,
69    C: CurveAffine,
70    E: EncodedChallenge<C>,
71    T: TranscriptRead<C, E>,
72    V: VerificationStrategy<'params, C>,
73>(
74    params: &'params Params<C>,
75    vk: &VerifyingKey<C>,
76    strategy: V,
77    instances: &[&[&[C::Scalar]]],
78    transcript: &mut T,
79) -> Result<V::Output, Error> {
80    // Check that instances matches the expected number of instance columns
81    for instances in instances.iter() {
82        if instances.len() != vk.cs.num_instance_columns {
83            return Err(Error::InvalidInstances);
84        }
85    }
86
87    let instance_commitments = instances
88        .iter()
89        .map(|instance| {
90            instance
91                .iter()
92                .map(|instance| {
93                    if instance.len() > params.n as usize - (vk.cs.blinding_factors() + 1) {
94                        return Err(Error::InstanceTooLarge);
95                    }
96                    let mut poly = instance.to_vec();
97                    poly.resize(params.n as usize, C::Scalar::ZERO);
98                    let poly = vk.domain.lagrange_from_vec(poly);
99
100                    Ok(params.commit_lagrange(&poly, Blind::default()).to_affine())
101                })
102                .collect::<Result<Vec<_>, _>>()
103        })
104        .collect::<Result<Vec<_>, _>>()?;
105
106    let num_proofs = instance_commitments.len();
107
108    // Hash verification key into transcript
109    vk.hash_into(transcript)?;
110
111    for instance_commitments in instance_commitments.iter() {
112        // Hash the instance (external) commitments into the transcript
113        for commitment in instance_commitments {
114            transcript.common_point(*commitment)?
115        }
116    }
117
118    let advice_commitments = (0..num_proofs)
119        .map(|_| -> Result<Vec<_>, _> {
120            // Hash the prover's advice commitments into the transcript
121            read_n_points(transcript, vk.cs.num_advice_columns)
122        })
123        .collect::<Result<Vec<_>, _>>()?;
124
125    // Sample theta challenge for keeping lookup columns linearly independent
126    let theta: ChallengeTheta<_> = transcript.squeeze_challenge_scalar();
127
128    let lookups_permuted = (0..num_proofs)
129        .map(|_| -> Result<Vec<_>, _> {
130            // Hash each lookup permuted commitment
131            vk.cs
132                .lookups
133                .iter()
134                .map(|argument| argument.read_permuted_commitments(transcript))
135                .collect::<Result<Vec<_>, _>>()
136        })
137        .collect::<Result<Vec<_>, _>>()?;
138
139    // Sample beta challenge
140    let beta: ChallengeBeta<_> = transcript.squeeze_challenge_scalar();
141
142    // Sample gamma challenge
143    let gamma: ChallengeGamma<_> = transcript.squeeze_challenge_scalar();
144
145    let permutations_committed = (0..num_proofs)
146        .map(|_| {
147            // Hash each permutation product commitment
148            vk.cs.permutation.read_product_commitments(vk, transcript)
149        })
150        .collect::<Result<Vec<_>, _>>()?;
151
152    let lookups_committed = lookups_permuted
153        .into_iter()
154        .map(|lookups| {
155            // Hash each lookup product commitment
156            lookups
157                .into_iter()
158                .map(|lookup| lookup.read_product_commitment(transcript))
159                .collect::<Result<Vec<_>, _>>()
160        })
161        .collect::<Result<Vec<_>, _>>()?;
162
163    let vanishing = vanishing::Argument::read_commitments_before_y(transcript)?;
164
165    // Sample y challenge, which keeps the gates linearly independent.
166    let y: ChallengeY<_> = transcript.squeeze_challenge_scalar();
167
168    let vanishing = vanishing.read_commitments_after_y(vk, transcript)?;
169
170    // Sample x challenge, which is used to ensure the circuit is
171    // satisfied with high probability.
172    let x: ChallengeX<_> = transcript.squeeze_challenge_scalar();
173    let instance_evals = (0..num_proofs)
174        .map(|_| -> Result<Vec<_>, _> { read_n_scalars(transcript, vk.cs.instance_queries.len()) })
175        .collect::<Result<Vec<_>, _>>()?;
176
177    let advice_evals = (0..num_proofs)
178        .map(|_| -> Result<Vec<_>, _> { read_n_scalars(transcript, vk.cs.advice_queries.len()) })
179        .collect::<Result<Vec<_>, _>>()?;
180
181    let fixed_evals = read_n_scalars(transcript, vk.cs.fixed_queries.len())?;
182
183    let vanishing = vanishing.evaluate_after_x(transcript)?;
184
185    let permutations_common = vk.permutation.evaluate(transcript)?;
186
187    let permutations_evaluated = permutations_committed
188        .into_iter()
189        .map(|permutation| permutation.evaluate(transcript))
190        .collect::<Result<Vec<_>, _>>()?;
191
192    let lookups_evaluated = lookups_committed
193        .into_iter()
194        .map(|lookups| -> Result<Vec<_>, _> {
195            lookups
196                .into_iter()
197                .map(|lookup| lookup.evaluate(transcript))
198                .collect::<Result<Vec<_>, _>>()
199        })
200        .collect::<Result<Vec<_>, _>>()?;
201
202    // This check ensures the circuit is satisfied so long as the polynomial
203    // commitments open to the correct values.
204    let vanishing = {
205        // x^n
206        let xn = x.pow(&[params.n, 0, 0, 0]);
207
208        let blinding_factors = vk.cs.blinding_factors();
209        let l_evals = vk
210            .domain
211            .l_i_range(*x, xn, (-((blinding_factors + 1) as i32))..=0);
212        assert_eq!(l_evals.len(), 2 + blinding_factors);
213        let l_last = l_evals[0];
214        let l_blind: C::Scalar = l_evals[1..(1 + blinding_factors)]
215            .iter()
216            .fold(C::Scalar::ZERO, |acc, eval| acc + eval);
217        let l_0 = l_evals[1 + blinding_factors];
218
219        // Compute the expected value of h(x)
220        let expressions = advice_evals
221            .iter()
222            .zip(instance_evals.iter())
223            .zip(permutations_evaluated.iter())
224            .zip(lookups_evaluated.iter())
225            .flat_map(|(((advice_evals, instance_evals), permutation), lookups)| {
226                let fixed_evals = &fixed_evals;
227                std::iter::empty()
228                    // Evaluate the circuit using the custom gates provided
229                    .chain(vk.cs.gates.iter().flat_map(move |gate| {
230                        gate.polynomials().iter().map(move |poly| {
231                            poly.evaluate(
232                                &|scalar| scalar,
233                                &|_| panic!("virtual selectors are removed during optimization"),
234                                &|query| fixed_evals[query.index],
235                                &|query| advice_evals[query.index],
236                                &|query| instance_evals[query.index],
237                                &|a| -a,
238                                &|a, b| a + &b,
239                                &|a, b| a * &b,
240                                &|a, scalar| a * &scalar,
241                            )
242                        })
243                    }))
244                    .chain(permutation.expressions(
245                        vk,
246                        &vk.cs.permutation,
247                        &permutations_common,
248                        advice_evals,
249                        fixed_evals,
250                        instance_evals,
251                        l_0,
252                        l_last,
253                        l_blind,
254                        beta,
255                        gamma,
256                        x,
257                    ))
258                    .chain(
259                        lookups
260                            .iter()
261                            .zip(vk.cs.lookups.iter())
262                            .flat_map(move |(p, argument)| {
263                                p.expressions(
264                                    l_0,
265                                    l_last,
266                                    l_blind,
267                                    argument,
268                                    theta,
269                                    beta,
270                                    gamma,
271                                    advice_evals,
272                                    fixed_evals,
273                                    instance_evals,
274                                )
275                            })
276                            .into_iter(),
277                    )
278            });
279
280        vanishing.verify(params, expressions, y, xn)
281    };
282
283    let queries = instance_commitments
284        .iter()
285        .zip(instance_evals.iter())
286        .zip(advice_commitments.iter())
287        .zip(advice_evals.iter())
288        .zip(permutations_evaluated.iter())
289        .zip(lookups_evaluated.iter())
290        .flat_map(
291            |(
292                (
293                    (((instance_commitments, instance_evals), advice_commitments), advice_evals),
294                    permutation,
295                ),
296                lookups,
297            )| {
298                iter::empty()
299                    .chain(vk.cs.instance_queries.iter().enumerate().map(
300                        move |(query_index, &(column, at))| {
301                            VerifierQuery::new_commitment(
302                                &instance_commitments[column.index()],
303                                vk.domain.rotate_omega(*x, at),
304                                instance_evals[query_index],
305                            )
306                        },
307                    ))
308                    .chain(vk.cs.advice_queries.iter().enumerate().map(
309                        move |(query_index, &(column, at))| {
310                            VerifierQuery::new_commitment(
311                                &advice_commitments[column.index()],
312                                vk.domain.rotate_omega(*x, at),
313                                advice_evals[query_index],
314                            )
315                        },
316                    ))
317                    .chain(permutation.queries(vk, x))
318                    .chain(
319                        lookups
320                            .iter()
321                            .flat_map(move |p| p.queries(vk, x))
322                            .into_iter(),
323                    )
324            },
325        )
326        .chain(
327            vk.cs
328                .fixed_queries
329                .iter()
330                .enumerate()
331                .map(|(query_index, &(column, at))| {
332                    VerifierQuery::new_commitment(
333                        &vk.fixed_commitments[column.index()],
334                        vk.domain.rotate_omega(*x, at),
335                        fixed_evals[query_index],
336                    )
337                }),
338        )
339        .chain(permutations_common.queries(&vk.permutation, x))
340        .chain(vanishing.queries(x));
341
342    // We are now convinced the circuit is satisfied so long as the
343    // polynomial commitments open to the correct values.
344    strategy.process(|msm| {
345        multiopen::verify_proof(params, transcript, queries, msm).map_err(|_| Error::Opening)
346    })
347}
