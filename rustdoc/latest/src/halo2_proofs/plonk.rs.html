38/// This is a verifying key which allows for the verification of proofs for a
39/// particular circuit.
40#[derive(Clone, Debug)]
41pub struct VerifyingKey<C: CurveAffine> {
42    domain: EvaluationDomain<C::Scalar>,
43    fixed_commitments: Vec<C>,
44    permutation: permutation::VerifyingKey<C>,
45    cs: ConstraintSystem<C::Scalar>,
46    /// Cached maximum degree of `cs` (which doesn't change after construction).
47    cs_degree: usize,
48    /// The representative of this `VerifyingKey` in transcripts.
49    transcript_repr: C::Scalar,
50}
51
52impl<C: CurveAffine> VerifyingKey<C>
53where
54    C::Scalar: FromUniformBytes<64>,
55{
56    fn from_parts(
57        domain: EvaluationDomain<C::Scalar>,
58        fixed_commitments: Vec<C>,
59        permutation: permutation::VerifyingKey<C>,
60        cs: ConstraintSystem<C::Scalar>,
61    ) -> Self {
62        // Compute cached values.
63        let cs_degree = cs.degree();
64
65        let mut vk = Self {
66            domain,
67            fixed_commitments,
68            permutation,
69            cs,
70            cs_degree,
71            // Temporary, this is not pinned.
72            transcript_repr: C::Scalar::ZERO,
73        };
74
75        let mut hasher = Blake2bParams::new()
76            .hash_length(64)
77            .personal(b"Halo2-Verify-Key")
78            .to_state();
79
80        let s = format!("{:?}", vk.pinned());
81
82        hasher.update(&(s.len() as u64).to_le_bytes());
83        hasher.update(s.as_bytes());
84
85        // Hash in final Blake2bState
86        vk.transcript_repr = C::Scalar::from_uniform_bytes(hasher.finalize().as_array());
87
88        vk
89    }
90}

92impl<C: CurveAffine> VerifyingKey<C> {
93    /// Hashes a verification key into a transcript.
94    pub fn hash_into<E: EncodedChallenge<C>, T: Transcript<C, E>>(
95        &self,
96        transcript: &mut T,
97    ) -> io::Result<()> {
98        transcript.common_scalar(self.transcript_repr)?;
99
100        Ok(())
101    }
102
103    /// Obtains a pinned representation of this verification key that contains
104    /// the minimal information necessary to reconstruct the verification key.
105    pub fn pinned(&self) -> PinnedVerificationKey<'_, C> {
106        PinnedVerificationKey {
107            base_modulus: C::Base::MODULUS,
108            scalar_modulus: C::Scalar::MODULUS,
109            domain: self.domain.pinned(),
110            fixed_commitments: &self.fixed_commitments,
111            permutation: &self.permutation,
112            cs: self.cs.pinned(),
113        }
114    }
115}

117/// Minimal representation of a verification key that can be used to identify
118/// its active contents.
119#[allow(dead_code)]
120#[derive(Debug)]
121pub struct PinnedVerificationKey<'a, C: CurveAffine> {
122    base_modulus: &'static str,
123    scalar_modulus: &'static str,
124    domain: PinnedEvaluationDomain<'a, C::Scalar>,
125    cs: PinnedConstraintSystem<'a, C::Scalar>,
126    fixed_commitments: &'a Vec<C>,
127    permutation: &'a permutation::VerifyingKey<C>,
128}
129/// This is a proving key which allows for the creation of proofs for a
130/// particular circuit.
131#[derive(Clone, Debug)]
132pub struct ProvingKey<C: CurveAffine> {
133    vk: VerifyingKey<C>,
134    l0: Polynomial<C::Scalar, ExtendedLagrangeCoeff>,
135    l_blind: Polynomial<C::Scalar, ExtendedLagrangeCoeff>,
136    l_last: Polynomial<C::Scalar, ExtendedLagrangeCoeff>,
137    fixed_values: Vec<Polynomial<C::Scalar, LagrangeCoeff>>,
138    fixed_polys: Vec<Polynomial<C::Scalar, Coeff>>,
139    fixed_cosets: Vec<Polynomial<C::Scalar, ExtendedLagrangeCoeff>>,
140    permutation: permutation::ProvingKey<C>,
141}
