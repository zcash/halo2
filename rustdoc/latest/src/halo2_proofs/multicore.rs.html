1//! An interface for dealing with the kinds of parallel computations involved in
2//! `halo2`. It's currently just a (very!) thin wrapper around `rayon` but may
3//! be extended in the future to allow for various parallelism strategies.
4
5#[cfg(all(
6    feature = "multicore",
7    target_arch = "wasm32",
8    not(target_feature = "atomics")
9))]
10compile_error!(
11    "The multicore feature flag is not supported on wasm32 architectures without atomics"
12);
13
14pub use maybe_rayon::{
15    iter::{IntoParallelIterator, ParallelIterator},
16    join, scope, Scope,

19#[cfg(feature = "multicore")]
20pub use maybe_rayon::{current_num_threads, iter::IndexedParallelIterator};
21
22#[cfg(not(feature = "multicore"))]
23pub fn current_num_threads() -> usize {
24    1
25}

27#[cfg(not(feature = "multicore"))]
28pub trait IndexedParallelIterator: std::iter::Iterator {}
29
30pub trait TryFoldAndReduce<T, E> {
31    /// Implements `iter.try_fold().try_reduce()` for `rayon::iter::ParallelIterator`,
32    /// falling back on `Iterator::try_fold` when the `multicore` feature flag is
33    /// disabled.
34    /// The `try_fold_and_reduce` function can only be called by a iter with
35    /// `Result<T, E>` item type because the `fold_op` must meet the trait
36    /// bounds of both `try_fold` and `try_reduce` from rayon.   
37    fn try_fold_and_reduce(
38        self,
39        identity: impl Fn() -> T + Send + Sync,
40        fold_op: impl Fn(T, Result<T, E>) -> Result<T, E> + Send + Sync,
41    ) -> Result<T, E>;
42}
43
44#[cfg(feature = "multicore")]
45impl<T, E, I> TryFoldAndReduce<T, E> for I
46where
47    T: Send + Sync,
48    E: Send + Sync,
49    I: maybe_rayon::iter::ParallelIterator<Item = Result<T, E>>,
50{
51    fn try_fold_and_reduce(
52        self,
53        identity: impl Fn() -> T + Send + Sync,
54        fold_op: impl Fn(T, Result<T, E>) -> Result<T, E> + Send + Sync,
55    ) -> Result<T, E> {
56        self.try_fold(&identity, &fold_op)
57            .try_reduce(&identity, |a, b| fold_op(a, Ok(b)))
58    }
59}
60
61#[cfg(not(feature = "multicore"))]
62impl<T, E, I> TryFoldAndReduce<T, E> for I
63where
64    I: std::iter::Iterator<Item = Result<T, E>>,
65{
66    fn try_fold_and_reduce(
67        mut self,
68        identity: impl Fn() -> T + Send + Sync,
69        fold_op: impl Fn(T, Result<T, E>) -> Result<T, E> + Send + Sync,
70    ) -> Result<T, E> {
71        self.try_fold(identity(), fold_op)
72    }
73}
