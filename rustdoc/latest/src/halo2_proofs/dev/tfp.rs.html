14/// A helper type that augments a [`FloorPlanner`] with [`tracing`] spans and events.
15///
16/// `TracingFloorPlanner` can be used to instrument your circuit and determine exactly
17/// what is happening during a particular run of keygen or proving. This can be useful for
18/// identifying unexpected non-determinism or changes to a circuit.
19///
20/// # No stability guarantees
21///
22/// The `tracing` output is intended for use during circuit development. It should not be
23/// considered production-stable, and the precise format or data exposed may change at any
24/// time.
25///
26/// # Examples
27///
28/// ```
29/// use ff::Field;
30/// use halo2_proofs::{
31///     circuit::{floor_planner, Layouter, Value},
32///     dev::TracingFloorPlanner,
33///     plonk::{Circuit, ConstraintSystem, Error},
34/// };
35///
36/// # struct MyCircuit<F: Field> {
37/// #     some_witness: Value<F>,
38/// # };
39/// # #[derive(Clone)]
40/// # struct MyConfig;
41/// impl<F: Field> Circuit<F> for MyCircuit<F> {
42///     // Wrap `TracingFloorPlanner` around your existing floor planner of choice.
43///     //type FloorPlanner = floor_planner::V1;
44///     type FloorPlanner = TracingFloorPlanner<floor_planner::V1>;
45///
46///     // The rest of your `Circuit` implementation is unchanged.
47///     type Config = MyConfig;
48///
49///     fn without_witnesses(&self) -> Self {
50///         Self { some_witness: Value::unknown() }
51///     }
52///
53///     fn configure(meta: &mut ConstraintSystem<F>) -> Self::Config {
54///         // ..
55/// #       todo!()
56///     }
57///
58///     fn synthesize(&self, config: Self::Config, layouter: impl Layouter<F>) -> Result<(), Error> {
59///         // ..
60/// #       todo!()
61///     }
62/// }
63///
64/// #[test]
65/// fn some_circuit_test() {
66///     // At the start of your test, enable tracing.
67///     tracing_subscriber::fmt()
68///         .with_max_level(tracing::Level::DEBUG)
69///         .with_ansi(false)
70///         .without_time()
71///         .init();
72///
73///     // Now when the rest of the test runs, you will get `tracing` output for every
74///     // operation that the circuit performs under the hood!
75/// }
76/// ```
77#[derive(Debug)]
78pub struct TracingFloorPlanner<P: FloorPlanner> {
79    _phantom: PhantomData<P>,
80}
81
82impl<P: FloorPlanner> FloorPlanner for TracingFloorPlanner<P> {
83    fn synthesize<F: Field, CS: Assignment<F>, C: Circuit<F>>(
84        cs: &mut CS,
85        circuit: &C,
86        config: C::Config,
87        constants: Vec<Column<Fixed>>,
88    ) -> Result<(), Error> {
89        P::synthesize(
90            &mut TracingAssignment::new(cs),
91            &TracingCircuit::borrowed(circuit),
92            config,
93            constants,
94        )
95    }
96}

98/// A helper type that augments a [`Circuit`] with [`tracing`] spans and events.
99enum TracingCircuit<'c, F: Field, C: Circuit<F>> {
100    Borrowed(&'c C, PhantomData<F>),
101    Owned(C, PhantomData<F>),
102}
103
104impl<'c, F: Field, C: Circuit<F>> TracingCircuit<'c, F, C> {
105    fn borrowed(circuit: &'c C) -> Self {
106        Self::Borrowed(circuit, PhantomData)
107    }
108
109    fn owned(circuit: C) -> Self {
110        Self::Owned(circuit, PhantomData)
111    }
112
113    fn inner_ref(&self) -> &C {
114        match self {
115            TracingCircuit::Borrowed(circuit, ..) => circuit,
116            TracingCircuit::Owned(circuit, ..) => circuit,
117        }
118    }
119}
120
121impl<'c, F: Field, C: Circuit<F>> Circuit<F> for TracingCircuit<'c, F, C> {
122    type Config = C::Config;
123    type FloorPlanner = C::FloorPlanner;
124
125    fn without_witnesses(&self) -> Self {
126        Self::owned(self.inner_ref().without_witnesses())
127    }
128
129    fn configure(meta: &mut ConstraintSystem<F>) -> Self::Config {
130        let _span = debug_span!("configure").entered();
131        C::configure(meta)
132    }
133
134    fn synthesize(&self, config: Self::Config, layouter: impl Layouter<F>) -> Result<(), Error> {
135        let _span = debug_span!("synthesize").entered();
136        self.inner_ref()
137            .synthesize(config, TracingLayouter::new(layouter))
138    }
139}

141/// A helper type that augments a [`Layouter`] with [`tracing`] spans and events.
142struct TracingLayouter<F: Field, L: Layouter<F>> {
143    layouter: L,
144    namespace_spans: Vec<EnteredSpan>,
145    _phantom: PhantomData<F>,
146}
147
148impl<F: Field, L: Layouter<F>> TracingLayouter<F, L> {
149    fn new(layouter: L) -> Self {
150        Self {
151            layouter,
152            namespace_spans: vec![],
153            _phantom: PhantomData,
154        }
155    }
156}

158impl<F: Field, L: Layouter<F>> Layouter<F> for TracingLayouter<F, L> {
159    type Root = Self;
160
161    fn assign_region<A, AR, N, NR>(&mut self, name: N, mut assignment: A) -> Result<AR, Error>
162    where
163        A: FnMut(Region<'_, F>) -> Result<AR, Error>,
164        N: Fn() -> NR,
165        NR: Into<String>,
166    {
167        let _span = debug_span!("region", name = name().into()).entered();
168        self.layouter.assign_region(name, |region| {
169            let mut region = TracingRegion(region);
170            let region: &mut dyn RegionLayouter<F> = &mut region;
171            assignment(region.into())
172        })
173    }
174
175    fn assign_table<A, N, NR>(&mut self, name: N, assignment: A) -> Result<(), Error>
176    where
177        A: FnMut(Table<'_, F>) -> Result<(), Error>,
178        N: Fn() -> NR,
179        NR: Into<String>,
180    {
181        let _span = debug_span!("table", name = name().into()).entered();
182        self.layouter.assign_table(name, assignment)
183    }
184
185    fn constrain_instance(
186        &mut self,
187        cell: Cell,
188        column: Column<Instance>,
189        row: usize,
190    ) -> Result<(), Error> {
191        self.layouter.constrain_instance(cell, column, row)
192    }
193
194    fn get_root(&mut self) -> &mut Self::Root {
195        self
196    }
197
198    fn push_namespace<NR, N>(&mut self, name_fn: N)
199    where
200        NR: Into<String>,
201        N: FnOnce() -> NR,
202    {
203        let name = name_fn().into();
204        self.namespace_spans.push(debug_span!("ns", name).entered());
205        self.layouter.push_namespace(|| name);
206    }
207
208    fn pop_namespace(&mut self, gadget_name: Option<String>) {
209        self.layouter.pop_namespace(gadget_name);
210        self.namespace_spans.pop();
211    }
212}

214fn debug_value_and_return_cell<F: Field, V: fmt::Debug>(value: AssignedCell<V, F>) -> Cell {

352impl<'cs, F: Field, CS: Assignment<F>> Assignment<F> for TracingAssignment<'cs, F, CS> {
353    fn enter_region<NR, N>(&mut self, name_fn: N)
354    where
355        NR: Into<String>,
356        N: FnOnce() -> NR,
357    {
358        self.in_region = true;
359        self.cs.enter_region(name_fn);
360    }
361
362    fn exit_region(&mut self) {
363        self.cs.exit_region();
364        self.in_region = false;
365    }

367    fn enable_selector<A, AR>(
368        &mut self,
369        annotation: A,
370        selector: &Selector,
371        row: usize,
372    ) -> Result<(), Error>
373    where
374        A: FnOnce() -> AR,
375        AR: Into<String>,
376    {
377        let annotation = annotation().into();
378        if self.in_region {
379            debug!(target: "position", row = row);
380        } else {
381            debug!(target: "enable_selector", name = annotation, row = row);
382        }
383        self.cs.enable_selector(|| annotation, selector, row)
384    }

390    }
