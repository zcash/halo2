14/// Verify a multi-opening proof
15pub fn verify_proof<
16    'r,
17    'params: 'r,
18    I,
19    C: CurveAffine,
20    E: EncodedChallenge<C>,
21    T: TranscriptRead<C, E>,
22>(
23    params: &'params Params<C>,
24    transcript: &mut T,
25    queries: I,
26    mut msm: MSM<'params, C>,
27) -> Result<Guard<'params, C, E>, Error>
28where
29    I: IntoIterator<Item = VerifierQuery<'r, 'params, C>> + Clone,
30{
31    // Sample x_1 for compressing openings at the same point sets together
32    let x_1: ChallengeX1<_> = transcript.squeeze_challenge_scalar();
33
34    // Sample a challenge x_2 for keeping the multi-point quotient
35    // polynomial terms linearly independent.
36    let x_2: ChallengeX2<_> = transcript.squeeze_challenge_scalar();
37
38    let (commitment_map, point_sets) =
39        construct_intermediate_sets(queries).ok_or(Error::OpeningError)?;
40
41    // Compress the commitments and expected evaluations at x together.
42    // using the challenge x_1
43    let mut q_commitments: Vec<_> = vec![
44        (params.empty_msm(), C::Scalar::ONE); // (accumulator, next x_1 power).
45        point_sets.len()];
46
47    // A vec of vecs of evals. The outer vec corresponds to the point set,
48    // while the inner vec corresponds to the points in a particular set.
49    let mut q_eval_sets = Vec::with_capacity(point_sets.len());
50    for point_set in point_sets.iter() {
51        q_eval_sets.push(vec![C::Scalar::ZERO; point_set.len()]);
52    }
53
54    {
55        let mut accumulate = |set_idx: usize, new_commitment, evals: Vec<C::Scalar>| {
56            let (q_commitment, x_1_power) = &mut q_commitments[set_idx];
57            match new_commitment {
58                CommitmentReference::Commitment(c) => {
59                    q_commitment.append_term(*x_1_power, *c);
60                }
61                CommitmentReference::MSM(msm) => {
62                    let mut msm = msm.clone();
63                    msm.scale(*x_1_power);
64                    q_commitment.add_msm(&msm);
65                }
66            }
67            for (eval, set_eval) in evals.iter().zip(q_eval_sets[set_idx].iter_mut()) {
68                *set_eval += (*eval) * (*x_1_power);
69            }
70            *x_1_power *= *x_1;
71        };
72
73        // Each commitment corresponds to evaluations at a set of points.
74        // For each set, we collapse each commitment's evals pointwise.
75        // Run in order of increasing x_1 powers.
76        for commitment_data in commitment_map.into_iter().rev() {
77            accumulate(
78                commitment_data.set_index,  // set_idx,
79                commitment_data.commitment, // commitment,
80                commitment_data.evals,      // evals
81            );
82        }
83    }
84
85    // Obtain the commitment to the multi-point quotient polynomial f(X).
86    let q_prime_commitment = transcript.read_point().map_err(|_| Error::SamplingError)?;
87
88    // Sample a challenge x_3 for checking that f(X) was committed to
89    // correctly.
90    let x_3: ChallengeX3<_> = transcript.squeeze_challenge_scalar();
91
92    // u is a vector containing the evaluations of the Q polynomial
93    // commitments at x_3
94    let mut u = Vec::with_capacity(q_eval_sets.len());
95    for _ in 0..q_eval_sets.len() {
96        u.push(transcript.read_scalar().map_err(|_| Error::SamplingError)?);
97    }
98
99    // We can compute the expected msm_eval at x_3 using the u provided
100    // by the prover and from x_2
101    let msm_eval = point_sets
102        .iter()
103        .zip(q_eval_sets.iter())
104        .zip(u.iter())
105        .fold(
106            C::Scalar::ZERO,
107            |msm_eval, ((points, evals), proof_eval)| {
108                let r_poly = lagrange_interpolate(points, evals);
109                let r_eval = eval_polynomial(&r_poly, *x_3);
110                let eval = points.iter().fold(*proof_eval - &r_eval, |eval, point| {
111                    eval * &(*x_3 - point).invert().unwrap()
112                });
113                msm_eval * &(*x_2) + &eval
114            },
115        );
116
117    // Sample a challenge x_4 that we will use to collapse the openings of
118    // the various remaining polynomials at x_3 together.
119    let x_4: ChallengeX4<_> = transcript.squeeze_challenge_scalar();
120
121    // Compute the final commitment that has to be opened
122    msm.append_term(C::Scalar::ONE, q_prime_commitment);
123    let (msm, v) = q_commitments.into_iter().zip(u.iter()).fold(
124        (msm, msm_eval),
125        |(mut msm, msm_eval), ((q_commitment, _), q_eval)| {
126            msm.scale(*x_4);
127            msm.add_msm(&q_commitment);
128            (msm, msm_eval * &(*x_4) + q_eval)
129        },
130    );
131
132    // Verify the opening proof
133    super::commitment::verify_proof(params, msm, transcript, *x_3, v)
134}

136impl<'a, 'b, C: CurveAffine> Query<C::Scalar> for VerifierQuery<'a, 'b, C> {
137    type Commitment = CommitmentReference<'a, 'b, C>;
138    type Eval = C::Scalar;
139
140    fn get_point(&self) -> C::Scalar {
141        self.point
142    }
143    fn get_eval(&self) -> C::Scalar {
144        self.eval
145    }
146    fn get_commitment(&self) -> Self::Commitment {
147        self.commitment
148    }
149}
