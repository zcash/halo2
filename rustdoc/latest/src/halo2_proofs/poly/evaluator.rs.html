17/// Returns `(chunk_size, num_chunks)` suitable for processing the given polynomial length
18/// in the current parallelization environment.
19fn get_chunk_params(poly_len: usize) -> (usize, usize) {
20    // Check the level of parallelization we have available.
21    let num_threads = multicore::current_num_threads();
22    // We scale the number of chunks by a constant factor, to ensure that if not all
23    // threads are available, we can achieve more uniform throughput and don't end up
24    // waiting on a couple of threads to process the last chunks.
25    let num_chunks = num_threads * 4;
26    // Calculate the ideal chunk size for the desired throughput. We use ceiling
27    // division to ensure the minimum chunk size is 1.
28    //     chunk_size = ceil(poly_len / num_chunks)
29    let chunk_size = (poly_len + num_chunks - 1) / num_chunks;
30    // Now re-calculate num_chunks from the actual chunk size.
31    //     num_chunks = ceil(poly_len / chunk_size)
32    let num_chunks = (poly_len + chunk_size - 1) / chunk_size;
33
34    (chunk_size, num_chunks)
35}

83/// An evaluation context for polynomial operations.
84///
85/// This context enables us to de-duplicate queries of circuit columns (and the rotations
86/// they might require), by storing a list of all the underlying polynomials involved in
87/// any query (which are almost certainly column polynomials). We use the context like so:
88///
89/// - We register each underlying polynomial with the evaluator, which returns a reference
90///   to it as a [`AstLeaf`].
91/// - The references are then used to build up a [`Ast`] that represents the overall
92///   operations to be applied to the polynomials.
93/// - Finally, we call [`Evaluator::evaluate`] passing in the [`Ast`].
94pub(crate) struct Evaluator<E, F: Field, B: Basis> {
95    polys: Vec<Polynomial<F, B>>,
96    _context: E,
97}
98
99/// Constructs a new `Evaluator`.
100///
101/// The `context` parameter is used to provide type safety for evaluators. It ensures that
102/// an evaluator will only be used to evaluate [`Ast`]s containing [`AstLeaf`]s obtained
103/// from itself. It should be set to the empty closure `|| {}`, because anonymous closures
104/// all have unique types.
105pub(crate) fn new_evaluator<E: Fn() + Clone, F: Field, B: Basis>(context: E) -> Evaluator<E, F, B> {
106    Evaluator {
107        polys: vec![],
108        _context: context,
109    }
110}

112impl<E, F: Field, B: Basis> Evaluator<E, F, B> {
113    /// Registers the given polynomial for use in this evaluation context.
114    ///
115    /// This API treats each registered polynomial as unique, even if the same polynomial
116    /// is added multiple times.
117    pub(crate) fn register_poly(&mut self, poly: Polynomial<F, B>) -> AstLeaf<E, B> {
118        let index = self.polys.len();
119        self.polys.push(poly);
120
121        AstLeaf {
122            index,
123            rotation: Rotation::cur(),
124            _evaluator: PhantomData::default(),
125        }
126    }
127
128    /// Evaluates the given polynomial operation against this context.
129    pub(crate) fn evaluate(
130        &self,
131        ast: &Ast<E, F, B>,
132        domain: &EvaluationDomain<F>,
133    ) -> Polynomial<F, B>
134    where
135        E: Copy + Send + Sync,
136        F: WithSmallOrderMulGroup<3>,
137        B: BasisOps,
138    {
139        // We're working in a single basis, so all polynomials are the same length.
140        let poly_len = self.polys.first().unwrap().len();
141        let (chunk_size, _num_chunks) = get_chunk_params(poly_len);
142
143        struct AstContext<'a, F: Field, B: Basis> {
144            domain: &'a EvaluationDomain<F>,
145            poly_len: usize,
146            chunk_size: usize,
147            chunk_index: usize,
148            polys: &'a [Polynomial<F, B>],
149        }
150
151        fn recurse<E, F: WithSmallOrderMulGroup<3>, B: BasisOps>(
152            ast: &Ast<E, F, B>,
153            ctx: &AstContext<'_, F, B>,
154        ) -> Vec<F> {
155            match ast {
156                Ast::Poly(leaf) => B::get_chunk_of_rotated(
157                    ctx.domain,
158                    ctx.chunk_size,
159                    ctx.chunk_index,
160                    &ctx.polys[leaf.index],
161                    leaf.rotation,
162                ),
163                Ast::Add(a, b) => {
164                    let mut lhs = recurse(a, ctx);
165                    let rhs = recurse(b, ctx);
166                    for (lhs, rhs) in lhs.iter_mut().zip(rhs.iter()) {
167                        *lhs += *rhs;
168                    }
169                    lhs
170                }
171                Ast::Mul(AstMul(a, b)) => {
172                    let mut lhs = recurse(a, ctx);
173                    let rhs = recurse(b, ctx);
174                    for (lhs, rhs) in lhs.iter_mut().zip(rhs.iter()) {
175                        *lhs *= *rhs;
176                    }
177                    lhs
178                }
179                Ast::Scale(a, scalar) => {
180                    let mut lhs = recurse(a, ctx);
181                    for lhs in lhs.iter_mut() {
182                        *lhs *= scalar;
183                    }
184                    lhs
185                }
186                Ast::DistributePowers(terms, base) => terms.iter().fold(
187                    B::constant_term(ctx.poly_len, ctx.chunk_size, ctx.chunk_index, F::ZERO),
188                    |mut acc, term| {
189                        let term = recurse(term, ctx);
190                        for (acc, term) in acc.iter_mut().zip(term) {
191                            *acc *= base;
192                            *acc += term;
193                        }
194                        acc
195                    },
196                ),
197                Ast::LinearTerm(scalar) => B::linear_term(
198                    ctx.domain,
199                    ctx.poly_len,
200                    ctx.chunk_size,
201                    ctx.chunk_index,
202                    *scalar,
203                ),
204                Ast::ConstantTerm(scalar) => {
205                    B::constant_term(ctx.poly_len, ctx.chunk_size, ctx.chunk_index, *scalar)
206                }
207            }
208        }
209
210        // Apply `ast` to each chunk in parallel, writing the result into an output
211        // polynomial.
212        let mut result = B::empty_poly(domain);
213        multicore::scope(|scope| {
214            for (chunk_index, out) in result.chunks_mut(chunk_size).enumerate() {
215                scope.spawn(move |_| {
216                    let ctx = AstContext {
217                        domain,
218                        poly_len,
219                        chunk_size,
220                        chunk_index,
221                        polys: &self.polys,
222                    };
223                    out.copy_from_slice(&recurse(ast, &ctx));
224                });
225            }
226        });
227        result
228    }
229}

248/// A polynomial operation backed by an [`Evaluator`].
249#[derive(Clone)]
250pub(crate) enum Ast<E, F: Field, B: Basis> {
251    Poly(AstLeaf<E, B>),
252    Add(Arc<Ast<E, F, B>>, Arc<Ast<E, F, B>>),
253    Mul(AstMul<E, F, B>),
254    Scale(Arc<Ast<E, F, B>>, F),
255    /// Represents a linear combination of a vector of nodes and the powers of a
256    /// field element, where the nodes are ordered from highest to lowest degree
257    /// terms.
258    DistributePowers(Arc<Vec<Ast<E, F, B>>>, F),
259    /// The degree-1 term of a polynomial.
260    ///
261    /// The field element is the coefficient of the term in the standard basis, not the
262    /// coefficient basis.
263    LinearTerm(F),
264    /// The degree-0 term of a polynomial.
265    ///
266    /// The field element is the same in both the standard and evaluation bases.
267    ConstantTerm(F),
268}

442/// Operations which can be performed over a given basis.
443pub(crate) trait BasisOps: Basis {
444    fn empty_poly<F: WithSmallOrderMulGroup<3>>(
445        domain: &EvaluationDomain<F>,
446    ) -> Polynomial<F, Self>;
447    fn constant_term<F: Field>(
448        poly_len: usize,
449        chunk_size: usize,
450        chunk_index: usize,
451        scalar: F,
452    ) -> Vec<F>;
453    fn linear_term<F: WithSmallOrderMulGroup<3>>(
454        domain: &EvaluationDomain<F>,
455        poly_len: usize,
456        chunk_size: usize,
457        chunk_index: usize,
458        scalar: F,
459    ) -> Vec<F>;
460    fn get_chunk_of_rotated<F: WithSmallOrderMulGroup<3>>(
461        domain: &EvaluationDomain<F>,
462        chunk_size: usize,
463        chunk_index: usize,
464        poly: &Polynomial<F, Self>,
465        rotation: Rotation,
466    ) -> Vec<F>;
467}

469impl BasisOps for Coeff {
470    fn empty_poly<F: WithSmallOrderMulGroup<3>>(
471        domain: &EvaluationDomain<F>,
472    ) -> Polynomial<F, Self> {
473        domain.empty_coeff()
474    }
475
476    fn constant_term<F: Field>(
477        poly_len: usize,
478        chunk_size: usize,
479        chunk_index: usize,
480        scalar: F,
481    ) -> Vec<F> {
482        let mut chunk = vec![F::ZERO; cmp::min(chunk_size, poly_len - chunk_size * chunk_index)];
483        if chunk_index == 0 {
484            chunk[0] = scalar;
485        }
486        chunk
487    }
488
489    fn linear_term<F: WithSmallOrderMulGroup<3>>(
490        _: &EvaluationDomain<F>,
491        poly_len: usize,
492        chunk_size: usize,
493        chunk_index: usize,
494        scalar: F,
495    ) -> Vec<F> {
496        let mut chunk = vec![F::ZERO; cmp::min(chunk_size, poly_len - chunk_size * chunk_index)];
497        // If the chunk size is 1 (e.g. if we have a small k and many threads), then the
498        // linear coefficient is the second chunk. Otherwise, the chunk size is greater
499        // than one, and the linear coefficient is the second element of the first chunk.
500        // Note that we check against the original chunk size, not the potentially-short
501        // actual size of the current chunk, because we want to know whether the size of
502        // the previous chunk was 1.
503        if chunk_size == 1 && chunk_index == 1 {
504            chunk[0] = scalar;
505        } else if chunk_index == 0 {
506            chunk[1] = scalar;
507        }
508        chunk
509    }
510
511    fn get_chunk_of_rotated<F: WithSmallOrderMulGroup<3>>(
512        _: &EvaluationDomain<F>,
513        _: usize,
514        _: usize,
515        _: &Polynomial<F, Self>,
516        _: Rotation,
517    ) -> Vec<F> {
518        panic!("Can't rotate polynomials in the standard basis")
519    }
520}

522impl BasisOps for LagrangeCoeff {
523    fn empty_poly<F: WithSmallOrderMulGroup<3>>(
524        domain: &EvaluationDomain<F>,
525    ) -> Polynomial<F, Self> {
526        domain.empty_lagrange()
527    }
528
529    fn constant_term<F: Field>(
530        poly_len: usize,
531        chunk_size: usize,
532        chunk_index: usize,
533        scalar: F,
534    ) -> Vec<F> {
535        vec![scalar; cmp::min(chunk_size, poly_len - chunk_size * chunk_index)]
536    }
537
538    fn linear_term<F: WithSmallOrderMulGroup<3>>(
539        domain: &EvaluationDomain<F>,
540        poly_len: usize,
541        chunk_size: usize,
542        chunk_index: usize,
543        scalar: F,
544    ) -> Vec<F> {
545        // Take every power of omega within the chunk, and multiply by scalar.
546        let omega = domain.get_omega();
547        let start = chunk_size * chunk_index;
548        (0..cmp::min(chunk_size, poly_len - start))
549            .scan(omega.pow_vartime([start as u64]) * scalar, |acc, _| {
550                let ret = *acc;
551                *acc *= omega;
552                Some(ret)
553            })
554            .collect()
555    }
556
557    fn get_chunk_of_rotated<F: WithSmallOrderMulGroup<3>>(
558        _: &EvaluationDomain<F>,
559        chunk_size: usize,
560        chunk_index: usize,
561        poly: &Polynomial<F, Self>,
562        rotation: Rotation,
563    ) -> Vec<F> {
564        poly.get_chunk_of_rotated(rotation, chunk_size, chunk_index)
565    }
566}

581        vec![scalar; cmp::min(chunk_size, poly_len - chunk_size * chunk_index)]

584    fn linear_term<F: WithSmallOrderMulGroup<3>>(
585        domain: &EvaluationDomain<F>,
586        poly_len: usize,
587        chunk_size: usize,
588        chunk_index: usize,
589        scalar: F,
590    ) -> Vec<F> {
591        // Take every power of the extended omega within the chunk, and multiply by scalar.
592        let omega = domain.get_extended_omega();
593        let start = chunk_size * chunk_index;
594        (0..cmp::min(chunk_size, poly_len - start))
595            .scan(
596                omega.pow_vartime([start as u64]) * F::ZETA * scalar,
597                |acc, _| {
598                    let ret = *acc;
599                    *acc *= omega;
600                    Some(ret)
601                },
602            )
603            .collect()
604    }
605
606    fn get_chunk_of_rotated<F: WithSmallOrderMulGroup<3>>(
607        domain: &EvaluationDomain<F>,
608        chunk_size: usize,
609        chunk_index: usize,
610        poly: &Polynomial<F, Self>,
611        rotation: Rotation,
612    ) -> Vec<F> {
613        domain.get_chunk_of_rotated_extended(poly, rotation, chunk_size, chunk_index)
614    }

634            Some(k) => k,
635            None => {
636                // We are on a machine with a power-of-two number of threads, and cannot
637                // trigger the bug.
638                eprintln!(
639                    "can't find a polynomial length for short_chunk_regression_test; skipping"
640                );
641                return;
642            }
643        };
644        eprintln!("Testing short-chunk regression with k = {}", k);
645
646        fn test_case<E: Copy + Send + Sync, B: BasisOps>(
647            k: u32,
648            mut evaluator: Evaluator<E, pallas::Base, B>,
649        ) {
650            // Instantiate the evaluator with a trivial polynomial.
651            let domain = EvaluationDomain::new(1, k);
652            evaluator.register_poly(B::empty_poly(&domain));
653
654            // With the bug present, these will panic.
655            let _ = evaluator.evaluate(&Ast::ConstantTerm(pallas::Base::ZERO), &domain);
656            let _ = evaluator.evaluate(&Ast::LinearTerm(pallas::Base::ZERO), &domain);
657        }
658
659        test_case(k, new_evaluator::<_, _, Coeff>(|| {}));
660        test_case(k, new_evaluator::<_, _, LagrangeCoeff>(|| {}));
661        test_case(k, new_evaluator::<_, _, ExtendedLagrangeCoeff>(|| {}));
662    }
