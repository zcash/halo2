52/// A polynomial query at a point
53#[derive(Debug, Clone)]
54pub struct VerifierQuery<'r, 'params: 'r, C: CurveAffine> {
55    /// point at which polynomial is queried
56    point: C::Scalar,
57    /// commitment to polynomial
58    commitment: CommitmentReference<'r, 'params, C>,

60    eval: C::Scalar,
61}

63impl<'r, 'params: 'r, C: CurveAffine> VerifierQuery<'r, 'params, C> {
64    /// Create a new verifier query based on a commitment
65    pub fn new_commitment(commitment: &'r C, point: C::Scalar, eval: C::Scalar) -> Self {
66        VerifierQuery {
67            point,
68            eval,
69            commitment: CommitmentReference::Commitment(commitment),
70        }
71    }
72
73    /// Create a new verifier query based on a linear combination of commitments
74    pub fn new_msm(
75        msm: &'r commitment::MSM<'params, C>,
76        point: C::Scalar,
77        eval: C::Scalar,
78    ) -> Self {
79        VerifierQuery {
80            point,
81            eval,
82            commitment: CommitmentReference::MSM(msm),
83        }
84    }

87#[allow(clippy::upper_case_acronyms)]
88#[derive(Copy, Clone, Debug)]
89enum CommitmentReference<'r, 'params: 'r, C: CurveAffine> {
90    Commitment(&'r C),
91    MSM(&'r commitment::MSM<'params, C>),
92}
93
94impl<'r, 'params: 'r, C: CurveAffine> PartialEq for CommitmentReference<'r, 'params, C> {
95    fn eq(&self, other: &Self) -> bool {
96        match (self, other) {
97            (&CommitmentReference::Commitment(a), &CommitmentReference::Commitment(b)) => {
98                std::ptr::eq(a, b)
99            }
100            (&CommitmentReference::MSM(a), &CommitmentReference::MSM(b)) => std::ptr::eq(a, b),
101            _ => false,
102        }
103    }
104}
105
106impl<'r, 'params: 'r, C: CurveAffine> Eq for CommitmentReference<'r, 'params, C> {}
107
108impl<'r, 'params: 'r, C: CurveAffine> Hash for CommitmentReference<'r, 'params, C> {
109    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
110        match *self {
111            CommitmentReference::Commitment(a) => std::ptr::hash(a, state),
112            CommitmentReference::MSM(a) => std::ptr::hash(a, state),
113        }
114    }

145type IntermediateSets<F, Q> = (
146    Vec<CommitmentData<<Q as Query<F>>::Eval, <Q as Query<F>>::Commitment>>,
147    Vec<Vec<F>>,
148);
149
150/// Returns `None` if `queries` contains two queries with the same point and commitment,
151/// but different evaluations.
152fn construct_intermediate_sets<F: Field + Ord, I, Q: Query<F>>(
153    queries: I,
154) -> Option<IntermediateSets<F, Q>>
155where
156    I: IntoIterator<Item = Q> + Clone,
157{
158    // Construct sets of unique commitments and corresponding information about
159    // their queries.
160    let mut commitment_map: IndexMap<Q::Commitment, CommitmentData<Option<Q::Eval>, ()>> =
161        IndexMap::new();
162
163    // Also construct mapping from a unique point to a point_index. This defines
164    // an ordering on the points.
165    let mut point_index_map = BTreeMap::new();
166
167    // Iterate over all of the queries, computing the ordering of the points
168    // while also creating new commitment data.
169    for query in queries.clone() {
170        let num_points = point_index_map.len();
171        let point_idx = point_index_map
172            .entry(query.get_point())
173            .or_insert(num_points);
174
175        commitment_map
176            .entry(query.get_commitment())
177            .or_insert_with(|| CommitmentData::new(()))
178            .point_indices
179            .push(*point_idx);
180    }
181
182    // Also construct inverse mapping from point_index to the point
183    let mut inverse_point_index_map = BTreeMap::new();
184    for (&point, &point_index) in point_index_map.iter() {
185        inverse_point_index_map.insert(point_index, point);
186    }
187
188    // Construct map of unique ordered point_idx_sets to their set_idx
189    let mut point_idx_sets = BTreeMap::new();
190    // Also construct mapping from commitment to point_idx_set
191    let mut commitment_set_map = IndexMap::new();
192
193    for (commitment, commitment_data) in commitment_map.iter_mut() {
194        let mut point_index_set = BTreeSet::new();
195        // Note that point_index_set is ordered, unlike point_indices
196        for &point_index in commitment_data.point_indices.iter() {
197            point_index_set.insert(point_index);
198        }
199        let len = point_index_set.len();
200
201        // Push point_index_set to CommitmentData for the relevant commitment
202        commitment_set_map.insert(*commitment, point_index_set.clone());
203
204        let num_sets = point_idx_sets.len();
205        point_idx_sets.entry(point_index_set).or_insert(num_sets);
206
207        // Initialise empty evals vec for each unique commitment
208        commitment_data.evals = vec![None; len];
209    }
210
211    // Populate set_index, evals and points for each commitment using point_idx_sets
212    for query in queries {
213        // An entry for every query's commitment was added to `commitment_map` the first
214        // time we iterated through `queries`.
215        let commitment_data = commitment_map
216            .get_mut(&query.get_commitment())
217            .expect("present by construction");
218
219        // The index of the point at which the commitment is queried
220        let point_index = point_index_map.get(&query.get_point()).unwrap();
221
222        // The point_index_set at which the commitment was queried
223        let point_index_set = commitment_set_map
224            .get(&query.get_commitment())
225            .expect("present by construction");
226
227        // The set_index of the point_index_set
228        let set_index = point_idx_sets.get(point_index_set).unwrap();
229        commitment_data.set_index = *set_index;
230        let point_index_set: Vec<usize> = point_index_set.iter().cloned().collect();
231
232        // The offset of the point_index in the point_index_set
233        let point_index_in_set = point_index_set
234            .iter()
235            .position(|i| i == point_index)
236            .unwrap();
237
238        // Insert the eval using the ordering of the point_index_set
239        let eval = commitment_data
240            .evals
241            .get_mut(point_index_in_set)
242            .expect("valid index");
243        if eval.is_none() {
244            *eval = Some(query.get_eval());
245        } else {
246            // Caller tried to provide two different evaluations for the same
247            // commitment. Permitting this would be unsound.
248            return None;
249        }
250    }
251
252    let commitment_map = commitment_map
253        .into_iter()
254        .map(|(commitment, commitment_data)| CommitmentData {
255            commitment,
256            set_index: commitment_data.set_index,
257            point_indices: commitment_data.point_indices,
258            evals: commitment_data
259                .evals
260                .into_iter()
261                .map(|eval| eval.expect("logic ensures all evals are set"))
262                .collect(),
263        })
264        .collect();
265
266    // Get actual points in each point set
267    let mut point_sets: Vec<Vec<F>> = vec![Vec::new(); point_idx_sets.len()];
268    for (point_idx_set, &set_idx) in point_idx_sets.iter() {
269        for &point_idx in point_idx_set.iter() {
270            let point = inverse_point_index_map.get(&point_idx).unwrap();
271            point_sets[set_idx].push(*point);
272        }
273    }
274
275    Some((commitment_map, point_sets))
276}

278#[test]
279fn test_roundtrip() {
280    use group::Curve;
281    use rand_core::OsRng;
282
283    use super::commitment::{Blind, Params};
284    use crate::arithmetic::eval_polynomial;
285    use crate::pasta::{EqAffine, Fp};
286    use crate::transcript::Challenge255;
287
288    const K: u32 = 4;
289
290    let params: Params<EqAffine> = Params::new(K);
291    let domain = EvaluationDomain::new(1, K);
292    let rng = OsRng;
293
294    let mut ax = domain.empty_coeff();
295    for (i, a) in ax.iter_mut().enumerate() {
296        *a = Fp::from(10 + i as u64);
297    }
298
299    let mut bx = domain.empty_coeff();
300    for (i, a) in bx.iter_mut().enumerate() {
301        *a = Fp::from(100 + i as u64);
302    }
303
304    let mut cx = domain.empty_coeff();
305    for (i, a) in cx.iter_mut().enumerate() {
306        *a = Fp::from(100 + i as u64);
307    }
308
309    let blind = Blind(Fp::random(rng));
310
311    let a = params.commit(&ax, blind).to_affine();
312    let b = params.commit(&bx, blind).to_affine();
313    let c = params.commit(&cx, blind).to_affine();
314
315    let x = Fp::random(rng);
316    let y = Fp::random(rng);
317    let avx = eval_polynomial(&ax, x);
318    let bvx = eval_polynomial(&bx, x);
319    let cvy = eval_polynomial(&cx, y);
320
321    let mut transcript = crate::transcript::Blake2bWrite::<_, _, Challenge255<_>>::init(vec![]);
322    create_proof(
323        &params,
324        rng,
325        &mut transcript,
326        std::iter::empty()
327            .chain(Some(ProverQuery {
328                point: x,
329                poly: &ax,
330                blind,
331            }))
332            .chain(Some(ProverQuery {
333                point: x,
334                poly: &bx,
335                blind,
336            }))
337            .chain(Some(ProverQuery {
338                point: y,
339                poly: &cx,
340                blind,
341            })),
342    )
343    .unwrap();
344    let proof = transcript.finalize();
345
346    {
347        let mut proof = &proof[..];
348        let mut transcript =
349            crate::transcript::Blake2bRead::<_, _, Challenge255<_>>::init(&mut proof);
350        let msm = params.empty_msm();
351
352        let guard = verify_proof(
353            &params,
354            &mut transcript,
355            std::iter::empty()
356                .chain(Some(VerifierQuery::new_commitment(&a, x, avx)))
357                .chain(Some(VerifierQuery::new_commitment(&b, x, avx))) // NB: wrong!
358                .chain(Some(VerifierQuery::new_commitment(&c, y, cvy))),
359            msm,
360        )
361        .unwrap();
362
363        // Should fail.
364        assert!(!guard.use_challenges().eval());
365    }
366
367    {
368        let mut proof = &proof[..];
369
370        let mut transcript =
371            crate::transcript::Blake2bRead::<_, _, Challenge255<_>>::init(&mut proof);
372        let msm = params.empty_msm();
373
374        let guard = verify_proof(
375            &params,
376            &mut transcript,
377            std::iter::empty()
378                .chain(Some(VerifierQuery::new_commitment(&a, x, avx)))
379                .chain(Some(VerifierQuery::new_commitment(&b, x, bvx)))
380                .chain(Some(VerifierQuery::new_commitment(&c, y, cvy))),
381            msm,
382        )
383        .unwrap();
384
385        // Should succeed.
386        assert!(guard.use_challenges().eval());
387    }
388}
