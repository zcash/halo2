16/// This structure contains precomputed constants and other details needed for
17/// performing operations on an evaluation domain of size $2^k$ and an extended
18/// domain of size $2^{k} * j$ with $j \neq 0$.
19#[derive(Clone, Debug)]
20pub struct EvaluationDomain<F: Field> {
21    n: u64,
22    k: u32,
23    extended_k: u32,
24    omega: F,
25    omega_inv: F,
26    extended_omega: F,
27    extended_omega_inv: F,
28    g_coset: F,
29    g_coset_inv: F,
30    quotient_poly_degree: u64,
31    ifft_divisor: F,
32    extended_ifft_divisor: F,
33    t_evaluations: Vec<F>,
34    barycentric_weight: F,
35}

38    /// This constructs a new evaluation domain object based on the provided
39    /// values $j, k$.
40    pub fn new(j: u32, k: u32) -> Self {
41        // quotient_poly_degree * params.n - 1 is the degree of the quotient polynomial
42        let quotient_poly_degree = (j - 1) as u64;
43
44        // n = 2^k
45        let n = 1u64 << k;
46
47        // We need to work within an extended domain, not params.k but params.k + i
48        // for some integer i such that 2^(params.k + i) is sufficiently large to
49        // describe the quotient polynomial.
50        let mut extended_k = k;
51        while (1 << extended_k) < (n * quotient_poly_degree) {
52            extended_k += 1;
53        }
54
55        // ensure extended_k <= S
56        assert!(extended_k <= F::S);
57
58        let mut extended_omega = F::ROOT_OF_UNITY;
59
60        // Get extended_omega, the 2^{extended_k}'th root of unity
61        // The loop computes extended_omega = omega^{2 ^ (S - extended_k)}
62        // Notice that extended_omega ^ {2 ^ extended_k} = omega ^ {2^S} = 1.
63        for _ in extended_k..F::S {
64            extended_omega = extended_omega.square();
65        }
66        let extended_omega = extended_omega;
67        let mut extended_omega_inv = extended_omega; // Inversion computed later
68
69        // Get omega, the 2^{k}'th root of unity (i.e. n'th root of unity)
70        // The loop computes omega = extended_omega ^ {2 ^ (extended_k - k)}
71        //           = (omega^{2 ^ (S - extended_k)})  ^ {2 ^ (extended_k - k)}
72        //           = omega ^ {2 ^ (S - k)}.
73        // Notice that omega ^ {2^k} = omega ^ {2^S} = 1.
74        let mut omega = extended_omega;
75        for _ in k..extended_k {
76            omega = omega.square();
77        }
78        let omega = omega;
79        let mut omega_inv = omega; // Inversion computed later
80
81        // We use zeta here because we know it generates a coset, and it's available
82        // already.
83        // The coset evaluation domain is:
84        // zeta {1, extended_omega, extended_omega^2, ..., extended_omega^{(2^extended_k) - 1}}
85        let g_coset = F::ZETA;
86        let g_coset_inv = g_coset.square();
87
88        let mut t_evaluations = Vec::with_capacity(1 << (extended_k - k));
89        {
90            // Compute the evaluations of t(X) = X^n - 1 in the coset evaluation domain.
91            // We don't have to compute all of them, because it will repeat.
92            let orig = F::ZETA.pow_vartime([n]);
93            let step = extended_omega.pow_vartime([n]);
94            let mut cur = orig;
95            loop {
96                t_evaluations.push(cur);
97                cur *= &step;
98                if cur == orig {
99                    break;
100                }
101            }
102            assert_eq!(t_evaluations.len(), 1 << (extended_k - k));
103
104            // Subtract 1 from each to give us t_evaluations[i] = t(zeta * extended_omega^i)
105            for coeff in &mut t_evaluations {
106                *coeff -= &F::ONE;
107            }
108
109            // Invert, because we're dividing by this polynomial.
110            // We invert in a batch, below.
111        }
112
113        let mut ifft_divisor = F::from(1 << k); // Inversion computed later
114        let mut extended_ifft_divisor = F::from(1 << extended_k); // Inversion computed later
115
116        // The barycentric weight of 1 over the evaluation domain
117        // 1 / \prod_{i != 0} (1 - omega^i)
118        let mut barycentric_weight = F::from(n); // Inversion computed later
119
120        // Compute batch inversion
121        t_evaluations
122            .iter_mut()
123            .chain(Some(&mut ifft_divisor))
124            .chain(Some(&mut extended_ifft_divisor))
125            .chain(Some(&mut barycentric_weight))
126            .chain(Some(&mut extended_omega_inv))
127            .chain(Some(&mut omega_inv))
128            .batch_invert();
129
130        EvaluationDomain {
131            n,
132            k,
133            extended_k,
134            omega,
135            omega_inv,
136            extended_omega,
137            extended_omega_inv,
138            g_coset,
139            g_coset_inv,
140            quotient_poly_degree,
141            ifft_divisor,
142            extended_ifft_divisor,
143            t_evaluations,
144            barycentric_weight,
145        }
146    }

239    /// This takes us from an n-length coefficient vector into a coset of the extended
240    /// evaluation domain, rotating by `rotation` if desired.
241    pub fn coeff_to_extended(
242        &self,
243        mut a: Polynomial<F, Coeff>,
244    ) -> Polynomial<F, ExtendedLagrangeCoeff> {
245        assert_eq!(a.values.len(), 1 << self.k);
246
247        self.distribute_powers_zeta(&mut a.values, true);
248        a.values.resize(self.extended_len(), F::ZERO);
249        best_fft(&mut a.values, self.extended_omega, self.extended_k);
250
251        Polynomial {
252            values: a.values,
253            _marker: PhantomData,
254        }

257    /// Rotate the extended domain polynomial over the original domain.
258    pub fn rotate_extended(
259        &self,
260        poly: &Polynomial<F, ExtendedLagrangeCoeff>,
261        rotation: Rotation,
262    ) -> Polynomial<F, ExtendedLagrangeCoeff> {
263        let new_rotation = ((1 << (self.extended_k - self.k)) * rotation.0.abs()) as usize;
264
265        let mut poly = poly.clone();
266
267        if rotation.0 >= 0 {
268            poly.values.rotate_left(new_rotation);
269        } else {
270            poly.values.rotate_right(new_rotation);
271        }
272
273        poly
274    }

297    /// This takes us from the extended evaluation domain and gets us the
298    /// quotient polynomial coefficients.
299    ///
300    /// This function will panic if the provided vector is not the correct
301    /// length.
302    // TODO/FIXME: caller should be responsible for truncating
303    pub fn extended_to_coeff(&self, mut a: Polynomial<F, ExtendedLagrangeCoeff>) -> Vec<F> {
304        assert_eq!(a.values.len(), self.extended_len());
305
306        // Inverse FFT
307        Self::ifft(
308            &mut a.values,
309            self.extended_omega_inv,
310            self.extended_k,
311            self.extended_ifft_divisor,
312        );
313
314        // Distribute powers to move from coset; opposite from the
315        // transformation we performed earlier.
316        self.distribute_powers_zeta(&mut a.values, false);
317
318        // Truncate it to match the size of the quotient polynomial; the
319        // evaluation domain might be slightly larger than necessary because
320        // it always lies on a power-of-two boundary.
321        a.values
322            .truncate((&self.n * self.quotient_poly_degree) as usize);
323
324        a.values
325    }

327    /// This divides the polynomial (in the extended domain) by the vanishing
328    /// polynomial of the $2^k$ size domain.
329    pub fn divide_by_vanishing_poly(
330        &self,
331        mut a: Polynomial<F, ExtendedLagrangeCoeff>,
332    ) -> Polynomial<F, ExtendedLagrangeCoeff> {
333        assert_eq!(a.values.len(), self.extended_len());
334
335        // Divide to obtain the quotient polynomial in the coset evaluation
336        // domain.
337        parallelize(&mut a.values, |h, mut index| {
338            for h in h {
339                *h *= &self.t_evaluations[index % self.t_evaluations.len()];
340                index += 1;
341            }
342        });
343
344        Polynomial {
345            values: a.values,
346            _marker: PhantomData,
347        }
348    }

350    /// Given a slice of group elements `[a_0, a_1, a_2, ...]`, this returns
351    /// `[a_0, [zeta]a_1, [zeta^2]a_2, a_3, [zeta]a_4, [zeta^2]a_5, a_6, ...]`,
352    /// where zeta is a cube root of unity in the multiplicative subgroup with
353    /// order (p - 1), i.e. zeta^3 = 1.
354    ///
355    /// `into_coset` should be set to `true` when moving into the coset,
356    /// and `false` when moving out. This toggles the choice of `zeta`.
357    fn distribute_powers_zeta(&self, a: &mut [F], into_coset: bool) {
358        let coset_powers = if into_coset {
359            [self.g_coset, self.g_coset_inv]
360        } else {
361            [self.g_coset_inv, self.g_coset]
362        };
363        parallelize(a, |a, mut index| {
364            for a in a {
365                // Distribute powers to move into/from coset
366                let i = index % (coset_powers.len() + 1);
367                if i != 0 {
368                    *a *= &coset_powers[i - 1];
369                }
370                index += 1;
371            }
372        });
373    }

375    fn ifft(a: &mut [F], omega_inv: F, log_n: u32, divisor: F) {
376        best_fft(a, omega_inv, log_n);
377        parallelize(a, |a, _| {
378            for a in a {
379                // Finish iFFT
380                *a *= &divisor;
381            }
382        });
383    }

406    /// Multiplies a value by some power of $\omega$, essentially rotating over
407    /// the domain.
408    pub fn rotate_omega(&self, value: F, rotation: Rotation) -> F {
409        let mut point = value;
410        if rotation.0 >= 0 {
411            point *= &self.get_omega().pow_vartime([rotation.0 as u64]);
412        } else {
413            point *= &self
414                .get_omega_inv()
415                .pow_vartime([(rotation.0 as i64).unsigned_abs()]);
416        }
417        point
418    }

420    /// Computes evaluations (at the point `x`, where `xn = x^n`) of Lagrange
421    /// basis polynomials `l_i(X)` defined such that `l_i(omega^i) = 1` and
422    /// `l_i(omega^j) = 0` for all `j != i` at each provided rotation `i`.
423    ///
424    /// # Implementation
425    ///
426    /// The polynomial
427    ///     $$\prod_{j=0,j \neq i}^{n - 1} (X - \omega^j)$$
428    /// has a root at all points in the domain except $\omega^i$, where it evaluates to
429    ///     $$\prod_{j=0,j \neq i}^{n - 1} (\omega^i - \omega^j)$$
430    /// and so we divide that polynomial by this value to obtain $l_i(X)$. Since
431    ///     $$\prod_{j=0,j \neq i}^{n - 1} (X - \omega^j)
432    ///       = \frac{X^n - 1}{X - \omega^i}$$
433    /// then $l_i(x)$ for some $x$ is evaluated as
434    ///     $$\left(\frac{x^n - 1}{x - \omega^i}\right)
435    ///       \cdot \left(\frac{1}{\prod_{j=0,j \neq i}^{n - 1} (\omega^i - \omega^j)}\right).$$
436    /// We refer to
437    ///     $$1 \over \prod_{j=0,j \neq i}^{n - 1} (\omega^i - \omega^j)$$
438    /// as the barycentric weight of $\omega^i$.
439    ///
440    /// We know that for $i = 0$
441    ///     $$\frac{1}{\prod_{j=0,j \neq i}^{n - 1} (\omega^i - \omega^j)} = \frac{1}{n}.$$
442    ///
443    /// If we multiply $(1 / n)$ by $\omega^i$ then we obtain
444    ///     $$\frac{1}{\prod_{j=0,j \neq 0}^{n - 1} (\omega^i - \omega^j)}
445    ///       = \frac{1}{\prod_{j=0,j \neq i}^{n - 1} (\omega^i - \omega^j)}$$
446    /// which is the barycentric weight of $\omega^i$.
447    pub fn l_i_range<I: IntoIterator<Item = i32> + Clone>(
448        &self,
449        x: F,
450        xn: F,
451        rotations: I,
452    ) -> Vec<F> {
453        let mut results;
454        {
455            let rotations = rotations.clone().into_iter();
456            results = Vec::with_capacity(rotations.size_hint().1.unwrap_or(0));
457            for rotation in rotations {
458                let rotation = Rotation(rotation);
459                let result = x - self.rotate_omega(F::ONE, rotation);
460                results.push(result);
461            }
462            results.iter_mut().batch_invert();
463        }
464
465        let common = (xn - F::ONE) * self.barycentric_weight;
466        for (rotation, result) in rotations.into_iter().zip(results.iter_mut()) {
467            let rotation = Rotation(rotation);
468            *result = self.rotate_omega(*result * common, rotation);
469        }
470
471        results
472    }

500#[test]
501fn test_rotate() {
502    use rand_core::OsRng;
503
504    use crate::arithmetic::eval_polynomial;
505    use crate::pasta::pallas::Scalar;
506
507    let domain = EvaluationDomain::<Scalar>::new(1, 3);
508    let rng = OsRng;
509
510    let mut poly = domain.empty_lagrange();
511    assert_eq!(poly.len(), 8);
512    for value in poly.iter_mut() {
513        *value = Scalar::random(rng);
514    }
515
516    let poly_rotated_cur = poly.rotate(Rotation::cur());
517    let poly_rotated_next = poly.rotate(Rotation::next());
518    let poly_rotated_prev = poly.rotate(Rotation::prev());
519
520    let poly = domain.lagrange_to_coeff(poly);
521    let poly_rotated_cur = domain.lagrange_to_coeff(poly_rotated_cur);
522    let poly_rotated_next = domain.lagrange_to_coeff(poly_rotated_next);
523    let poly_rotated_prev = domain.lagrange_to_coeff(poly_rotated_prev);
524
525    let x = Scalar::random(rng);
526
527    assert_eq!(
528        eval_polynomial(&poly[..], x),
529        eval_polynomial(&poly_rotated_cur[..], x)
530    );
531    assert_eq!(
532        eval_polynomial(&poly[..], x * domain.omega),
533        eval_polynomial(&poly_rotated_next[..], x)
534    );
535    assert_eq!(
536        eval_polynomial(&poly[..], x * domain.omega_inv),
537        eval_polynomial(&poly_rotated_prev[..], x)
538    );
539}

542fn test_l_i() {
543    use rand_core::OsRng;
544
545    use crate::arithmetic::{eval_polynomial, lagrange_interpolate};
546    use crate::pasta::pallas::Scalar;
547    let domain = EvaluationDomain::<Scalar>::new(1, 3);
548
549    let mut l = vec![];
550    let mut points = vec![];
551    for i in 0..8 {
552        points.push(domain.omega.pow(&[i, 0, 0, 0]));
553    }
554    for i in 0..8 {
555        let mut l_i = vec![Scalar::zero(); 8];
556        l_i[i] = Scalar::ONE;
557        let l_i = lagrange_interpolate(&points[..], &l_i[..]);
558        l.push(l_i);
559    }
560
561    let x = Scalar::random(OsRng);
562    let xn = x.pow(&[8, 0, 0, 0]);
563
564    let evaluations = domain.l_i_range(x, xn, -7..=7);
565    for i in 0..8 {
566        assert_eq!(eval_polynomial(&l[i][..], x), evaluations[7 + i]);
567        assert_eq!(eval_polynomial(&l[(8 - i) % 8][..], x), evaluations[7 - i]);
568    }

576    let k = 11;
