16/// Table containing independent generators S[0..2^k]
17#[derive(Eq, PartialEq, Copy, Clone, Debug)]
18pub struct GeneratorTableConfig {
19    pub table_idx: TableColumn,
20    pub table_x: TableColumn,
21    pub table_y: TableColumn,
22}
23
24impl GeneratorTableConfig {
25    #[allow(clippy::too_many_arguments)]
26    #[allow(non_snake_case)]
27    /// Even though the lookup table can be used in other parts of the circuit,
28    /// this specific configuration sets up Sinsemilla-specific constraints
29    /// controlled by `q_sinsemilla`, and would likely not apply to other chips.
30    pub fn configure<Hash, Commit, F, Lookup>(
31        meta: &mut ConstraintSystem<pallas::Base>,
32        config: super::SinsemillaConfig<Hash, Commit, F, Lookup>,
33    ) where
34        Hash: HashDomains<pallas::Affine>,
35        F: FixedPoints<pallas::Affine>,
36        Commit: CommitDomains<pallas::Affine, F, Hash>,
37        Lookup: PallasLookupRangeCheck,
38    {
39        let (table_idx, table_x, table_y) = (
40            config.generator_table.table_idx,
41            config.generator_table.table_x,
42            config.generator_table.table_y,
43        );
44
45        // https://p.z.cash/halo2-0.1:sinsemilla-constraints?partial
46        meta.lookup(|meta| {
47            let q_s1 = meta.query_selector(config.q_sinsemilla1);
48            let q_s2 = meta.query_fixed(config.q_sinsemilla2);
49            let q_s3 = config.q_s3(meta);
50            let q_run = q_s2 - q_s3;
51
52            // m_{i+1} = z_{i} - 2^K * q_{run,i} * z_{i + 1}
53            // Note that the message words m_i's are 1-indexed while the
54            // running sum z_i's are 0-indexed.
55            let word = {
56                let z_cur = meta.query_advice(config.bits, Rotation::cur());
57                let z_next = meta.query_advice(config.bits, Rotation::next());
58                z_cur - (q_run * z_next * pallas::Base::from(1 << sinsemilla::K))
59            };
60
61            let x_p = meta.query_advice(config.double_and_add.x_p, Rotation::cur());
62
63            // y_{p,i} = (Y_{A,i} / 2) - lambda1 * (x_{A,i} - x_{P,i})
64            let y_p = {
65                let lambda1 = meta.query_advice(config.double_and_add.lambda_1, Rotation::cur());
66                let x_a = meta.query_advice(config.double_and_add.x_a, Rotation::cur());
67                let Y_A = config.double_and_add.Y_A(meta, Rotation::cur());
68
69                (Y_A * pallas::Base::TWO_INV) - (lambda1 * (x_a - x_p.clone()))
70            };
71
72            // Lookup expressions default to the first entry when `q_s1`
73            // is not enabled.
74            let (init_x, init_y) = SINSEMILLA_S[0];
75            let not_q_s1 = Expression::Constant(pallas::Base::one()) - q_s1.clone();
76
77            let m = q_s1.clone() * word; // The first table index is 0.
78            let x_p = q_s1.clone() * x_p + not_q_s1.clone() * init_x;
79            let y_p = q_s1 * y_p + not_q_s1 * init_y;
80
81            vec![(m, table_idx), (x_p, table_x), (y_p, table_y)]
82        });
83    }
84
85    /// Load the generator table into the circuit.
86    pub fn load<F, const K: usize>(
87        &self,
88        lookup_config: impl LookupRangeCheck<F, K>,
89        layouter: &mut impl Layouter<pallas::Base>,
90    ) -> Result<(), Error>
91    where
92        F: PrimeFieldBits,
93    {
94        lookup_config.load(self, layouter)
95    }
