61/// Chip implementing `MerkleInstructions`.
62///
63/// This chip specifically implements `MerkleInstructions::hash_layer` as the `MerkleCRH`
64/// function `hash = SinsemillaHash(Q, ùëô‚ãÜ || left‚ãÜ || right‚ãÜ)`, where:
65/// - `ùëô‚ãÜ = I2LEBSP_10(l)`
66/// - `left‚ãÜ = I2LEBSP_255(left)`
67/// - `right‚ãÜ = I2LEBSP_255(right)`
68///
69/// This chip does **NOT** constrain `left‚ãÜ` and `right‚ãÜ` to be canonical encodings of
70/// `left` and `right`.
71#[derive(Clone, Debug, PartialEq, Eq)]
72pub struct MerkleChip<Hash, Commit, Fixed, Lookup = PallasLookupRangeCheckConfig>
73where
74    Hash: HashDomains<pallas::Affine>,
75    Fixed: FixedPoints<pallas::Affine>,
76    Commit: CommitDomains<pallas::Affine, Fixed, Hash>,
77    Lookup: PallasLookupRangeCheck,
78{
79    config: MerkleConfig<Hash, Commit, Fixed, Lookup>,
80}

101impl<Hash, Commit, F, Lookup> MerkleChip<Hash, Commit, F, Lookup>
102where
103    Hash: HashDomains<pallas::Affine>,
104    F: FixedPoints<pallas::Affine>,
105    Commit: CommitDomains<pallas::Affine, F, Hash>,
106    Lookup: PallasLookupRangeCheck,
107{
108    /// Configures the [`MerkleChip`].
109    pub fn configure(
110        meta: &mut ConstraintSystem<pallas::Base>,
111        sinsemilla_config: SinsemillaConfig<Hash, Commit, F, Lookup>,
112    ) -> MerkleConfig<Hash, Commit, F, Lookup> {
113        // All five advice columns are equality-enabled by SinsemillaConfig.
114        let advices = sinsemilla_config.advices();
115        let cond_swap_config = CondSwapChip::configure(meta, advices);
116
117        // This selector enables the decomposition gate.
118        let q_decompose = meta.selector();
119
120        // Check that pieces have been decomposed correctly for Sinsemilla hash.
121        // <https://zips.z.cash/protocol/protocol.pdf#orchardmerklecrh>
122        //
123        // a = a_0||a_1 = l || (bits 0..=239 of left)
124        // b = b_0||b_1||b_2
125        //   = (bits 240..=249 of left) || (bits 250..=254 of left) || (bits 0..=4 of right)
126        // c = bits 5..=254 of right
127        //
128        // The message pieces `a`, `b`, `c` are constrained by Sinsemilla to be
129        // 250 bits, 20 bits, and 250 bits respectively.
130        //
131        // The pieces and subpieces are arranged in the following configuration:
132        // |  A_0  |  A_1  |  A_2  |  A_3  |  A_4  | q_decompose |
133        // -------------------------------------------------------
134        // |   a   |   b   |   c   |  left | right |      1      |
135        // |  z1_a |  z1_b |  b_1  |  b_2  |   l   |      0      |
136        meta.create_gate("Decomposition check", |meta| {
137            let q_decompose = meta.query_selector(q_decompose);
138            let l_whole = meta.query_advice(advices[4], Rotation::next());
139
140            let two_pow_5 = pallas::Base::from(1 << 5);
141            let two_pow_10 = two_pow_5.square();
142
143            // a_whole is constrained by Sinsemilla to be 250 bits.
144            let a_whole = meta.query_advice(advices[0], Rotation::cur());
145            // b_whole is constrained by Sinsemilla to be 20 bits.
146            let b_whole = meta.query_advice(advices[1], Rotation::cur());
147            // c_whole is constrained by Sinsemilla to be 250 bits.
148            let c_whole = meta.query_advice(advices[2], Rotation::cur());
149            let left_node = meta.query_advice(advices[3], Rotation::cur());
150            let right_node = meta.query_advice(advices[4], Rotation::cur());
151
152            // a = a_0||a_1 = l || (bits 0..=239 of left)
153            //
154            // z_1 of SinsemillaHash(a) = a_1
155            // => a_0 = a - (a_1 * 2^10)
156            let z1_a = meta.query_advice(advices[0], Rotation::next());
157            let a_1 = z1_a;
158            // Derive a_0 (constrained by SinsemillaHash to be 10 bits)
159            let a_0 = a_whole - a_1.clone() * two_pow_10;
160
161            // b = b_0||b_1||b_2
162            //   = (bits 240..=249 of left) || (bits 250..=254 of left) || (bits 0..=4 of right)
163            // The Orchard specification allows this representation to be non-canonical.
164            // <https://zips.z.cash/protocol/protocol.pdf#merklepath>
165            //
166            //    z_1 of SinsemillaHash(b) = b_1 + 2^5 b_2
167            // => b_0 = b - (z1_b * 2^10)
168            let z1_b = meta.query_advice(advices[1], Rotation::next());
169            // b_1 has been constrained to be 5 bits outside this gate.
170            let b_1 = meta.query_advice(advices[2], Rotation::next());
171            // b_2 has been constrained to be 5 bits outside this gate.
172            let b_2 = meta.query_advice(advices[3], Rotation::next());
173            // Constrain b_1 + 2^5 b_2 = z1_b
174            // https://p.z.cash/halo2-0.1:sinsemilla-merkle-crh-bit-lengths?partial
175            let b1_b2_check = z1_b.clone() - (b_1.clone() + b_2.clone() * two_pow_5);
176            // Derive b_0 (constrained by SinsemillaHash to be 10 bits)
177            let b_0 = b_whole - (z1_b * two_pow_10);
178
179            // Check that left = a_1 (240 bits) || b_0 (10 bits) || b_1 (5 bits)
180            // https://p.z.cash/halo2-0.1:sinsemilla-merkle-crh-decomposition?partial
181            let left_check = {
182                let reconstructed = {
183                    let two_pow_240 = pallas::Base::from_u128(1 << 120).square();
184                    a_1 + (b_0 + b_1 * two_pow_10) * two_pow_240
185                };
186                reconstructed - left_node
187            };
188
189            // Check that right = b_2 (5 bits) || c (250 bits)
190            // The Orchard specification allows this representation to be non-canonical.
191            // <https://zips.z.cash/protocol/protocol.pdf#merklepath>
192            // https://p.z.cash/halo2-0.1:sinsemilla-merkle-crh-decomposition?partial
193            let right_check = b_2 + c_whole * two_pow_5 - right_node;
194
195            Constraints::with_selector(
196                q_decompose,
197                [
198                    ("l_check", a_0 - l_whole),
199                    ("left_check", left_check),
200                    ("right_check", right_check),
201                    ("b1_b2_check", b1_b2_check),
202                ],
203            )
204        });
205
206        MerkleConfig {
207            advices,
208            q_decompose,
209            cond_swap_config,
210            sinsemilla_config,
211        }
212    }

217    }

220impl<Hash, Commit, Fixed, Lookup, const MERKLE_DEPTH: usize>
221    MerkleInstructions<pallas::Affine, MERKLE_DEPTH, { sinsemilla::K }, { sinsemilla::C }>
222    for MerkleChip<Hash, Commit, Fixed, Lookup>
223where
224    Hash: HashDomains<pallas::Affine> + Eq,
225    Fixed: FixedPoints<pallas::Affine>,
226    Commit: CommitDomains<pallas::Affine, Fixed, Hash> + Eq,
227    Lookup: PallasLookupRangeCheck,
228{
229    #[allow(non_snake_case)]
230    fn hash_layer(
231        &self,
232        mut layouter: impl Layouter<pallas::Base>,
233        Q: pallas::Affine,
234        // l = MERKLE_DEPTH - layer - 1
235        l: usize,
236        left: Self::Var,
237        right: Self::Var,
238    ) -> Result<Self::Var, Error> {
239        let config = self.config().clone();
240
241        // We need to hash `l || left || right`, where `l` is a 10-bit value.
242        // We allow `left` and `right` to be non-canonical 255-bit encodings.
243        //
244        // a = a_0||a_1 = l || (bits 0..=239 of left)
245        // b = b_0||b_1||b_2
246        //   = (bits 240..=249 of left) || (bits 250..=254 of left) || (bits 0..=4 of right)
247        // c = bits 5..=254 of right
248        //
249        // We start by witnessing all of the individual pieces, and range-constraining the
250        // short pieces b_1 and b_2.
251        //
252        // https://p.z.cash/halo2-0.1:sinsemilla-merkle-crh-bit-lengths?partial
253
254        // `a = a_0||a_1` = `l` || (bits 0..=239 of `left`)
255        let a = MessagePiece::from_subpieces(
256            self.clone(),
257            layouter.namespace(|| "Witness a = a_0 || a_1"),
258            [
259                RangeConstrained::bitrange_of(Value::known(&pallas::Base::from(l as u64)), 0..10),
260                RangeConstrained::bitrange_of(left.value(), 0..240),
261            ],
262        )?;
263
264        // b = b_0 || b_1 || b_2
265        //   = (bits 240..=249 of left) || (bits 250..=254 of left) || (bits 0..=4 of right)
266        let (b_1, b_2, b) = {
267            // b_0 = (bits 240..=249 of `left`)
268            let b_0 = RangeConstrained::bitrange_of(left.value(), 240..250);
269
270            // b_1 = (bits 250..=254 of `left`)
271            // Constrain b_1 to 5 bits.
272            let b_1 = RangeConstrained::witness_short(
273                &config.sinsemilla_config.lookup_config(),
274                layouter.namespace(|| "b_1"),
275                left.value(),
276                250..(pallas::Base::NUM_BITS as usize),
277            )?;
278
279            // b_2 = (bits 0..=4 of `right`)
280            // Constrain b_2 to 5 bits.
281            let b_2 = RangeConstrained::witness_short(
282                &config.sinsemilla_config.lookup_config(),
283                layouter.namespace(|| "b_2"),
284                right.value(),
285                0..5,
286            )?;
287
288            let b = MessagePiece::from_subpieces(
289                self.clone(),
290                layouter.namespace(|| "Witness b = b_0 || b_1 || b_2"),
291                [b_0, b_1.value(), b_2.value()],
292            )?;
293
294            (b_1, b_2, b)
295        };
296
297        // c = bits 5..=254 of `right`
298        let c = MessagePiece::from_subpieces(
299            self.clone(),
300            layouter.namespace(|| "Witness c"),
301            [RangeConstrained::bitrange_of(
302                right.value(),
303                5..(pallas::Base::NUM_BITS as usize),
304            )],
305        )?;
306
307        // hash = SinsemillaHash(Q, ùëô‚ãÜ || left‚ãÜ || right‚ãÜ)
308        //
309        // `hash = ‚ä•` is handled internally to `SinsemillaChip::hash_to_point`: incomplete
310        // addition constraints allows ‚ä• to occur, and then during synthesis it detects
311        // these edge cases and raises an error (aborting proof creation).
312        //
313        // Note that MerkleCRH as-defined maps ‚ä• to 0. This is for completeness outside
314        // the circuit (so that the ‚ä• does not propagate into the type system). The chip
315        // explicitly doesn't map ‚ä• to 0; in fact it cannot, as doing so would require
316        // constraints that amount to using complete addition. The rationale for excluding
317        // this map is the same as why Sinsemilla uses incomplete addition: this situation
318        // yields a nontrivial discrete log relation, and by assumption it is hard to find
319        // these.
320        //
321        // https://p.z.cash/proto:merkle-crh-orchard
322        let (point, zs) = self.hash_to_point(
323            layouter.namespace(|| format!("hash at l = {}", l)),
324            Q,
325            vec![a.inner(), b.inner(), c.inner()].into(),
326        )?;
327        let hash = Self::extract(&point);
328
329        // `SinsemillaChip::hash_to_point` returns the running sum for each `MessagePiece`.
330        // Grab the outputs we need for the decomposition constraints.
331        let z1_a = zs[0][1].clone();
332        let z1_b = zs[1][1].clone();
333
334        // Check that the pieces have been decomposed properly.
335        //
336        // The pieces and subpieces are arranged in the following configuration:
337        // |  A_0  |  A_1  |  A_2  |  A_3  |  A_4  | q_decompose |
338        // -------------------------------------------------------
339        // |   a   |   b   |   c   |  left | right |      1      |
340        // |  z1_a |  z1_b |  b_1  |  b_2  |   l   |      0      |
341        {
342            layouter.assign_region(
343                || "Check piece decomposition",
344                |mut region| {
345                    // Set the fixed column `l` to the current l.
346                    // Recall that l = MERKLE_DEPTH - layer - 1.
347                    // The layer with 2^n nodes is called "layer n".
348                    config.q_decompose.enable(&mut region, 0)?;
349                    region.assign_advice_from_constant(
350                        || format!("l {}", l),
351                        config.advices[4],
352                        1,
353                        pallas::Base::from(l as u64),
354                    )?;
355
356                    // Offset 0
357                    // Copy and assign `a` at the correct position.
358                    a.inner().cell_value().copy_advice(
359                        || "copy a",
360                        &mut region,
361                        config.advices[0],
362                        0,
363                    )?;
364                    // Copy and assign `b` at the correct position.
365                    b.inner().cell_value().copy_advice(
366                        || "copy b",
367                        &mut region,
368                        config.advices[1],
369                        0,
370                    )?;
371                    // Copy and assign `c` at the correct position.
372                    c.inner().cell_value().copy_advice(
373                        || "copy c",
374                        &mut region,
375                        config.advices[2],
376                        0,
377                    )?;
378                    // Copy and assign the left node at the correct position.
379                    left.copy_advice(|| "left", &mut region, config.advices[3], 0)?;
380                    // Copy and assign the right node at the correct position.
381                    right.copy_advice(|| "right", &mut region, config.advices[4], 0)?;
382
383                    // Offset 1
384                    // Copy and assign z_1 of SinsemillaHash(a) = a_1
385                    z1_a.copy_advice(|| "z1_a", &mut region, config.advices[0], 1)?;
386                    // Copy and assign z_1 of SinsemillaHash(b) = b_1
387                    z1_b.copy_advice(|| "z1_b", &mut region, config.advices[1], 1)?;
388                    // Copy `b_1`, which has been constrained to be a 5-bit value
389                    b_1.inner()
390                        .copy_advice(|| "b_1", &mut region, config.advices[2], 1)?;
391                    // Copy `b_2`, which has been constrained to be a 5-bit value
392                    b_2.inner()
393                        .copy_advice(|| "b_2", &mut region, config.advices[3], 1)?;
394
395                    Ok(())
396                },
397            )?;
398        }
399
400        // Check layer hash output against Sinsemilla primitives hash
401        #[cfg(test)]
402        {
403            use crate::{sinsemilla::primitives::HashDomain, utilities::i2lebsp};
404
405            use group::ff::PrimeFieldBits;
406
407            left.value()
408                .zip(right.value())
409                .zip(hash.value())
410                .assert_if_known(|((left, right), hash)| {
411                    let l = i2lebsp::<10>(l as u64);
412                    let left: Vec<_> = left
413                        .to_le_bits()
414                        .iter()
415                        .by_vals()
416                        .take(pallas::Base::NUM_BITS as usize)
417                        .collect();
418                    let right: Vec<_> = right
419                        .to_le_bits()
420                        .iter()
421                        .by_vals()
422                        .take(pallas::Base::NUM_BITS as usize)
423                        .collect();
424                    let merkle_crh = HashDomain::from_Q(Q.into());
425
426                    let mut message = l.to_vec();
427                    message.extend_from_slice(&left);
428                    message.extend_from_slice(&right);
429
430                    let expected = merkle_crh.hash(message.into_iter()).unwrap();
431
432                    expected.to_repr() == hash.to_repr()
433                });
434        }
435
436        Ok(hash)
437    }
438}
