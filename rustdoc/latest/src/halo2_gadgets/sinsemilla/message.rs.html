1//! Gadget and chips for the Sinsemilla hash function.
2use ff::{Field, PrimeFieldBits};
3use halo2_proofs::circuit::{AssignedCell, Cell, Value};
4use std::fmt::Debug;
5
6/// A [`Message`] composed of several [`MessagePiece`]s.
7#[derive(Clone, Debug)]
8pub struct Message<F: Field, const K: usize, const MAX_WORDS: usize>(Vec<MessagePiece<F, K>>);
9
10impl<F: PrimeFieldBits, const K: usize, const MAX_WORDS: usize> From<Vec<MessagePiece<F, K>>>
11    for Message<F, K, MAX_WORDS>
12{
13    fn from(pieces: Vec<MessagePiece<F, K>>) -> Self {
14        // A message cannot contain more than `MAX_WORDS` words.
15        assert!(pieces.iter().map(|piece| piece.num_words()).sum::<usize>() < MAX_WORDS);
16        Message(pieces)
17    }
18}
19
20impl<F: PrimeFieldBits, const K: usize, const MAX_WORDS: usize> std::ops::Deref
21    for Message<F, K, MAX_WORDS>
22{
23    type Target = [MessagePiece<F, K>];
24
25    fn deref(&self) -> &[MessagePiece<F, K>] {
26        &self.0
27    }
28}
29
30/// A [`MessagePiece`] of some bitlength.
31///
32/// The piece must fit within a base field element, which means its length
33/// cannot exceed the base field's `NUM_BITS`.
34#[derive(Clone, Debug)]
35pub struct MessagePiece<F: Field, const K: usize> {
36    cell_value: AssignedCell<F, F>,
37    /// The number of K-bit words in this message piece.
38    num_words: usize,
39}
40
41impl<F: PrimeFieldBits, const K: usize> MessagePiece<F, K> {
42    pub fn new(cell_value: AssignedCell<F, F>, num_words: usize) -> Self {
43        assert!(num_words * K < F::NUM_BITS as usize);
44        Self {
45            cell_value,
46            num_words,
47        }
48    }
49
50    pub fn num_words(&self) -> usize {
51        self.num_words
52    }
53
54    pub fn cell(&self) -> Cell {
55        self.cell_value.cell()
56    }
57
58    pub fn field_elem(&self) -> Value<F> {
59        self.cell_value.value().cloned()
60    }
61
62    pub fn cell_value(&self) -> AssignedCell<F, F> {
63        self.cell_value.clone()
64    }
65}
