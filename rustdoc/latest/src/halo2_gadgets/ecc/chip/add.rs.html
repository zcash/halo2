77    fn create_gate(&self, meta: &mut ConstraintSystem<pallas::Base>) {
78        // https://p.z.cash/halo2-0.1:ecc-complete-addition
79        meta.create_gate("complete addition", |meta| {
80            let q_add = meta.query_selector(self.q_add);
81            let x_p = meta.query_advice(self.x_p, Rotation::cur());
82            let y_p = meta.query_advice(self.y_p, Rotation::cur());
83            let x_q = meta.query_advice(self.x_qr, Rotation::cur());
84            let y_q = meta.query_advice(self.y_qr, Rotation::cur());
85            let x_r = meta.query_advice(self.x_qr, Rotation::next());
86            let y_r = meta.query_advice(self.y_qr, Rotation::next());
87            let lambda = meta.query_advice(self.lambda, Rotation::cur());
88
89            // Œ± = inv0(x_q - x_p)
90            let alpha = meta.query_advice(self.alpha, Rotation::cur());
91            // Œ≤ = inv0(x_p)
92            let beta = meta.query_advice(self.beta, Rotation::cur());
93            // Œ≥ = inv0(x_q)
94            let gamma = meta.query_advice(self.gamma, Rotation::cur());
95            // Œ¥ = inv0(y_p + y_q) if x_q = x_p, 0 otherwise
96            let delta = meta.query_advice(self.delta, Rotation::cur());
97
98            // Useful composite expressions
99            // (x_q ‚àí x_p)
100            let x_q_minus_x_p = x_q.clone() - x_p.clone();
101            // (x_p - x_r)
102            let x_p_minus_x_r = x_p.clone() - x_r.clone();
103            // (y_q + y_p)
104            let y_q_plus_y_p = y_q.clone() + y_p.clone();
105            // Œ± ‚ãÖ(x_q - x_p)
106            let if_alpha = x_q_minus_x_p.clone() * alpha;
107            // Œ≤ ‚ãÖ x_p
108            let if_beta = x_p.clone() * beta;
109            // Œ≥ ‚ãÖ x_q
110            let if_gamma = x_q.clone() * gamma;
111            // Œ¥ ‚ãÖ(y_q + y_p)
112            let if_delta = y_q_plus_y_p.clone() * delta;
113
114            // Useful constants
115            let one = Expression::Constant(pallas::Base::one());
116            let two = Expression::Constant(pallas::Base::from(2));
117            let three = Expression::Constant(pallas::Base::from(3));
118
119            // (x_q ‚àí x_p)‚ãÖ((x_q ‚àí x_p)‚ãÖŒª ‚àí (y_q‚àíy_p)) = 0
120            let poly1 = {
121                let y_q_minus_y_p = y_q.clone() - y_p.clone(); // (y_q ‚àí y_p)
122                let incomplete = x_q_minus_x_p.clone() * lambda.clone() - y_q_minus_y_p; // (x_q ‚àí x_p)‚ãÖŒª ‚àí (y_q‚àíy_p)
123
124                // q_add ‚ãÖ(x_q ‚àí x_p)‚ãÖ((x_q ‚àí x_p)‚ãÖŒª ‚àí (y_q‚àíy_p))
125                x_q_minus_x_p.clone() * incomplete
126            };
127
128            // (1 - (x_q - x_p)‚ãÖŒ±)‚ãÖ(2y_p ‚ãÖŒª - 3x_p^2) = 0
129            let poly2 = {
130                let three_x_p_sq = three * x_p.clone().square(); // 3x_p^2
131                let two_y_p = two * y_p.clone(); // 2y_p
132                let tangent_line = two_y_p * lambda.clone() - three_x_p_sq; // (2y_p ‚ãÖŒª - 3x_p^2)
133
134                // q_add ‚ãÖ(1 - (x_q - x_p)‚ãÖŒ±)‚ãÖ(2y_p ‚ãÖŒª - 3x_p^2)
135                (one.clone() - if_alpha.clone()) * tangent_line
136            };
137
138            // (Œª^2 - x_p - x_q - x_r)
139            let nonexceptional_x_r =
140                lambda.clone().square() - x_p.clone() - x_q.clone() - x_r.clone();
141            // (Œª ‚ãÖ(x_p - x_r) - y_p - y_r)
142            let nonexceptional_y_r = lambda * x_p_minus_x_r - y_p.clone() - y_r.clone();
143
144            // x_p‚ãÖx_q‚ãÖ(x_q - x_p)‚ãÖ(Œª^2 - x_p - x_q - x_r) = 0
145            let poly3a =
146                x_p.clone() * x_q.clone() * x_q_minus_x_p.clone() * nonexceptional_x_r.clone();
147
148            // x_p‚ãÖx_q‚ãÖ(x_q - x_p)‚ãÖ(Œª ‚ãÖ(x_p - x_r) - y_p - y_r) = 0
149            let poly3b = x_p.clone() * x_q.clone() * x_q_minus_x_p * nonexceptional_y_r.clone();
150
151            // x_p‚ãÖx_q‚ãÖ(y_q + y_p)‚ãÖ(Œª^2 - x_p - x_q - x_r) = 0
152            let poly3c = x_p.clone() * x_q.clone() * y_q_plus_y_p.clone() * nonexceptional_x_r;
153
154            // x_p‚ãÖx_q‚ãÖ(y_q + y_p)‚ãÖ(Œª ‚ãÖ(x_p - x_r) - y_p - y_r) = 0
155            let poly3d = x_p.clone() * x_q.clone() * y_q_plus_y_p * nonexceptional_y_r;
156
157            // (1 - x_p * Œ≤) * (x_r - x_q) = 0
158            let poly4a = (one.clone() - if_beta.clone()) * (x_r.clone() - x_q);
159
160            // (1 - x_p * Œ≤) * (y_r - y_q) = 0
161            let poly4b = (one.clone() - if_beta) * (y_r.clone() - y_q);
162
163            // (1 - x_q * Œ≥) * (x_r - x_p) = 0
164            let poly5a = (one.clone() - if_gamma.clone()) * (x_r.clone() - x_p);
165
166            // (1 - x_q * Œ≥) * (y_r - y_p) = 0
167            let poly5b = (one.clone() - if_gamma) * (y_r.clone() - y_p);
168
169            // ((1 - (x_q - x_p) * Œ± - (y_q + y_p) * Œ¥)) * x_r
170            let poly6a = (one.clone() - if_alpha.clone() - if_delta.clone()) * x_r;
171
172            // ((1 - (x_q - x_p) * Œ± - (y_q + y_p) * Œ¥)) * y_r
173            let poly6b = (one - if_alpha - if_delta) * y_r;
174
175            Constraints::with_selector(
176                q_add,
177                [
178                    ("1", poly1),
179                    ("2", poly2),
180                    ("3a", poly3a),
181                    ("3b", poly3b),
182                    ("3c", poly3c),
183                    ("3d", poly3d),
184                    ("4a", poly4a),
185                    ("4b", poly4b),
186                    ("5a", poly5a),
187                    ("5b", poly5b),
188                    ("6a", poly6a),
189                    ("6b", poly6b),
190                ],
191            )
192        });
193    }

195    pub(super) fn assign_region(
196        &self,
197        p: &EccPoint,
198        q: &EccPoint,
199        offset: usize,
200        region: &mut Region<'_, pallas::Base>,
201    ) -> Result<EccPoint, Error> {
202        // Enable `q_add` selector
203        self.q_add.enable(region, offset)?;
204
205        // Copy point `p` into `x_p`, `y_p` columns
206        p.x.copy_advice(|| "x_p", region, self.x_p, offset)?;
207        p.y.copy_advice(|| "y_p", region, self.y_p, offset)?;
208
209        // Copy point `q` into `x_qr`, `y_qr` columns
210        q.x.copy_advice(|| "x_q", region, self.x_qr, offset)?;
211        q.y.copy_advice(|| "y_q", region, self.y_qr, offset)?;
212
213        let (x_p, y_p) = (p.x.value(), p.y.value());
214        let (x_q, y_q) = (q.x.value(), q.y.value());
215
216        // Assign Œ± = inv0(x_q - x_p)
217        let alpha = (x_q - x_p).invert();
218        region.assign_advice(|| "Œ±", self.alpha, offset, || alpha)?;
219
220        // Assign Œ≤ = inv0(x_p)
221        let beta = x_p.invert();
222        region.assign_advice(|| "Œ≤", self.beta, offset, || beta)?;
223
224        // Assign Œ≥ = inv0(x_q)
225        let gamma = x_q.invert();
226        region.assign_advice(|| "Œ≥", self.gamma, offset, || gamma)?;
227
228        // Assign Œ¥ = inv0(y_q + y_p) if x_q = x_p, 0 otherwise
229        let delta = x_p
230            .zip(x_q)
231            .zip(y_p)
232            .zip(y_q)
233            .map(|(((x_p, x_q), y_p), y_q)| {
234                if x_q == x_p {
235                    (y_q + y_p).invert()
236                } else {
237                    Assigned::Zero
238                }
239            });
240        region.assign_advice(|| "Œ¥", self.delta, offset, || delta)?;
241
242        #[allow(clippy::collapsible_else_if)]
243        // Assign lambda
244        let lambda =
245            x_p.zip(y_p)
246                .zip(x_q)
247                .zip(y_q)
248                .zip(alpha)
249                .map(|((((x_p, y_p), x_q), y_q), alpha)| {
250                    if x_q != x_p {
251                        // Œª = (y_q - y_p)/(x_q - x_p)
252                        // Here, alpha = inv0(x_q - x_p), which suffices since we
253                        // know that x_q != x_p in this branch.
254                        (y_q - y_p) * alpha
255                    } else {
256                        if !y_p.is_zero_vartime() {
257                            // 3(x_p)^2
258                            let three_x_p_sq = x_p.square() * pallas::Base::from(3);
259                            // 1 / 2(y_p)
260                            let inv_two_y_p = y_p.invert() * pallas::Base::TWO_INV;
261                            // Œª = 3(x_p)^2 / 2(y_p)
262                            three_x_p_sq * inv_two_y_p
263                        } else {
264                            Assigned::Zero
265                        }
266                    }
267                });
268        region.assign_advice(|| "Œª", self.lambda, offset, || lambda)?;
269
270        // Calculate (x_r, y_r)
271        let r =
272            x_p.zip(y_p)
273                .zip(x_q)
274                .zip(y_q)
275                .zip(lambda)
276                .map(|((((x_p, y_p), x_q), y_q), lambda)| {
277                    {
278                        if x_p.is_zero_vartime() {
279                            // 0 + Q = Q
280                            (*x_q, *y_q)
281                        } else if x_q.is_zero_vartime() {
282                            // P + 0 = P
283                            (*x_p, *y_p)
284                        } else if (x_q == x_p) && (*y_q == -y_p) {
285                            // P + (-P) maps to (0,0)
286                            (Assigned::Zero, Assigned::Zero)
287                        } else {
288                            // x_r = Œª^2 - x_p - x_q
289                            let x_r = lambda.square() - x_p - x_q;
290                            // y_r = Œª(x_p - x_r) - y_p
291                            let y_r = lambda * (x_p - x_r) - y_p;
292                            (x_r, y_r)
293                        }
294                    }
295                });
296
297        // Assign x_r
298        let x_r = r.map(|r| r.0);
299        let x_r_cell = region.assign_advice(|| "x_r", self.x_qr, offset + 1, || x_r)?;
300
301        // Assign y_r
302        let y_r = r.map(|r| r.1);
303        let y_r_cell = region.assign_advice(|| "y_r", self.y_qr, offset + 1, || y_r)?;
304
305        let result = EccPoint::from_coordinates_unchecked(x_r_cell, y_r_cell);
306
307        #[cfg(test)]
308        // Check that the correct sum is obtained.
309        {
310            use group::Curve;
311
312            let p = p.point();
313            let q = q.point();
314            let real_sum = p.zip(q).map(|(p, q)| p + q);
315            let result = result.point();
316
317            real_sum
318                .zip(result)
319                .assert_if_known(|(real_sum, result)| &real_sum.to_affine() == result);
320        }
321
322        Ok(result)
323    }

337    #[allow(clippy::too_many_arguments)]
338    pub fn test_add<
339        EccChip: EccInstructions<pallas::Affine, Point = EccPoint> + Clone + Eq + std::fmt::Debug,
340    >(
341        chip: EccChip,
342        mut layouter: impl Layouter<pallas::Base>,
343        p_val: pallas::Affine,
344        p: &NonIdentityPoint<pallas::Affine, EccChip>,
345        q_val: pallas::Affine,
346        q: &NonIdentityPoint<pallas::Affine, EccChip>,
347        p_neg: &NonIdentityPoint<pallas::Affine, EccChip>,
348    ) -> Result<(), Error> {
349        // Make sure P and Q are not the same point.
350        assert_ne!(p_val, q_val);
351
352        // Check complete addition P + (-P)
353        let zero = {
354            let result = p.add(layouter.namespace(|| "P + (-P)"), p_neg)?;
355            result
356                .inner()
357                .is_identity()
358                .assert_if_known(|is_identity| *is_identity);
359            result
360        };
361
362        // Check complete addition ùí™ + ùí™
363        {
364            let result = zero.add(layouter.namespace(|| "ùí™ + ùí™"), &zero)?;
365            result.constrain_equal(layouter.namespace(|| "ùí™ + ùí™ = ùí™"), &zero)?;
366        }
367
368        // Check P + Q
369        {
370            let result = p.add(layouter.namespace(|| "P + Q"), q)?;
371            let witnessed_result = NonIdentityPoint::new(
372                chip.clone(),
373                layouter.namespace(|| "witnessed P + Q"),
374                Value::known((p_val + q_val).to_affine()),
375            )?;
376            result.constrain_equal(layouter.namespace(|| "constrain P + Q"), &witnessed_result)?;
377        }
378
379        // P + P
380        {
381            let result = p.add(layouter.namespace(|| "P + P"), p)?;
382            let witnessed_result = NonIdentityPoint::new(
383                chip.clone(),
384                layouter.namespace(|| "witnessed P + P"),
385                Value::known((p_val + p_val).to_affine()),
386            )?;
387            result.constrain_equal(layouter.namespace(|| "constrain P + P"), &witnessed_result)?;
388        }
389
390        // P + ùí™
391        {
392            let result = p.add(layouter.namespace(|| "P + ùí™"), &zero)?;
393            result.constrain_equal(layouter.namespace(|| "P + ùí™ = P"), p)?;
394        }
395
396        // ùí™ + P
397        {
398            let result = zero.add(layouter.namespace(|| "ùí™ + P"), p)?;
399            result.constrain_equal(layouter.namespace(|| "ùí™ + P = P"), p)?;
400        }
401
402        // (x, y) + (Œ∂x, y) should behave like normal P + Q.
403        let endo_p = p_val.to_curve().endo();
404        let endo_p = NonIdentityPoint::new(
405            chip.clone(),
406            layouter.namespace(|| "endo(P)"),
407            Value::known(endo_p.to_affine()),
408        )?;
409        p.add(layouter.namespace(|| "P + endo(P)"), &endo_p)?;
410
411        // (x, y) + (Œ∂x, -y) should also behave like normal P + Q.
412        let endo_p_neg = (-p_val).to_curve().endo();
413        let endo_p_neg = NonIdentityPoint::new(
414            chip.clone(),
415            layouter.namespace(|| "endo(-P)"),
416            Value::known(endo_p_neg.to_affine()),
417        )?;
418        p.add(layouter.namespace(|| "P + endo(-P)"), &endo_p_neg)?;
419
420        // (x, y) + ((Œ∂^2)x, y)
421        let endo_2_p = p_val.to_curve().endo().endo();
422        let endo_2_p = NonIdentityPoint::new(
423            chip.clone(),
424            layouter.namespace(|| "endo^2(P)"),
425            Value::known(endo_2_p.to_affine()),
426        )?;
427        p.add(layouter.namespace(|| "P + endo^2(P)"), &endo_2_p)?;
428
429        // (x, y) + ((Œ∂^2)x, -y)
430        let endo_2_p_neg = (-p_val).to_curve().endo().endo();
431        let endo_2_p_neg = NonIdentityPoint::new(
432            chip,
433            layouter.namespace(|| "endo^2(-P)"),
434            Value::known(endo_2_p_neg.to_affine()),
435        )?;
436        p.add(layouter.namespace(|| "P + endo^2(-P)"), &endo_2_p_neg)?;
437
438        Ok(())
439    }
