13#[derive(Copy, Clone, Debug, Eq, PartialEq)]
14pub struct Config {
15    // Selector used to constrain the cells used in complete addition.
16    q_mul_decompose_var: Selector,
17    // Advice column used to decompose scalar in complete addition.
18    pub z_complete: Column<Advice>,
19    // Configuration used in complete addition
20    add_config: add::Config,
21}

23impl Config {
24    pub(super) fn configure(
25        meta: &mut ConstraintSystem<pallas::Base>,
26        z_complete: Column<Advice>,
27        add_config: add::Config,
28    ) -> Self {
29        meta.enable_equality(z_complete);
30
31        let config = Self {
32            q_mul_decompose_var: meta.selector(),
33            z_complete,
34            add_config,
35        };
36
37        config.create_gate(meta);
38
39        config
40    }

42    /// Gate used to check scalar decomposition is correct.
43    /// This is used to check the bits used in complete addition, since the incomplete
44    /// addition gate (controlled by `q_mul`) already checks scalar decomposition for
45    /// the other bits.
46    fn create_gate(&self, meta: &mut ConstraintSystem<pallas::Base>) {
47        // | y_p | z_complete |
48        // --------------------
49        // | y_p | z_{i + 1}  |
50        // |     | base_y     |
51        // |     | z_i        |
52        // https://p.z.cash/halo2-0.1:ecc-var-mul-complete-gate
53        meta.create_gate(
54            "Decompose scalar for complete bits of variable-base mul",
55            |meta| {
56                let q_mul_decompose_var = meta.query_selector(self.q_mul_decompose_var);
57                // z_{i + 1}
58                let z_prev = meta.query_advice(self.z_complete, Rotation::prev());
59                // z_i
60                let z_next = meta.query_advice(self.z_complete, Rotation::next());
61
62                // k_{i} = z_{i} - 2⋅z_{i+1}
63                let k = z_next - Expression::Constant(pallas::Base::from(2)) * z_prev;
64                // (k_i) ⋅ (1 - k_i) = 0
65                let bool_check = bool_check(k.clone());
66
67                // base_y
68                let base_y = meta.query_advice(self.z_complete, Rotation::cur());
69                // y_p
70                let y_p = meta.query_advice(self.add_config.y_p, Rotation::prev());
71
72                // k_i = 0 => y_p = -base_y
73                // k_i = 1 => y_p = base_y
74                let y_switch = ternary(k, base_y.clone() - y_p.clone(), base_y + y_p);
75
76                Constraints::with_selector(
77                    q_mul_decompose_var,
78                    [("bool_check", bool_check), ("y_switch", y_switch)],
79                )
80            },
81        );
82    }

84    #[allow(clippy::type_complexity)]
85    #[allow(non_snake_case)]
86    #[allow(clippy::too_many_arguments)]
87    pub(super) fn assign_region(
88        &self,
89        region: &mut Region<'_, pallas::Base>,
90        offset: usize,
91        bits: &[Value<bool>],
92        base: &EccPoint,
93        x_a: X<pallas::Base>,
94        y_a: Y<pallas::Base>,
95        z: Z<pallas::Base>,
96    ) -> Result<(EccPoint, Vec<Z<pallas::Base>>), Error> {
97        // Make sure we have the correct number of bits for the complete addition
98        // part of variable-base scalar mul.
99        assert_eq!(bits.len(), COMPLETE_RANGE.len());
100
101        // Enable selectors for complete range
102        for row in 0..COMPLETE_RANGE.len() {
103            // Each iteration uses 2 rows (two complete additions)
104            let row = 2 * row;
105            // Check scalar decomposition for each iteration. Since the gate enabled by
106            // `q_mul_decompose_var` queries the previous row, we enable the selector on
107            // `row + offset + 1` (instead of `row + offset`).
108            self.q_mul_decompose_var.enable(region, row + offset + 1)?;
109        }
110
111        // Use x_a, y_a output from incomplete addition
112        let mut acc = EccPoint::from_coordinates_unchecked(x_a.0, y_a.0);
113
114        // Copy running sum `z` from incomplete addition
115        let mut z = {
116            let z = z.copy_advice(
117                || "Copy `z` running sum from incomplete addition",
118                region,
119                self.z_complete,
120                offset,
121            )?;
122            Z(z)
123        };
124
125        // Store interstitial running sum `z`s in vector
126        let mut zs: Vec<Z<pallas::Base>> = Vec::with_capacity(bits.len());
127
128        // Complete addition
129        for (iter, k) in bits.iter().enumerate() {
130            // Each iteration uses 2 rows (two complete additions)
131            let row = 2 * iter;
132
133            // | x_p | y_p | x_qr  | y_qr  | z_complete |
134            // ------------------------------------------
135            // | U_x | U_y | acc_x | acc_y | z_{i + 1}  | row + offset
136            // |acc_x|acc_y|acc+U_x|acc+U_y| base_y     |
137            // |     |     | res_x | res_y | z_i        |
138
139            // Update `z`.
140            z = {
141                // z_next = z_cur * 2 + k_next
142                let z_val = z.value() * Value::known(pallas::Base::from(2))
143                    + k.map(|k| pallas::Base::from(k as u64));
144                let z_cell =
145                    region.assign_advice(|| "z", self.z_complete, row + offset + 2, || z_val)?;
146                Z(z_cell)
147            };
148            zs.push(z.clone());
149
150            // Assign `y_p` for complete addition.
151            let y_p = {
152                let base_y = base.y.copy_advice(
153                    || "Copy `base.y`",
154                    region,
155                    self.z_complete,
156                    row + offset + 1,
157                )?;
158
159                // If the bit is set, use `y`; if the bit is not set, use `-y`
160                let y_p =
161                    base_y
162                        .value()
163                        .cloned()
164                        .zip(k.as_ref())
165                        .map(|(base_y, k)| if !k { -base_y } else { base_y });
166
167                // Assign the conditionally-negated y coordinate into the cell it will be
168                // used from by both the complete addition gate, and the decomposition and
169                // conditional negation gate.
170                //
171                // The complete addition gate will copy this cell onto itself. This is
172                // fine because we are just assigning the same value to the same cell
173                // twice, and then applying an equality constraint between the cell and
174                // itself (which the permutation argument treats as a no-op).
175                region.assign_advice(|| "y_p", self.add_config.y_p, row + offset, || y_p)?
176            };
177
178            // U = P if the bit is set; U = -P is the bit is not set.
179            let U = EccPoint::from_coordinates_unchecked(base.x.clone(), y_p);
180
181            // Acc + U
182            let tmp_acc = self
183                .add_config
184                .assign_region(&U, &acc, row + offset, region)?;
185
186            // Acc + U + Acc
187            acc = self
188                .add_config
189                .assign_region(&acc, &tmp_acc, row + offset + 1, region)?;
190        }
191        Ok((acc, zs))
192    }
