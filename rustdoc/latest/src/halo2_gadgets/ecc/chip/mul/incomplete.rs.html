15/// A helper struct for implementing single-row double-and-add using incomplete addition.
16#[derive(Copy, Clone, Debug, Eq, PartialEq)]
17pub(crate) struct DoubleAndAdd {
18    // x-coordinate of the accumulator in each double-and-add iteration.
19    pub(crate) x_a: Column<Advice>,
20    // x-coordinate of the point being added in each double-and-add iteration.
21    pub(crate) x_p: Column<Advice>,
22    // lambda1 in each double-and-add iteration.
23    pub(crate) lambda_1: Column<Advice>,
24    // lambda2 in each double-and-add iteration.
25    pub(crate) lambda_2: Column<Advice>,
26}
27
28impl DoubleAndAdd {
29    /// Derives the expression `x_r = lambda_1^2 - x_a - x_p`.
30    pub(crate) fn x_r(
31        &self,
32        meta: &mut VirtualCells<pallas::Base>,
33        rotation: Rotation,
34    ) -> Expression<pallas::Base> {
35        let x_a = meta.query_advice(self.x_a, rotation);
36        let x_p = meta.query_advice(self.x_p, rotation);
37        let lambda_1 = meta.query_advice(self.lambda_1, rotation);
38        lambda_1.square() - x_a - x_p
39    }
40
41    /// Derives the expression `Y_A = (lambda_1 + lambda_2) * (x_a - x_r)`.
42    ///
43    /// Note that this is missing the factor of `1/2`; the Sinsemilla constraints factor
44    /// it out, so we leave it up to the caller to handle it.
45    #[allow(non_snake_case)]
46    pub(crate) fn Y_A(
47        &self,
48        meta: &mut VirtualCells<pallas::Base>,
49        rotation: Rotation,
50    ) -> Expression<pallas::Base> {
51        let x_a = meta.query_advice(self.x_a, rotation);
52        let lambda_1 = meta.query_advice(self.lambda_1, rotation);
53        let lambda_2 = meta.query_advice(self.lambda_2, rotation);
54        (lambda_1 + lambda_2) * (x_a - self.x_r(meta, rotation))
55    }
56}

58#[derive(Copy, Clone, Debug, Eq, PartialEq)]
59pub(crate) struct Config<const NUM_BITS: usize> {
60    // Selector constraining the first row of incomplete addition.
61    pub(super) q_mul_1: Selector,
62    // Selector constraining the main loop of incomplete addition.
63    pub(super) q_mul_2: Selector,
64    // Selector constraining the last row of incomplete addition.
65    pub(super) q_mul_3: Selector,
66    // Cumulative sum used to decompose the scalar.
67    pub(super) z: Column<Advice>,
68    // Logic specific to merged double-and-add.
69    pub(super) double_and_add: DoubleAndAdd,
70    // y-coordinate of the point being added in each double-and-add iteration.
71    pub(super) y_p: Column<Advice>,
72}

74impl<const NUM_BITS: usize> Config<NUM_BITS> {
75    pub(super) fn configure(
76        meta: &mut ConstraintSystem<pallas::Base>,
77        z: Column<Advice>,
78        x_a: Column<Advice>,
79        x_p: Column<Advice>,
80        y_p: Column<Advice>,
81        lambda_1: Column<Advice>,
82        lambda_2: Column<Advice>,
83    ) -> Self {
84        meta.enable_equality(z);
85        meta.enable_equality(lambda_1);
86
87        let config = Self {
88            q_mul_1: meta.selector(),
89            q_mul_2: meta.selector(),
90            q_mul_3: meta.selector(),
91            z,
92            double_and_add: DoubleAndAdd {
93                x_a,
94                x_p,
95                lambda_1,
96                lambda_2,
97            },
98            y_p,
99        };
100
101        config.create_gate(meta);
102
103        config
104    }

106    // Gate for incomplete addition part of variable-base scalar multiplication.
107    fn create_gate(&self, meta: &mut ConstraintSystem<pallas::Base>) {
108        // Closure to compute x_{R,i} = λ_{1,i}^2 - x_{A,i} - x_{P,i}
109        let x_r = |meta: &mut VirtualCells<pallas::Base>, rotation: Rotation| {
110            self.double_and_add.x_r(meta, rotation)
111        };
112
113        // Closure to compute y_{A,i} = (λ_{1,i} + λ_{2,i}) * (x_{A,i} - x_{R,i}) / 2
114        let y_a = |meta: &mut VirtualCells<pallas::Base>, rotation: Rotation| {
115            self.double_and_add.Y_A(meta, rotation) * pallas::Base::TWO_INV
116        };
117
118        // Constraints used for q_mul_{2, 3} == 1
119        // https://p.z.cash/halo2-0.1:ecc-var-mul-incomplete-main-loop?partial
120        // https://p.z.cash/halo2-0.1:ecc-var-mul-incomplete-last-row?partial
121        let for_loop = |meta: &mut VirtualCells<pallas::Base>,
122                        y_a_next: Expression<pallas::Base>| {
123            let one = Expression::Constant(pallas::Base::one());
124
125            // z_i
126            let z_cur = meta.query_advice(self.z, Rotation::cur());
127            // z_{i+1}
128            let z_prev = meta.query_advice(self.z, Rotation::prev());
129            // x_{A,i}
130            let x_a_cur = meta.query_advice(self.double_and_add.x_a, Rotation::cur());
131            // x_{A,i-1}
132            let x_a_next = meta.query_advice(self.double_and_add.x_a, Rotation::next());
133            // x_{P,i}
134            let x_p_cur = meta.query_advice(self.double_and_add.x_p, Rotation::cur());
135            // y_{P,i}
136            let y_p_cur = meta.query_advice(self.y_p, Rotation::cur());
137            // λ_{1,i}
138            let lambda1_cur = meta.query_advice(self.double_and_add.lambda_1, Rotation::cur());
139            // λ_{2,i}
140            let lambda2_cur = meta.query_advice(self.double_and_add.lambda_2, Rotation::cur());
141
142            let y_a_cur = y_a(meta, Rotation::cur());
143
144            // The current bit in the scalar decomposition, k_i = z_i - 2⋅z_{i+1}.
145            // Recall that we assigned the cumulative variable `z_i` in descending order,
146            // i from n down to 0. So z_{i+1} corresponds to the `z_prev` query.
147            let k = z_cur - z_prev * pallas::Base::from(2);
148            // Check booleanity of decomposition.
149            let bool_check = bool_check(k.clone());
150
151            // λ_{1,i}⋅(x_{A,i} − x_{P,i}) − y_{A,i} + (2k_i - 1) y_{P,i} = 0
152            let gradient_1 = lambda1_cur * (x_a_cur.clone() - x_p_cur) - y_a_cur.clone()
153                + (k * pallas::Base::from(2) - one) * y_p_cur;
154
155            // λ_{2,i}^2 − x_{A,i-1} − x_{R,i} − x_{A,i} = 0
156            let secant_line = lambda2_cur.clone().square()
157                - x_a_next.clone()
158                - x_r(meta, Rotation::cur())
159                - x_a_cur.clone();
160
161            // λ_{2,i}⋅(x_{A,i} − x_{A,i-1}) − y_{A,i} − y_{A,i-1} = 0
162            let gradient_2 = lambda2_cur * (x_a_cur - x_a_next) - y_a_cur - y_a_next;
163
164            std::iter::empty()
165                .chain(Some(("bool_check", bool_check)))
166                .chain(Some(("gradient_1", gradient_1)))
167                .chain(Some(("secant_line", secant_line)))
168                .chain(Some(("gradient_2", gradient_2)))
169        };
170
171        // q_mul_1 == 1 checks
172        // https://p.z.cash/halo2-0.1:ecc-var-mul-incomplete-first-row
173        meta.create_gate("q_mul_1 == 1 checks", |meta| {
174            let q_mul_1 = meta.query_selector(self.q_mul_1);
175
176            let y_a_next = y_a(meta, Rotation::next());
177            let y_a_witnessed = meta.query_advice(self.double_and_add.lambda_1, Rotation::cur());
178            Constraints::with_selector(q_mul_1, Some(("init y_a", y_a_witnessed - y_a_next)))
179        });
180
181        // q_mul_2 == 1 checks
182        // https://p.z.cash/halo2-0.1:ecc-var-mul-incomplete-main-loop?partial
183        meta.create_gate("q_mul_2 == 1 checks", |meta| {
184            let q_mul_2 = meta.query_selector(self.q_mul_2);
185
186            let y_a_next = y_a(meta, Rotation::next());
187
188            // x_{P,i}
189            let x_p_cur = meta.query_advice(self.double_and_add.x_p, Rotation::cur());
190            // x_{P,i-1}
191            let x_p_next = meta.query_advice(self.double_and_add.x_p, Rotation::next());
192            // y_{P,i}
193            let y_p_cur = meta.query_advice(self.y_p, Rotation::cur());
194            // y_{P,i-1}
195            let y_p_next = meta.query_advice(self.y_p, Rotation::next());
196
197            // The base used in double-and-add remains constant. We check that its
198            // x- and y- coordinates are the same throughout.
199            let x_p_check = x_p_cur - x_p_next;
200            let y_p_check = y_p_cur - y_p_next;
201
202            Constraints::with_selector(
203                q_mul_2,
204                std::iter::empty()
205                    .chain(Some(("x_p_check", x_p_check)))
206                    .chain(Some(("y_p_check", y_p_check)))
207                    .chain(for_loop(meta, y_a_next)),
208            )
209        });
210
211        // q_mul_3 == 1 checks
212        // https://p.z.cash/halo2-0.1:ecc-var-mul-incomplete-last-row?partial
213        meta.create_gate("q_mul_3 == 1 checks", |meta| {
214            let q_mul_3 = meta.query_selector(self.q_mul_3);
215            let y_a_final = meta.query_advice(self.double_and_add.lambda_1, Rotation::next());
216            Constraints::with_selector(q_mul_3, for_loop(meta, y_a_final))
217        });
218    }

220    /// We perform incomplete addition on all but the last three bits of the
221    /// decomposed scalar.
222    /// We split the bits in the incomplete addition range into "hi" and "lo"
223    /// halves and process them side by side, using the same rows but with
224    /// non-overlapping columns. The base is never the identity point even at
225    /// the boundary between halves.
226    /// Returns (x, y, z).
227    #[allow(clippy::type_complexity)]
228    pub(super) fn double_and_add(
229        &self,
230        region: &mut Region<'_, pallas::Base>,
231        offset: usize,
232        base: &NonIdentityEccPoint,
233        bits: &[Value<bool>],
234        acc: (X<pallas::Base>, Y<pallas::Base>, Z<pallas::Base>),
235    ) -> Result<(X<pallas::Base>, Y<pallas::Base>, Vec<Z<pallas::Base>>), Error> {
236        // Check that we have the correct number of bits for this double-and-add.
237        assert_eq!(bits.len(), NUM_BITS);
238
239        // Handle exceptional cases
240        let (x_p, y_p) = (base.x.value().cloned(), base.y.value().cloned());
241        let (x_a, y_a) = (acc.0.value().cloned(), acc.1.value().cloned());
242
243        x_a.zip(y_a)
244            .zip(x_p.zip(y_p))
245            .error_if_known_and(|((x_a, y_a), (x_p, y_p))| {
246                // A is point at infinity
247                (x_p.is_zero_vartime() && y_p.is_zero_vartime())
248                // Q is point at infinity
249                || (x_a.is_zero_vartime() && y_a.is_zero_vartime())
250                // x_p = x_a
251                || (x_p == x_a)
252            })?;
253
254        // Set q_mul values
255        {
256            // q_mul_1 = 1 on offset 0
257            self.q_mul_1.enable(region, offset)?;
258
259            let offset = offset + 1;
260            // q_mul_2 = 1 on all rows after offset 0, excluding the last row.
261            for idx in 0..(NUM_BITS - 1) {
262                self.q_mul_2.enable(region, offset + idx)?;
263            }
264
265            // q_mul_3 = 1 on the last row.
266            self.q_mul_3.enable(region, offset + NUM_BITS - 1)?;
267        }
268
269        // Initialise double-and-add
270        let (mut x_a, mut y_a, mut z) = {
271            // Initialise the running `z` sum for the scalar bits.
272            let z = acc.2.copy_advice(|| "starting z", region, self.z, offset)?;
273
274            // Initialise acc
275            let x_a = acc.0.copy_advice(
276                || "starting x_a",
277                region,
278                self.double_and_add.x_a,
279                offset + 1,
280            )?;
281            let y_a = acc.1.copy_advice(
282                || "starting y_a",
283                region,
284                self.double_and_add.lambda_1,
285                offset,
286            )?;
287
288            (x_a, y_a.value().cloned(), z)
289        };
290
291        // Increase offset by 1; we used row 0 for initializing `z`.
292        let offset = offset + 1;
293
294        // Initialise vector to store all interstitial `z` running sum values.
295        let mut zs: Vec<Z<pallas::Base>> = Vec::with_capacity(bits.len());
296
297        // Incomplete addition
298        for (row, k) in bits.iter().enumerate() {
299            // z_{i} = 2 * z_{i+1} + k_i
300            // https://p.z.cash/halo2-0.1:ecc-var-mul-witness-scalar?partial
301            let z_val = z
302                .value()
303                .zip(k.as_ref())
304                .map(|(z_val, k)| pallas::Base::from(2) * z_val + pallas::Base::from(*k as u64));
305            z = region.assign_advice(|| "z", self.z, row + offset, || z_val)?;
306            zs.push(Z(z.clone()));
307
308            // Assign `x_p`, `y_p`
309            region.assign_advice(|| "x_p", self.double_and_add.x_p, row + offset, || x_p)?;
310            region.assign_advice(|| "y_p", self.y_p, row + offset, || y_p)?;
311
312            // If the bit is set, use `y`; if the bit is not set, use `-y`
313            let y_p = y_p
314                .zip(k.as_ref())
315                .map(|(y_p, k)| if !k { -y_p } else { y_p });
316
317            // Compute and assign λ1⋅(x_A − x_P) = y_A − y_P
318            let lambda1 = y_a
319                .zip(y_p)
320                .zip(x_a.value())
321                .zip(x_p)
322                .map(|(((y_a, y_p), x_a), x_p)| (y_a - y_p) * (x_a - x_p).invert());
323            region.assign_advice(
324                || "lambda1",
325                self.double_and_add.lambda_1,
326                row + offset,
327                || lambda1,
328            )?;
329
330            // x_R = λ1^2 - x_A - x_P
331            let x_r = lambda1
332                .zip(x_a.value())
333                .zip(x_p)
334                .map(|((lambda1, x_a), x_p)| lambda1.square() - x_a - x_p);
335
336            // λ2 = (2(y_A) / (x_A - x_R)) - λ1
337            let lambda2 =
338                lambda1
339                    .zip(y_a)
340                    .zip(x_a.value())
341                    .zip(x_r)
342                    .map(|(((lambda1, y_a), x_a), x_r)| {
343                        y_a * pallas::Base::from(2) * (x_a - x_r).invert() - lambda1
344                    });
345            region.assign_advice(
346                || "lambda2",
347                self.double_and_add.lambda_2,
348                row + offset,
349                || lambda2,
350            )?;
351
352            // Compute and assign `x_a` for the next row
353            let x_a_new = lambda2.square() - x_a.value() - x_r;
354            y_a = lambda2 * (x_a.value() - x_a_new) - y_a;
355            let x_a_val = x_a_new;
356            x_a = region.assign_advice(
357                || "x_a",
358                self.double_and_add.x_a,
359                row + offset + 1,
360                || x_a_val,
361            )?;
362        }
363
364        // Witness final y_a
365        let y_a = region.assign_advice(
366            || "y_a",
367            self.double_and_add.lambda_1,
368            offset + NUM_BITS,
369            || y_a,
370        )?;
371
372        Ok((X(x_a), Y(y_a), zs))
373    }
