11/// Window size for fixed-base scalar multiplication
12pub const FIXED_BASE_WINDOW_SIZE: usize = 3;
13
14/// $2^{`FIXED_BASE_WINDOW_SIZE`}$
15pub const H: usize = 1 << FIXED_BASE_WINDOW_SIZE;
16
17/// Number of windows for a full-width scalar
18pub const NUM_WINDOWS: usize =
19    (pallas::Scalar::NUM_BITS as usize + FIXED_BASE_WINDOW_SIZE - 1) / FIXED_BASE_WINDOW_SIZE;
20
21/// Number of windows for a short signed scalar
22pub const NUM_WINDOWS_SHORT: usize =
23    (L_SCALAR_SHORT + FIXED_BASE_WINDOW_SIZE - 1) / FIXED_BASE_WINDOW_SIZE;
24
25/// $\ell_\mathsf{value}$
26/// Number of bits in an unsigned short scalar.
27pub(crate) const L_SCALAR_SHORT: usize = 64;

37/// For each fixed base, we calculate its scalar multiples in three-bit windows.
38/// Each window will have $2^3 = 8$ points. The tables are computed as described in
39/// [the Halo 2 book](https://zcash.github.io/halo2/design/gadgets/ecc/fixed-base-scalar-mul.html#load-fixed-base).
40fn compute_window_table<C: CurveAffine>(base: C, num_windows: usize) -> Vec<[C; H]> {
41    let mut window_table: Vec<[C; H]> = Vec::with_capacity(num_windows);
42
43    // Generate window table entries for all windows but the last.
44    // For these first `num_windows - 1` windows, we compute the multiple [(k+2)*(2^3)^w]B.
45    // Here, w ranges from [0..`num_windows - 1`)
46    for w in 0..(num_windows - 1) {
47        window_table.push(
48            (0..H)
49                .map(|k| {
50                    // scalar = (k+2)*(8^w)
51                    let scalar = C::Scalar::from(k as u64 + 2)
52                        * C::Scalar::from(H as u64).pow(&[w as u64, 0, 0, 0]);
53                    (base * scalar).to_affine()
54                })
55                .collect::<ArrayVec<C, H>>()
56                .into_inner()
57                .unwrap(),
58        );
59    }
60
61    // Generate window table entries for the last window, w = `num_windows - 1`.
62    // For the last window, we compute [k * (2^3)^w - sum]B, where sum is defined
63    // as sum = \sum_{j = 0}^{`num_windows - 2`} 2^{3j+1}
64    let sum = (0..(num_windows - 1)).fold(C::Scalar::ZERO, |acc, j| {
65        acc + C::Scalar::from(2).pow(&[FIXED_BASE_WINDOW_SIZE as u64 * j as u64 + 1, 0, 0, 0])
66    });
67    window_table.push(
68        (0..H)
69            .map(|k| {
70                // scalar = k * (2^3)^w - sum, where w = `num_windows - 1`
71                let scalar = C::Scalar::from(k as u64)
72                    * C::Scalar::from(H as u64).pow(&[(num_windows - 1) as u64, 0, 0, 0])
73                    - sum;
74                (base * scalar).to_affine()
75            })
76            .collect::<ArrayVec<C, H>>()
77            .into_inner()
78            .unwrap(),
79    );
80
81    window_table
82}

84/// For each window, we interpolate the $x$-coordinate.
85/// Here, we pre-compute and store the coefficients of the interpolation polynomial.
86pub fn compute_lagrange_coeffs<C: CurveAffine>(base: C, num_windows: usize) -> Vec<[C::Base; H]> {
87    // We are interpolating over the 3-bit window, k \in [0..8)
88    let points: Vec<_> = (0..H).map(|i| C::Base::from(i as u64)).collect();
89
90    let window_table = compute_window_table(base, num_windows);
91
92    window_table
93        .iter()
94        .map(|window_points| {
95            let x_window_points: Vec<_> = window_points
96                .iter()
97                .map(|point| *point.coordinates().unwrap().x())
98                .collect();
99            lagrange_interpolate(&points, &x_window_points)
100                .into_iter()
101                .collect::<ArrayVec<C::Base, H>>()
102                .into_inner()
103                .unwrap()
104        })
105        .collect()
106}

108/// For each window, $z$ is a field element such that for each point $(x, y)$ in the window:
109/// - $z + y = u^2$ (some square in the field); and
110/// - $z - y$ is not a square.
111/// If successful, return a vector of `(z: u64, us: [C::Base; H])` for each window.
112///
113/// This function was used to generate the `z`s and `u`s for the Orchard fixed
114/// bases. The outputs of this function have been stored as constants, and it
115/// is not called anywhere in this codebase. However, we keep this function here
116/// as a utility for those who wish to use it with different parameters.
117pub fn find_zs_and_us<C: CurveAffine>(
118    base: C,
119    num_windows: usize,
120) -> Option<Vec<(u64, [C::Base; H])>> {
121    // Closure to find z and u's for one window
122    let find_z_and_us = |window_points: &[C]| {
123        assert_eq!(H, window_points.len());
124
125        let ys: Vec<_> = window_points
126            .iter()
127            .map(|point| *point.coordinates().unwrap().y())
128            .collect();
129        (0..(1000 * (1 << (2 * H)))).find_map(|z| {
130            ys.iter()
131                .map(|&y| {
132                    if (-y + C::Base::from(z)).sqrt().is_none().into() {
133                        (y + C::Base::from(z)).sqrt().into()
134                    } else {
135                        None
136                    }
137                })
138                .collect::<Option<ArrayVec<C::Base, H>>>()
139                .map(|us| (z, us.into_inner().unwrap()))
140        })
141    };
142
143    let window_table = compute_window_table(base, num_windows);
144    window_table
145        .iter()
146        .map(|window_points| find_z_and_us(window_points))
147        .collect()
148}

150/// Test that the z-values and u-values satisfy the conditions:
151///      1. z + y = u^2,
152///      2. z - y is not a square
153/// for the y-coordinate of each fixed-base multiple in each window.
154#[cfg(any(test, feature = "test-dependencies"))]
155#[cfg_attr(docsrs, doc(cfg(feature = "test-dependencies")))]
156pub fn test_zs_and_us<C: CurveAffine>(base: C, z: &[u64], u: &[[[u8; 32]; H]], num_windows: usize) {
157    let window_table = compute_window_table(base, num_windows);
158
159    for ((u, z), window_points) in u.iter().zip(z.iter()).zip(window_table) {
160        for (u, point) in u.iter().zip(window_points.iter()) {
161            let y = *point.coordinates().unwrap().y();
162            let mut u_repr = <C::Base as PrimeField>::Repr::default();
163            u_repr.as_mut().copy_from_slice(u);
164            let u = C::Base::from_repr(u_repr).unwrap();
165            assert_eq!(C::Base::from(*z) + y, u * u); // allow either square root
166            assert!(bool::from((C::Base::from(*z) - y).sqrt().is_none()));
167        }
168    }
169}

171/// Test that Lagrange interpolation coefficients reproduce the correct x-coordinate
172/// for each fixed-base multiple in each window.
173#[cfg(any(test, feature = "test-dependencies"))]
174#[cfg_attr(docsrs, doc(cfg(feature = "test-dependencies")))]
175pub fn test_lagrange_coeffs<C: CurveAffine>(base: C, num_windows: usize) {
176    /// Evaluate y = f(x) given the coefficients of f(x)
177    fn evaluate<C: CurveAffine>(x: u8, coeffs: &[C::Base]) -> C::Base {
178        let x = C::Base::from(x as u64);
179        coeffs
180            .iter()
181            .rev()
182            .cloned()
183            .reduce(|acc, coeff| acc * x + coeff)
184            .unwrap_or(C::Base::ZERO)
185    }
186
187    let lagrange_coeffs = compute_lagrange_coeffs(base, num_windows);
188
189    // Check first 84 windows, i.e. `k_0, k_1, ..., k_83`
190    for (idx, coeffs) in lagrange_coeffs[0..(num_windows - 1)].iter().enumerate() {
191        // Test each three-bit chunk in this window.
192        for bits in 0..(H as u8) {
193            {
194                // Interpolate the x-coordinate using this window's coefficients
195                let interpolated_x = evaluate::<C>(bits, coeffs);
196
197                // Compute the actual x-coordinate of the multiple [(k+2)*(8^w)]B.
198                let point = base
199                    * C::Scalar::from(bits as u64 + 2)
200                    * C::Scalar::from(H as u64).pow(&[idx as u64, 0, 0, 0]);
201                let x = *point.to_affine().coordinates().unwrap().x();
202
203                // Check that the interpolated x-coordinate matches the actual one.
204                assert_eq!(x, interpolated_x);
205            }
206        }
207    }
208
209    // Check last window.
210    for bits in 0..(H as u8) {
211        // Interpolate the x-coordinate using the last window's coefficients
212        let interpolated_x = evaluate::<C>(bits, &lagrange_coeffs[num_windows - 1]);
213
214        // Compute the actual x-coordinate of the multiple [k * (8^84) - offset]B,
215        // where offset = \sum_{j = 0}^{83} 2^{3j+1}
216        let offset = (0..(num_windows - 1)).fold(C::Scalar::ZERO, |acc, w| {
217            acc + C::Scalar::from(2).pow(&[FIXED_BASE_WINDOW_SIZE as u64 * w as u64 + 1, 0, 0, 0])
218        });
219        let scalar = C::Scalar::from(bits as u64)
220            * C::Scalar::from(H as u64).pow(&[(num_windows - 1) as u64, 0, 0, 0])
221            - offset;
222        let point = base * scalar;
223        let x = *point.to_affine().coordinates().unwrap().x();
224
225        // Check that the interpolated x-coordinate matches the actual one.
226        assert_eq!(x, interpolated_x);
227    }
228}

240        /// Generate an arbitrary Pallas point.
241        pub fn arb_point()(bytes in prop::array::uniform32(0u8..)) -> pallas::Point {
242            // Instead of rejecting out-of-range bytes, let's reduce them.
243            let mut buf = [0; 64];
244            buf[..32].copy_from_slice(&bytes);
245            let scalar = pallas::Scalar::from_uniform_bytes(&buf);
246            pallas::Point::generator() * scalar
247        }
248    }
