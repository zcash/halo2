32    pub(crate) fn configure(
33        meta: &mut ConstraintSystem<pallas::Base>,
34        canon_advices: [Column<Advice>; 3],
35        lookup_config: Lookup,
36        super_config: super::Config<Fixed>,
37    ) -> Self {
38        for advice in canon_advices.iter() {
39            meta.enable_equality(*advice);
40        }
41
42        let config = Self {
43            q_mul_fixed_base_field: meta.selector(),
44            canon_advices,
45            lookup_config,
46            super_config,
47        };
48
49        let add_incomplete_advices = config.super_config.add_incomplete_config.advice_columns();
50        for canon_advice in config.canon_advices.iter() {
51            assert!(
52                !add_incomplete_advices.contains(canon_advice),
53                "Deconflict canon_advice columns with incomplete addition columns."
54            );
55        }
56
57        config.create_gate(meta);
58
59        config
60    }

62    fn create_gate(&self, meta: &mut ConstraintSystem<pallas::Base>) {
63        // Check that the base field element is canonical.
64        // https://p.z.cash/halo2-0.1:ecc-fixed-mul-base-canonicity
65        meta.create_gate("Canonicity checks", |meta| {
66            let q_mul_fixed_base_field = meta.query_selector(self.q_mul_fixed_base_field);
67
68            let alpha = meta.query_advice(self.canon_advices[0], Rotation::prev());
69            // The last three bits of α.
70            let z_84_alpha = meta.query_advice(self.canon_advices[2], Rotation::prev());
71
72            // Decompose α into three pieces, in little-endian order:
73            //            α = α_0 (252 bits)  || α_1 (2 bits) || α_2 (1 bit).
74            //
75            // α_0 is derived, not witnessed.
76            let alpha_0 = {
77                let two_pow_252 = pallas::Base::from_u128(1 << 126).square();
78                alpha - (z_84_alpha.clone() * two_pow_252)
79            };
80            let alpha_1 = meta.query_advice(self.canon_advices[1], Rotation::cur());
81            let alpha_2 = meta.query_advice(self.canon_advices[2], Rotation::cur());
82
83            let alpha_0_prime = meta.query_advice(self.canon_advices[0], Rotation::cur());
84            let z_13_alpha_0_prime = meta.query_advice(self.canon_advices[0], Rotation::next());
85            let z_44_alpha = meta.query_advice(self.canon_advices[1], Rotation::next());
86            let z_43_alpha = meta.query_advice(self.canon_advices[2], Rotation::next());
87
88            let decomposition_checks = {
89                // Range-constrain α_1 to be 2 bits
90                let alpha_1_range_check = range_check(alpha_1.clone(), 1 << 2);
91                // Boolean-constrain α_2
92                let alpha_2_range_check = bool_check(alpha_2.clone());
93                // Check that α_1 + 2^2 α_2 = z_84_alpha
94                let z_84_alpha_check = z_84_alpha.clone()
95                    - (alpha_1.clone() + alpha_2.clone() * pallas::Base::from(1 << 2));
96
97                std::iter::empty()
98                    .chain(Some(("alpha_1_range_check", alpha_1_range_check)))
99                    .chain(Some(("alpha_2_range_check", alpha_2_range_check)))
100                    .chain(Some(("z_84_alpha_check", z_84_alpha_check)))
101            };
102
103            // Check α_0_prime = α_0 + 2^130 - t_p
104            let alpha_0_prime_check = {
105                let two_pow_130 = Expression::Constant(pallas::Base::from_u128(1 << 65).square());
106                let t_p = Expression::Constant(pallas::Base::from_u128(T_P));
107                alpha_0_prime - (alpha_0 + two_pow_130 - t_p)
108            };
109
110            // We want to enforce canonicity of a 255-bit base field element, α.
111            // That is, we want to check that 0 ≤ α < p, where p is Pallas base
112            // field modulus p = 2^254 + t_p
113            //                 = 2^254 + 45560315531419706090280762371685220353.
114            // Note that t_p < 2^130.
115            //
116            // α has been decomposed into three pieces in little-endian order:
117            //            α = α_0 (252 bits)  || α_1 (2 bits) || α_2 (1 bit).
118            //              = α_0 + 2^252 α_1 + 2^254 α_2.
119            //
120            // If the MSB α_2 = 1, then:
121            //      - α_2 = 1 => α_1 = 0, and
122            //      - α_2 = 1 => α_0 < t_p. To enforce this:
123            //          - α_2 = 1 => 0 ≤ α_0 < 2^130
124            //                - alpha_0_hi_120 = 0 (constrain α_0 to be 132 bits)
125            //                - a_43 = 0 or 1 (constrain α_0[130..=131] to be 0)
126            //          - α_2 = 1 => 0 ≤ α_0 + 2^130 - t_p < 2^130
127            //                    => 13 ten-bit lookups of α_0 + 2^130 - t_p
128            //                    => z_13_alpha_0_prime = 0
129            //
130            let canon_checks = {
131                // alpha_0_hi_120 = z_44 - 2^120 z_84
132                let alpha_0_hi_120 = {
133                    let two_pow_120 =
134                        Expression::Constant(pallas::Base::from_u128(1 << 60).square());
135                    z_44_alpha.clone() - z_84_alpha * two_pow_120
136                };
137                // a_43 = z_43 - (2^3)z_44
138                let a_43 = z_43_alpha - z_44_alpha * *H_BASE;
139
140                std::iter::empty()
141                    .chain(Some(("MSB = 1 => alpha_1 = 0", alpha_2.clone() * alpha_1)))
142                    .chain(Some((
143                        "MSB = 1 => alpha_0_hi_120 = 0",
144                        alpha_2.clone() * alpha_0_hi_120,
145                    )))
146                    .chain(Some((
147                        "MSB = 1 => a_43 = 0 or 1",
148                        alpha_2.clone() * bool_check(a_43),
149                    )))
150                    .chain(Some((
151                        "MSB = 1 => z_13_alpha_0_prime = 0",
152                        alpha_2 * z_13_alpha_0_prime,
153                    )))
154            };
155
156            Constraints::with_selector(
157                q_mul_fixed_base_field,
158                canon_checks
159                    .chain(decomposition_checks)
160                    .chain(Some(("alpha_0_prime check", alpha_0_prime_check))),
161            )
162        });
163    }

165    pub fn assign(
166        &self,
167        mut layouter: impl Layouter<pallas::Base>,
168        scalar: AssignedCell<pallas::Base, pallas::Base>,
169        base: &<Fixed as FixedPoints<pallas::Affine>>::Base,
170    ) -> Result<EccPoint, Error>
171    where
172        <Fixed as FixedPoints<pallas::Affine>>::Base: super::super::FixedPoint<pallas::Affine>,
173    {
174        let (scalar, acc, mul_b) = layouter.assign_region(
175            || "Base-field elem fixed-base mul (incomplete addition)",
176            |mut region| {
177                let offset = 0;
178
179                // Decompose scalar
180                let scalar = {
181                    let running_sum = self.super_config.running_sum_config.copy_decompose(
182                        &mut region,
183                        offset,
184                        scalar.clone(),
185                        true,
186                        pallas::Base::NUM_BITS as usize,
187                        NUM_WINDOWS,
188                    )?;
189                    EccBaseFieldElemFixed {
190                        base_field_elem: running_sum[0].clone(),
191                        running_sum: (*running_sum).as_slice().try_into().unwrap(),
192                    }
193                };
194
195                let (acc, mul_b) = self.super_config.assign_region_inner::<_, NUM_WINDOWS>(
196                    &mut region,
197                    offset,
198                    &(&scalar).into(),
199                    base,
200                    self.super_config.running_sum_config.q_range_check(),
201                )?;
202
203                Ok((scalar, acc, mul_b))
204            },
205        )?;
206
207        // Add to the accumulator and return the final result as `[scalar]B`.
208        let result = layouter.assign_region(
209            || "Base-field elem fixed-base mul (complete addition)",
210            |mut region| {
211                self.super_config.add_config.assign_region(
212                    &mul_b.clone().into(),
213                    &acc.clone().into(),
214                    0,
215                    &mut region,
216                )
217            },
218        )?;
219
220        #[cfg(test)]
221        // Check that the correct multiple is obtained.
222        {
223            use super::super::FixedPoint;
224            use group::Curve;
225
226            let scalar = &scalar
227                .base_field_elem()
228                .value()
229                .map(|scalar| pallas::Scalar::from_repr(scalar.to_repr()).unwrap());
230            let real_mul = scalar.map(|scalar| base.generator() * scalar);
231            let result = result.point();
232
233            real_mul
234                .zip(result)
235                .assert_if_known(|(real_mul, result)| &real_mul.to_affine() == result);
236        }
237
238        // We want to enforce canonicity of a 255-bit base field element, α.
239        // That is, we want to check that 0 ≤ α < p, where p is Pallas base
240        // field modulus p = 2^254 + t_p
241        //                 = 2^254 + 45560315531419706090280762371685220353.
242        // Note that t_p < 2^130.
243        //
244        // α has been decomposed into three pieces in little-endian order:
245        //            α = α_0 (252 bits)  || α_1 (2 bits) || α_2 (1 bit).
246        //              = α_0 + 2^252 α_1 + 2^254 α_2.
247        //
248        // If the MSB α_2 = 1, then:
249        //      - α_2 = 1 => α_1 = 0, and
250        //      - α_2 = 1 => α_0 < t_p. To enforce this:
251        //      - α_2 = 1 => 0 ≤ α_0 < 2^130
252        //                => 13 ten-bit lookups of α_0
253        //      - α_2 = 1 => 0 ≤ α_0 + 2^130 - t_p < 2^130
254        //                => 13 ten-bit lookups of α_0 + 2^130 - t_p
255        //                => z_13_alpha_0_prime = 0
256        //
257        let (alpha, running_sum) = (scalar.base_field_elem, &scalar.running_sum);
258        let z_43_alpha = running_sum[43].clone();
259        let z_44_alpha = running_sum[44].clone();
260        let z_84_alpha = running_sum[84].clone();
261
262        // α_0 = α - z_84_alpha * 2^252
263        let alpha_0 = alpha
264            .value()
265            .zip(z_84_alpha.value())
266            .map(|(alpha, z_84_alpha)| {
267                let two_pow_252 = pallas::Base::from_u128(1 << 126).square();
268                alpha - z_84_alpha * two_pow_252
269            });
270
271        let (alpha_0_prime, z_13_alpha_0_prime) = {
272            // alpha_0_prime = alpha + 2^130 - t_p.
273            let alpha_0_prime = alpha_0.map(|alpha_0| {
274                let two_pow_130 = pallas::Base::from_u128(1 << 65).square();
275                let t_p = pallas::Base::from_u128(T_P);
276                alpha_0 + two_pow_130 - t_p
277            });
278            let zs = self.lookup_config.witness_check(
279                layouter.namespace(|| "Lookup range check alpha_0 + 2^130 - t_p"),
280                alpha_0_prime,
281                13,
282                false,
283            )?;
284            let alpha_0_prime = zs[0].clone();
285
286            (alpha_0_prime, zs[13].clone())
287        };
288
289        layouter.assign_region(
290            || "Canonicity checks",
291            |mut region| {
292                // Activate canonicity check gate
293                self.q_mul_fixed_base_field.enable(&mut region, 1)?;
294
295                // Offset 0
296                {
297                    let offset = 0;
298
299                    // Copy α
300                    alpha.copy_advice(|| "Copy α", &mut region, self.canon_advices[0], offset)?;
301
302                    // z_84_alpha = the top three bits of alpha.
303                    z_84_alpha.copy_advice(
304                        || "Copy z_84_alpha",
305                        &mut region,
306                        self.canon_advices[2],
307                        offset,
308                    )?;
309                }
310
311                // Offset 1
312                {
313                    let offset = 1;
314                    // Copy alpha_0_prime = alpha_0 + 2^130 - t_p.
315                    // We constrain this in the custom gate to be derived correctly.
316                    alpha_0_prime.copy_advice(
317                        || "Copy α_0 + 2^130 - t_p",
318                        &mut region,
319                        self.canon_advices[0],
320                        offset,
321                    )?;
322
323                    // Decompose α into three pieces,
324                    //     α = α_0 (252 bits)  || α_1 (2 bits) || α_2 (1 bit).
325                    // We only need to witness α_1 and α_2. α_0 is derived in the gate.
326                    // Witness α_1 = α[252..=253]
327                    let alpha_1 = alpha.value().map(|alpha| bitrange_subset(alpha, 252..254));
328                    region.assign_advice(
329                        || "α_1 = α[252..=253]",
330                        self.canon_advices[1],
331                        offset,
332                        || alpha_1,
333                    )?;
334
335                    // Witness the MSB α_2 = α[254]
336                    let alpha_2 = alpha.value().map(|alpha| bitrange_subset(alpha, 254..255));
337                    region.assign_advice(
338                        || "α_2 = α[254]",
339                        self.canon_advices[2],
340                        offset,
341                        || alpha_2,
342                    )?;
343                }
344
345                // Offset 2
346                {
347                    let offset = 2;
348                    // Copy z_13_alpha_0_prime
349                    z_13_alpha_0_prime.copy_advice(
350                        || "Copy z_13_alpha_0_prime",
351                        &mut region,
352                        self.canon_advices[0],
353                        offset,
354                    )?;
355
356                    // Copy z_44_alpha
357                    z_44_alpha.copy_advice(
358                        || "Copy z_44_alpha",
359                        &mut region,
360                        self.canon_advices[1],
361                        offset,
362                    )?;
363
364                    // Copy z_43_alpha
365                    z_43_alpha.copy_advice(
366                        || "Copy z_43_alpha",
367                        &mut region,
368                        self.canon_advices[2],
369                        offset,
370                    )?;
371                }
372
373                Ok(())
374            },
375        )?;
376
377        Ok(result)
378    }

415    #[allow(clippy::op_ref)]
416    fn test_single_base<Lookup: PallasLookupRangeCheck>(
417        chip: EccChip<TestFixedBases, Lookup>,
418        mut layouter: impl Layouter<pallas::Base>,
419        base: FixedPointBaseField<pallas::Affine, EccChip<TestFixedBases, Lookup>>,
420        base_val: pallas::Affine,
421    ) -> Result<(), Error> {
422        let rng = OsRng;
423
424        let column = chip.config().advices[0];
425
426        fn constrain_equal_non_id<Lookup: PallasLookupRangeCheck>(
427            chip: EccChip<TestFixedBases, Lookup>,
428            mut layouter: impl Layouter<pallas::Base>,
429            base_val: pallas::Affine,
430            scalar_val: pallas::Base,
431            result: Point<pallas::Affine, EccChip<TestFixedBases, Lookup>>,
432        ) -> Result<(), Error> {
433            // Move scalar from base field into scalar field (which always fits for Pallas).
434            let scalar = pallas::Scalar::from_repr(scalar_val.to_repr()).unwrap();
435            let expected = NonIdentityPoint::new(
436                chip,
437                layouter.namespace(|| "expected point"),
438                Value::known((base_val * scalar).to_affine()),
439            )?;
440            result.constrain_equal(layouter.namespace(|| "constrain result"), &expected)
441        }
442
443        // [a]B
444        {
445            let scalar_fixed = pallas::Base::random(rng);
446            let result = {
447                let scalar_fixed = chip.load_private(
448                    layouter.namespace(|| "random base field element"),
449                    column,
450                    Value::known(scalar_fixed),
451                )?;
452                base.mul(layouter.namespace(|| "random [a]B"), scalar_fixed)?
453            };
454            constrain_equal_non_id(
455                chip.clone(),
456                layouter.namespace(|| "random [a]B"),
457                base_val,
458                scalar_fixed,
459                result,
460            )?;
461        }
462
463        // There is a single canonical sequence of window values for which a doubling occurs on the last step:
464        // 1333333333333333333333333333333333333333333333333333333333333333333333333333333333334 in octal.
465        // (There is another *non-canonical* sequence
466        // 5333333333333333333333333333333333333333332711161673731021062440252244051273333333333 in octal.)
467        {
468            let h = pallas::Base::from(H as u64);
469            let scalar_fixed = "1333333333333333333333333333333333333333333333333333333333333333333333333333333333334"
470                        .chars()
471                        .fold(pallas::Base::zero(), |acc, c| {
472                            acc * &h + &pallas::Base::from(c.to_digit(8).unwrap() as u64)
473                        });
474            let result = {
475                let scalar_fixed = chip.load_private(
476                    layouter.namespace(|| "mul with double"),
477                    column,
478                    Value::known(scalar_fixed),
479                )?;
480                base.mul(layouter.namespace(|| "mul with double"), scalar_fixed)?
481            };
482            constrain_equal_non_id(
483                chip.clone(),
484                layouter.namespace(|| "mul with double"),
485                base_val,
486                scalar_fixed,
487                result,
488            )?;
489        }
490
491        // [0]B should return (0,0) since it uses complete addition
492        // on the last step.
493        {
494            let scalar_fixed = pallas::Base::zero();
495            let result = {
496                let scalar_fixed = chip.load_private(
497                    layouter.namespace(|| "zero"),
498                    column,
499                    Value::known(scalar_fixed),
500                )?;
501                base.mul(layouter.namespace(|| "mul by zero"), scalar_fixed)?
502            };
503            result
504                .inner()
505                .is_identity()
506                .assert_if_known(|is_identity| *is_identity);
507        }
508
509        // [-1]B is the largest base field element
510        {
511            let scalar_fixed = -pallas::Base::one();
512            let result = {
513                let scalar_fixed = chip.load_private(
514                    layouter.namespace(|| "-1"),
515                    column,
516                    Value::known(scalar_fixed),
517                )?;
518                base.mul(layouter.namespace(|| "mul by -1"), scalar_fixed)?
519            };
520            constrain_equal_non_id(
521                chip,
522                layouter.namespace(|| "mul by -1"),
523                base_val,
524                scalar_fixed,
525                result,
526            )?;
527        }
528
529        Ok(())
530    }
