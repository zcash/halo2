35    fn create_gate(&self, meta: &mut ConstraintSystem<pallas::Base>) {
36        // Gate contains the following constraints:
37        // - https://p.z.cash/halo2-0.1:ecc-fixed-mul-short-msb
38        // - https://p.z.cash/halo2-0.1:ecc-fixed-mul-short-conditional-neg
39        meta.create_gate("Short fixed-base mul gate", |meta| {
40            let q_mul_fixed_short = meta.query_selector(self.q_mul_fixed_short);
41            let y_p = meta.query_advice(self.super_config.add_config.y_p, Rotation::cur());
42            let y_a = meta.query_advice(self.super_config.add_config.y_qr, Rotation::cur());
43            // z_21 = k_21
44            let last_window = meta.query_advice(self.super_config.u, Rotation::cur());
45            let sign = meta.query_advice(self.super_config.window, Rotation::cur());
46
47            let one = Expression::Constant(pallas::Base::one());
48
49            // Check that last window is either 0 or 1.
50            let last_window_check = bool_check(last_window);
51            // Check that sign is either 1 or -1.
52            let sign_check = sign.clone().square() - one;
53
54            // `(x_a, y_a)` is the result of `[m]B`, where `m` is the magnitude.
55            // We conditionally negate this result using `y_p = y_a * s`, where `s` is the sign.
56
57            // Check that the final `y_p = y_a` or `y_p = -y_a`
58            //
59            // This constraint is redundant / unnecessary, because `sign` is constrained
60            // to -1 or 1 by `sign_check`, and `negation_check` therefore permits a strict
61            // subset of the cases that this constraint permits.
62            let y_check = (y_p.clone() - y_a.clone()) * (y_p.clone() + y_a.clone());
63
64            // Check that the correct sign is witnessed s.t. sign * y_p = y_a
65            let negation_check = sign * y_p - y_a;
66
67            Constraints::with_selector(
68                q_mul_fixed_short,
69                [
70                    ("last_window_check", last_window_check),
71                    ("sign_check", sign_check),
72                    ("y_check", y_check),
73                    ("negation_check", negation_check),
74                ],
75            )
76        });
77    }

79    /// Constraints `magnitude` to be at most 66 bits.
80    ///
81    /// The final window is separately constrained to be a single bit, which completes the
82    /// 64-bit range constraint.
83    fn decompose(
84        &self,
85        region: &mut Region<'_, pallas::Base>,
86        offset: usize,
87        magnitude_sign: MagnitudeSign,
88    ) -> Result<EccScalarFixedShort, Error> {
89        let (magnitude, sign) = magnitude_sign;
90
91        // Decompose magnitude
92        let running_sum = self.super_config.running_sum_config.copy_decompose(
93            region,
94            offset,
95            magnitude.clone(),
96            true,
97            L_SCALAR_SHORT,
98            NUM_WINDOWS_SHORT,
99        )?;
100
101        Ok(EccScalarFixedShort {
102            magnitude,
103            sign,
104            running_sum: Some((*running_sum).as_slice().try_into().unwrap()),
105        })
106    }

108    pub fn assign(
109        &self,
110        mut layouter: impl Layouter<pallas::Base>,
111        scalar: &EccScalarFixedShort,
112        base: &<Fixed as FixedPoints<pallas::Affine>>::ShortScalar,
113    ) -> Result<(EccPoint, EccScalarFixedShort), Error>
114    where
115        <Fixed as FixedPoints<pallas::Affine>>::ShortScalar:
116            super::super::FixedPoint<pallas::Affine>,
117    {
118        let (scalar, acc, mul_b) = layouter.assign_region(
119            || "Short fixed-base mul (incomplete addition)",
120            |mut region| {
121                let offset = 0;
122
123                // Decompose the scalar
124                let scalar = match scalar.running_sum {
125                    None => self.decompose(
126                        &mut region,
127                        offset,
128                        (scalar.magnitude.clone(), scalar.sign.clone()),
129                    ),
130                    Some(_) => todo!("unimplemented for halo2_gadgets v0.1.0"),
131                }?;
132
133                let (acc, mul_b) = self
134                    .super_config
135                    .assign_region_inner::<_, NUM_WINDOWS_SHORT>(
136                        &mut region,
137                        offset,
138                        &(&scalar).into(),
139                        base,
140                        self.super_config.running_sum_config.q_range_check(),
141                    )?;
142
143                Ok((scalar, acc, mul_b))
144            },
145        )?;
146
147        // Last window
148        let result = layouter.assign_region(
149            || "Short fixed-base mul (most significant word)",
150            |mut region| {
151                let offset = 0;
152                // Add to the cumulative sum to get `[magnitude]B`.
153                let magnitude_mul = self.super_config.add_config.assign_region(
154                    &mul_b.clone().into(),
155                    &acc.clone().into(),
156                    offset,
157                    &mut region,
158                )?;
159
160                // Increase offset by 1 after complete addition
161                let offset = offset + 1;
162
163                // Copy sign to `window` column
164                let sign = scalar.sign.copy_advice(
165                    || "sign",
166                    &mut region,
167                    self.super_config.window,
168                    offset,
169                )?;
170
171                // Copy last window to `u` column.
172                // (Although the last window is not a `u` value; we are copying it into the `u`
173                // column because there is an available cell there.)
174                let z_21 = scalar.running_sum.as_ref().unwrap()[21].clone();
175                z_21.copy_advice(|| "last_window", &mut region, self.super_config.u, offset)?;
176
177                // Conditionally negate `y`-coordinate
178                let y_val = sign.value().and_then(|sign| {
179                    if sign == &-pallas::Base::one() {
180                        -magnitude_mul.y.value()
181                    } else {
182                        magnitude_mul.y.value().cloned()
183                    }
184                });
185
186                // Enable mul_fixed_short selector on final row
187                self.q_mul_fixed_short.enable(&mut region, offset)?;
188
189                // Assign final `y` to `y_p` column and return final point
190                let y_var = region.assign_advice(
191                    || "y_var",
192                    self.super_config.add_config.y_p,
193                    offset,
194                    || y_val,
195                )?;
196
197                Ok(EccPoint::from_coordinates_unchecked(magnitude_mul.x, y_var))
198            },
199        )?;
200
201        #[cfg(test)]
202        // Check that the correct multiple is obtained.
203        // This inlined test is only done for valid 64-bit magnitudes
204        // and valid +/- 1 signs.
205        // Invalid values result in constraint failures which are
206        // tested at the circuit-level.
207        {
208            use super::super::FixedPoint;
209            use group::{ff::PrimeField, Curve};
210
211            scalar
212                .magnitude
213                .value()
214                .zip(scalar.sign.value())
215                .zip(result.point())
216                .assert_if_known(|((magnitude, sign), result)| {
217                    let magnitude_is_valid =
218                        magnitude <= &&pallas::Base::from(0xFFFF_FFFF_FFFF_FFFFu64);
219                    let sign_is_valid = sign.square() == pallas::Base::one();
220                    // Only check the result if the magnitude and sign are valid.
221                    !(magnitude_is_valid && sign_is_valid) || {
222                        let scalar = {
223                            // Move magnitude from base field into scalar field (which always fits
224                            // for Pallas).
225                            let magnitude = pallas::Scalar::from_repr(magnitude.to_repr()).unwrap();
226
227                            let sign = if sign == &&pallas::Base::one() {
228                                pallas::Scalar::one()
229                            } else {
230                                -pallas::Scalar::one()
231                            };
232
233                            magnitude * sign
234                        };
235                        let real_mul = base.generator() * scalar;
236
237                        &real_mul.to_affine() == result
238                    }
239                });
240        }
241
242        Ok((result, scalar))
243    }

245    /// Multiply the point by sign, using the q_mul_fixed_short gate.
246    /// This constrains `sign` to be in {-1, 1}.
247    pub(crate) fn assign_scalar_sign(
248        &self,
249        mut layouter: impl Layouter<pallas::Base>,
250        sign: &AssignedCell<pallas::Base, pallas::Base>,
251        point: &EccPoint,
252    ) -> Result<EccPoint, Error> {
253        let signed_point = layouter.assign_region(
254            || "Signed point",
255            |mut region| {
256                let offset = 0;
257
258                // Enable mul_fixed_short selector to check the sign logic.
259                self.q_mul_fixed_short.enable(&mut region, offset)?;
260
261                // Set "last window" to 0 (this field is irrelevant here).
262                region.assign_advice_from_constant(
263                    || "u=0",
264                    self.super_config.u,
265                    offset,
266                    pallas::Base::zero(),
267                )?;
268
269                // Copy sign to `window` column
270                sign.copy_advice(|| "sign", &mut region, self.super_config.window, offset)?;
271
272                // Assign the input y-coordinate.
273                point.y.copy_advice(
274                    || "unsigned y",
275                    &mut region,
276                    self.super_config.add_config.y_qr,
277                    offset,
278                )?;
279
280                // Conditionally negate y-coordinate according to the value of sign
281                let signed_y_val = sign.value().and_then(|sign| {
282                    if sign == &-pallas::Base::one() {
283                        -point.y.value()
284                    } else {
285                        point.y.value().cloned()
286                    }
287                });
288
289                // Assign the output signed y-coordinate.
290                let signed_y = region.assign_advice(
291                    || "signed y",
292                    self.super_config.add_config.y_p,
293                    offset,
294                    || signed_y_val,
295                )?;
296
297                Ok(EccPoint {
298                    x: point.x.clone(),
299                    y: signed_y,
300                })
301            },
302        )?;
303
304        Ok(signed_point)
305    }

569    fn format_value(v: pallas::Base) -> String {
570        use ff::Field;
571        if v.is_zero_vartime() {
572            "0".into()
573        } else if v == pallas::Base::one() {
574            "1".into()
575        } else if v == -pallas::Base::one() {
576            "-1".into()
577        } else {
578            // Format value as hex.
579            let s = format!("{:?}", v);
580            // Remove leading zeroes.
581            let s = s.strip_prefix("0x").unwrap();
582            let s = s.trim_start_matches('0');
583            format!("0x{}", s)
584        }
585    }

755    pub(crate) fn test_mul_sign<Lookup: PallasLookupRangeCheck>(
756        chip: EccChip<TestFixedBases, Lookup>,
757        mut layouter: impl Layouter<pallas::Base>,
758    ) -> Result<(), Error> {
759        // Generate a random non-identity point P
760        let p_val = pallas::Point::random(rand::rngs::OsRng).to_affine();
761        let p = Point::new(
762            chip.clone(),
763            layouter.namespace(|| "P"),
764            Value::known(p_val),
765        )?;
766
767        // Create -P
768        let p_neg_val = -p_val;
769        let p_neg = Point::new(
770            chip.clone(),
771            layouter.namespace(|| "-P"),
772            Value::known(p_neg_val),
773        )?;
774
775        // Create the identity point
776        let identity = Point::new(
777            chip.clone(),
778            layouter.namespace(|| "identity"),
779            Value::known(pallas::Point::identity().to_affine()),
780        )?;
781
782        // Create -1 and 1 scalars
783        let pos_sign = chip.load_private(
784            layouter.namespace(|| "positive sign"),
785            chip.config().advices[0],
786            Value::known(pallas::Base::one()),
787        )?;
788        let neg_sign = chip.load_private(
789            layouter.namespace(|| "negative sign"),
790            chip.config().advices[1],
791            Value::known(-pallas::Base::one()),
792        )?;
793
794        // [1] P == P
795        {
796            let result = p.mul_sign(layouter.namespace(|| "[1] P"), &pos_sign)?;
797            result.constrain_equal(layouter.namespace(|| "constrain [1] P"), &p)?;
798        }
799
800        // [-1] P == -P
801        {
802            let result = p.mul_sign(layouter.namespace(|| "[1] P"), &neg_sign)?;
803            result.constrain_equal(layouter.namespace(|| "constrain [1] P"), &p_neg)?;
804        }
805
806        // [1] 0 == 0
807        {
808            let result = identity.mul_sign(layouter.namespace(|| "[1] O"), &pos_sign)?;
809            result.constrain_equal(layouter.namespace(|| "constrain [1] 0"), &identity)?;
810        }
811
812        // [-1] 0 == 0
813        {
814            let result = identity.mul_sign(layouter.namespace(|| "[-1] O"), &neg_sign)?;
815            result.constrain_equal(layouter.namespace(|| "constrain [1] 0"), &identity)?;
816        }
817
818        Ok(())
819    }
