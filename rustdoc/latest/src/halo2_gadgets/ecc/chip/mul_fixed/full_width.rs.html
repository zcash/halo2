20    pub(crate) fn configure(
21        meta: &mut ConstraintSystem<pallas::Base>,
22        super_config: super::Config<Fixed>,
23    ) -> Self {
24        let config = Self {
25            q_mul_fixed_full: meta.selector(),
26            super_config,
27        };
28
29        config.create_gate(meta);
30
31        config
32    }

34    fn create_gate(&self, meta: &mut ConstraintSystem<pallas::Base>) {
35        // Check that each window `k` is within 3 bits
36        // https://p.z.cash/halo2-0.1:ecc-fixed-mul-full-word
37        meta.create_gate("Full-width fixed-base scalar mul", |meta| {
38            let q_mul_fixed_full = meta.query_selector(self.q_mul_fixed_full);
39            let window = meta.query_advice(self.super_config.window, Rotation::cur());
40
41            Constraints::with_selector(
42                q_mul_fixed_full,
43                self.super_config
44                    .coords_check(meta, window.clone())
45                    .into_iter()
46                    // Constrain each window to a 3-bit value:
47                    // window * (1 - window) * ... * (7 - window)
48                    .chain(Some(("window range check", range_check(window, H)))),
49            )
50        });
51    }

53    /// Witnesses the given scalar as `NUM_WINDOWS` 3-bit windows.
54    ///
55    /// The scalar is allowed to be non-canonical.
56    fn witness(
57        &self,
58        region: &mut Region<'_, pallas::Base>,
59        offset: usize,
60        scalar: Value<pallas::Scalar>,
61    ) -> Result<EccScalarFixed, Error> {
62        let windows = self.decompose_scalar_fixed::<{ pallas::Scalar::NUM_BITS as usize }>(
63            scalar, offset, region,
64        )?;
65
66        Ok(EccScalarFixed {
67            value: scalar,
68            windows: Some(windows),
69        })
70    }

72    /// Witnesses the given scalar as `NUM_WINDOWS` 3-bit windows.
73    ///
74    /// The scalar is allowed to be non-canonical.
75    fn decompose_scalar_fixed<const SCALAR_NUM_BITS: usize>(
76        &self,
77        scalar: Value<pallas::Scalar>,
78        offset: usize,
79        region: &mut Region<'_, pallas::Base>,
80    ) -> Result<ArrayVec<AssignedCell<pallas::Base, pallas::Base>, NUM_WINDOWS>, Error> {
81        // Enable `q_mul_fixed_full` selector
82        for idx in 0..NUM_WINDOWS {
83            self.q_mul_fixed_full.enable(region, offset + idx)?;
84        }
85
86        // Decompose scalar into `k-bit` windows
87        let scalar_windows: Value<Vec<u8>> = scalar.map(|scalar| {
88            decompose_word::<pallas::Scalar>(&scalar, SCALAR_NUM_BITS, FIXED_BASE_WINDOW_SIZE)
89        });
90
91        // Transpose `Value<Vec<u8>>` into `Vec<Value<pallas::Base>>`.
92        let scalar_windows = scalar_windows
93            .map(|windows| {
94                windows
95                    .into_iter()
96                    .map(|window| pallas::Base::from(window as u64))
97            })
98            .transpose_vec(NUM_WINDOWS);
99
100        // Store the scalar decomposition
101        let mut windows: ArrayVec<AssignedCell<pallas::Base, pallas::Base>, NUM_WINDOWS> =
102            ArrayVec::new();
103        for (idx, window) in scalar_windows.into_iter().enumerate() {
104            let window_cell = region.assign_advice(
105                || format!("k[{:?}]", offset + idx),
106                self.super_config.window,
107                offset + idx,
108                || window,
109            )?;
110            windows.push(window_cell);
111        }
112
113        Ok(windows)
114    }

116    pub fn assign(
117        &self,
118        mut layouter: impl Layouter<pallas::Base>,
119        scalar: &EccScalarFixed,
120        base: &<Fixed as FixedPoints<pallas::Affine>>::FullScalar,
121    ) -> Result<(EccPoint, EccScalarFixed), Error>
122    where
123        <Fixed as FixedPoints<pallas::Affine>>::FullScalar:
124            super::super::FixedPoint<pallas::Affine>,
125    {
126        let (scalar, acc, mul_b) = layouter.assign_region(
127            || "Full-width fixed-base mul (incomplete addition)",
128            |mut region| {
129                let offset = 0;
130
131                // Lazily witness the scalar.
132                let scalar = match scalar.windows {
133                    None => self.witness(&mut region, offset, scalar.value),
134                    Some(_) => todo!("unimplemented for halo2_gadgets v0.1.0"),
135                }?;
136
137                let (acc, mul_b) = self.super_config.assign_region_inner::<_, NUM_WINDOWS>(
138                    &mut region,
139                    offset,
140                    &(&scalar).into(),
141                    base,
142                    self.q_mul_fixed_full,
143                )?;
144
145                Ok((scalar, acc, mul_b))
146            },
147        )?;
148
149        // Add to the accumulator and return the final result as `[scalar]B`.
150        let result = layouter.assign_region(
151            || "Full-width fixed-base mul (last window, complete addition)",
152            |mut region| {
153                self.super_config.add_config.assign_region(
154                    &mul_b.clone().into(),
155                    &acc.clone().into(),
156                    0,
157                    &mut region,
158                )
159            },
160        )?;
161
162        #[cfg(test)]
163        // Check that the correct multiple is obtained.
164        {
165            use super::super::FixedPoint;
166            use group::Curve;
167
168            let real_mul = scalar.value.map(|scalar| base.generator() * scalar);
169            let result = result.point();
170
171            real_mul
172                .zip(result)
173                .assert_if_known(|(real_mul, result)| &real_mul.to_affine() == result);
174        }
175
176        Ok((result, scalar))
177    }

214    #[allow(clippy::op_ref)]
215    fn test_single_base<Lookup: PallasLookupRangeCheck>(
216        chip: EccChip<TestFixedBases, Lookup>,
217        mut layouter: impl Layouter<pallas::Base>,
218        base: FixedPoint<pallas::Affine, EccChip<TestFixedBases, Lookup>>,
219        base_val: pallas::Affine,
220    ) -> Result<(), Error> {
221        fn constrain_equal_non_id<Lookup: PallasLookupRangeCheck>(
222            chip: EccChip<TestFixedBases, Lookup>,
223            mut layouter: impl Layouter<pallas::Base>,
224            base_val: pallas::Affine,
225            scalar_val: pallas::Scalar,
226            result: Point<pallas::Affine, EccChip<TestFixedBases, Lookup>>,
227        ) -> Result<(), Error> {
228            let expected = NonIdentityPoint::new(
229                chip,
230                layouter.namespace(|| "expected point"),
231                Value::known((base_val * scalar_val).to_affine()),
232            )?;
233            result.constrain_equal(layouter.namespace(|| "constrain result"), &expected)
234        }
235
236        // [a]B
237        {
238            let scalar_fixed = pallas::Scalar::random(OsRng);
239            let by = ScalarFixed::new(
240                chip.clone(),
241                layouter.namespace(|| "random a"),
242                Value::known(scalar_fixed),
243            )?;
244
245            let (result, _) = base.mul(layouter.namespace(|| "random [a]B"), by)?;
246            constrain_equal_non_id(
247                chip.clone(),
248                layouter.namespace(|| "random [a]B"),
249                base_val,
250                scalar_fixed,
251                result,
252            )?;
253        }
254
255        // There is a single canonical sequence of window values for which a doubling occurs on the last step:
256        // 1333333333333333333333333333333333333333333333333333333333333333333333333333333333334 in octal.
257        // (There is another *non-canonical* sequence
258        // 5333333333333333333333333333333333333333332711161673731021062440252244051273333333333 in octal.)
259        {
260            const LAST_DOUBLING: &str = "1333333333333333333333333333333333333333333333333333333333333333333333333333333333334";
261            let h = pallas::Scalar::from(H as u64);
262            let scalar_fixed = LAST_DOUBLING
263                .chars()
264                .fold(pallas::Scalar::zero(), |acc, c| {
265                    acc * &h + &pallas::Scalar::from(c.to_digit(8).unwrap() as u64)
266                });
267            let by = ScalarFixed::new(
268                chip.clone(),
269                layouter.namespace(|| LAST_DOUBLING),
270                Value::known(scalar_fixed),
271            )?;
272            let (result, _) = base.mul(layouter.namespace(|| "mul with double"), by)?;
273
274            constrain_equal_non_id(
275                chip.clone(),
276                layouter.namespace(|| "mul with double"),
277                base_val,
278                scalar_fixed,
279                result,
280            )?;
281        }
282
283        // [0]B should return (0,0) since it uses complete addition
284        // on the last step.
285        {
286            let scalar_fixed = pallas::Scalar::zero();
287            let zero = ScalarFixed::new(
288                chip.clone(),
289                layouter.namespace(|| "0"),
290                Value::known(scalar_fixed),
291            )?;
292            let (result, _) = base.mul(layouter.namespace(|| "mul by zero"), zero)?;
293            result
294                .inner()
295                .is_identity()
296                .assert_if_known(|is_identity| *is_identity);
297        }
298
299        // [-1]B is the largest scalar field element.
300        {
301            let scalar_fixed = -pallas::Scalar::one();
302            let neg_1 = ScalarFixed::new(
303                chip.clone(),
304                layouter.namespace(|| "-1"),
305                Value::known(scalar_fixed),
306            )?;
307            let (result, _) = base.mul(layouter.namespace(|| "mul by -1"), neg_1)?;
308            constrain_equal_non_id(
309                chip,
310                layouter.namespace(|| "mul by -1"),
311                base_val,
312                scalar_fixed,
313                result,
314            )?;
315        }
316
317        Ok(())
318    }
