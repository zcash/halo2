25/// Number of bits for which complete addition needs to be used in variable-base
26/// scalar multiplication
27const NUM_COMPLETE_BITS: usize = 3;
28
29// Bits used in incomplete addition. k_{254} to k_{4} inclusive
30const INCOMPLETE_LEN: usize = pallas::Scalar::NUM_BITS as usize - 1 - NUM_COMPLETE_BITS;
31
32// Bits k_{254} to k_{4} inclusive are used in incomplete addition.
33// The `hi` half is k_{254} to k_{130} inclusive (length 125 bits).
34// (It is a coincidence that k_{130} matches the boundary of the
35// overflow check described in [the book](https://zcash.github.io/halo2/design/gadgets/ecc/var-base-scalar-mul.html#overflow-check).)
36const INCOMPLETE_HI_RANGE: Range<usize> = 0..INCOMPLETE_HI_LEN;
37const INCOMPLETE_HI_LEN: usize = INCOMPLETE_LEN / 2;
38
39// Bits k_{254} to k_{4} inclusive are used in incomplete addition.
40// The `lo` half is k_{129} to k_{4} inclusive (length 126 bits).
41const INCOMPLETE_LO_RANGE: Range<usize> = INCOMPLETE_HI_LEN..INCOMPLETE_LEN;
42const INCOMPLETE_LO_LEN: usize = INCOMPLETE_LEN - INCOMPLETE_HI_LEN;
43
44// Bits k_{3} to k_{1} inclusive are used in complete addition.
45// Bit k_{0} is handled separately.
46const COMPLETE_RANGE: Range<usize> = INCOMPLETE_LEN..(INCOMPLETE_LEN + NUM_COMPLETE_BITS);

65    pub(super) fn configure(
66        meta: &mut ConstraintSystem<pallas::Base>,
67        add_config: add::Config,
68        lookup_config: Lookup,
69        advices: [Column<Advice>; 10],
70    ) -> Self {
71        let hi_config = incomplete::Config::configure(
72            meta, advices[9], advices[3], advices[0], advices[1], advices[4], advices[5],
73        );
74        let lo_config = incomplete::Config::configure(
75            meta, advices[6], advices[7], advices[0], advices[1], advices[8], advices[2],
76        );
77        let complete_config = complete::Config::configure(meta, advices[9], add_config);
78        let overflow_config =
79            overflow::Config::configure(meta, lookup_config, advices[6..9].try_into().unwrap());
80
81        let config = Self {
82            q_mul_lsb: meta.selector(),
83            add_config,
84            hi_config,
85            lo_config,
86            complete_config,
87            overflow_config,
88        };
89
90        config.create_gate(meta);
91
92        assert_eq!(
93            config.hi_config.double_and_add.x_p, config.lo_config.double_and_add.x_p,
94            "x_p is shared across hi and lo halves."
95        );
96        assert_eq!(
97            config.hi_config.y_p, config.lo_config.y_p,
98            "y_p is shared across hi and lo halves."
99        );
100
101        // For both hi_config and lo_config:
102        // z and lambda1 are assigned on the same row as the add_config output.
103        // Therefore, z and lambda1 must not overlap with add_config.x_qr, add_config.y_qr.
104        let add_config_outputs = config.add_config.output_columns();
105        {
106            assert!(
107                !add_config_outputs.contains(&config.hi_config.z),
108                "incomplete config z cannot overlap with complete addition columns."
109            );
110            assert!(
111                !add_config_outputs.contains(&config.hi_config.double_and_add.lambda_1),
112                "incomplete config lambda1 cannot overlap with complete addition columns."
113            );
114        }
115        {
116            assert!(
117                !add_config_outputs.contains(&config.lo_config.z),
118                "incomplete config z cannot overlap with complete addition columns."
119            );
120            assert!(
121                !add_config_outputs.contains(&config.lo_config.double_and_add.lambda_1),
122                "incomplete config lambda1 cannot overlap with complete addition columns."
123            );
124        }
125
126        config
127    }

129    fn create_gate(&self, meta: &mut ConstraintSystem<pallas::Base>) {
130        // If `lsb` is 0, (x, y) = (x_p, -y_p). If `lsb` is 1, (x, y) = (0,0).
131        // https://p.z.cash/halo2-0.1:ecc-var-mul-lsb-gate?partial
132        meta.create_gate("LSB check", |meta| {
133            let q_mul_lsb = meta.query_selector(self.q_mul_lsb);
134
135            let z_1 = meta.query_advice(self.complete_config.z_complete, Rotation::cur());
136            let z_0 = meta.query_advice(self.complete_config.z_complete, Rotation::next());
137            let x_p = meta.query_advice(self.add_config.x_p, Rotation::cur());
138            let y_p = meta.query_advice(self.add_config.y_p, Rotation::cur());
139            let base_x = meta.query_advice(self.add_config.x_p, Rotation::next());
140            let base_y = meta.query_advice(self.add_config.y_p, Rotation::next());
141
142            //    z_0 = 2 * z_1 + k_0
143            // => k_0 = z_0 - 2 * z_1
144            let lsb = z_0 - z_1 * pallas::Base::from(2);
145
146            let bool_check = bool_check(lsb.clone());
147
148            // `lsb` = 0 => (x_p, y_p) = (x, -y)
149            // `lsb` = 1 => (x_p, y_p) = (0,0)
150            let lsb_x = ternary(lsb.clone(), x_p.clone(), x_p - base_x);
151            let lsb_y = ternary(lsb, y_p.clone(), y_p + base_y);
152
153            Constraints::with_selector(
154                q_mul_lsb,
155                [
156                    ("bool_check", bool_check),
157                    ("lsb_x", lsb_x),
158                    ("lsb_y", lsb_y),
159                ],
160            )
161        });
162    }

164    pub(super) fn assign(
165        &self,
166        mut layouter: impl Layouter<pallas::Base>,
167        alpha: AssignedCell<pallas::Base, pallas::Base>,
168        base: &NonIdentityEccPoint,
169    ) -> Result<(EccPoint, ScalarVar), Error> {
170        let (result, zs): (EccPoint, Vec<Z<pallas::Base>>) = layouter.assign_region(
171            || "variable-base scalar mul",
172            |mut region| {
173                let offset = 0;
174
175                // Case `base` into an `EccPoint` for later use.
176                let base_point: EccPoint = base.clone().into();
177
178                // Decompose `k = alpha + t_q` bitwise (big-endian bit order).
179                let bits = decompose_for_scalar_mul(alpha.value());
180
181                // Define ranges for each part of the algorithm.
182                let bits_incomplete_hi = &bits[INCOMPLETE_HI_RANGE];
183                let bits_incomplete_lo = &bits[INCOMPLETE_LO_RANGE];
184                let lsb = bits[pallas::Scalar::NUM_BITS as usize - 1];
185
186                // Initialize the accumulator `acc = [2]base` using complete addition.
187                let acc =
188                    self.add_config
189                        .assign_region(&base_point, &base_point, offset, &mut region)?;
190
191                // Increase the offset by 1 after complete addition.
192                let offset = offset + 1;
193
194                // Initialize the running sum for scalar decomposition to zero.
195                //
196                // `incomplete::Config::double_and_add` will copy this cell directly into
197                // itself. This is fine because we are just assigning the same value to
198                // the same cell twice, and then applying an equality constraint between
199                // the cell and itself (which the permutation argument treats as a no-op).
200                let z_init = Z(region.assign_advice_from_constant(
201                    || "z_init = 0",
202                    self.hi_config.z,
203                    offset,
204                    pallas::Base::zero(),
205                )?);
206
207                // Double-and-add (incomplete addition) for the `hi` half of the scalar decomposition
208                let (x_a, y_a, zs_incomplete_hi) = self.hi_config.double_and_add(
209                    &mut region,
210                    offset,
211                    base,
212                    bits_incomplete_hi,
213                    (X(acc.x), Y(acc.y), z_init.clone()),
214                )?;
215
216                // Double-and-add (incomplete addition) for the `lo` half of the scalar decomposition
217                let z = zs_incomplete_hi.last().expect("should not be empty");
218                let (x_a, y_a, zs_incomplete_lo) = self.lo_config.double_and_add(
219                    &mut region,
220                    offset,
221                    base,
222                    bits_incomplete_lo,
223                    (x_a, y_a, z.clone()),
224                )?;
225
226                // Move from incomplete addition to complete addition.
227                // Inside incomplete::double_and_add, the offset was increased once after initialization
228                // of the running sum.
229                // Then, the final assignment of double-and-add was made on row + offset + 1.
230                // Outside of incomplete addition, we must account for these offset increases by adding
231                // 2 to the incomplete addition length.
232                assert!(INCOMPLETE_LO_RANGE.len() >= INCOMPLETE_HI_RANGE.len());
233                let offset = offset + INCOMPLETE_LO_RANGE.len() + 2;
234
235                // Complete addition
236                let (acc, zs_complete) = {
237                    let z = zs_incomplete_lo.last().expect("should not be empty");
238                    // Bits used in complete addition. k_{3} to k_{1} inclusive
239                    // The LSB k_{0} is handled separately.
240                    let bits_complete = &bits[COMPLETE_RANGE];
241                    self.complete_config.assign_region(
242                        &mut region,
243                        offset,
244                        bits_complete,
245                        &base_point,
246                        x_a,
247                        y_a,
248                        z.clone(),
249                    )?
250                };
251
252                // Each iteration of the complete addition uses two rows.
253                let offset = offset + COMPLETE_RANGE.len() * 2;
254
255                // Process the least significant bit
256                let z_1 = zs_complete.last().unwrap().clone();
257                let (result, z_0) = self.process_lsb(&mut region, offset, base, acc, z_1, lsb)?;
258
259                #[cfg(test)]
260                // Check that the correct multiple is obtained.
261                {
262                    use group::Curve;
263
264                    let base = base.point();
265                    let alpha = alpha
266                        .value()
267                        .map(|alpha| pallas::Scalar::from_repr(alpha.to_repr()).unwrap());
268                    let real_mul = base.zip(alpha).map(|(base, alpha)| base * alpha);
269                    let result = result.point();
270
271                    real_mul
272                        .zip(result)
273                        .assert_if_known(|(real_mul, result)| &real_mul.to_affine() == result);
274                }
275
276                let zs = {
277                    let mut zs = std::iter::empty()
278                        .chain(Some(z_init))
279                        .chain(zs_incomplete_hi.into_iter())
280                        .chain(zs_incomplete_lo.into_iter())
281                        .chain(zs_complete.into_iter())
282                        .chain(Some(z_0))
283                        .collect::<Vec<_>>();
284                    assert_eq!(zs.len(), pallas::Scalar::NUM_BITS as usize + 1);
285
286                    // This reverses zs to give us [z_0, z_1, ..., z_{254}, z_{255}].
287                    zs.reverse();
288                    zs
289                };
290
291                Ok((result, zs))
292            },
293        )?;
294
295        self.overflow_config.overflow_check(
296            layouter.namespace(|| "overflow check"),
297            alpha.clone(),
298            &zs,
299        )?;
300
301        Ok((result, ScalarVar::BaseFieldElem(alpha)))
302    }
303
304    /// Processes the final scalar bit `k_0`.
305    ///
306    /// Assumptions for this sub-region:
307    /// - `acc_x` and `acc_y` are assigned in row `offset` by the previous complete
308    ///   addition. They will be copied into themselves.
309    /// - `z_1 is assigned in row `offset` by the mul::complete region assignment. We only
310    ///   use its value here.
311    ///
312    /// `x_p` and `y_p` are assigned here, and then copied into themselves by the complete
313    /// addition subregion.
314    ///
315    /// ```text
316    /// | x_p  | y_p  | acc_x | acc_y | complete addition  | z_1 | q_mul_lsb = 1
317    /// |base_x|base_y| res_x | res_y |   |   |    |   |   | z_0 |
318    /// ```
319    ///
320    /// [Specification](https://p.z.cash/halo2-0.1:ecc-var-mul-lsb-gate?partial).
321    fn process_lsb(
322        &self,
323        region: &mut Region<'_, pallas::Base>,
324        offset: usize,
325        base: &NonIdentityEccPoint,
326        acc: EccPoint,
327        z_1: Z<pallas::Base>,
328        lsb: Value<bool>,
329    ) -> Result<(EccPoint, Z<pallas::Base>), Error> {
330        // Enforce switching logic on LSB using a custom gate
331        self.q_mul_lsb.enable(region, offset)?;
332
333        // z_1 has been assigned at (z_complete, offset).
334        // Assign z_0 = 2â‹…z_1 + k_0
335        let z_0 = {
336            let z_0_val = z_1.value().zip(lsb).map(|(z_1, lsb)| {
337                let lsb = pallas::Base::from(lsb as u64);
338                z_1 * pallas::Base::from(2) + lsb
339            });
340            let z_0_cell = region.assign_advice(
341                || "z_0",
342                self.complete_config.z_complete,
343                offset + 1,
344                || z_0_val,
345            )?;
346
347            Z(z_0_cell)
348        };
349
350        // Copy in `base_x`, `base_y` to use in the LSB gate
351        base.x()
352            .copy_advice(|| "copy base_x", region, self.add_config.x_p, offset + 1)?;
353        base.y()
354            .copy_advice(|| "copy base_y", region, self.add_config.y_p, offset + 1)?;
355
356        // If `lsb` is 0, return `Acc + (-P)`. If `lsb` is 1, simply return `Acc + 0`.
357        let x = lsb.and_then(|lsb| {
358            if !lsb {
359                base.x.value().cloned()
360            } else {
361                Value::known(Assigned::Zero)
362            }
363        });
364
365        let y = lsb.and_then(|lsb| {
366            if !lsb {
367                -base.y.value()
368            } else {
369                Value::known(Assigned::Zero)
370            }
371        });
372
373        let x_cell = region.assign_advice(|| "x", self.add_config.x_p, offset, || x)?;
374        let y_cell = region.assign_advice(|| "y", self.add_config.y_p, offset, || y)?;
375
376        let p = EccPoint::from_coordinates_unchecked(x_cell, y_cell);
377
378        // Return the result of the final complete addition as `[scalar]B`
379        let result = self.add_config.assign_region(&p, &acc, offset, region)?;
380
381        Ok((result, z_0))
382    }
383}

418// https://p.z.cash/halo2-0.1:ecc-var-mul-witness-scalar?partial
419#[allow(clippy::assign_op_pattern)]
420#[allow(clippy::ptr_offset_with_cast)]
421fn decompose_for_scalar_mul(scalar: Value<&pallas::Base>) -> Vec<Value<bool>> {
422    construct_uint! {
423        struct U256(4);
424    }
425
426    let bitstring = scalar.map(|scalar| {
427        // We use `k = scalar + t_q` in the double-and-add algorithm, where
428        // the scalar field `F_q = 2^254 + t_q`.
429        // Note that the addition `scalar + t_q` is not reduced.
430        //
431        let scalar = U256::from_little_endian(&scalar.to_repr());
432        let t_q = U256::from_little_endian(&T_Q.to_le_bytes());
433        let k = scalar + t_q;
434
435        // Little-endian bit representation of `k`.
436        let bitstring = {
437            let mut le_bytes = [0u8; 32];
438            k.to_little_endian(&mut le_bytes);
439            le_bytes
440                .into_iter()
441                .flat_map(|byte| (0..8).map(move |shift| (byte >> shift) % 2 == 1))
442        };
443
444        // Take the first 255 bits.
445        bitstring
446            .take(pallas::Scalar::NUM_BITS as usize)
447            .collect::<Vec<_>>()
448    });
449
450    // Transpose.
451    let mut bitstring = bitstring.transpose_vec(pallas::Scalar::NUM_BITS as usize);
452    // Reverse to get the big-endian bit representation.
453    bitstring.reverse();
454    bitstring
455}

487        fn constrain_equal_non_id<
488            EccChip: EccInstructions<pallas::Affine, Point = EccPoint> + Clone + Eq + std::fmt::Debug,
