19/// The running sum $[z_0, ..., z_W]$. If created in strict mode, $z_W = 0$.
20#[derive(Debug)]
21pub struct RunningSum<F: PrimeFieldBits>(Vec<AssignedCell<F, F>>);
22impl<F: PrimeFieldBits> std::ops::Deref for RunningSum<F> {
23    type Target = Vec<AssignedCell<F, F>>;
24
25    fn deref(&self) -> &Vec<AssignedCell<F, F>> {
26        &self.0
27    }
28}

30impl<F: PrimeFieldBits> RangeConstrained<F, AssignedCell<F, F>> {
31    /// Witnesses a subset of the bits in `value` and constrains them to be the correct
32    /// number of bits.
33    ///
34    /// # Panics
35    ///
36    /// Panics if `bitrange.len() >= K`.
37    pub fn witness_short<const K: usize, Lookup: LookupRangeCheck<F, K>>(
38        lookup_config: &Lookup,
39        layouter: impl Layouter<F>,
40        value: Value<&F>,
41        bitrange: Range<usize>,
42    ) -> Result<Self, Error> {
43        let num_bits = bitrange.len();
44        assert!(num_bits < K);
45
46        // Witness the subset and constrain it to be the correct number of bits.
47        lookup_config
48            .witness_short_check(
49                layouter,
50                value.map(|value| bitrange_subset(value, bitrange)),
51                num_bits,
52            )
53            .map(|inner| Self {
54                inner,
55                num_bits,
56                _phantom: PhantomData::default(),
57            })
58    }
59}

61/// Configuration that provides methods for a lookup range check.
62#[derive(Eq, PartialEq, Debug, Clone, Copy)]
63pub struct LookupRangeCheckConfig<F: PrimeFieldBits, const K: usize> {
64    q_lookup: Selector,
65    q_running: Selector,
66    q_bitshift: Selector,
67    running_sum: Column<Advice>,
68    table_idx: TableColumn,
69    _marker: PhantomData<F>,

72/// Trait that provides common methods for a lookup range check.
73pub trait LookupRangeCheck<F: PrimeFieldBits, const K: usize>: Eq + Copy + Debug {
74    /// Returns a reference to the `LookupRangeCheckConfig` instance.
75    fn config(&self) -> &LookupRangeCheckConfig<F, K>;
76
77    /// The `running_sum` advice column breaks the field element into `K`-bit
78    /// words. It is used to construct the input expression to the lookup
79    /// argument.
80    ///
81    /// The `table_idx` fixed column contains values from [0..2^K). Looking up
82    /// a value in `table_idx` constrains it to be within this range. The table
83    /// can be loaded outside this helper.
84    ///
85    /// # Side-effects
86    ///
87    /// Both the `running_sum` and `constants` columns will be equality-enabled.
88    fn configure(
89        meta: &mut ConstraintSystem<F>,
90        running_sum: Column<Advice>,
91        table_idx: TableColumn,
92    ) -> Self
93    where
94        Self: Sized;
95
96    /// Load the generator table into the circuit.
97    fn load(
98        &self,
99        generator_table_config: &GeneratorTableConfig,
100        layouter: &mut impl Layouter<pallas::Base>,
101    ) -> Result<(), Error>;
102
103    #[cfg(test)]
104    /// Fill `table_idx` and `table_range_check_tag`.
105    ///
106    /// This is only used in testing for now, since the Sinsemilla chip provides a pre-loaded table
107    /// in the Orchard context.
108    fn load_range_check_table(&self, layouter: &mut impl Layouter<F>) -> Result<(), Error>;
109
110    /// Constrain `x` to be a NUM_BITS word.
111    ///
112    /// `element` must have been assigned to `self.config().running_sum` at offset 0.
113    fn short_range_check(
114        &self,
115        region: &mut Region<'_, F>,
116        element: AssignedCell<F, F>,
117        num_bits: usize,
118    ) -> Result<(), Error>;
119
120    /// Range check on an existing cell that is copied into this helper.
121    ///
122    /// Returns an error if `element` is not in a column that was passed to
123    /// [`ConstraintSystem::enable_equality`] during circuit configuration.
124    fn copy_check(
125        &self,
126        mut layouter: impl Layouter<F>,
127        element: AssignedCell<F, F>,
128        num_words: usize,
129        strict: bool,
130    ) -> Result<RunningSum<F>, Error> {
131        layouter.assign_region(
132            || format!("{:?} words range check", num_words),
133            |mut region| {
134                // Copy `element` and initialize running sum `z_0 = element` to decompose it.
135                let z_0 =
136                    element.copy_advice(|| "z_0", &mut region, self.config().running_sum, 0)?;
137                self.range_check(&mut region, z_0, num_words, strict)
138            },
139        )
140    }
141
142    /// Range check on a value that is witnessed in this helper.
143    fn witness_check(
144        &self,
145        mut layouter: impl Layouter<F>,
146        value: Value<F>,
147        num_words: usize,
148        strict: bool,
149    ) -> Result<RunningSum<F>, Error> {
150        layouter.assign_region(
151            || "Witness element",
152            |mut region| {
153                let z_0 = region.assign_advice(
154                    || "Witness element",
155                    self.config().running_sum,
156                    0,
157                    || value,
158                )?;
159                self.range_check(&mut region, z_0, num_words, strict)
160            },
161        )
162    }
163
164    /// If `strict` is set to "true", the field element must fit into
165    /// `num_words * K` bits. In other words, the the final cumulative sum `z_{num_words}`
166    /// must be zero.
167    ///
168    /// If `strict` is set to "false", the final `z_{num_words}` is not constrained.
169    ///
170    /// `element` must have been assigned to `self.running_sum` at offset 0.
171    fn range_check(
172        &self,
173        region: &mut Region<'_, F>,
174        element: AssignedCell<F, F>,
175        num_words: usize,
176        strict: bool,
177    ) -> Result<RunningSum<F>, Error> {
178        // `num_words` must fit into a single field element.
179        assert!(num_words * K <= F::CAPACITY as usize);
180        let num_bits = num_words * K;
181
182        // Chunk the first num_bits bits into K-bit words.
183        let words = {
184            // Take first num_bits bits of `element`.
185            let bits = element.value().map(|element| {
186                element
187                    .to_le_bits()
188                    .into_iter()
189                    .take(num_bits)
190                    .collect::<Vec<_>>()
191            });
192
193            bits.map(|bits| {
194                bits.chunks_exact(K)
195                    .map(|word| F::from(lebs2ip::<K>(&(word.try_into().unwrap()))))
196                    .collect::<Vec<_>>()
197            })
198            .transpose_vec(num_words)
199        };
200
201        let mut zs = vec![element.clone()];
202
203        // Assign cumulative sum such that
204        //          z_i = 2^{K}⋅z_{i + 1} + a_i
205        // => z_{i + 1} = (z_i - a_i) / (2^K)
206        //
207        // For `element` = a_0 + 2^10 a_1 + ... + 2^{120} a_{12}}, initialize z_0 = `element`.
208        // If `element` fits in 130 bits, we end up with z_{13} = 0.
209        let mut z = element;
210        let inv_two_pow_k = F::from(1u64 << K).invert().unwrap();
211        for (idx, word) in words.iter().enumerate() {
212            // Enable q_lookup on this row
213            self.config().q_lookup.enable(region, idx)?;
214            // Enable q_running on this row
215            self.config().q_running.enable(region, idx)?;
216
217            // z_next = (z_cur - m_cur) / 2^K
218            z = {
219                let z_val = z
220                    .value()
221                    .zip(*word)
222                    .map(|(z, word)| (*z - word) * inv_two_pow_k);
223
224                // Assign z_next
225                region.assign_advice(
226                    || format!("z_{:?}", idx + 1),
227                    self.config().running_sum,
228                    idx + 1,
229                    || z_val,
230                )?
231            };
232            zs.push(z.clone());
233        }
234
235        if strict {
236            // Constrain the final `z` to be zero.
237            region.constrain_constant(zs.last().unwrap().cell(), F::ZERO)?;
238        }
239
240        Ok(RunningSum(zs))
241    }
242
243    /// Short range check on an existing cell that is copied into this helper.
244    ///
245    /// # Panics
246    ///
247    /// Panics if NUM_BITS is equal to or larger than K.
248    fn copy_short_check(
249        &self,
250        mut layouter: impl Layouter<F>,
251        element: AssignedCell<F, F>,
252        num_bits: usize,
253    ) -> Result<(), Error> {
254        assert!(num_bits < K);
255        layouter.assign_region(
256            || format!("Range check {:?} bits", num_bits),
257            |mut region| {
258                // Copy `element` to use in the k-bit lookup.
259                let element =
260                    element.copy_advice(|| "element", &mut region, self.config().running_sum, 0)?;
261                self.short_range_check(&mut region, element, num_bits)
262            },
263        )
264    }
265
266    /// Short range check on value that is witnessed in this helper.
267    ///
268    /// # Panics
269    ///
270    /// Panics if num_bits is larger than K.
271    fn witness_short_check(
272        &self,
273        mut layouter: impl Layouter<F>,
274        element: Value<F>,
275        num_bits: usize,
276    ) -> Result<AssignedCell<F, F>, Error> {
277        assert!(num_bits <= K);
278        layouter.assign_region(
279            || format!("Range check {:?} bits", num_bits),
280            |mut region| {
281                // Witness `element` to use in the k-bit lookup.
282                let element = region.assign_advice(
283                    || "Witness element",
284                    self.config().running_sum,
285                    0,
286                    || element,
287                )?;
288
289                self.short_range_check(&mut region, element.clone(), num_bits)?;
290
291                Ok(element)
292            },
293        )
294    }
295}

297impl<F: PrimeFieldBits, const K: usize> LookupRangeCheck<F, K> for LookupRangeCheckConfig<F, K> {
298    fn config(&self) -> &LookupRangeCheckConfig<F, K> {
299        self
300    }
301
302    /// The `running_sum` advice column breaks the field element into `K`-bit
303    /// words. It is used to construct the input expression to the lookup
304    /// argument.
305    ///
306    /// The `table_idx` fixed column contains values from [0..2^K). Looking up
307    /// a value in `table_idx` constrains it to be within this range. The table
308    /// can be loaded outside this helper.
309    ///
310    /// # Side-effects
311    ///
312    /// Both the `running_sum` and `constants` columns will be equality-enabled.
313    fn configure(
314        meta: &mut ConstraintSystem<F>,
315        running_sum: Column<Advice>,
316        table_idx: TableColumn,
317    ) -> Self {
318        meta.enable_equality(running_sum);
319
320        let q_lookup = meta.complex_selector();
321        let q_running = meta.complex_selector();
322        let q_bitshift = meta.selector();
323
324        let config = LookupRangeCheckConfig {
325            q_lookup,
326            q_running,
327            q_bitshift,
328            running_sum,
329            table_idx,
330            _marker: PhantomData,
331        };
332
333        // https://p.z.cash/halo2-0.1:decompose-combined-lookup
334        meta.lookup(|meta| {
335            let q_lookup = meta.query_selector(config.q_lookup);
336            let q_running = meta.query_selector(config.q_running);
337            let z_cur = meta.query_advice(config.running_sum, Rotation::cur());
338            let one = Expression::Constant(F::ONE);
339
340            // In the case of a running sum decomposition, we recover the word from
341            // the difference of the running sums:
342            //    z_i = 2^{K}⋅z_{i + 1} + a_i
343            // => a_i = z_i - 2^{K}⋅z_{i + 1}
344            let running_sum_lookup = {
345                let running_sum_word = {
346                    let z_next = meta.query_advice(config.running_sum, Rotation::next());
347                    z_cur.clone() - z_next * F::from(1 << K)
348                };
349
350                q_running.clone() * running_sum_word
351            };
352
353            // In the short range check, the word is directly witnessed.
354            let short_lookup = {
355                let short_word = z_cur;
356                let q_short = one - q_running;
357
358                q_short * short_word
359            };
360
361            // Combine the running sum and short lookups:
362            vec![(
363                q_lookup * (running_sum_lookup + short_lookup),
364                config.table_idx,
365            )]
366        });
367
368        // For short lookups, check that the word has been shifted by the correct number of bits.
369        // https://p.z.cash/halo2-0.1:decompose-short-lookup
370        meta.create_gate("Short lookup bitshift", |meta| {
371            let q_bitshift = meta.query_selector(config.q_bitshift);
372            let word = meta.query_advice(config.running_sum, Rotation::prev());
373            let shifted_word = meta.query_advice(config.running_sum, Rotation::cur());
374            let inv_two_pow_s = meta.query_advice(config.running_sum, Rotation::next());
375
376            let two_pow_k = F::from(1 << K);
377
378            // shifted_word = word * 2^{K-s}
379            //              = word * 2^K * inv_two_pow_s
380            Constraints::with_selector(
381                q_bitshift,
382                Some(word * two_pow_k * inv_two_pow_s - shifted_word),
383            )
384        });
385
386        config
387    }
388
389    /// Load the generator table into the circuit.
390    ///
391    /// | table_idx |     table_x    |     table_y    |
392    /// ------------------------------------------------
393    /// |     0     |    X(S\[0\])   |    Y(S\[0\])   |
394    /// |     1     |    X(S\[1\])   |    Y(S\[1\])   |
395    /// |    ...    |      ...       |       ...      |
396    /// |   2^10-1  | X(S\[2^10-1\]) | Y(S\[2^10-1\]) |
397    fn load(
398        &self,
399        generator_table_config: &GeneratorTableConfig,
400        layouter: &mut impl Layouter<pallas::Base>,
401    ) -> Result<(), Error> {
402        layouter.assign_table(
403            || "generator_table",
404            |mut table| {
405                for (index, (x, y)) in SINSEMILLA_S.iter().enumerate() {
406                    table.assign_cell(
407                        || "table_idx",
408                        generator_table_config.table_idx,
409                        index,
410                        || Value::known(pallas::Base::from(index as u64)),
411                    )?;
412                    table.assign_cell(
413                        || "table_x",
414                        generator_table_config.table_x,
415                        index,
416                        || Value::known(*x),
417                    )?;
418                    table.assign_cell(
419                        || "table_y",
420                        generator_table_config.table_y,
421                        index,
422                        || Value::known(*y),
423                    )?;
424                }
425                Ok(())
426            },
427        )
428    }
429
430    #[cfg(test)]
431    // Fill `table_idx` and `table_range_check_tag`.
432    // This is only used in testing for now, since the Sinsemilla chip provides a pre-loaded table
433    // in the Orchard context.
434    fn load_range_check_table(&self, layouter: &mut impl Layouter<F>) -> Result<(), Error> {
435        layouter.assign_table(
436            || "table_idx",
437            |mut table| {
438                // We generate the row values lazily (we only need them during keygen).
439                for index in 0..(1 << K) {
440                    table.assign_cell(
441                        || "table_idx",
442                        self.table_idx,
443                        index,
444                        || Value::known(F::from(index as u64)),
445                    )?;
446                }
447                Ok(())
448            },
449        )
450    }

452    /// Constrain `x` to be a NUM_BITS word.
453    ///
454    /// `element` must have been assigned to `self.running_sum` at offset 0.
455    fn short_range_check(
456        &self,
457        region: &mut Region<'_, F>,
458        element: AssignedCell<F, F>,
459        num_bits: usize,
460    ) -> Result<(), Error> {
461        // Enable lookup for `element`.
462        self.q_lookup.enable(region, 0)?;
463
464        // Enable lookup for shifted element, to constrain it to 10 bits.
465        self.q_lookup.enable(region, 1)?;
466
467        // Check element has been shifted by the correct number of bits.
468        self.q_bitshift.enable(region, 1)?;
469
470        // Assign shifted `element * 2^{K - num_bits}`
471        let shifted = element.value().into_field() * F::from(1 << (K - num_bits));
472
473        region.assign_advice(
474            || format!("element * 2^({}-{})", K, num_bits),
475            self.running_sum,
476            1,
477            || shifted,
478        )?;
479
480        // Assign 2^{-num_bits} from a fixed column.
481        let inv_two_pow_s = F::from(1 << num_bits).invert().unwrap();
482        region.assign_advice_from_constant(
483            || format!("2^(-{})", num_bits),
484            self.running_sum,
485            2,
486            inv_two_pow_s,
487        )?;
488
489        Ok(())
490    }
491}

504/// Configuration that provides methods for an efficient 4, 5, and K-bit lookup range check.
505#[derive(Eq, PartialEq, Debug, Clone, Copy)]
506pub struct LookupRangeCheck4_5BConfig<F: PrimeFieldBits, const K: usize> {
507    base: LookupRangeCheckConfig<F, K>,
508    q_range_check_4: Selector,
509    q_range_check_5: Selector,
510    table_range_check_tag: TableColumn,
511}

513impl<F: PrimeFieldBits, const K: usize> LookupRangeCheck4_5BConfig<F, K> {
514    /// The `running_sum` advice column breaks the field element into `K`-bit
515    /// words. It is used to construct the input expression to the lookup
516    /// argument.
517    ///
518    /// The `table_idx` fixed column contains values from [0..2^K). Looking up
519    /// a value in `table_idx` constrains it to be within this range. The table
520    /// can be loaded outside this helper.
521    ///
522    /// # Side effects
523    ///
524    /// Both the `running_sum` and `constants` columns will be equality-enabled.
525    pub fn configure_with_tag(
526        meta: &mut ConstraintSystem<F>,
527        running_sum: Column<Advice>,
528        table_idx: TableColumn,
529        table_range_check_tag: TableColumn,
530    ) -> Self {
531        meta.enable_equality(running_sum);
532
533        let q_lookup = meta.complex_selector();
534        let q_running = meta.complex_selector();
535        let q_bitshift = meta.selector();
536        let q_range_check_4 = meta.complex_selector();
537        let q_range_check_5 = meta.complex_selector();
538
539        let config = LookupRangeCheck4_5BConfig {
540            base: LookupRangeCheckConfig {
541                q_lookup,
542                q_running,
543                q_bitshift,
544                running_sum,
545                table_idx,
546                _marker: PhantomData,
547            },
548            q_range_check_4,
549            q_range_check_5,
550            table_range_check_tag,
551        };
552
553        // https://p.z.cash/halo2-0.1:decompose-combined-lookup
554        meta.lookup(|meta| {
555            let q_lookup = meta.query_selector(config.base.q_lookup);
556            let q_running = meta.query_selector(config.base.q_running);
557            let q_range_check_4 = meta.query_selector(config.q_range_check_4);
558            let q_range_check_5 = meta.query_selector(config.q_range_check_5);
559
560            let z_cur = meta.query_advice(config.base.running_sum, Rotation::cur());
561            let one = Expression::Constant(F::ONE);
562
563            // In the case of a running sum decomposition, we recover the word from
564            // the difference of the running sums:
565            //    z_i = 2^{K}⋅z_{i + 1} + a_i
566            // => a_i = z_i - 2^{K}⋅z_{i + 1}
567            let running_sum_lookup = {
568                let running_sum_word = {
569                    let z_next = meta.query_advice(config.base.running_sum, Rotation::next());
570                    z_cur.clone() - z_next * F::from(1 << K)
571                };
572
573                q_running.clone() * running_sum_word
574            };
575
576            // In the short range check, the word is directly witnessed.
577            let short_lookup = {
578                let short_word = z_cur.clone();
579                let q_short = one.clone() - q_running;
580
581                q_short * short_word
582            };
583
584            // q_range_check is equal to
585            // - 1 if q_range_check_4 = 1 or q_range_check_5 = 1
586            // - 0 otherwise
587            let q_range_check = one.clone()
588                - (one.clone() - q_range_check_4.clone()) * (one.clone() - q_range_check_5.clone());
589
590            // num_bits is equal to
591            // - 5 if q_range_check_5 = 1
592            // - 4 if q_range_check_4 = 1 and q_range_check_5 = 0
593            // - 0 otherwise
594            let num_bits = q_range_check_5.clone() * Expression::Constant(F::from(5_u64))
595                + (one.clone() - q_range_check_5)
596                    * q_range_check_4
597                    * Expression::Constant(F::from(4_u64));
598
599            // Combine the running sum, short lookups and optimized range checks:
600            vec![
601                (
602                    q_lookup.clone()
603                        * ((one - q_range_check.clone()) * (running_sum_lookup + short_lookup)
604                            + q_range_check.clone() * z_cur),
605                    config.base.table_idx,
606                ),
607                (
608                    q_lookup * q_range_check * num_bits,
609                    config.table_range_check_tag,
610                ),
611            ]
612        });
613
614        // For short lookups, check that the word has been shifted by the correct number of bits.
615        // https://p.z.cash/halo2-0.1:decompose-short-lookup
616        meta.create_gate("Short lookup bitshift", |meta| {
617            let q_bitshift = meta.query_selector(config.base.q_bitshift);
618            let word = meta.query_advice(config.base.running_sum, Rotation::prev());
619            let shifted_word = meta.query_advice(config.base.running_sum, Rotation::cur());
620            let inv_two_pow_s = meta.query_advice(config.base.running_sum, Rotation::next());
621
622            let two_pow_k = F::from(1 << K);
623
624            // shifted_word = word * 2^{K-s}
625            //              = word * 2^K * inv_two_pow_s
626            Constraints::with_selector(
627                q_bitshift,
628                Some(word * two_pow_k * inv_two_pow_s - shifted_word),
629            )
630        });
631
632        config
633    }
634}

647    ) -> Self {
648        let table_range_check_tag = meta.lookup_table_column();
649        Self::configure_with_tag(meta, running_sum, table_idx, table_range_check_tag)
650    }

652    /// Load the generator table into the circuit.
653    ///
654    /// | table_idx |     table_x    |     table_y    | table_range_check_tag |
655    /// -------------------------------------------------------------------
656    /// |     0     |    X(S\[0\])   |    Y(S\[0\])   |           0           |
657    /// |     1     |    X(S\[1\])   |    Y(S\[1\])   |           0           |
658    /// |    ...    |      ...       |       ...      |           0           |
659    /// |   2^10-1  | X(S\[2^10-1\]) | Y(S\[2^10-1\]) |           0           |
660    /// |     0     |    X(S\[0\])   |    Y(S\[0\])   |           4           |
661    /// |     1     |    X(S\[1\])   |    Y(S\[1\])   |           4           |
662    /// |    ...    |       ...      |       ...      |           4           |
663    /// |   2^4-1   | X(S\[2^4-1\])  | Y(S\[2^4-1\])  |           4           |
664    /// |     0     |    X(S\[0\])   |    Y(S\[0\])   |           5           |
665    /// |     1     |    X(S\[1\])   |    Y(S\[1\])   |           5           |
666    /// |    ...    |       ...      |       ...      |           5           |
667    /// |   2^5-1   | X(S\[2^5-1\])  | Y(S\[2^5-1\])  |           5           |
668    ///
669    /// # Rationale
670    ///
671    /// While it is possible to avoid duplicated rows by using tags and storing them in an advice
672    /// column, as done in the spread table of the SHA-256 gadget, applying this technique to the
673    /// range check lookup table would significantly increase complexity.
674    ///
675    /// Indeed, in the range check lookup tables, tags are currently computed on the fly and are not
676    /// stored in advice columns. Adopting the no duplicated rows approach would require modifying
677    /// the lookup chip to include an advice column for tags and updating the public API accordingly.
678    /// This tag column would need to be populated each time a range check lookup is performed.
679    /// Additionally, the current lookup range check implementation involves combinations of
680    /// multiple lookups, further complicating this approach.
681    ///
682    /// The performance benefit would be minimal: our current table has 1072 rows (2^10 + 2^4 + 2^5),
683    /// and switching to the no duplicated-rows method would reduce it only slightly to 1024 rows
684    /// (2^10), while adding more logic and infrastructure to manage the tags.
685    ///
686    /// Therefore, we retain the simpler duplicated-rows format for clarity and maintainability.
687    fn load(
688        &self,
689        generator_table_config: &GeneratorTableConfig,
690        layouter: &mut impl Layouter<pallas::Base>,
691    ) -> Result<(), Error> {
692        layouter.assign_table(
693            || "generator_table",
694            |mut table| {
695                for (index, (x, y)) in SINSEMILLA_S.iter().enumerate() {
696                    table.assign_cell(
697                        || "table_idx",
698                        generator_table_config.table_idx,
699                        index,
700                        || Value::known(pallas::Base::from(index as u64)),
701                    )?;
702                    table.assign_cell(
703                        || "table_x",
704                        generator_table_config.table_x,
705                        index,
706                        || Value::known(*x),
707                    )?;
708                    table.assign_cell(
709                        || "table_y",
710                        generator_table_config.table_y,
711                        index,
712                        || Value::known(*y),
713                    )?;
714
715                    table.assign_cell(
716                        || "table_range_check_tag",
717                        self.table_range_check_tag,
718                        index,
719                        || Value::known(pallas::Base::zero()),
720                    )?;
721                    if index < (1 << 4) {
722                        let new_index = index + (1 << sinsemilla::K);
723                        table.assign_cell(
724                            || "table_idx",
725                            generator_table_config.table_idx,
726                            new_index,
727                            || Value::known(pallas::Base::from(index as u64)),
728                        )?;
729                        table.assign_cell(
730                            || "table_x",
731                            generator_table_config.table_x,
732                            new_index,
733                            || Value::known(*x),
734                        )?;
735                        table.assign_cell(
736                            || "table_y",
737                            generator_table_config.table_y,
738                            new_index,
739                            || Value::known(*y),
740                        )?;
741                        table.assign_cell(
742                            || "table_range_check_tag",
743                            self.table_range_check_tag,
744                            new_index,
745                            || Value::known(pallas::Base::from(4_u64)),
746                        )?;
747                    }
748                    if index < (1 << 5) {
749                        let new_index = index + (1 << 10) + (1 << 4);
750                        table.assign_cell(
751                            || "table_idx",
752                            generator_table_config.table_idx,
753                            new_index,
754                            || Value::known(pallas::Base::from(index as u64)),
755                        )?;
756                        table.assign_cell(
757                            || "table_x",
758                            generator_table_config.table_x,
759                            new_index,
760                            || Value::known(*x),
761                        )?;
762                        table.assign_cell(
763                            || "table_y",
764                            generator_table_config.table_y,
765                            new_index,
766                            || Value::known(*y),
767                        )?;
768                        table.assign_cell(
769                            || "table_range_check_tag",
770                            self.table_range_check_tag,
771                            new_index,
772                            || Value::known(pallas::Base::from(5_u64)),
773                        )?;
774                    }
775                }
776                Ok(())
777            },
778        )
779    }

783    // This is only used in testing for now, since the Sinsemilla chip provides a pre-loaded table

785    fn load_range_check_table(&self, layouter: &mut impl Layouter<F>) -> Result<(), Error> {
786        layouter.assign_table(
787            || "table_idx",
788            |mut table| {
789                let mut assign_cells =
790                    |offset: usize, table_size, value: u64| -> Result<usize, Error> {
791                        for index in 0..table_size {
792                            let new_index = index + offset;
793                            table.assign_cell(
794                                || "table_idx",
795                                self.base.table_idx,
796                                new_index,
797                                || Value::known(F::from(index as u64)),
798                            )?;
799                            table.assign_cell(
800                                || "table_range_check_tag",
801                                self.table_range_check_tag,
802                                new_index,
803                                || Value::known(F::from(value)),
804                            )?;
805                        }
806                        Ok(offset + table_size)
807                    };
808
809                // We generate the row values lazily (we only need them during keygen).
810                let mut offset = 0;
811
812                //annotation: &'v (dyn Fn() -> String + 'v),
813                //column: TableColumn,
814                //offset: usize,
815                //to: &'v mut (dyn FnMut() -> Value<Assigned<F>> + 'v),
816
817                offset = assign_cells(offset, 1 << K, 0)?;
818                offset = assign_cells(offset, 1 << 4, 4)?;
819                assign_cells(offset, 1 << 5, 5)?;
820
821                Ok(())
822            },
823        )
824    }

826    /// Constrain `x` to be a NUM_BITS word.
827    ///
828    /// `element` must have been assigned to `self.running_sum` at offset 0.
829    fn short_range_check(
830        &self,
831        region: &mut Region<'_, F>,
832        element: AssignedCell<F, F>,
833        num_bits: usize,
834    ) -> Result<(), Error> {
835        match num_bits {
836            4 => {
837                self.base.q_lookup.enable(region, 0)?;
838                self.q_range_check_4.enable(region, 0)?;
839                Ok(())
840            }
841
842            5 => {
843                self.base.q_lookup.enable(region, 0)?;
844                self.q_range_check_5.enable(region, 0)?;
845                Ok(())
846            }
847
848            _ => self.base.short_range_check(region, element, num_bits),
849        }
850    }

898    impl<F: PrimeFieldBits, Lookup: LookupRangeCheck<F, K> + std::clone::Clone> Circuit<F>
899        for MyLookupCircuit<F, Lookup>
900    {
901        type Config = Lookup;
902        type FloorPlanner = SimpleFloorPlanner;
903
904        fn without_witnesses(&self) -> Self {
905            MyLookupCircuit::new(self.num_words)
906        }
907
908        fn configure(meta: &mut ConstraintSystem<F>) -> Self::Config {
909            let running_sum = meta.advice_column();
910            let table_idx = meta.lookup_table_column();
911            let constants = meta.fixed_column();
912            meta.enable_constant(constants);
913
914            Lookup::configure(meta, running_sum, table_idx)
915        }
916
917        fn synthesize(
918            &self,
919            config: Self::Config,
920            mut layouter: impl Layouter<F>,
921        ) -> Result<(), Error> {
922            // Load table_idx
923            config.load_range_check_table(&mut layouter)?;
924
925            // Lookup constraining element to be no longer than num_words * K bits.
926            let elements_and_expected_final_zs = [
927                (F::from((1 << (self.num_words * K)) - 1), F::ZERO, true), // a word that is within self.num_words * K bits long
928                (F::from(1 << (self.num_words * K)), F::ONE, false), // a word that is just over self.num_words * K bits long
929            ];
930
931            fn expected_zs<F: PrimeFieldBits, const K: usize>(
932                element: F,
933                num_words: usize,
934            ) -> Vec<F> {
935                let chunks = {
936                    element
937                        .to_le_bits()
938                        .iter()
939                        .by_vals()
940                        .take(num_words * K)
941                        .collect::<Vec<_>>()
942                        .chunks_exact(K)
943                        .map(|chunk| F::from(lebs2ip::<K>(chunk.try_into().unwrap())))
944                        .collect::<Vec<_>>()
945                };
946                let expected_zs = {
947                    let inv_two_pow_k = F::from(1 << K).invert().unwrap();
948                    chunks.iter().fold(vec![element], |mut zs, a_i| {
949                        // z_{i + 1} = (z_i - a_i) / 2^{K}
950                        let z = (zs[zs.len() - 1] - a_i) * inv_two_pow_k;
951                        zs.push(z);
952                        zs
953                    })
954                };
955                expected_zs
956            }
957
958            for (element, expected_final_z, strict) in elements_and_expected_final_zs.iter() {
959                let expected_zs = expected_zs::<F, K>(*element, self.num_words);
960
961                let zs = config.witness_check(
962                    layouter.namespace(|| format!("Lookup {:?}", self.num_words)),
963                    Value::known(*element),
964                    self.num_words,
965                    *strict,
966                )?;
967
968                assert_eq!(*expected_zs.last().unwrap(), *expected_final_z);
969
970                for (expected_z, z) in expected_zs.into_iter().zip(zs.iter()) {
971                    z.value().assert_if_known(|z| &&expected_z == z);
972                }
973            }
974            Ok(())
975        }
976    }
