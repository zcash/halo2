14    // Decompose `A,B,C,D` words
15    // (2, 11, 9, 10)-bit chunks
16    #[allow(clippy::too_many_arguments)]
17    pub fn s_decompose_abcd(
18        s_decompose_abcd: Expression<F>,
19        a: Expression<F>,
20        spread_a: Expression<F>,
21        b: Expression<F>,
22        spread_b: Expression<F>,
23        tag_b: Expression<F>,
24        c_lo: Expression<F>,
25        spread_c_lo: Expression<F>,
26        c_mid: Expression<F>,
27        spread_c_mid: Expression<F>,
28        c_hi: Expression<F>,
29        spread_c_hi: Expression<F>,
30        d: Expression<F>,
31        spread_d: Expression<F>,
32        tag_d: Expression<F>,
33        word_lo: Expression<F>,
34        spread_word_lo: Expression<F>,
35        word_hi: Expression<F>,
36        spread_word_hi: Expression<F>,
37    ) -> Constraints<
38        F,
39        (&'static str, Expression<F>),
40        impl Iterator<Item = (&'static str, Expression<F>)>,
41    > {
42        let check_spread_and_range =
43            Gate::three_bit_spread_and_range(c_lo.clone(), spread_c_lo.clone())
44                .chain(Gate::three_bit_spread_and_range(
45                    c_mid.clone(),
46                    spread_c_mid.clone(),
47                ))
48                .chain(Gate::three_bit_spread_and_range(
49                    c_hi.clone(),
50                    spread_c_hi.clone(),
51                ))
52                .chain(Gate::two_bit_spread_and_range(a.clone(), spread_a.clone()));
53        let range_check_tag_b = Gate::range_check(tag_b, 0, 2);
54        let range_check_tag_d = Gate::range_check(tag_d, 0, 1);
55        let dense_check = a
56            + b * F::from(1 << 2)
57            + c_lo * F::from(1 << 13)
58            + c_mid * F::from(1 << 16)
59            + c_hi * F::from(1 << 19)
60            + d * F::from(1 << 22)
61            + word_lo * (-F::ONE)
62            + word_hi * F::from(1 << 16) * (-F::ONE);
63        let spread_check = spread_a
64            + spread_b * F::from(1 << 4)
65            + spread_c_lo * F::from(1 << 26)
66            + spread_c_mid * F::from(1 << 32)
67            + spread_c_hi * F::from(1 << 38)
68            + spread_d * F::from(1 << 44)
69            + spread_word_lo * (-F::ONE)
70            + spread_word_hi * F::from(1 << 32) * (-F::ONE);
71
72        Constraints::with_selector(
73            s_decompose_abcd,
74            check_spread_and_range
75                .chain(Some(("range_check_tag_b", range_check_tag_b)))
76                .chain(Some(("range_check_tag_d", range_check_tag_d)))
77                .chain(Some(("dense_check", dense_check)))
78                .chain(Some(("spread_check", spread_check))),
79        )
80    }

135            + spread_a_hi * F::from(1 << 6)
136            + spread_b_lo * F::from(1 << 12)
137            + spread_b_hi * F::from(1 << 16)
138            + spread_c * F::from(1 << 22)

153    // s_upper_sigma_0 on abcd words
154    // (2, 11, 9, 10)-bit chunks
155    #[allow(clippy::too_many_arguments)]
156    pub fn s_upper_sigma_0(
157        s_upper_sigma_0: Expression<F>,
158        spread_r0_even: Expression<F>,
159        spread_r0_odd: Expression<F>,
160        spread_r1_even: Expression<F>,
161        spread_r1_odd: Expression<F>,
162        spread_a: Expression<F>,
163        spread_b: Expression<F>,
164        spread_c_lo: Expression<F>,
165        spread_c_mid: Expression<F>,
166        spread_c_hi: Expression<F>,
167        spread_d: Expression<F>,
168    ) -> Option<(&'static str, Expression<F>)> {
169        let spread_witness = spread_r0_even
170            + spread_r0_odd * F::from(2)
171            + (spread_r1_even + spread_r1_odd * F::from(2)) * F::from(1 << 32);
172        let xor_0 = spread_b.clone()
173            + spread_c_lo.clone() * F::from(1 << 22)
174            + spread_c_mid.clone() * F::from(1 << 28)
175            + spread_c_hi.clone() * F::from(1 << 34)
176            + spread_d.clone() * F::from(1 << 40)
177            + spread_a.clone() * F::from(1 << 60);
178        let xor_1 = spread_c_lo.clone()
179            + spread_c_mid.clone() * F::from(1 << 6)
180            + spread_c_hi.clone() * F::from(1 << 12)
181            + spread_d.clone() * F::from(1 << 18)
182            + spread_a.clone() * F::from(1 << 38)
183            + spread_b.clone() * F::from(1 << 42);
184        let xor_2 = spread_d
185            + spread_a * F::from(1 << 20)
186            + spread_b * F::from(1 << 24)
187            + spread_c_lo * F::from(1 << 46)
188            + spread_c_mid * F::from(1 << 52)
189            + spread_c_hi * F::from(1 << 58);
190        let xor = xor_0 + xor_1 + xor_2;
191        let check = spread_witness + (xor * -F::ONE);
192
193        Some(("s_upper_sigma_0", s_upper_sigma_0 * check))
194    }

199    pub fn s_upper_sigma_1(
200        s_upper_sigma_1: Expression<F>,
201        spread_r0_even: Expression<F>,
202        spread_r0_odd: Expression<F>,
203        spread_r1_even: Expression<F>,
204        spread_r1_odd: Expression<F>,
205        spread_a_lo: Expression<F>,
206        spread_a_hi: Expression<F>,
207        spread_b_lo: Expression<F>,
208        spread_b_hi: Expression<F>,
209        spread_c: Expression<F>,
210        spread_d: Expression<F>,
211    ) -> Option<(&'static str, Expression<F>)> {
212        let spread_witness = spread_r0_even
213            + spread_r0_odd * F::from(2)
214            + (spread_r1_even + spread_r1_odd * F::from(2)) * F::from(1 << 32);
215
216        let xor_0 = spread_b_lo.clone()
217            + spread_b_hi.clone() * F::from(1 << 4)
218            + spread_c.clone() * F::from(1 << 10)
219            + spread_d.clone() * F::from(1 << 38)
220            + spread_a_lo.clone() * F::from(1 << 52)
221            + spread_a_hi.clone() * F::from(1 << 58);
222        let xor_1 = spread_c.clone()
223            + spread_d.clone() * F::from(1 << 28)
224            + spread_a_lo.clone() * F::from(1 << 42)
225            + spread_a_hi.clone() * F::from(1 << 48)
226            + spread_b_lo.clone() * F::from(1 << 54)
227            + spread_b_hi.clone() * F::from(1 << 58);
228        let xor_2 = spread_d
229            + spread_a_lo * F::from(1 << 14)
230            + spread_a_hi * F::from(1 << 20)
231            + spread_b_lo * F::from(1 << 26)
232            + spread_b_hi * F::from(1 << 30)
233            + spread_c * F::from(1 << 36);
234        let xor = xor_0 + xor_1 + xor_2;
235        let check = spread_witness + (xor * -F::ONE);
236
237        Some(("s_upper_sigma_1", s_upper_sigma_1 * check))
238    }

240    // First part of choice gate on (E, F, G), E ∧ F
241    #[allow(clippy::too_many_arguments)]
242    pub fn s_ch(
243        s_ch: Expression<F>,
244        spread_p0_even: Expression<F>,
245        spread_p0_odd: Expression<F>,
246        spread_p1_even: Expression<F>,
247        spread_p1_odd: Expression<F>,
248        spread_e_lo: Expression<F>,
249        spread_e_hi: Expression<F>,
250        spread_f_lo: Expression<F>,
251        spread_f_hi: Expression<F>,
252    ) -> Option<(&'static str, Expression<F>)> {
253        let lhs_lo = spread_e_lo + spread_f_lo;
254        let lhs_hi = spread_e_hi + spread_f_hi;
255        let lhs = lhs_lo + lhs_hi * F::from(1 << 32);
256
257        let rhs_even = spread_p0_even + spread_p1_even * F::from(1 << 32);
258        let rhs_odd = spread_p0_odd + spread_p1_odd * F::from(1 << 32);
259        let rhs = rhs_even + rhs_odd * F::from(2);
260
261        let check = lhs + rhs * -F::ONE;
262
263        Some(("s_ch", s_ch * check))
264    }

266    // Second part of Choice gate on (E, F, G), ¬E ∧ G
267    #[allow(clippy::too_many_arguments)]
268    pub fn s_ch_neg(
269        s_ch_neg: Expression<F>,
270        spread_q0_even: Expression<F>,
271        spread_q0_odd: Expression<F>,
272        spread_q1_even: Expression<F>,
273        spread_q1_odd: Expression<F>,
274        spread_e_lo: Expression<F>,
275        spread_e_hi: Expression<F>,
276        spread_e_neg_lo: Expression<F>,
277        spread_e_neg_hi: Expression<F>,
278        spread_g_lo: Expression<F>,
279        spread_g_hi: Expression<F>,
280    ) -> Constraints<
281        F,
282        (&'static str, Expression<F>),
283        impl Iterator<Item = (&'static str, Expression<F>)>,
284    > {
285        let neg_check = {
286            let evens = Self::ones() * F::from(MASK_EVEN_32 as u64);
287            // evens - spread_e_lo = spread_e_neg_lo
288            let lo_check = spread_e_neg_lo.clone() + spread_e_lo + (evens.clone() * (-F::ONE));
289            // evens - spread_e_hi = spread_e_neg_hi
290            let hi_check = spread_e_neg_hi.clone() + spread_e_hi + (evens * (-F::ONE));
291
292            std::iter::empty()
293                .chain(Some(("lo_check", lo_check)))
294                .chain(Some(("hi_check", hi_check)))
295        };
296
297        let lhs_lo = spread_e_neg_lo + spread_g_lo;
298        let lhs_hi = spread_e_neg_hi + spread_g_hi;
299        let lhs = lhs_lo + lhs_hi * F::from(1 << 32);
300
301        let rhs_even = spread_q0_even + spread_q1_even * F::from(1 << 32);
302        let rhs_odd = spread_q0_odd + spread_q1_odd * F::from(1 << 32);
303        let rhs = rhs_even + rhs_odd * F::from(2);
304
305        Constraints::with_selector(s_ch_neg, neg_check.chain(Some(("s_ch_neg", lhs - rhs))))
306    }

308    // Majority gate on (A, B, C)
309    #[allow(clippy::too_many_arguments)]
310    pub fn s_maj(
311        s_maj: Expression<F>,
312        spread_m_0_even: Expression<F>,
313        spread_m_0_odd: Expression<F>,
314        spread_m_1_even: Expression<F>,
315        spread_m_1_odd: Expression<F>,
316        spread_a_lo: Expression<F>,
317        spread_a_hi: Expression<F>,
318        spread_b_lo: Expression<F>,
319        spread_b_hi: Expression<F>,
320        spread_c_lo: Expression<F>,
321        spread_c_hi: Expression<F>,
322    ) -> Option<(&'static str, Expression<F>)> {
323        let maj_even = spread_m_0_even + spread_m_1_even * F::from(1 << 32);
324        let maj_odd = spread_m_0_odd + spread_m_1_odd * F::from(1 << 32);
325        let maj = maj_even + maj_odd * F::from(2);
326
327        let a = spread_a_lo + spread_a_hi * F::from(1 << 32);
328        let b = spread_b_lo + spread_b_hi * F::from(1 << 32);
329        let c = spread_c_lo + spread_c_hi * F::from(1 << 32);
330        let sum = a + b + c;
331
332        Some(("maj", s_maj * (sum - maj)))
333    }

335    // s_h_prime to get H' = H + Ch(E, F, G) + s_upper_sigma_1(E) + K + W
336    #[allow(clippy::too_many_arguments)]
337    pub fn s_h_prime(
338        s_h_prime: Expression<F>,
339        h_prime_lo: Expression<F>,
340        h_prime_hi: Expression<F>,
341        h_prime_carry: Expression<F>,
342        sigma_e_lo: Expression<F>,
343        sigma_e_hi: Expression<F>,
344        ch_lo: Expression<F>,
345        ch_hi: Expression<F>,
346        ch_neg_lo: Expression<F>,
347        ch_neg_hi: Expression<F>,
348        h_lo: Expression<F>,
349        h_hi: Expression<F>,
350        k_lo: Expression<F>,
351        k_hi: Expression<F>,
352        w_lo: Expression<F>,
353        w_hi: Expression<F>,
354    ) -> Option<(&'static str, Expression<F>)> {
355        let lo = h_lo + ch_lo + ch_neg_lo + sigma_e_lo + k_lo + w_lo;
356        let hi = h_hi + ch_hi + ch_neg_hi + sigma_e_hi + k_hi + w_hi;
357
358        let sum = lo + hi * F::from(1 << 16);
359        let h_prime = h_prime_lo + h_prime_hi * F::from(1 << 16);
360
361        let check = sum - (h_prime_carry * F::from(1 << 32)) - h_prime;
362
363        Some(("s_h_prime", s_h_prime * check))
364    }

368    pub fn s_a_new(

370        a_new_lo: Expression<F>,

372        a_new_carry: Expression<F>,

382        let sum = lo + hi * F::from(1 << 16);
383        let a_new = a_new_lo + a_new_hi * F::from(1 << 16);
384
385        let check = sum - (a_new_carry * F::from(1 << 32)) - a_new;
386
387        Some(("s_a_new", s_a_new * check))

412    // s_digest on final round
413    #[allow(clippy::too_many_arguments)]
414    pub fn s_digest(
415        s_digest: Expression<F>,
416        lo_0: Expression<F>,
417        hi_0: Expression<F>,
418        word_0: Expression<F>,
419        lo_1: Expression<F>,
420        hi_1: Expression<F>,
421        word_1: Expression<F>,
422        lo_2: Expression<F>,
423        hi_2: Expression<F>,
424        word_2: Expression<F>,
425        lo_3: Expression<F>,
426        hi_3: Expression<F>,
427        word_3: Expression<F>,
428    ) -> impl IntoIterator<Item = Constraint<F>> {
429        let check_lo_hi = |lo: Expression<F>, hi: Expression<F>, word: Expression<F>| {
430            lo + hi * F::from(1 << 16) - word
431        };
432
433        Constraints::with_selector(
434            s_digest,
435            [
436                ("check_lo_hi_0", check_lo_hi(lo_0, hi_0, word_0)),
437                ("check_lo_hi_1", check_lo_hi(lo_1, hi_1, word_1)),
438                ("check_lo_hi_2", check_lo_hi(lo_2, hi_2, word_2)),
439                ("check_lo_hi_3", check_lo_hi(lo_3, hi_3, word_3)),
440            ],
441        )
442    }
