108/// Returns starting row number of a compression round
109pub fn get_round_row(round_idx: RoundIdx) -> usize {
110    match round_idx {
111        RoundIdx::Init => 0,
112        RoundIdx::Main(MainRoundIdx(idx)) => {
113            assert!(idx < 64);
114            idx * SUBREGION_MAIN_WORD
115        }
116    }
117}

221            row + 1,

242            a_5,

370    pub(super) fn assign_upper_sigma_0(
371        &self,
372        region: &mut Region<'_, pallas::Base>,
373        round_idx: MainRoundIdx,
374        word: AbcdVar,
375    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
376        // Rename these here for ease of matching the gates to the specification.
377        let a_3 = self.extras[0];
378        let a_4 = self.extras[1];
379        let a_5 = self.message_schedule;
380
381        let row = get_upper_sigma_0_row(round_idx);
382
383        self.s_upper_sigma_0.enable(region, row)?;
384
385        // Assign `spread_a` and copy constraint
386        word.a
387            .spread
388            .copy_advice(|| "spread_a", region, a_3, row + 1)?;
389        // Assign `spread_b` and copy constraint
390        word.b.spread.copy_advice(|| "spread_b", region, a_5, row)?;
391        // Assign `spread_c_lo` and copy constraint
392        word.c_lo
393            .spread
394            .copy_advice(|| "spread_c_lo", region, a_3, row - 1)?;
395        // Assign `spread_c_mid` and copy constraint
396        word.c_mid
397            .spread
398            .copy_advice(|| "spread_c_mid", region, a_4, row - 1)?;
399        // Assign `spread_c_hi` and copy constraint
400        word.c_hi
401            .spread
402            .copy_advice(|| "spread_c_hi", region, a_4, row + 1)?;
403        // Assign `spread_d` and copy constraint
404        word.d.spread.copy_advice(|| "spread_d", region, a_4, row)?;
405
406        // Calculate R_0^{even}, R_0^{odd}, R_1^{even}, R_1^{odd}
407        let r = word.xor_upper_sigma();
408        let r_0: Value<[bool; 32]> = r.map(|r| r[..32].try_into().unwrap());
409        let r_0_even = r_0.map(even_bits);
410        let r_0_odd = r_0.map(odd_bits);
411
412        let r_1: Value<[bool; 32]> = r.map(|r| r[32..].try_into().unwrap());
413        let r_1_even = r_1.map(even_bits);
414        let r_1_odd = r_1.map(odd_bits);
415
416        self.assign_sigma_outputs(
417            region,
418            &self.lookup,
419            a_3,
420            row,
421            r_0_even,
422            r_0_odd,
423            r_1_even,
424            r_1_odd,
425        )
426    }

512    pub(super) fn assign_ch(
513        &self,
514        region: &mut Region<'_, pallas::Base>,
515        round_idx: MainRoundIdx,
516        spread_halves_e: RoundWordSpread,
517        spread_halves_f: RoundWordSpread,
518    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
519        let a_3 = self.extras[0];
520        let a_4 = self.extras[1];
521
522        let row = get_ch_row(round_idx);
523
524        self.s_ch.enable(region, row)?;
525
526        // Assign and copy spread_e_lo, spread_e_hi
527        spread_halves_e
528            .0
529            .copy_advice(|| "spread_e_lo", region, a_3, row - 1)?;
530        spread_halves_e
531            .1
532            .copy_advice(|| "spread_e_hi", region, a_4, row - 1)?;
533
534        // Assign and copy spread_f_lo, spread_f_hi
535        spread_halves_f
536            .0
537            .copy_advice(|| "spread_f_lo", region, a_3, row + 1)?;
538        spread_halves_f
539            .1
540            .copy_advice(|| "spread_f_hi", region, a_4, row + 1)?;
541
542        let p: Value<[bool; 64]> = spread_halves_e
543            .value()
544            .zip(spread_halves_f.value())
545            .map(|(e, f)| i2lebsp(e + f));
546
547        let p_0: Value<[bool; 32]> = p.map(|p| p[..32].try_into().unwrap());
548        let p_0_even = p_0.map(even_bits);
549        let p_0_odd = p_0.map(odd_bits);
550
551        let p_1: Value<[bool; 32]> = p.map(|p| p[32..].try_into().unwrap());
552        let p_1_even = p_1.map(even_bits);
553        let p_1_odd = p_1.map(odd_bits);
554
555        self.assign_ch_outputs(region, row, p_0_even, p_0_odd, p_1_even, p_1_odd)
556    }

558    pub(super) fn assign_ch_neg(
559        &self,
560        region: &mut Region<'_, pallas::Base>,
561        round_idx: MainRoundIdx,
562        spread_halves_e: RoundWordSpread,
563        spread_halves_g: RoundWordSpread,
564    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
565        let row = get_ch_neg_row(round_idx);
566
567        self.s_ch_neg.enable(region, row)?;
568
569        let a_3 = self.extras[0];
570        let a_4 = self.extras[1];
571        let a_5 = self.message_schedule;
572
573        // Assign and copy spread_e_lo, spread_e_hi
574        spread_halves_e
575            .0
576            .copy_advice(|| "spread_e_lo", region, a_5, row - 1)?;
577        spread_halves_e
578            .1
579            .copy_advice(|| "spread_e_hi", region, a_5, row)?;
580
581        // Assign and copy spread_g_lo, spread_g_hi
582        spread_halves_g
583            .0
584            .copy_advice(|| "spread_g_lo", region, a_3, row + 1)?;
585        spread_halves_g
586            .1
587            .copy_advice(|| "spread_g_hi", region, a_4, row + 1)?;
588
589        // Calculate neg_e_lo
590        let spread_neg_e_lo = spread_halves_e
591            .0
592            .value()
593            .map(|spread_e_lo| negate_spread(spread_e_lo.0));
594        // Assign spread_neg_e_lo
595        AssignedBits::<32>::assign_bits(
596            region,
597            || "spread_neg_e_lo",
598            a_3,
599            row - 1,
600            spread_neg_e_lo,
601        )?;
602
603        // Calculate neg_e_hi
604        let spread_neg_e_hi = spread_halves_e
605            .1
606            .value()
607            .map(|spread_e_hi| negate_spread(spread_e_hi.0));
608        // Assign spread_neg_e_hi
609        AssignedBits::<32>::assign_bits(
610            region,
611            || "spread_neg_e_hi",
612            a_4,
613            row - 1,
614            spread_neg_e_hi,
615        )?;
616
617        let p: Value<[bool; 64]> = {
618            let spread_neg_e = spread_neg_e_lo
619                .zip(spread_neg_e_hi)
620                .map(|(lo, hi)| lebs2ip(&lo) + (1 << 32) * lebs2ip(&hi));
621            spread_neg_e
622                .zip(spread_halves_g.value())
623                .map(|(neg_e, g)| i2lebsp(neg_e + g))
624        };
625
626        let p_0: Value<[bool; 32]> = p.map(|p| p[..32].try_into().unwrap());
627        let p_0_even = p_0.map(even_bits);
628        let p_0_odd = p_0.map(odd_bits);
629
630        let p_1: Value<[bool; 32]> = p.map(|p| p[32..].try_into().unwrap());
631        let p_1_even = p_1.map(even_bits);
632        let p_1_odd = p_1.map(odd_bits);
633
634        self.assign_ch_outputs(region, row, p_0_even, p_0_odd, p_1_even, p_1_odd)
635    }

645    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
646        let a_3 = self.extras[0];
647        let (_even, odd) = self.assign_spread_outputs(
648            region,
649            &self.lookup,
650            a_3,
651            row,
652            r_0_even,
653            r_0_odd,
654            r_1_even,
655            r_1_odd,
656        )?;
657
658        Ok(odd)
659    }

661    pub(super) fn assign_maj(
662        &self,
663        region: &mut Region<'_, pallas::Base>,
664        round_idx: MainRoundIdx,
665        spread_halves_a: RoundWordSpread,
666        spread_halves_b: RoundWordSpread,
667        spread_halves_c: RoundWordSpread,
668    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
669        let a_4 = self.extras[1];
670        let a_5 = self.message_schedule;
671
672        let row = get_maj_row(round_idx);
673
674        self.s_maj.enable(region, row)?;
675
676        // Assign and copy spread_a_lo, spread_a_hi
677        spread_halves_a
678            .0
679            .copy_advice(|| "spread_a_lo", region, a_4, row - 1)?;
680        spread_halves_a
681            .1
682            .copy_advice(|| "spread_a_hi", region, a_5, row - 1)?;
683
684        // Assign and copy spread_b_lo, spread_b_hi
685        spread_halves_b
686            .0
687            .copy_advice(|| "spread_b_lo", region, a_4, row)?;
688        spread_halves_b
689            .1
690            .copy_advice(|| "spread_b_hi", region, a_5, row)?;
691
692        // Assign and copy spread_c_lo, spread_c_hi
693        spread_halves_c
694            .0
695            .copy_advice(|| "spread_c_lo", region, a_4, row + 1)?;
696        spread_halves_c
697            .1
698            .copy_advice(|| "spread_c_hi", region, a_5, row + 1)?;
699
700        let m: Value<[bool; 64]> = spread_halves_a
701            .value()
702            .zip(spread_halves_b.value())
703            .zip(spread_halves_c.value())
704            .map(|((a, b), c)| i2lebsp(a + b + c));
705
706        let m_0: Value<[bool; 32]> = m.map(|m| m[..32].try_into().unwrap());
707        let m_0_even = m_0.map(even_bits);
708        let m_0_odd = m_0.map(odd_bits);
709
710        let m_1: Value<[bool; 32]> = m.map(|m| m[32..].try_into().unwrap());
711        let m_1_even = m_1.map(even_bits);
712        let m_1_odd = m_1.map(odd_bits);
713
714        self.assign_maj_outputs(region, row, m_0_even, m_0_odd, m_1_even, m_1_odd)
715    }

717    // s_h_prime to get H' = H + Ch(E, F, G) + s_upper_sigma_1(E) + K + W
718    #[allow(clippy::too_many_arguments)]
719    pub(super) fn assign_h_prime(
720        &self,
721        region: &mut Region<'_, pallas::Base>,
722        round_idx: MainRoundIdx,
723        h: RoundWordDense,
724        ch: (AssignedBits<16>, AssignedBits<16>),
725        ch_neg: (AssignedBits<16>, AssignedBits<16>),
726        sigma_1: (AssignedBits<16>, AssignedBits<16>),
727        k: u32,
728        w: &(AssignedBits<16>, AssignedBits<16>),
729    ) -> Result<RoundWordDense, Error> {
730        let row = get_h_prime_row(round_idx);
731        self.s_h_prime.enable(region, row)?;
732
733        let a_4 = self.extras[1];
734        let a_5 = self.message_schedule;
735        let a_6 = self.extras[2];
736        let a_7 = self.extras[3];
737        let a_8 = self.extras[4];
738        let a_9 = self.extras[5];
739
740        // Assign and copy h
741        h.0.copy_advice(|| "h_lo", region, a_7, row - 1)?;
742        h.1.copy_advice(|| "h_hi", region, a_7, row)?;
743
744        // Assign and copy sigma_1
745        sigma_1.0.copy_advice(|| "sigma_1_lo", region, a_4, row)?;
746        sigma_1.1.copy_advice(|| "sigma_1_hi", region, a_5, row)?;
747
748        // Assign k
749        let k: [bool; 32] = i2lebsp(k.into());
750        let k_lo: [bool; 16] = k[..16].try_into().unwrap();
751        let k_hi: [bool; 16] = k[16..].try_into().unwrap();
752        {
753            AssignedBits::<16>::assign_bits(region, || "k_lo", a_6, row - 1, Value::known(k_lo))?;
754            AssignedBits::<16>::assign_bits(region, || "k_hi", a_6, row, Value::known(k_hi))?;
755        }
756
757        // Assign and copy w
758        w.0.copy_advice(|| "w_lo", region, a_8, row - 1)?;
759        w.1.copy_advice(|| "w_hi", region, a_8, row)?;
760
761        // Assign and copy ch
762        ch.1.copy_advice(|| "ch_neg_hi", region, a_6, row + 1)?;
763
764        // Assign and copy ch_neg
765        ch_neg.0.copy_advice(|| "ch_neg_lo", region, a_5, row - 1)?;
766        ch_neg.1.copy_advice(|| "ch_neg_hi", region, a_5, row + 1)?;
767
768        // Assign h_prime_lo, h_prime_hi, h_prime_carry
769        {
770            let (h_prime, h_prime_carry) = sum_with_carry(vec![
771                (h.0.value_u16(), h.1.value_u16()),
772                (ch.0.value_u16(), ch.1.value_u16()),
773                (ch_neg.0.value_u16(), ch_neg.1.value_u16()),
774                (sigma_1.0.value_u16(), sigma_1.1.value_u16()),
775                (
776                    Value::known(lebs2ip(&k_lo) as u16),
777                    Value::known(lebs2ip(&k_hi) as u16),
778                ),
779                (w.0.value_u16(), w.1.value_u16()),
780            ]);
781
782            region.assign_advice(
783                || "h_prime_carry",
784                a_9,
785                row + 1,
786                || h_prime_carry.map(pallas::Base::from),
787            )?;
788
789            let h_prime: Value<[bool; 32]> = h_prime.map(|w| i2lebsp(w.into()));
790            let h_prime_lo: Value<[bool; 16]> = h_prime.map(|w| w[..16].try_into().unwrap());
791            let h_prime_hi: Value<[bool; 16]> = h_prime.map(|w| w[16..].try_into().unwrap());
792
793            let h_prime_lo =
794                AssignedBits::<16>::assign_bits(region, || "h_prime_lo", a_7, row + 1, h_prime_lo)?;
795            let h_prime_hi =
796                AssignedBits::<16>::assign_bits(region, || "h_prime_hi", a_8, row + 1, h_prime_hi)?;
797
798            Ok((h_prime_lo, h_prime_hi).into())
799        }
800    }

893    pub fn assign_word_halves_dense(
894        &self,
895        region: &mut Region<'_, pallas::Base>,
896        lo_row: usize,
897        lo_col: Column<Advice>,
898        hi_row: usize,
899        hi_col: Column<Advice>,

901    ) -> Result<RoundWordDense, Error> {
902        let word: Value<[bool; 32]> = word.map(|w| i2lebsp(w.into()));
903
904        let lo = {
905            let lo: Value<[bool; 16]> = word.map(|w| w[..16].try_into().unwrap());

907        };
908

911            AssignedBits::<16>::assign_bits(region, || "hi", hi_col, hi_row, hi)?
912        };
913
914        Ok((lo, hi).into())
915    }

917    // Assign hi and lo halves for both dense and spread versions of a word
918    #[allow(clippy::type_complexity)]
919    pub fn assign_word_halves(
920        &self,
921        region: &mut Region<'_, pallas::Base>,
922        row: usize,
923        word: Value<u32>,
924    ) -> Result<(RoundWordDense, RoundWordSpread), Error> {
925        // Rename these here for ease of matching the gates to the specification.
926        let a_7 = self.extras[3];
927        let a_8 = self.extras[4];
928
929        let word: Value<[bool; 32]> = word.map(|w| i2lebsp(w.into()));
930        let lo: Value<[bool; 16]> = word.map(|w| w[..16].try_into().unwrap());
931        let hi: Value<[bool; 16]> = word.map(|w| w[16..].try_into().unwrap());
932
933        let w_lo = SpreadVar::without_lookup(region, a_7, row, a_8, row, lo.map(SpreadWord::new))?;
934        let w_hi =
935            SpreadVar::without_lookup(region, a_7, row + 1, a_8, row + 1, hi.map(SpreadWord::new))?;
936
937        Ok((
938            (w_lo.dense, w_hi.dense).into(),
939            (w_lo.spread, w_hi.spread).into(),
940        ))

944#[allow(clippy::many_single_char_names)]
945pub fn match_state(
946    state: State,
947) -> (
948    RoundWordA,
949    RoundWord,
950    RoundWord,
951    RoundWordDense,
952    RoundWordE,
953    RoundWord,
954    RoundWord,
955    RoundWordDense,
956) {
957    let a = match state.a {
958        Some(StateWord::A(a)) => a,
959        _ => unreachable!(),
960    };
961    let b = match state.b {
962        Some(StateWord::B(b)) => b,
963        _ => unreachable!(),
964    };
965    let c = match state.c {
966        Some(StateWord::C(c)) => c,
967        _ => unreachable!(),
968    };
969    let d = match state.d {
970        Some(StateWord::D(d)) => d,
971        _ => unreachable!(),
972    };
973    let e = match state.e {
974        Some(StateWord::E(e)) => e,
975        _ => unreachable!(),
976    };
977    let f = match state.f {
978        Some(StateWord::F(f)) => f,
979        _ => unreachable!(),
980    };
981    let g = match state.g {
982        Some(StateWord::G(g)) => g,
983        _ => unreachable!(),
984    };
985    let h = match state.h {
986        Some(StateWord::H(h)) => h,
987        _ => unreachable!(),
988    };
989
990    (a, b, c, d, e, f, g, h)
