62impl MessageScheduleConfig {
63    /// Configures the message schedule.
64    ///
65    /// `message_schedule` is the column into which the message schedule will be placed.
66    /// The caller must create appropriate permutations in order to load schedule words
67    /// into the compression rounds.
68    ///
69    /// `extras` contains columns that the message schedule will only use for internal
70    /// gates, and will not place any constraints on (such as lookup constraints) outside
71    /// itself.
72    #[allow(clippy::many_single_char_names)]
73    pub(super) fn configure(
74        meta: &mut ConstraintSystem<pallas::Base>,
75        lookup: SpreadInputs,
76        message_schedule: Column<Advice>,
77        extras: [Column<Advice>; 6],
78    ) -> Self {
79        // Create fixed columns for the selectors we will require.
80        let s_word = meta.selector();
81        let s_decompose_0 = meta.selector();
82        let s_decompose_1 = meta.selector();
83        let s_decompose_2 = meta.selector();
84        let s_decompose_3 = meta.selector();
85        let s_lower_sigma_0 = meta.selector();
86        let s_lower_sigma_1 = meta.selector();
87        let s_lower_sigma_0_v2 = meta.selector();
88        let s_lower_sigma_1_v2 = meta.selector();
89
90        // Rename these here for ease of matching the gates to the specification.
91        let a_0 = lookup.tag;
92        let a_1 = lookup.dense;
93        let a_2 = lookup.spread;
94        let a_3 = extras[0];
95        let a_4 = extras[1];
96        let a_5 = message_schedule;
97        let a_6 = extras[2];
98        let a_7 = extras[3];
99        let a_8 = extras[4];
100        let a_9 = extras[5];
101
102        // s_word for W_[16..64]
103        meta.create_gate("s_word for W_[16..64]", |meta| {
104            let s_word = meta.query_selector(s_word);
105
106            let sigma_0_lo = meta.query_advice(a_6, Rotation::prev());
107            let sigma_0_hi = meta.query_advice(a_6, Rotation::cur());
108
109            let sigma_1_lo = meta.query_advice(a_7, Rotation::prev());
110            let sigma_1_hi = meta.query_advice(a_7, Rotation::cur());
111
112            let w_minus_9_lo = meta.query_advice(a_8, Rotation::prev());
113            let w_minus_9_hi = meta.query_advice(a_8, Rotation::cur());
114
115            let w_minus_16_lo = meta.query_advice(a_3, Rotation::prev());
116            let w_minus_16_hi = meta.query_advice(a_4, Rotation::prev());
117
118            let word = meta.query_advice(a_5, Rotation::cur());
119            let carry = meta.query_advice(a_9, Rotation::cur());
120
121            ScheduleGate::s_word(
122                s_word,
123                sigma_0_lo,
124                sigma_0_hi,
125                sigma_1_lo,
126                sigma_1_hi,
127                w_minus_9_lo,
128                w_minus_9_hi,
129                w_minus_16_lo,
130                w_minus_16_hi,
131                word,
132                carry,
133            )
134        });
135
136        // s_decompose_0 for all words
137        meta.create_gate("s_decompose_0", |meta| {
138            let s_decompose_0 = meta.query_selector(s_decompose_0);
139            let lo = meta.query_advice(a_3, Rotation::cur());
140            let hi = meta.query_advice(a_4, Rotation::cur());
141            let word = meta.query_advice(a_5, Rotation::cur());
142
143            ScheduleGate::s_decompose_0(s_decompose_0, lo, hi, word)
144        });
145
146        // s_decompose_1 for W_[1..14]
147        // (3, 4, 11, 14)-bit chunks
148        meta.create_gate("s_decompose_1", |meta| {
149            let s_decompose_1 = meta.query_selector(s_decompose_1);
150            let a = meta.query_advice(a_3, Rotation::next()); // 3-bit chunk
151            let b = meta.query_advice(a_4, Rotation::next()); // 4-bit chunk
152            let c = meta.query_advice(a_1, Rotation::next()); // 11-bit chunk
153            let tag_c = meta.query_advice(a_0, Rotation::next());
154            let d = meta.query_advice(a_1, Rotation::cur()); // 14-bit chunk
155            let tag_d = meta.query_advice(a_0, Rotation::cur());
156            let word = meta.query_advice(a_5, Rotation::cur());
157
158            ScheduleGate::s_decompose_1(s_decompose_1, a, b, c, tag_c, d, tag_d, word)
159        });
160
161        // s_decompose_2 for W_[14..49]
162        // (3, 4, 3, 7, 1, 1, 13)-bit chunks
163        meta.create_gate("s_decompose_2", |meta| {
164            let s_decompose_2 = meta.query_selector(s_decompose_2);
165            let a = meta.query_advice(a_3, Rotation::prev()); // 3-bit chunk
166            let b = meta.query_advice(a_1, Rotation::next()); // 4-bit chunk
167            let c = meta.query_advice(a_4, Rotation::prev()); // 3-bit chunk
168            let d = meta.query_advice(a_1, Rotation::cur()); // 7-bit chunk
169            let tag_d = meta.query_advice(a_0, Rotation::cur());
170            let e = meta.query_advice(a_3, Rotation::next()); // 1-bit chunk
171            let f = meta.query_advice(a_4, Rotation::next()); // 1-bit chunk
172            let g = meta.query_advice(a_1, Rotation::prev()); // 13-bit chunk
173            let tag_g = meta.query_advice(a_0, Rotation::prev());
174            let word = meta.query_advice(a_5, Rotation::cur());
175
176            ScheduleGate::s_decompose_2(s_decompose_2, a, b, c, d, tag_d, e, f, g, tag_g, word)
177        });
178
179        // s_decompose_3 for W_49 to W_61
180        // (10, 7, 2, 13)-bit chunks
181        meta.create_gate("s_decompose_3", |meta| {
182            let s_decompose_3 = meta.query_selector(s_decompose_3);
183            let a = meta.query_advice(a_1, Rotation::next()); // 10-bit chunk
184            let tag_a = meta.query_advice(a_0, Rotation::next());
185            let b = meta.query_advice(a_4, Rotation::next()); // 7-bit chunk
186            let c = meta.query_advice(a_3, Rotation::next()); // 2-bit chunk
187            let d = meta.query_advice(a_1, Rotation::cur()); // 13-bit chunk
188            let tag_d = meta.query_advice(a_0, Rotation::cur());
189            let word = meta.query_advice(a_5, Rotation::cur());
190
191            ScheduleGate::s_decompose_3(s_decompose_3, a, tag_a, b, c, d, tag_d, word)
192        });
193
194        // sigma_0 v1 on W_[1..14]
195        // (3, 4, 11, 14)-bit chunks
196        meta.create_gate("sigma_0 v1", |meta| {
197            ScheduleGate::s_lower_sigma_0(
198                meta.query_selector(s_lower_sigma_0),
199                meta.query_advice(a_2, Rotation::prev()), // spread_r0_even
200                meta.query_advice(a_2, Rotation::cur()),  // spread_r0_odd
201                meta.query_advice(a_2, Rotation::next()), // spread_r1_even
202                meta.query_advice(a_3, Rotation::cur()),  // spread_r1_odd
203                meta.query_advice(a_5, Rotation::next()), // a
204                meta.query_advice(a_6, Rotation::next()), // spread_a
205                meta.query_advice(a_6, Rotation::cur()),  // b
206                meta.query_advice(a_3, Rotation::prev()), // b_lo
207                meta.query_advice(a_4, Rotation::prev()), // spread_b_lo
208                meta.query_advice(a_5, Rotation::prev()), // b_hi
209                meta.query_advice(a_6, Rotation::prev()), // spread_b_hi
210                meta.query_advice(a_4, Rotation::cur()),  // spread_c
211                meta.query_advice(a_5, Rotation::cur()),  // spread_d
212            )
213        });
214
215        // sigma_0 v2 on W_[14..49]
216        // (3, 4, 3, 7, 1, 1, 13)-bit chunks
217        meta.create_gate("sigma_0 v2", |meta| {
218            ScheduleGate::s_lower_sigma_0_v2(
219                meta.query_selector(s_lower_sigma_0_v2),
220                meta.query_advice(a_2, Rotation::prev()), // spread_r0_even
221                meta.query_advice(a_2, Rotation::cur()),  // spread_r0_odd
222                meta.query_advice(a_2, Rotation::next()), // spread_r1_even
223                meta.query_advice(a_3, Rotation::cur()),  // spread_r1_odd
224                meta.query_advice(a_3, Rotation::next()), // a
225                meta.query_advice(a_4, Rotation::next()), // spread_a
226                meta.query_advice(a_6, Rotation::cur()),  // b
227                meta.query_advice(a_3, Rotation::prev()), // b_lo
228                meta.query_advice(a_4, Rotation::prev()), // spread_b_lo
229                meta.query_advice(a_5, Rotation::prev()), // b_hi
230                meta.query_advice(a_6, Rotation::prev()), // spread_b_hi
231                meta.query_advice(a_5, Rotation::next()), // c
232                meta.query_advice(a_6, Rotation::next()), // spread_c
233                meta.query_advice(a_4, Rotation::cur()),  // spread_d
234                meta.query_advice(a_7, Rotation::cur()),  // spread_e
235                meta.query_advice(a_7, Rotation::next()), // spread_f
236                meta.query_advice(a_5, Rotation::cur()),  // spread_g
237            )
238        });

241        // (3, 4, 3, 7, 1, 1, 13)-bit chunks
242        meta.create_gate("sigma_1 v2", |meta| {
243            ScheduleGate::s_lower_sigma_1_v2(
244                meta.query_selector(s_lower_sigma_1_v2),
245                meta.query_advice(a_2, Rotation::prev()), // spread_r0_even
246                meta.query_advice(a_2, Rotation::cur()),  // spread_r0_odd
247                meta.query_advice(a_2, Rotation::next()), // spread_r1_even
248                meta.query_advice(a_3, Rotation::cur()),  // spread_r1_odd
249                meta.query_advice(a_3, Rotation::next()), // a
250                meta.query_advice(a_4, Rotation::next()), // spread_a
251                meta.query_advice(a_6, Rotation::cur()),  // b
252                meta.query_advice(a_3, Rotation::prev()), // b_lo
253                meta.query_advice(a_4, Rotation::prev()), // spread_b_lo
254                meta.query_advice(a_5, Rotation::prev()), // b_hi
255                meta.query_advice(a_6, Rotation::prev()), // spread_b_hi
256                meta.query_advice(a_5, Rotation::next()), // c
257                meta.query_advice(a_6, Rotation::next()), // spread_c
258                meta.query_advice(a_4, Rotation::cur()),  // spread_d
259                meta.query_advice(a_7, Rotation::cur()),  // spread_e

273                meta.query_advice(a_3, Rotation::cur()),  // spread_r1_odd
274                meta.query_advice(a_4, Rotation::cur()),  // spread_a
275                meta.query_advice(a_6, Rotation::cur()),  // b
276                meta.query_advice(a_3, Rotation::prev()), // b_lo
277                meta.query_advice(a_4, Rotation::prev()), // spread_b_lo
278                meta.query_advice(a_5, Rotation::prev()), // b_mid
279                meta.query_advice(a_6, Rotation::prev()), // spread_b_mid

304    #[allow(clippy::type_complexity)]
305    pub(super) fn process(
306        &self,
307        layouter: &mut impl Layouter<pallas::Base>,
308        input: [BlockWord; BLOCK_SIZE],
309    ) -> Result<
310        (
311            [MessageWord; ROUNDS],
312            [(AssignedBits<16>, AssignedBits<16>); ROUNDS],
313        ),
314        Error,
315    > {
316        let mut w = Vec::<MessageWord>::with_capacity(ROUNDS);
317        let mut w_halves = Vec::<(AssignedBits<16>, AssignedBits<16>)>::with_capacity(ROUNDS);
318
319        layouter.assign_region(
320            || "process message block",
321            |mut region| {
322                w = Vec::<MessageWord>::with_capacity(ROUNDS);
323                w_halves = Vec::<(AssignedBits<16>, AssignedBits<16>)>::with_capacity(ROUNDS);
324
325                // Assign all fixed columns
326                for index in 1..14 {
327                    let row = get_word_row(index);
328                    self.s_decompose_1.enable(&mut region, row)?;
329                    self.s_lower_sigma_0.enable(&mut region, row + 3)?;
330                }
331
332                for index in 14..49 {
333                    let row = get_word_row(index);
334                    self.s_decompose_2.enable(&mut region, row)?;
335                    self.s_lower_sigma_0_v2.enable(&mut region, row + 3)?;
336                    self.s_lower_sigma_1_v2
337                        .enable(&mut region, row + SIGMA_0_V2_ROWS + 3)?;
338
339                    let new_word_idx = index + 2;
340                    self.s_word
341                        .enable(&mut region, get_word_row(new_word_idx - 16) + 1)?;
342                }
343
344                for index in 49..62 {
345                    let row = get_word_row(index);
346                    self.s_decompose_3.enable(&mut region, row)?;
347                    self.s_lower_sigma_1.enable(&mut region, row + 3)?;
348
349                    let new_word_idx = index + 2;
350                    self.s_word
351                        .enable(&mut region, get_word_row(new_word_idx - 16) + 1)?;
352                }
353
354                for index in 0..64 {
355                    let row = get_word_row(index);
356                    self.s_decompose_0.enable(&mut region, row)?;
357                }
358
359                // Assign W[0..16]
360                for (i, word) in input.iter().enumerate() {
361                    let (word, halves) = self.assign_word_and_halves(&mut region, word.0, i)?;
362                    w.push(MessageWord(word));
363                    w_halves.push(halves);
364                }
365
366                // Returns the output of sigma_0 on W_[1..14]
367                let lower_sigma_0_output = self.assign_subregion1(&mut region, &input[1..14])?;
368
369                // sigma_0_v2 and sigma_1_v2 on W_[14..49]
370                // Returns the output of sigma_0_v2 on W_[36..49], to be used in subregion3
371                let lower_sigma_0_v2_output = self.assign_subregion2(
372                    &mut region,
373                    lower_sigma_0_output,
374                    &mut w,
375                    &mut w_halves,
376                )?;
377
378                // sigma_1 v1 on W[49..62]
379                self.assign_subregion3(
380                    &mut region,
381                    lower_sigma_0_v2_output,
382                    &mut w,
383                    &mut w_halves,
384                )?;
385
386                Ok(())
387            },
388        )?;
389
390        Ok((w.try_into().unwrap(), w_halves.try_into().unwrap()))
391    }

416                MyCircuit {}
417            }

420                Table16Chip::configure(meta)
421            }
422
423            fn synthesize(
424                &self,
425                config: Self::Config,
426                mut layouter: impl Layouter<pallas::Base>,
427            ) -> Result<(), Error> {
428                // Load lookup table
429                SpreadTableChip::load(config.lookup.clone(), &mut layouter)?;
430
431                // Provide input
432                // Test vector: "abc"
433                let inputs: [BlockWord; BLOCK_SIZE] = msg_schedule_test_input();
434
435                // Run message_scheduler to get W_[0..64]
436                let (w, _) = config.message_schedule.process(&mut layouter, inputs)?;
437                for (word, test_word) in w.iter().zip(MSG_SCHEDULE_TEST_OUTPUT.iter()) {
438                    word.value().assert_if_known(|bits| {
439                        let word: u32 = lebs2ip(bits) as u32;
440                        word == *test_word
441                    });
442                }
443                Ok(())
444            }
445        }

447        let circuit: MyCircuit = MyCircuit {};
448
449        let prover = match MockProver::<pallas::Base>::run(17, &circuit, vec![]) {
450            Ok(prover) => prover,
451            Err(e) => panic!("{:?}", e),

453        assert_eq!(prover.verify(), Ok(()));
