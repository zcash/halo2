5/// The sequence of bits representing a u64 in little-endian order.
6///
7/// # Panics
8///
9/// Panics if the expected length of the sequence `NUM_BITS` exceeds
10/// 64.
11pub fn i2lebsp<const NUM_BITS: usize>(int: u64) -> [bool; NUM_BITS] {
12    /// Takes in an FnMut closure and returns a constant-length array with elements of
13    /// type `Output`.
14    fn gen_const_array<Output: Copy + Default, const LEN: usize>(
15        closure: impl FnMut(usize) -> Output,
16    ) -> [Output; LEN] {
17        gen_const_array_with_default(Default::default(), closure)
18    }
19
20    fn gen_const_array_with_default<Output: Copy, const LEN: usize>(
21        default_value: Output,
22        closure: impl FnMut(usize) -> Output,
23    ) -> [Output; LEN] {
24        let mut ret: [Output; LEN] = [default_value; LEN];
25        for (bit, val) in ret.iter_mut().zip((0..LEN).map(closure)) {
26            *bit = val;
27        }
28        ret
29    }
30
31    assert!(NUM_BITS <= 64);
32    gen_const_array(|mask: usize| (int & (1 << mask)) != 0)
33}

35/// Returns the integer representation of a little-endian bit-array.
36/// Panics if the number of bits exceeds 64.
37pub fn lebs2ip<const K: usize>(bits: &[bool; K]) -> u64 {
38    assert!(K <= 64);
39    bits.iter()
40        .enumerate()
41        .fold(0u64, |acc, (i, b)| acc + if *b { 1 << i } else { 0 })
42}

44/// Helper function that interleaves a little-endian bit-array with zeros
45/// in the odd indices. That is, it takes the array
46///         [b_0, b_1, ..., b_n]
47/// to
48///         [b_0, 0, b_1, 0, ..., b_n, 0].
49/// Panics if bit-array is longer than 16 bits.
50pub fn spread_bits<const DENSE: usize, const SPREAD: usize>(
51    bits: impl Into<[bool; DENSE]>,
52) -> [bool; SPREAD] {
53    assert_eq!(DENSE * 2, SPREAD);
54    assert!(DENSE <= 16);
55
56    let bits: [bool; DENSE] = bits.into();
57    let mut spread = [false; SPREAD];
58
59    for (idx, bit) in bits.iter().enumerate() {
60        spread[idx * 2] = *bit;
61    }
62
63    spread
64}

66/// Negates the even bits in a spread bit-array.
67pub fn negate_spread<const LEN: usize>(arr: [bool; LEN]) -> [bool; LEN] {
68    assert_eq!(LEN % 2, 0);
69
70    let mut neg = arr;
71    for even_idx in (0..LEN).step_by(2) {
72        let odd_idx = even_idx + 1;
73        assert!(!arr[odd_idx]);
74
75        neg[even_idx] = !arr[even_idx];
76    }
77
78    neg
79}

81/// Returns even bits in a bit-array
82pub fn even_bits<const LEN: usize, const HALF: usize>(bits: [bool; LEN]) -> [bool; HALF] {
83    assert_eq!(LEN % 2, 0);
84    let mut even_bits = [false; HALF];
85    for idx in 0..HALF {
86        even_bits[idx] = bits[idx * 2]
87    }
88    even_bits
89}

91/// Returns odd bits in a bit-array
92pub fn odd_bits<const LEN: usize, const HALF: usize>(bits: [bool; LEN]) -> [bool; HALF] {
93    assert_eq!(LEN % 2, 0);
94    let mut odd_bits = [false; HALF];
95    for idx in 0..HALF {
96        odd_bits[idx] = bits[idx * 2 + 1]
97    }
98    odd_bits

101/// Given a vector of words as vec![(lo: u16, hi: u16)], returns their sum: u32, along
102/// with a carry bit.
103pub fn sum_with_carry(words: Vec<(Value<u16>, Value<u16>)>) -> (Value<u32>, Value<u64>) {
104    let words_lo: Value<Vec<u64>> = words.iter().map(|(lo, _)| lo.map(|lo| lo as u64)).collect();
105    let words_hi: Value<Vec<u64>> = words.iter().map(|(_, hi)| hi.map(|hi| hi as u64)).collect();
106
107    let sum: Value<u64> = {
108        let sum_lo: Value<u64> = words_lo.map(|vec| vec.iter().sum());
109        let sum_hi: Value<u64> = words_hi.map(|vec| vec.iter().sum());
110        sum_lo.zip(sum_hi).map(|(lo, hi)| lo + (1 << 16) * hi)
111    };
112
113    let carry = sum.map(|sum| sum >> 32);
114    let sum = sum.map(|sum| sum as u32);
115
116    (sum, carry)
117}
