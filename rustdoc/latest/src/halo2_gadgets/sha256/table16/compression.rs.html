23pub trait UpperSigmaVar<
24    const A_LEN: usize,
25    const B_LEN: usize,
26    const C_LEN: usize,
27    const D_LEN: usize,
28>
29{
30    fn spread_a(&self) -> Value<[bool; A_LEN]>;
31    fn spread_b(&self) -> Value<[bool; B_LEN]>;
32    fn spread_c(&self) -> Value<[bool; C_LEN]>;
33    fn spread_d(&self) -> Value<[bool; D_LEN]>;
34
35    fn xor_upper_sigma(&self) -> Value<[bool; 64]> {
36        self.spread_a()
37            .zip(self.spread_b())
38            .zip(self.spread_c())
39            .zip(self.spread_d())
40            .map(|(((a, b), c), d)| {
41                let xor_0 = b
42                    .iter()
43                    .chain(c.iter())
44                    .chain(d.iter())
45                    .chain(a.iter())
46                    .copied()
47                    .collect::<Vec<_>>();
48                let xor_1 = c
49                    .iter()
50                    .chain(d.iter())
51                    .chain(a.iter())
52                    .chain(b.iter())
53                    .copied()
54                    .collect::<Vec<_>>();
55                let xor_2 = d
56                    .iter()
57                    .chain(a.iter())
58                    .chain(b.iter())
59                    .chain(c.iter())
60                    .copied()
61                    .collect::<Vec<_>>();
62
63                let xor_0 = lebs2ip::<64>(&xor_0.try_into().unwrap());
64                let xor_1 = lebs2ip::<64>(&xor_1.try_into().unwrap());
65                let xor_2 = lebs2ip::<64>(&xor_2.try_into().unwrap());
66
67                i2lebsp(xor_0 + xor_1 + xor_2)
68            })
69    }
70}

72/// A variable that represents the `[A,B,C,D]` words of the SHA-256 internal state.
73///
74/// The structure of this variable is influenced by the following factors:
75/// - In `Σ_0(A)` we need `A` to be split into pieces `(a,b,c,d)` of lengths `(2,11,9,10)`
76///   bits respectively (counting from the little end), as well as their spread forms.
77/// - `Maj(A,B,C)` requires having the bits of each input in spread form. For `A` we can
78///   reuse the pieces from `Σ_0(A)`. Since `B` and `C` are assigned from `A` and `B`
79///   respectively in each round, we therefore also have the same pieces in earlier rows.
80///   We align the columns to make it efficient to copy-constrain these forms where they
81///   are needed.
82#[derive(Clone, Debug)]
83pub struct AbcdVar {
84    a: SpreadVar<2, 4>,
85    b: SpreadVar<11, 22>,
86    c_lo: SpreadVar<3, 6>,
87    c_mid: SpreadVar<3, 6>,
88    c_hi: SpreadVar<3, 6>,
89    d: SpreadVar<10, 20>,
90}
91
92impl AbcdVar {
93    fn a_range() -> Range<usize> {
94        0..2
95    }
96
97    fn b_range() -> Range<usize> {
98        2..13
99    }
100
101    fn c_lo_range() -> Range<usize> {
102        13..16
103    }
104
105    fn c_mid_range() -> Range<usize> {
106        16..19
107    }
108
109    fn c_hi_range() -> Range<usize> {
110        19..22
111    }
112
113    fn d_range() -> Range<usize> {
114        22..32
115    }
116
117    fn pieces(val: u32) -> Vec<Vec<bool>> {
118        let val: [bool; 32] = i2lebsp(val.into());
119        vec![
120            val[Self::a_range()].to_vec(),
121            val[Self::b_range()].to_vec(),
122            val[Self::c_lo_range()].to_vec(),
123            val[Self::c_mid_range()].to_vec(),
124            val[Self::c_hi_range()].to_vec(),
125            val[Self::d_range()].to_vec(),
126        ]
127    }
128}

158    }
159}

259#[derive(Clone, Debug)]
260pub struct RoundWordDense(AssignedBits<16>, AssignedBits<16>);
261
262impl From<(AssignedBits<16>, AssignedBits<16>)> for RoundWordDense {
263    fn from(halves: (AssignedBits<16>, AssignedBits<16>)) -> Self {
264        Self(halves.0, halves.1)
265    }
266}
267
268impl RoundWordDense {
269    pub fn value(&self) -> Value<u32> {
270        self.0
271            .value_u16()
272            .zip(self.1.value_u16())
273            .map(|(lo, hi)| lo as u32 + (1 << 16) * hi as u32)
274    }
275}

458impl CompressionConfig {
459    pub(super) fn configure(
460        meta: &mut ConstraintSystem<pallas::Base>,
461        lookup: SpreadInputs,
462        message_schedule: Column<Advice>,
463        extras: [Column<Advice>; 6],
464    ) -> Self {
465        let s_ch = meta.selector();
466        let s_ch_neg = meta.selector();
467        let s_maj = meta.selector();
468        let s_h_prime = meta.selector();
469        let s_a_new = meta.selector();
470        let s_e_new = meta.selector();
471
472        let s_upper_sigma_0 = meta.selector();
473        let s_upper_sigma_1 = meta.selector();
474
475        // Decomposition gate for AbcdVar
476        let s_decompose_abcd = meta.selector();
477        // Decomposition gate for EfghVar
478        let s_decompose_efgh = meta.selector();
479
480        let s_digest = meta.selector();
481
482        // Rename these here for ease of matching the gates to the specification.
483        let a_0 = lookup.tag;
484        let a_1 = lookup.dense;
485        let a_2 = lookup.spread;
486        let a_3 = extras[0];
487        let a_4 = extras[1];
488        let a_5 = message_schedule;
489        let a_6 = extras[2];
490        let a_7 = extras[3];
491        let a_8 = extras[4];
492        let a_9 = extras[5];
493
494        // Decompose `A,B,C,D` words into (2, 11, 9, 10)-bit chunks.
495        // `c` is split into (3, 3, 3)-bit c_lo, c_mid, c_hi.
496        meta.create_gate("decompose ABCD", |meta| {
497            let s_decompose_abcd = meta.query_selector(s_decompose_abcd);
498            let a = meta.query_advice(a_3, Rotation::next()); // 2-bit chunk
499            let spread_a = meta.query_advice(a_4, Rotation::next());
500            let b = meta.query_advice(a_1, Rotation::cur()); // 11-bit chunk
501            let spread_b = meta.query_advice(a_2, Rotation::cur());
502            let tag_b = meta.query_advice(a_0, Rotation::cur());
503            let c_lo = meta.query_advice(a_3, Rotation::cur()); // 3-bit chunk
504            let spread_c_lo = meta.query_advice(a_4, Rotation::cur());
505            let c_mid = meta.query_advice(a_5, Rotation::cur()); // 3-bit chunk
506            let spread_c_mid = meta.query_advice(a_6, Rotation::cur());
507            let c_hi = meta.query_advice(a_5, Rotation::next()); // 3-bit chunk
508            let spread_c_hi = meta.query_advice(a_6, Rotation::next());
509            let d = meta.query_advice(a_1, Rotation::next()); // 7-bit chunk
510            let spread_d = meta.query_advice(a_2, Rotation::next());
511            let tag_d = meta.query_advice(a_0, Rotation::next());
512            let word_lo = meta.query_advice(a_7, Rotation::cur());
513            let spread_word_lo = meta.query_advice(a_8, Rotation::cur());
514            let word_hi = meta.query_advice(a_7, Rotation::next());
515            let spread_word_hi = meta.query_advice(a_8, Rotation::next());
516
517            CompressionGate::s_decompose_abcd(
518                s_decompose_abcd,
519                a,
520                spread_a,
521                b,
522                spread_b,
523                tag_b,
524                c_lo,
525                spread_c_lo,
526                c_mid,
527                spread_c_mid,
528                c_hi,
529                spread_c_hi,
530                d,
531                spread_d,
532                tag_d,
533                word_lo,
534                spread_word_lo,
535                word_hi,
536                spread_word_hi,
537            )
538        });
539
540        // Decompose `E,F,G,H` words into (6, 5, 14, 7)-bit chunks.
541        // `a` is split into (3, 3)-bit a_lo, a_hi
542        // `b` is split into (2, 3)-bit b_lo, b_hi
543        meta.create_gate("Decompose EFGH", |meta| {
544            let s_decompose_efgh = meta.query_selector(s_decompose_efgh);
545            let a_lo = meta.query_advice(a_3, Rotation::next()); // 3-bit chunk
546            let spread_a_lo = meta.query_advice(a_4, Rotation::next());
547            let a_hi = meta.query_advice(a_5, Rotation::next()); // 3-bit chunk
548            let spread_a_hi = meta.query_advice(a_6, Rotation::next());
549            let b_lo = meta.query_advice(a_3, Rotation::cur()); // 2-bit chunk
550            let spread_b_lo = meta.query_advice(a_4, Rotation::cur());
551            let b_hi = meta.query_advice(a_5, Rotation::cur()); // 3-bit chunk
552            let spread_b_hi = meta.query_advice(a_6, Rotation::cur());
553            let c = meta.query_advice(a_1, Rotation::next()); // 14-bit chunk
554            let spread_c = meta.query_advice(a_2, Rotation::next());
555            let tag_c = meta.query_advice(a_0, Rotation::next());
556            let d = meta.query_advice(a_1, Rotation::cur()); // 7-bit chunk
557            let spread_d = meta.query_advice(a_2, Rotation::cur());
558            let tag_d = meta.query_advice(a_0, Rotation::cur());
559            let word_lo = meta.query_advice(a_7, Rotation::cur());
560            let spread_word_lo = meta.query_advice(a_8, Rotation::cur());
561            let word_hi = meta.query_advice(a_7, Rotation::next());
562            let spread_word_hi = meta.query_advice(a_8, Rotation::next());
563
564            CompressionGate::s_decompose_efgh(
565                s_decompose_efgh,
566                a_lo,
567                spread_a_lo,
568                a_hi,
569                spread_a_hi,
570                b_lo,
571                spread_b_lo,
572                b_hi,
573                spread_b_hi,
574                c,
575                spread_c,
576                tag_c,
577                d,
578                spread_d,
579                tag_d,
580                word_lo,
581                spread_word_lo,
582                word_hi,
583                spread_word_hi,
584            )
585        });
586
587        // s_upper_sigma_0 on abcd words
588        // (2, 11, 9, 10)-bit chunks
589        meta.create_gate("s_upper_sigma_0", |meta| {
590            let s_upper_sigma_0 = meta.query_selector(s_upper_sigma_0);
591            let spread_r0_even = meta.query_advice(a_2, Rotation::prev());
592            let spread_r0_odd = meta.query_advice(a_2, Rotation::cur());
593            let spread_r1_even = meta.query_advice(a_2, Rotation::next());
594            let spread_r1_odd = meta.query_advice(a_3, Rotation::cur());
595
596            let spread_a = meta.query_advice(a_3, Rotation::next());
597            let spread_b = meta.query_advice(a_5, Rotation::cur());
598            let spread_c_lo = meta.query_advice(a_3, Rotation::prev());
599            let spread_c_mid = meta.query_advice(a_4, Rotation::prev());
600            let spread_c_hi = meta.query_advice(a_4, Rotation::next());
601            let spread_d = meta.query_advice(a_4, Rotation::cur());
602
603            CompressionGate::s_upper_sigma_0(
604                s_upper_sigma_0,
605                spread_r0_even,
606                spread_r0_odd,
607                spread_r1_even,
608                spread_r1_odd,
609                spread_a,
610                spread_b,
611                spread_c_lo,
612                spread_c_mid,
613                spread_c_hi,
614                spread_d,
615            )
616        });
617
618        // s_upper_sigma_1 on efgh words
619        // (6, 5, 14, 7)-bit chunks
620        meta.create_gate("s_upper_sigma_1", |meta| {
621            let s_upper_sigma_1 = meta.query_selector(s_upper_sigma_1);
622            let spread_r0_even = meta.query_advice(a_2, Rotation::prev());
623            let spread_r0_odd = meta.query_advice(a_2, Rotation::cur());
624            let spread_r1_even = meta.query_advice(a_2, Rotation::next());
625            let spread_r1_odd = meta.query_advice(a_3, Rotation::cur());
626            let spread_a_lo = meta.query_advice(a_3, Rotation::next());
627            let spread_a_hi = meta.query_advice(a_4, Rotation::next());
628            let spread_b_lo = meta.query_advice(a_3, Rotation::prev());
629            let spread_b_hi = meta.query_advice(a_4, Rotation::prev());
630            let spread_c = meta.query_advice(a_5, Rotation::cur());
631            let spread_d = meta.query_advice(a_4, Rotation::cur());
632
633            CompressionGate::s_upper_sigma_1(
634                s_upper_sigma_1,
635                spread_r0_even,
636                spread_r0_odd,
637                spread_r1_even,
638                spread_r1_odd,
639                spread_a_lo,
640                spread_a_hi,
641                spread_b_lo,
642                spread_b_hi,
643                spread_c,
644                spread_d,
645            )
646        });
647
648        // s_ch on efgh words
649        // First part of choice gate on (E, F, G), E ∧ F
650        meta.create_gate("s_ch", |meta| {

862    /// Initialize compression with a constant Initialization Vector of 32-byte words.
863    /// Returns an initialized state.
864    pub(super) fn initialize_with_iv(
865        &self,
866        layouter: &mut impl Layouter<pallas::Base>,
867        init_state: [u32; STATE],
868    ) -> Result<State, Error> {
869        let mut new_state = State::empty_state();
870        layouter.assign_region(
871            || "initialize_with_iv",
872            |mut region| {
873                new_state = self.initialize_iv(&mut region, init_state)?;
874                Ok(())
875            },
876        )?;
877        Ok(new_state)
878    }
879
880    /// Initialize compression with some initialized state. This could be a state
881    /// output from a previous compression round.
882    pub(super) fn initialize_with_state(
883        &self,
884        layouter: &mut impl Layouter<pallas::Base>,
885        init_state: State,
886    ) -> Result<State, Error> {
887        let mut new_state = State::empty_state();
888        layouter.assign_region(
889            || "initialize_with_state",
890            |mut region| {
891                new_state = self.initialize_state(&mut region, init_state.clone())?;
892                Ok(())
893            },
894        )?;
895        Ok(new_state)
896    }

898    /// Given an initialized state and a message schedule, perform 64 compression rounds.
899    pub(super) fn compress(
900        &self,
901        layouter: &mut impl Layouter<pallas::Base>,
902        initialized_state: State,
903        w_halves: [(AssignedBits<16>, AssignedBits<16>); ROUNDS],
904    ) -> Result<State, Error> {
905        let mut state = State::empty_state();
906        layouter.assign_region(
907            || "compress",
908            |mut region| {
909                state = initialized_state.clone();
910                for (idx, w_halves) in w_halves.iter().enumerate() {
911                    state = self.assign_round(&mut region, idx.into(), state.clone(), w_halves)?;
912                }
913                Ok(())
914            },
915        )?;
916        Ok(state)
917    }

919    /// After the final round, convert the state into the final digest.
920    pub(super) fn digest(
921        &self,
922        layouter: &mut impl Layouter<pallas::Base>,
923        state: State,
924    ) -> Result<[BlockWord; DIGEST_SIZE], Error> {
925        let mut digest = [BlockWord(Value::known(0)); DIGEST_SIZE];
926        layouter.assign_region(
927            || "digest",
928            |mut region| {
929                digest = self.assign_digest(&mut region, state.clone())?;
930
931                Ok(())
932            },
933        )?;
934        Ok(digest)
935    }
