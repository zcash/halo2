55    fn xor_sigma_0(&self) -> Value<[bool; 64]> {
56        self.spread_a()
57            .zip(self.spread_b())
58            .zip(self.spread_c())
59            .zip(self.spread_d())
60            .zip(self.spread_e())
61            .zip(self.spread_f())
62            .zip(self.spread_g())
63            .map(|((((((a, b), c), d), e), f), g)| {
64                let xor_0 = b
65                    .iter()
66                    .chain(c.iter())
67                    .chain(d.iter())
68                    .chain(e.iter())
69                    .chain(f.iter())
70                    .chain(g.iter())
71                    .chain(std::iter::repeat(&false).take(6))
72                    .copied()
73                    .collect::<Vec<_>>();
74
75                let xor_1 = c
76                    .iter()
77                    .chain(d.iter())
78                    .chain(e.iter())
79                    .chain(f.iter())
80                    .chain(g.iter())
81                    .chain(a.iter())
82                    .chain(b.iter())
83                    .copied()
84                    .collect::<Vec<_>>();
85
86                let xor_2 = f
87                    .iter()
88                    .chain(g.iter())
89                    .chain(a.iter())
90                    .chain(b.iter())
91                    .chain(c.iter())
92                    .chain(d.iter())
93                    .chain(e.iter())
94                    .copied()
95                    .collect::<Vec<_>>();
96
97                let xor_0 = lebs2ip::<64>(&xor_0.try_into().unwrap());
98                let xor_1 = lebs2ip::<64>(&xor_1.try_into().unwrap());
99                let xor_2 = lebs2ip::<64>(&xor_2.try_into().unwrap());
100
101                i2lebsp(xor_0 + xor_1 + xor_2)
102            })
103    }

156    pub fn assign_subregion2(
157        &self,
158        region: &mut Region<'_, pallas::Base>,
159        lower_sigma_0_output: Vec<(AssignedBits<16>, AssignedBits<16>)>,
160        w: &mut Vec<MessageWord>,
161        w_halves: &mut Vec<(AssignedBits<16>, AssignedBits<16>)>,
162    ) -> Result<Vec<(AssignedBits<16>, AssignedBits<16>)>, Error> {
163        let a_5 = self.message_schedule;
164        let a_6 = self.extras[2];
165        let a_7 = self.extras[3];
166        let a_8 = self.extras[4];
167        let a_9 = self.extras[5];
168
169        let mut lower_sigma_0_v2_results =
170            Vec::<(AssignedBits<16>, AssignedBits<16>)>::with_capacity(SUBREGION_2_LEN);
171        let mut lower_sigma_1_v2_results =
172            Vec::<(AssignedBits<16>, AssignedBits<16>)>::with_capacity(SUBREGION_2_LEN);
173
174        // Closure to compose new word
175        // W_i = sigma_1(W_{i - 2}) + W_{i - 7} + sigma_0(W_{i - 15}) + W_{i - 16}
176        // e.g. W_16 = sigma_1(W_14) + W_9 + sigma_0(W_1) + W_0
177
178        // sigma_0(W_[1..14]) will be used to get the new W_[16..29]
179        // sigma_0_v2(W_[14..36]) will be used to get the new W_[29..51]
180        // sigma_1_v2(W_[14..49]) will be used to get the W_[16..51]
181        // The lowest-index words involved will be W_[0..13]
182        let mut new_word = |idx: usize,
183                            sigma_0_output: &(AssignedBits<16>, AssignedBits<16>)|
184         -> Result<Vec<(AssignedBits<16>, AssignedBits<16>)>, Error> {
185            // Decompose word into (3, 4, 3, 7, 1, 1, 13)-bit chunks
186            let word = self.decompose_word(region, w[idx].value(), idx)?;
187
188            // sigma_0 v2 and sigma_1 v2 on word
189            lower_sigma_0_v2_results.push(self.lower_sigma_0_v2(region, word.clone())?);
190            lower_sigma_1_v2_results.push(self.lower_sigma_1_v2(region, word)?);
191
192            let new_word_idx = idx + 2;
193
194            // Copy sigma_0(W_{i - 15}) output from Subregion 1
195            sigma_0_output.0.copy_advice(
196                || format!("sigma_0(W_{})_lo", new_word_idx - 15),
197                region,
198                a_6,
199                get_word_row(new_word_idx - 16),
200            )?;
201            sigma_0_output.1.copy_advice(
202                || format!("sigma_0(W_{})_hi", new_word_idx - 15),
203                region,
204                a_6,
205                get_word_row(new_word_idx - 16) + 1,
206            )?;
207
208            // Copy sigma_1(W_{i - 2})
209            lower_sigma_1_v2_results[new_word_idx - 16].0.copy_advice(
210                || format!("sigma_1(W_{})_lo", new_word_idx - 2),
211                region,
212                a_7,
213                get_word_row(new_word_idx - 16),
214            )?;
215            lower_sigma_1_v2_results[new_word_idx - 16].1.copy_advice(
216                || format!("sigma_1(W_{})_hi", new_word_idx - 2),
217                region,
218                a_7,
219                get_word_row(new_word_idx - 16) + 1,
220            )?;
221
222            // Copy W_{i - 7}
223            w_halves[new_word_idx - 7].0.copy_advice(
224                || format!("W_{}_lo", new_word_idx - 7),
225                region,
226                a_8,
227                get_word_row(new_word_idx - 16),
228            )?;
229            w_halves[new_word_idx - 7].1.copy_advice(
230                || format!("W_{}_hi", new_word_idx - 7),
231                region,
232                a_8,
233                get_word_row(new_word_idx - 16) + 1,
234            )?;
235
236            // Calculate W_i, carry_i
237            let (word, carry) = sum_with_carry(vec![
238                (
239                    lower_sigma_1_v2_results[new_word_idx - 16].0.value_u16(),
240                    lower_sigma_1_v2_results[new_word_idx - 16].1.value_u16(),
241                ),
242                (
243                    w_halves[new_word_idx - 7].0.value_u16(),
244                    w_halves[new_word_idx - 7].1.value_u16(),
245                ),
246                (sigma_0_output.0.value_u16(), sigma_0_output.1.value_u16()),
247                (
248                    w_halves[new_word_idx - 16].0.value_u16(),
249                    w_halves[new_word_idx - 16].1.value_u16(),
250                ),
251            ]);
252
253            // Assign W_i, carry_i
254            region.assign_advice(
255                || format!("W_{}", new_word_idx),
256                a_5,
257                get_word_row(new_word_idx - 16) + 1,
258                || word.map(|word| pallas::Base::from(word as u64)),
259            )?;
260            region.assign_advice(
261                || format!("carry_{}", new_word_idx),
262                a_9,
263                get_word_row(new_word_idx - 16) + 1,
264                || carry.map(pallas::Base::from),
265            )?;
266            let (word, halves) = self.assign_word_and_halves(region, word, new_word_idx)?;
267            w.push(MessageWord(word));
268            w_halves.push(halves);
269
270            Ok(lower_sigma_0_v2_results.clone())
271        };
272
273        let mut tmp_lower_sigma_0_v2_results: Vec<(AssignedBits<16>, AssignedBits<16>)> =
274            Vec::with_capacity(SUBREGION_2_LEN);
275
276        // Use up all the output from Subregion 1 lower_sigma_0
277        for i in 14..27 {
278            tmp_lower_sigma_0_v2_results = new_word(i, &lower_sigma_0_output[i - 14])?;
279        }
280
281        for i in 27..49 {
282            tmp_lower_sigma_0_v2_results =
283                new_word(i, &tmp_lower_sigma_0_v2_results[i + 2 - 15 - 14])?;
284        }
285
286        // Return lower_sigma_0_v2 output for W_[36..49]
287        Ok(lower_sigma_0_v2_results.split_off(36 - 14))
288    }

290    /// Pieces of length [3, 4, 3, 7, 1, 1, 13]
291    fn decompose_word(
292        &self,
293        region: &mut Region<'_, pallas::Base>,
294        word: Value<&Bits<32>>,
295        index: usize,
296    ) -> Result<Subregion2Word, Error> {
297        let row = get_word_row(index);
298
299        let pieces = word.map(|word| {
300            vec![
301                word[0..3].to_vec(),
302                word[3..7].to_vec(),
303                word[7..10].to_vec(),
304                word[10..17].to_vec(),
305                vec![word[17]],
306                vec![word[18]],
307                word[19..32].to_vec(),
308            ]
309        });
310        let pieces = pieces.transpose_vec(7);
311
312        // Rename these here for ease of matching the gates to the specification.
313        let a_3 = self.extras[0];
314        let a_4 = self.extras[1];
315
316        // Assign `a` (3-bit piece)
317        let a = AssignedBits::<3>::assign_bits(region, || "a", a_3, row - 1, pieces[0].clone())?;
318
319        // Assign `b` (4-bit piece) lookup
320        let spread_b: Value<SpreadWord<4, 8>> = pieces[1].clone().map(SpreadWord::try_new);
321        let spread_b = SpreadVar::with_lookup(region, &self.lookup, row + 1, spread_b)?;
322
323        // Assign `c` (3-bit piece)
324        let c = AssignedBits::<3>::assign_bits(region, || "c", a_4, row - 1, pieces[2].clone())?;
325
326        // Assign `d` (7-bit piece) lookup
327        let spread_d: Value<SpreadWord<7, 14>> = pieces[3].clone().map(SpreadWord::try_new);
328        let spread_d = SpreadVar::with_lookup(region, &self.lookup, row, spread_d)?;
329
330        // Assign `e` (1-bit piece)
331        let e = AssignedBits::<1>::assign_bits(region, || "e", a_3, row + 1, pieces[4].clone())?;
332
333        // Assign `f` (1-bit piece)
334        let f = AssignedBits::<1>::assign_bits(region, || "f", a_4, row + 1, pieces[5].clone())?;
335
336        // Assign `g` (13-bit piece) lookup
337        let spread_g = pieces[6].clone().map(SpreadWord::try_new);
338        let spread_g = SpreadVar::with_lookup(region, &self.lookup, row - 1, spread_g)?;
339
340        Ok(Subregion2Word {
341            index,
342            a,
343            b: spread_b.dense,
344            c,
345            _d: spread_d.dense,
346            e,
347            f,
348            _g: spread_g.dense,
349            spread_d: spread_d.spread,
350            spread_g: spread_g.spread,
351        })
352    }

354    /// A word in subregion 2
355    /// (3, 4, 3, 7, 1, 1, 13)-bit chunks
356    #[allow(clippy::type_complexity)]
357    fn assign_lower_sigma_v2_pieces(
358        &self,
359        region: &mut Region<'_, pallas::Base>,
360        row: usize,
361        word: &Subregion2Word,
362    ) -> Result<(), Error> {
363        let a_3 = self.extras[0];
364        let a_4 = self.extras[1];
365        let a_5 = self.message_schedule;
366        let a_6 = self.extras[2];
367        let a_7 = self.extras[3];
368
369        // Assign `a` and copy constraint
370        word.a.copy_advice(|| "a", region, a_3, row + 1)?;
371
372        // Witness `spread_a`
373        AssignedBits::<6>::assign_bits(region, || "spread_a", a_4, row + 1, word.spread_a())?;
374
375        // Split `b` (4-bit chunk) into `b_hi` and `b_lo`
376        // Assign `b_lo`, `spread_b_lo`
377
378        let b_lo: Value<[bool; 2]> = word.b.value().map(|b| b.0[..2].try_into().unwrap());
379        let spread_b_lo = b_lo.map(spread_bits);
380        {
381            AssignedBits::<2>::assign_bits(region, || "b_lo", a_3, row - 1, b_lo)?;
382
383            AssignedBits::<4>::assign_bits(region, || "spread_b_lo", a_4, row - 1, spread_b_lo)?;
384        };
385
386        // Split `b` (2-bit chunk) into `b_hi` and `b_lo`
387        // Assign `b_hi`, `spread_b_hi`
388        let b_hi: Value<[bool; 2]> = word.b.value().map(|b| b.0[2..].try_into().unwrap());
389        let spread_b_hi = b_hi.map(spread_bits);
390        {
391            AssignedBits::<2>::assign_bits(region, || "b_hi", a_5, row - 1, b_hi)?;
392
393            AssignedBits::<4>::assign_bits(region, || "spread_b_hi", a_6, row - 1, spread_b_hi)?;
394        };
395
396        // Assign `b` and copy constraint
397        word.b.copy_advice(|| "b", region, a_6, row)?;
398
399        // Assign `c` and copy constraint
400        word.c.copy_advice(|| "c", region, a_5, row + 1)?;
401
402        // Witness `spread_c`
403        AssignedBits::<6>::assign_bits(region, || "spread_c", a_6, row + 1, word.spread_c())?;
404
405        // Assign `spread_d` and copy constraint
406        word.spread_d.copy_advice(|| "spread_d", region, a_4, row)?;
407
408        // Assign `e` and copy constraint
409        word.e.copy_advice(|| "e", region, a_7, row)?;
410
411        // Assign `f` and copy constraint
412        word.f.copy_advice(|| "f", region, a_7, row + 1)?;
413
414        // Assign `spread_g` and copy constraint
415        word.spread_g.copy_advice(|| "spread_g", region, a_5, row)?;
416
417        Ok(())
418    }

420    fn lower_sigma_0_v2(
421        &self,
422        region: &mut Region<'_, pallas::Base>,
423        word: Subregion2Word,
424    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
425        let a_3 = self.extras[0];
426        let row = get_word_row(word.index) + 3;
427
428        // Assign lower sigma_v2 pieces
429        self.assign_lower_sigma_v2_pieces(region, row, &word)?;
430
431        // Calculate R_0^{even}, R_0^{odd}, R_1^{even}, R_1^{odd}
432        let r = word.xor_sigma_0();
433        let r_0: Value<[bool; 32]> = r.map(|r| r[..32].try_into().unwrap());
434        let r_0_even = r_0.map(even_bits);
435        let r_0_odd = r_0.map(odd_bits);
436
437        let r_1: Value<[bool; 32]> = r.map(|r| r[32..].try_into().unwrap());
438        let r_1_even = r_1.map(even_bits);
439        let r_1_odd = r_1.map(odd_bits);
440
441        self.assign_sigma_outputs(
442            region,
443            &self.lookup,
444            a_3,
445            row,
446            r_0_even,
447            r_0_odd,
448            r_1_even,
449            r_1_odd,
450        )
451    }
