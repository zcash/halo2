10// A word in subregion 1
11// (3, 4, 11, 14)-bit chunks
12#[derive(Debug)]
13pub struct Subregion1Word {
14    index: usize,
15    a: AssignedBits<3>,
16    b: AssignedBits<4>,
17    _c: AssignedBits<11>,
18    _d: AssignedBits<14>,
19    spread_c: AssignedBits<22>,
20    spread_d: AssignedBits<28>,
21}
22
23impl Subregion1Word {
24    fn spread_a(&self) -> Value<[bool; 6]> {
25        self.a.value().map(|v| v.spread())
26    }
27
28    fn spread_b(&self) -> Value<[bool; 8]> {
29        self.b.value().map(|v| v.spread())
30    }
31
32    fn spread_c(&self) -> Value<[bool; 22]> {
33        self.spread_c.value().map(|v| v.0)
34    }
35
36    fn spread_d(&self) -> Value<[bool; 28]> {
37        self.spread_d.value().map(|v| v.0)
38    }
39
40    fn xor_lower_sigma_0(&self) -> Value<[bool; 64]> {
41        self.spread_a()
42            .zip(self.spread_b())
43            .zip(self.spread_c())
44            .zip(self.spread_d())
45            .map(|(((a, b), c), d)| {
46                let xor_0 = b
47                    .iter()
48                    .chain(c.iter())
49                    .chain(d.iter())
50                    .chain(std::iter::repeat(&false).take(6))
51                    .copied()
52                    .collect::<Vec<_>>();
53                let xor_1 = c
54                    .iter()
55                    .chain(d.iter())
56                    .chain(a.iter())
57                    .chain(b.iter())
58                    .copied()
59                    .collect::<Vec<_>>();
60                let xor_2 = d
61                    .iter()
62                    .chain(a.iter())
63                    .chain(b.iter())
64                    .chain(c.iter())
65                    .copied()
66                    .collect::<Vec<_>>();
67
68                let xor_0 = lebs2ip::<64>(&xor_0.try_into().unwrap());
69                let xor_1 = lebs2ip::<64>(&xor_1.try_into().unwrap());
70                let xor_2 = lebs2ip::<64>(&xor_2.try_into().unwrap());
71
72                i2lebsp(xor_0 + xor_1 + xor_2)
73            })
74    }
75}

77impl MessageScheduleConfig {
78    pub fn assign_subregion1(
79        &self,
80        region: &mut Region<'_, pallas::Base>,
81        input: &[BlockWord],
82    ) -> Result<Vec<(AssignedBits<16>, AssignedBits<16>)>, Error> {
83        assert_eq!(input.len(), SUBREGION_1_LEN);
84        Ok(input
85            .iter()
86            .enumerate()
87            .map(|(idx, word)| {
88                // s_decompose_1 on W_[1..14]
89                let subregion1_word = self
90                    .decompose_subregion1_word(
91                        region,
92                        word.0.map(|word| i2lebsp(word.into())),
93                        idx + 1,
94                    )
95                    .unwrap();
96
97                // lower_sigma_0 on W_[1..14]
98                self.lower_sigma_0(region, subregion1_word).unwrap()
99            })
100            .collect::<Vec<_>>())
101    }

103    /// Pieces of length [3, 4, 11, 14]
104    fn decompose_subregion1_word(
105        &self,
106        region: &mut Region<'_, pallas::Base>,
107        word: Value<[bool; 32]>,
108        index: usize,
109    ) -> Result<Subregion1Word, Error> {
110        let row = get_word_row(index);
111
112        // Rename these here for ease of matching the gates to the specification.
113        let a_3 = self.extras[0];
114        let a_4 = self.extras[1];
115
116        let pieces = word.map(|word| {
117            vec![
118                word[0..3].to_vec(),
119                word[3..7].to_vec(),
120                word[7..18].to_vec(),
121                word[18..32].to_vec(),
122            ]
123        });
124        let pieces = pieces.transpose_vec(4);
125
126        // Assign `a` (3-bit piece)
127        let a =
128            AssignedBits::<3>::assign_bits(region, || "word_a", a_3, row + 1, pieces[0].clone())?;
129        // Assign `b` (4-bit piece)
130        let b =
131            AssignedBits::<4>::assign_bits(region, || "word_b", a_4, row + 1, pieces[1].clone())?;
132
133        // Assign `c` (11-bit piece) lookup
134        let spread_c = pieces[2].clone().map(SpreadWord::try_new);
135        let spread_c = SpreadVar::with_lookup(region, &self.lookup, row + 1, spread_c)?;
136
137        // Assign `d` (14-bit piece) lookup
138        let spread_d = pieces[3].clone().map(SpreadWord::try_new);
139        let spread_d = SpreadVar::with_lookup(region, &self.lookup, row, spread_d)?;
140
141        Ok(Subregion1Word {
142            index,
143            a,
144            b,
145            _c: spread_c.dense,
146            _d: spread_d.dense,
147            spread_c: spread_c.spread,
148            spread_d: spread_d.spread,
149        })
150    }

152    // sigma_0 v1 on a word in W_1 to W_13
153    // (3, 4, 11, 14)-bit chunks
154    fn lower_sigma_0(
155        &self,
156        region: &mut Region<'_, pallas::Base>,
157        word: Subregion1Word,
158    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
159        let a_3 = self.extras[0];
160        let a_4 = self.extras[1];
161        let a_5 = self.message_schedule;
162        let a_6 = self.extras[2];
163
164        let row = get_word_row(word.index) + 3;
165
166        // Assign `a` and copy constraint
167        word.a.copy_advice(|| "a", region, a_5, row + 1)?;
168
169        // Witness `spread_a`
170        let spread_a = word.a.value().map(|bits| spread_bits(bits.0));
171        AssignedBits::<6>::assign_bits(region, || "spread_a", a_6, row + 1, spread_a)?;
172
173        // Split `b` (4-bit chunk) into `b_hi` and `b_lo`
174        // Assign `b_lo`, `spread_b_lo`
175        let b_lo: Value<[bool; 2]> = word.b.value().map(|b| b.0[..2].try_into().unwrap());
176        let spread_b_lo = b_lo.map(spread_bits);
177        {
178            AssignedBits::<2>::assign_bits(region, || "b_lo", a_3, row - 1, b_lo)?;
179
180            AssignedBits::<4>::assign_bits(region, || "spread_b_lo", a_4, row - 1, spread_b_lo)?;
181        };
182
183        // Split `b` (2-bit chunk) into `b_hi` and `b_lo`
184        // Assign `b_hi`, `spread_b_hi`
185        let b_hi: Value<[bool; 2]> = word.b.value().map(|b| b.0[2..].try_into().unwrap());
186        let spread_b_hi = b_hi.map(spread_bits);
187        {
188            AssignedBits::<2>::assign_bits(region, || "b_hi", a_5, row - 1, b_hi)?;
189
190            AssignedBits::<4>::assign_bits(region, || "spread_b_hi", a_6, row - 1, spread_b_hi)?;
191        };
192
193        // Assign `b` and copy constraint
194        word.b.copy_advice(|| "b", region, a_6, row)?;
195
196        // Assign `spread_c` and copy constraint
197        word.spread_c.copy_advice(|| "spread_c", region, a_4, row)?;
198
199        // Assign `spread_d` and copy constraint
200        word.spread_d.copy_advice(|| "spread_d", region, a_5, row)?;
201
202        // Calculate R_0^{even}, R_0^{odd}, R_1^{even}, R_1^{odd}
203        let r = word.xor_lower_sigma_0();
204        let r_0: Value<[bool; 32]> = r.map(|r| r[..32].try_into().unwrap());
205        let r_0_even = r_0.map(even_bits);
206        let r_0_odd = r_0.map(odd_bits);
207
208        let r_1: Value<[bool; 32]> = r.map(|r| r[32..].try_into().unwrap());
209        let r_1_even = r_1.map(even_bits);
210        let r_1_odd = r_1.map(odd_bits);
211
212        self.assign_sigma_outputs(
213            region,
214            &self.lookup,
215            a_3,
216            row,
217            r_0_even,
218            r_0_odd,
219            r_1_even,
220            r_1_odd,
221        )
222    }
