10// A word in subregion 3
11// (10, 7, 2, 13)-bit chunks
12pub struct Subregion3Word {
13    index: usize,
14    #[allow(dead_code)]
15    a: AssignedBits<10>,
16    b: AssignedBits<7>,
17    c: AssignedBits<2>,
18    #[allow(dead_code)]
19    d: AssignedBits<13>,
20    spread_a: AssignedBits<20>,
21    spread_d: AssignedBits<26>,
22}
23
24impl Subregion3Word {
25    fn spread_a(&self) -> Value<[bool; 20]> {
26        self.spread_a.value().map(|v| v.0)
27    }
28
29    fn spread_b(&self) -> Value<[bool; 14]> {
30        self.b.value().map(|v| v.spread())
31    }
32
33    fn spread_c(&self) -> Value<[bool; 4]> {
34        self.c.value().map(|v| v.spread())
35    }
36
37    fn spread_d(&self) -> Value<[bool; 26]> {
38        self.spread_d.value().map(|v| v.0)
39    }
40
41    fn xor_lower_sigma_1(&self) -> Value<[bool; 64]> {
42        self.spread_a()
43            .zip(self.spread_b())
44            .zip(self.spread_c())
45            .zip(self.spread_d())
46            .map(|(((a, b), c), d)| {
47                let xor_0 = b
48                    .iter()
49                    .chain(c.iter())
50                    .chain(d.iter())
51                    .chain(std::iter::repeat(&false).take(20))
52                    .copied()
53                    .collect::<Vec<_>>();
54
55                let xor_1 = c
56                    .iter()
57                    .chain(d.iter())
58                    .chain(a.iter())
59                    .chain(b.iter())
60                    .copied()
61                    .collect::<Vec<_>>();
62                let xor_2 = d
63                    .iter()
64                    .chain(a.iter())
65                    .chain(b.iter())
66                    .chain(c.iter())
67                    .copied()
68                    .collect::<Vec<_>>();
69
70                let xor_0 = lebs2ip::<64>(&xor_0.try_into().unwrap());
71                let xor_1 = lebs2ip::<64>(&xor_1.try_into().unwrap());
72                let xor_2 = lebs2ip::<64>(&xor_2.try_into().unwrap());
73
74                i2lebsp(xor_0 + xor_1 + xor_2)
75            })
76    }
77}

79impl MessageScheduleConfig {
80    // W_[49..62]
81    pub fn assign_subregion3(
82        &self,
83        region: &mut Region<'_, pallas::Base>,
84        lower_sigma_0_v2_output: Vec<(AssignedBits<16>, AssignedBits<16>)>,
85        w: &mut Vec<MessageWord>,
86        w_halves: &mut Vec<(AssignedBits<16>, AssignedBits<16>)>,
87    ) -> Result<(), Error> {
88        let a_5 = self.message_schedule;
89        let a_6 = self.extras[2];
90        let a_7 = self.extras[3];
91        let a_8 = self.extras[4];
92        let a_9 = self.extras[5];
93
94        // Closure to compose new word
95        // W_i = sigma_1(W_{i - 2}) + W_{i - 7} + sigma_0(W_{i - 15}) + W_{i - 16}
96        // e.g. W_51 = sigma_1(W_49) + W_44 + sigma_0(W_36) + W_35
97
98        // sigma_0_v2(W_[36..49]) will be used to get the new W_[51..64]
99        // sigma_1(W_[49..62]) will also be used to get the W_[51..64]
100        // The lowest-index words involved will be W_[35..58]
101        let mut new_word = |idx: usize| -> Result<(), Error> {
102            // Decompose word into (10, 7, 2, 13)-bit chunks
103            let subregion3_word = self.decompose_subregion3_word(region, w[idx].value(), idx)?;
104
105            // sigma_1 on subregion3_word
106            let (r_0_even, r_1_even) = self.lower_sigma_1(region, subregion3_word)?;
107
108            let new_word_idx = idx + 2;
109
110            // Copy sigma_0_v2(W_{i - 15}) output from Subregion 2
111            lower_sigma_0_v2_output[idx - 49].0.copy_advice(
112                || format!("sigma_0(W_{})_lo", new_word_idx - 15),
113                region,
114                a_6,
115                get_word_row(new_word_idx - 16),
116            )?;
117            lower_sigma_0_v2_output[idx - 49].1.copy_advice(
118                || format!("sigma_0(W_{})_hi", new_word_idx - 15),
119                region,
120                a_6,
121                get_word_row(new_word_idx - 16) + 1,
122            )?;
123
124            // Copy sigma_1(W_{i - 2})
125            r_0_even.copy_advice(
126                || format!("sigma_1(W_{})_lo", new_word_idx - 2),
127                region,
128                a_7,
129                get_word_row(new_word_idx - 16),
130            )?;
131            r_1_even.copy_advice(
132                || format!("sigma_1(W_{})_hi", new_word_idx - 2),
133                region,
134                a_7,
135                get_word_row(new_word_idx - 16) + 1,
136            )?;
137
138            // Copy W_{i - 7}
139            w_halves[new_word_idx - 7].0.copy_advice(
140                || format!("W_{}_lo", new_word_idx - 7),
141                region,
142                a_8,
143                get_word_row(new_word_idx - 16),
144            )?;
145            w_halves[new_word_idx - 7].1.copy_advice(
146                || format!("W_{}_hi", new_word_idx - 7),
147                region,
148                a_8,
149                get_word_row(new_word_idx - 16) + 1,
150            )?;
151
152            // Calculate W_i, carry_i
153            let (word, carry) = sum_with_carry(vec![
154                (r_0_even.value_u16(), r_1_even.value_u16()),
155                (
156                    w_halves[new_word_idx - 7].0.value_u16(),
157                    w_halves[new_word_idx - 7].1.value_u16(),
158                ),
159                (
160                    lower_sigma_0_v2_output[idx - 49].0.value_u16(),
161                    lower_sigma_0_v2_output[idx - 49].1.value_u16(),
162                ),
163                (
164                    w_halves[new_word_idx - 16].0.value_u16(),
165                    w_halves[new_word_idx - 16].1.value_u16(),
166                ),
167            ]);
168
169            // Assign W_i, carry_i
170            region.assign_advice(
171                || format!("W_{}", new_word_idx),
172                a_5,
173                get_word_row(new_word_idx - 16) + 1,
174                || word.map(|word| pallas::Base::from(word as u64)),
175            )?;
176            region.assign_advice(
177                || format!("carry_{}", new_word_idx),
178                a_9,
179                get_word_row(new_word_idx - 16) + 1,
180                || carry.map(pallas::Base::from),
181            )?;
182            let (word, halves) = self.assign_word_and_halves(region, word, new_word_idx)?;
183            w.push(MessageWord(word));
184            w_halves.push(halves);
185
186            Ok(())
187        };
188
189        for i in 49..62 {
190            new_word(i)?;
191        }
192
193        Ok(())
194    }

196    /// Pieces of length [10, 7, 2, 13]
197    fn decompose_subregion3_word(
198        &self,
199        region: &mut Region<'_, pallas::Base>,
200        word: Value<&Bits<32>>,
201        index: usize,
202    ) -> Result<Subregion3Word, Error> {
203        let row = get_word_row(index);
204
205        // Rename these here for ease of matching the gates to the specification.
206        let a_3 = self.extras[0];
207        let a_4 = self.extras[1];
208
209        let pieces = word.map(|word| {
210            vec![
211                word[0..10].to_vec(),
212                word[10..17].to_vec(),
213                word[17..19].to_vec(),
214                word[19..32].to_vec(),
215            ]
216        });
217        let pieces = pieces.transpose_vec(4);
218
219        // Assign `a` (10-bit piece)
220        let spread_a = pieces[0].clone().map(SpreadWord::try_new);
221        let spread_a = SpreadVar::with_lookup(region, &self.lookup, row + 1, spread_a)?;
222
223        // Assign `b` (7-bit piece)
224        let b = AssignedBits::<7>::assign_bits(region, || "b", a_4, row + 1, pieces[1].clone())?;
225
226        // Assign `c` (2-bit piece)
227        let c = AssignedBits::<2>::assign_bits(region, || "c", a_3, row + 1, pieces[2].clone())?;
228
229        // Assign `d` (13-bit piece) lookup
230        let spread_d = pieces[3].clone().map(SpreadWord::try_new);
231        let spread_d = SpreadVar::with_lookup(region, &self.lookup, row, spread_d)?;
232
233        Ok(Subregion3Word {
234            index,
235            a: spread_a.dense,
236            b,
237            c,
238            d: spread_d.dense,
239            spread_a: spread_a.spread,
240            spread_d: spread_d.spread,
241        })
242    }

244    fn lower_sigma_1(
245        &self,
246        region: &mut Region<'_, pallas::Base>,
247        word: Subregion3Word,
248    ) -> Result<(AssignedBits<16>, AssignedBits<16>), Error> {
249        let a_3 = self.extras[0];
250        let a_4 = self.extras[1];
251        let a_5 = self.message_schedule;
252        let a_6 = self.extras[2];
253
254        let row = get_word_row(word.index) + 3;
255
256        // Assign `spread_a` and copy constraint
257        word.spread_a.copy_advice(|| "spread_a", region, a_4, row)?;
258
259        // Split `b` (7-bit chunk) into (2, 2, 3)-bit `b_lo`, `b_mid` and `b_hi`.
260        // Assign `b_lo`, `spread_b_lo`, `b_mid`, `spread_b_mid`, `b_hi`, `spread_b_hi`.
261
262        // b_lo (2-bit chunk)
263        {
264            let b_lo: Value<[bool; 2]> = word.b.value().map(|v| v[0..2].try_into().unwrap());
265            let b_lo = b_lo.map(SpreadWord::<2, 4>::new);
266            SpreadVar::without_lookup(region, a_3, row - 1, a_4, row - 1, b_lo)?;
267        }
268
269        // b_mid (2-bit chunk)
270        {
271            let b_mid: Value<[bool; 2]> = word.b.value().map(|v| v[2..4].try_into().unwrap());
272            let b_mid = b_mid.map(SpreadWord::<2, 4>::new);
273            SpreadVar::without_lookup(region, a_5, row - 1, a_6, row - 1, b_mid)?;
274        }
275
276        // b_hi (3-bit chunk)
277        {
278            let b_hi: Value<[bool; 3]> = word.b.value().map(|v| v[4..7].try_into().unwrap());
279            let b_hi = b_hi.map(SpreadWord::<3, 6>::new);
280            SpreadVar::without_lookup(region, a_5, row + 1, a_6, row + 1, b_hi)?;
281        }
282
283        // Assign `b` and copy constraint
284        word.b.copy_advice(|| "b", region, a_6, row)?;
285
286        // Assign `c` and copy constraint
287        word.c.copy_advice(|| "c", region, a_3, row + 1)?;
288
289        // Witness `spread_c`
290        {
291            let spread_c = word.c.value().map(spread_bits);
292            AssignedBits::<4>::assign_bits(region, || "spread_c", a_4, row + 1, spread_c)?;
293        }
294
295        // Assign `spread_d` and copy constraint
296        word.spread_d.copy_advice(|| "spread_d", region, a_5, row)?;
297
298        // (10, 7, 2, 13)
299        // Calculate R_0^{even}, R_0^{odd}, R_1^{even}, R_1^{odd}
300        let r = word.xor_lower_sigma_1();
301        let r_0: Value<[bool; 32]> = r.map(|r| r[..32].try_into().unwrap());
302        let r_0_even = r_0.map(even_bits);
303        let r_0_odd = r_0.map(odd_bits);
304
305        let r_1: Value<[bool; 32]> = r.map(|r| r[32..].try_into().unwrap());
306        let r_1_even = r_1.map(even_bits);
307        let r_1_odd = r_1.map(odd_bits);
308
309        self.assign_sigma_outputs(
310            region,
311            &self.lookup,
312            a_3,
313            row,
314            r_0_even,
315            r_0_odd,
316            r_1_even,
317            r_1_odd,
318        )
319    }
