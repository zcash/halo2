37/// Returns row number of a word
38pub fn get_word_row(word_idx: usize) -> usize {
39    assert!(word_idx <= 63);
40    if word_idx == 0 {
41        0
42    } else if (1..=13).contains(&word_idx) {
43        SUBREGION_0_ROWS + SUBREGION_1_WORD * (word_idx - 1)
44    } else if (14..=48).contains(&word_idx) {
45        SUBREGION_0_ROWS + SUBREGION_1_ROWS + SUBREGION_2_WORD * (word_idx - 14) + 1
46    } else if (49..=61).contains(&word_idx) {
47        SUBREGION_0_ROWS + SUBREGION_1_ROWS + SUBREGION_2_ROWS + SUBREGION_3_WORD * (word_idx - 49)
48    } else {
49        SUBREGION_0_ROWS
50            + SUBREGION_1_ROWS
51            + SUBREGION_2_ROWS
52            + SUBREGION_3_ROWS
53            + DECOMPOSE_0_ROWS * (word_idx - 62)
54    }
55}

148impl MessageScheduleConfig {
149    // Assign a word and its hi and lo halves
150    pub fn assign_word_and_halves(
151        &self,
152        region: &mut Region<'_, pallas::Base>,
153        word: Value<u32>,
154        word_idx: usize,
155    ) -> Result<(AssignedBits<32>, (AssignedBits<16>, AssignedBits<16>)), Error> {
156        // Rename these here for ease of matching the gates to the specification.
157        let a_3 = self.extras[0];
158        let a_4 = self.extras[1];
159
160        let row = get_word_row(word_idx);
161
162        let w_lo = {
163            let w_lo_val = word.map(|word| word as u16);
164            AssignedBits::<16>::assign(region, || format!("W_{}_lo", word_idx), a_3, row, w_lo_val)?
165        };
166        let w_hi = {
167            let w_hi_val = word.map(|word| (word >> 16) as u16);
168            AssignedBits::<16>::assign(region, || format!("W_{}_hi", word_idx), a_4, row, w_hi_val)?
169        };
170
171        let word = AssignedBits::<32>::assign(
172            region,
173            || format!("W_{}", word_idx),
174            self.message_schedule,
175            row,
176            word,
177        )?;
178
179        Ok((word, (w_lo, w_hi)))
180    }
181}
