7pub(super) fn generate_mds<F: FromUniformBytes<64> + Ord, const T: usize>(
8    grain: &mut Grain<F>,
9    mut select: usize,
10) -> (Mds<F, T>, Mds<F, T>) {
11    let (xs, ys, mds) = loop {
12        // Generate two [F; T] arrays of unique field elements.
13        let (xs, ys) = loop {
14            let mut vals: Vec<_> = (0..2 * T)
15                .map(|_| grain.next_field_element_without_rejection())
16                .collect();
17
18            // Check that we have unique field elements.
19            let mut unique = vals.clone();
20            unique.sort_unstable();
21            unique.dedup();
22            if vals.len() == unique.len() {
23                let rhs = vals.split_off(T);
24                break (vals, rhs);
25            }
26        };
27
28        // We need to ensure that the MDS is secure. Instead of checking the MDS against
29        // the relevant algorithms directly, we witness a fixed number of MDS matrices
30        // that we need to sample from the given Grain state before obtaining a secure
31        // matrix. This can be determined out-of-band via the reference implementation in
32        // Sage.
33        if select != 0 {
34            select -= 1;
35            continue;
36        }
37
38        // Generate a Cauchy matrix, with elements a_ij in the form:
39        //     a_ij = 1/(x_i + y_j); x_i + y_j != 0
40        //
41        // It would be much easier to use the alternate definition:
42        //     a_ij = 1/(x_i - y_j); x_i - y_j != 0
43        //
44        // These are clearly equivalent on `y <- -y`, but it is easier to work with the
45        // negative formulation, because ensuring that xs âˆª ys is unique implies that
46        // x_i - y_j != 0 by construction (whereas the positive case does not hold). It
47        // also makes computation of the matrix inverse simpler below (the theorem used
48        // was formulated for the negative definition).
49        //
50        // However, the Poseidon paper and reference impl use the positive formulation,
51        // and we want to rely on the reference impl for MDS security, so we use the same
52        // formulation.
53        let mut mds = [[F::ZERO; T]; T];
54        #[allow(clippy::needless_range_loop)]
55        for i in 0..T {
56            for j in 0..T {
57                let sum = xs[i] + ys[j];
58                // We leverage the secure MDS selection counter to also check this.
59                assert!(!sum.is_zero_vartime());
60                mds[i][j] = sum.invert().unwrap();
61            }
62        }
63
64        break (xs, ys, mds);
65    };
66
67    // Compute the inverse. All square Cauchy matrices have a non-zero determinant and
68    // thus are invertible. The inverse for a Cauchy matrix of the form:
69    //
70    //     a_ij = 1/(x_i - y_j); x_i - y_j != 0
71    //
72    // has elements b_ij given by:
73    //
74    //     b_ij = (x_j - y_i) A_j(y_i) B_i(x_j)    (Schechter 1959, Theorem 1)
75    //
76    // where A_i(x) and B_i(x) are the Lagrange polynomials for xs and ys respectively.
77    //
78    // We adapt this to the positive Cauchy formulation by negating ys.
79    let mut mds_inv = [[F::ZERO; T]; T];
80    let l = |xs: &[F], j, x: F| {
81        let x_j = xs[j];
82        xs.iter().enumerate().fold(F::ONE, |acc, (m, x_m)| {
83            if m == j {
84                acc
85            } else {
86                // We hard-code the type, to avoid spurious "cannot infer type" rustc errors.
87                let denominator: F = x_j - x_m;
88
89                // We can invert freely; by construction, the elements of xs are distinct.
90                let denominator_inverted: F = denominator.invert().unwrap();
91
92                acc * (x - x_m) * denominator_inverted
93            }
94        })
95    };
96    let neg_ys: Vec<_> = ys.iter().map(|y| -*y).collect();
97    for i in 0..T {
98        for j in 0..T {
99            mds_inv[i][j] = (xs[j] - neg_ys[i]) * l(&xs, j, neg_ys[i]) * l(&neg_ys, i, xs[j]);
100        }
101    }
102
103    (mds, mds_inv)
104}

113    #[test]
114    fn poseidon_mds() {
115        const T: usize = 3;
116        let mut grain = Grain::new(super::super::grain::SboxType::Pow, T as u16, 8, 56);
117        let (mds, mds_inv) = generate_mds::<Fp, T>(&mut grain, 0);
118
119        // Verify that MDS * MDS^-1 = I.
120        #[allow(clippy::needless_range_loop)]
121        for i in 0..T {
122            for j in 0..T {
123                let expected = if i == j { Fp::ONE } else { Fp::ZERO };
124                assert_eq!(
125                    (0..T).fold(Fp::ZERO, |acc, k| acc + (mds[i][k] * mds_inv[k][j])),
126                    expected
127                );
128            }
129        }
130    }
